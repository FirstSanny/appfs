24 def rate(hit, num):
25  """Return the fraction of `hit`/`num`, as a string."""
26  if num == 0:
27  return "1"
28  else:
29  return "%.4g" % (float(hit) / num)

47  def report(self, morfs, outfile=None):
48  """Generate a Cobertura-compatible XML report for `morfs`.
49 
50  `morfs` is a list of modules or file names.
51 
52  `outfile` is a file object to write the XML to.
53 
54  """
55  # Initial setup.
56  outfile = outfile or sys.stdout
57 
58  # Create the DOM that will store the data.
59  impl = xml.dom.minidom.getDOMImplementation()
60  self.xml_out = impl.createDocument(None, "coverage", None)
61 
62  # Write header stuff.
63  xcoverage = self.xml_out.documentElement
64  xcoverage.setAttribute("version", __version__)
65  xcoverage.setAttribute("timestamp", str(int(time.time()*1000)))
66  xcoverage.appendChild(self.xml_out.createComment(
67  " Generated by coverage.py: %s " % __url__
68  ))
69  xcoverage.appendChild(self.xml_out.createComment(" Based on %s " % DTD_URL))
70 
71  # Call xml_file for each file in the data.
72  self.report_files(self.xml_file, morfs)
73 
74  xsources = self.xml_out.createElement("sources")
75  xcoverage.appendChild(xsources)
76 
77  # Populate the XML DOM with the source info.
78  for path in sorted(self.source_paths):
79  xsource = self.xml_out.createElement("source")
80  xsources.appendChild(xsource)
81  txt = self.xml_out.createTextNode(path)
82  xsource.appendChild(txt)
83 
84  lnum_tot, lhits_tot = 0, 0
85  bnum_tot, bhits_tot = 0, 0
86 
87  xpackages = self.xml_out.createElement("packages")
88  xcoverage.appendChild(xpackages)
89 
90  # Populate the XML DOM with the package info.
91  for pkg_name, pkg_data in sorted(iitems(self.packages)):
92  class_elts, lhits, lnum, bhits, bnum = pkg_data
93  xpackage = self.xml_out.createElement("package")
94  xpackages.appendChild(xpackage)
95  xclasses = self.xml_out.createElement("classes")
96  xpackage.appendChild(xclasses)
97  for _, class_elt in sorted(iitems(class_elts)):
98  xclasses.appendChild(class_elt)
99  xpackage.setAttribute("name", pkg_name.replace(os.sep, '.'))
100  xpackage.setAttribute("line-rate", rate(lhits, lnum))
101  if self.has_arcs:
102  branch_rate = rate(bhits, bnum)
103  else:
104  branch_rate = "0"
105  xpackage.setAttribute("branch-rate", branch_rate)
106  xpackage.setAttribute("complexity", "0")
107 
108  lnum_tot += lnum
109  lhits_tot += lhits
110  bnum_tot += bnum
111  bhits_tot += bhits
112 
113  xcoverage.setAttribute("lines-valid", str(lnum_tot))
114  xcoverage.setAttribute("lines-covered", str(lhits_tot))
115  xcoverage.setAttribute("line-rate", rate(lhits_tot, lnum_tot))
116  if self.has_arcs:

118  xcoverage.setAttribute("branches-covered", str(bhits_tot))
119  xcoverage.setAttribute("branch-rate", rate(bhits_tot, bnum_tot))
120  else:
121  xcoverage.setAttribute("branches-covered", "0")
122  xcoverage.setAttribute("branches-valid", "0")
123  xcoverage.setAttribute("branch-rate", "0")
124  xcoverage.setAttribute("complexity", "0")
125 
126  # Use the DOM to write the output file.
127  out = self.xml_out.toprettyxml()
128  if env.PY2:
129  out = out.encode("utf8")
130  outfile.write(out)
131

133  denom = lnum_tot + bnum_tot

140  def xml_file(self, fr, analysis):
141  """Add to the XML report for a single file."""
142 
143  # Create the 'lines' and 'package' XML elements, which
144  # are populated later. Note that a package == a directory.
145  filename = fr.filename.replace("\\", "/")
146  for source_path in self.source_paths:
147  if filename.startswith(source_path.replace("\\", "/") + "/"):
148  rel_name = filename[len(source_path)+1:]
149  break
150  else:
151  rel_name = fr.relative_filename()
152 
153  dirname = os.path.dirname(rel_name) or u"."
154  dirname = "/".join(dirname.split("/")[:self.config.xml_package_depth])
155  package_name = dirname.replace("/", ".")
156 
157  if rel_name != fr.filename:
158  self.source_paths.add(fr.filename[:-len(rel_name)].rstrip(r"\/"))
159  package = self.packages.setdefault(package_name, [{}, 0, 0, 0, 0])
160 
161  xclass = self.xml_out.createElement("class")
162 
163  xclass.appendChild(self.xml_out.createElement("methods"))
164 
165  xlines = self.xml_out.createElement("lines")
166  xclass.appendChild(xlines)
167 
168  xclass.setAttribute("name", os.path.relpath(rel_name, dirname))
169  xclass.setAttribute("filename", rel_name.replace("\\", "/"))
170  xclass.setAttribute("complexity", "0")
171 
172  branch_stats = analysis.branch_stats()
173  missing_branch_arcs = analysis.missing_branch_arcs()
174 
175  # For each statement, create an XML 'line' element.
176  for line in sorted(analysis.statements):
177  xline = self.xml_out.createElement("line")
178  xline.setAttribute("number", str(line))
179 
180  # Q: can we get info about the number of times a statement is
181  # executed? If so, that should be recorded here.
182  xline.setAttribute("hits", str(int(line not in analysis.missing)))
183 
184  if self.has_arcs:
185  if line in branch_stats:
186  total, taken = branch_stats[line]
187  xline.setAttribute("branch", "true")
188  xline.setAttribute(
189  "condition-coverage",
190  "%d%% (%d/%d)" % (100*taken//total, taken, total)
191  )
192  if line in missing_branch_arcs:
193  annlines = ["exit" if b < 0 else str(b) for b in missing_branch_arcs[line]]
194  xline.setAttribute("missing-branches", ",".join(annlines))
195  xlines.appendChild(xline)
196 
197  class_lines = len(analysis.statements)
198  class_hits = class_lines - len(analysis.missing)
199 
200  if self.has_arcs:
201  class_branches = sum(t for t, k in branch_stats.values())
202  missing_branches = sum(t - k for t, k in branch_stats.values())
203  class_br_hits = class_branches - missing_branches
204  else:
205  class_branches = 0.0
206  class_br_hits = 0.0
207 
208  # Finalize the statistics that are collected in the XML DOM.
209  xclass.setAttribute("line-rate", rate(class_hits, class_lines))
210  if self.has_arcs:
211  branch_rate = rate(class_br_hits, class_branches)
212  else:
213  branch_rate = "0"
214  xclass.setAttribute("branch-rate", branch_rate)
215 
216  package[0][rel_name] = xclass
217  package[1] += class_hits
218  package[2] += class_lines
219  package[3] += class_br_hits
220  package[4] += class_branches
