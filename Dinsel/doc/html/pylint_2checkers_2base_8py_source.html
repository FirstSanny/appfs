167 def _determine_function_name_type(node, config=None):
168  """Determine the name type whose regex the a function's name should match.
169 
170  :param node: A function node.
171  :type node: astroid.node_classes.NodeNG
172  :param config: Configuration from which to pull additional property classes.
173  :type config: :class:`optparse.Values`
174 
175  :returns: One of ('function', 'method', 'attr')
176  :rtype: str
177  """
178  property_classes, property_names = _get_properties(config)
179  if not node.is_method():
180  return 'function'
181  if node.decorators:
182  decorators = node.decorators.nodes
183  else:
184  decorators = []
185  for decorator in decorators:
186  # If the function is a property (decorated with @property

193  return 'attr'

195  # form, treat it like an attribute as well.
