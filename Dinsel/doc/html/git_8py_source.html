36  if url:

259  def get_url_rev(self):
260  """
261  Prefixes stub URLs like 'user@hostname:user/repo.git' with 'ssh://'.
262  That's required because although they use SSH they sometimes doesn't
263  work with a ssh:// scheme (e.g. Github). But we need a scheme for
264  parsing. Hence we remove it again afterwards and return it as a stub.
265  """
266  if '://' not in self.url:
267  assert 'file:' not in self.url
268  self.url = self.url.replace('git+', 'git+ssh://')
269  url, rev = super(Git, self).get_url_rev()
270  url = url.replace('ssh://', '')
271  else:
272  url, rev = super(Git, self).get_url_rev()
273 
274  return url, rev
