56 def handle_requires(metadata, pkg_info, key):
57  """
58  Place the runtime requirements from pkg_info into metadata.
59  """
60  may_requires = OrderedDefaultDict(list)
61  for value in sorted(pkg_info.get_all(key)):
62  extra_match = EXTRA_RE.search(value)
63  if extra_match:
64  groupdict = extra_match.groupdict()
65  condition = groupdict['condition']
66  extra = groupdict['extra']
67  package = groupdict['package']
68  if condition.endswith(' and '):
69  condition = condition[:-5]
70  else:
71  condition, extra = None, None
72  package = value
73  key = MayRequiresKey(condition, extra)
74  may_requires[key].append(package)
75 
76  if may_requires:
77  metadata['run_requires'] = []
78  def sort_key(item):
79  # Both condition and extra could be None, which can't be compared
80  # against strings in Python 3.
81  key, value = item
82  if key.condition is None:
83  return ''
84  return key.condition
85  for key, value in sorted(may_requires.items(), key=sort_key):
86  may_requirement = OrderedDict((('requires', value),))
87  if key.extra:
88  may_requirement['extra'] = key.extra
89  if key.condition:
90  may_requirement['environment'] = key.condition
91  metadata['run_requires'].append(may_requirement)

95  metadata['extras'].extend([key.extra for key in may_requires.keys() if key.extra])

240 def generate_requirements(extras_require):
241  """
242  Convert requirements from a setup()-style dictionary to ('Requires-Dist', 'requirement')
243  and ('Provides-Extra', 'extra') tuples.
244 
245  extras_require is a dictionary of {extra: [requirements]} as passed to setup(),
246  using the empty extra {'': [requirements]} to hold install_requires.
247  """
248  for extra, depends in extras_require.items():
249  condition = ''
250  if extra and ':' in extra: # setuptools extra:condition syntax
251  extra, condition = extra.split(':', 1)
252  extra = pkg_resources.safe_extra(extra)
253  if extra:
254  yield ('Provides-Extra', extra)
255  if condition:
256  condition += " and "
257  condition += "extra == '%s'" % extra
258  if condition:
259  condition = '; ' + condition
260  for new_req in convert_requirements(depends):
261  yield ('Requires-Dist', new_req + condition)
