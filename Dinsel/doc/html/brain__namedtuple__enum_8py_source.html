29 def infer_func_form(node, base_type, context=None, enum=False):
30  """Specific inference function for namedtuple or Python 3 enum. """
31  def infer_first(node):
32  if node is util.Uninferable:
33  raise UseInferenceDefault
34  try:
35  value = next(node.infer(context=context))
36  if value is util.Uninferable:
37  raise UseInferenceDefault()
38  else:
39  return value
40  except StopIteration:
41  raise InferenceError()
42 
43  # node is a Call node, class name as first argument and generated class
44  # attributes as second argument
45  if len(node.args) != 2:
46  # something weird here, go back to class implementation
47  raise UseInferenceDefault()
48  # namedtuple or enums list of attributes can be a list of strings or a
49  # whitespace-separate string
50  try:
51  name = infer_first(node.args[0]).value
52  names = infer_first(node.args[1])
53  try:
54  attributes = names.value.replace(',', ' ').split()
55  except AttributeError:
56  if not enum:
57  attributes = [infer_first(const).value for const in names.elts]
58  else:
59  # Enums supports either iterator of (name, value) pairs
60  # or mappings.
61  # TODO: support only list, tuples and mappings.
62  if hasattr(names, 'items') and isinstance(names.items, list):
63  attributes = [infer_first(const[0]).value
64  for const in names.items
65  if isinstance(const[0], nodes.Const)]
66  elif hasattr(names, 'elts'):
67  # Enums can support either ["a", "b", "c"]
68  # or [("a", 1), ("b", 2), ...], but they can't
69  # be mixed.
70  if all(isinstance(const, nodes.Tuple)
71  for const in names.elts):
72  attributes = [infer_first(const.elts[0]).value
73  for const in names.elts
74  if isinstance(const, nodes.Tuple)]
75  else:
76  attributes = [infer_first(const).value
77  for const in names.elts]
78  else:
79  raise AttributeError
80  if not attributes:
81  raise AttributeError
82  except (AttributeError, exceptions.InferenceError):
83  raise UseInferenceDefault()
84 
85  # If we can't infer the name of the class, don't crash, up to this point
86  # we know it is a namedtuple anyway.
87  name = name or 'Uninferable'
88  # we want to return a Class node instance with proper attributes set
89  class_node = nodes.ClassDef(name, 'docstring')
90  class_node.parent = node.parent
91  # set base class=tuple
92  class_node.bases.append(base_type)
93  # XXX add __init__(*attributes) method
94  for attr in attributes:
95  fake_node = nodes.EmptyNode()
96  fake_node.parent = class_node
97  fake_node.attrname = attr
98  class_node.instance_attrs[attr] = [fake_node]
99  return class_node, name, attributes

114 def infer_named_tuple(node, context=None):
115  """Specific inference function for namedtuple Call node"""
116  class_node, name, attributes = infer_func_form(node, nodes.Tuple._proxied,
117  context=context)
118  call_site = arguments.CallSite.from_call(node)
119  func = next(extract_node('import collections; collections.namedtuple').infer())
120  try:
121  rename = next(call_site.infer_argument(func, 'rename', context)).bool_value()
122  except InferenceError:
123  rename = False
124 
125  if rename:
126  attributes = _get_renamed_namedtuple_atributes(attributes)
127 
128  field_def = (" {name} = property(lambda self: self[{index:d}], "
129  "doc='Alias for field number {index:d}')")
130  field_defs = '\n'.join(field_def.format(name=name, index=index)
131  for index, name in enumerate(attributes))
132  fake = AstroidBuilder(MANAGER).string_build('''
133 class %(name)s(tuple):
134  __slots__ = ()
135  _fields = %(fields)r
136  def _asdict(self):
137  return self.__dict__
138  @classmethod
139  def _make(cls, iterable, new=tuple.__new__, len=len):
140  return new(cls, iterable)
141  def _replace(self, **kwds):
142  return self
143  def __getnewargs__(self):
144  return tuple(self)
145 %(field_defs)s
146  ''' % {'name': name, 'fields': attributes, 'field_defs': field_defs})
147  class_node.locals['_asdict'] = fake.body[0].locals['_asdict']
148  class_node.locals['_make'] = fake.body[0].locals['_make']
149  class_node.locals['_replace'] = fake.body[0].locals['_replace']
150  class_node.locals['_fields'] = fake.body[0].locals['_fields']
151  for attr in attributes:
152  class_node.locals[attr] = fake.body[0].locals[attr]
153  # we use UseInferenceDefault, we can't be a generator so return an iterator
154  return iter([class_node])

157 def _get_renamed_namedtuple_atributes(field_names):
158  names = list(field_names)
159  seen = set()
160  for i, name in enumerate(field_names):
161  if (not all(c.isalnum() or c == '_' for c in name) or keyword.iskeyword(name)
162  or not name or name[0].isdigit() or name.startswith('_') or name in seen):
163  names[i] = '_%d' % i
164  seen.add(name)
165  return tuple(names)

168 def infer_enum(node, context=None):
169  """ Specific inference function for enum Call node. """
170  enum_meta = extract_node('''
171  class EnumMeta(object):
172  'docstring'
173  def __call__(self, node):
174  class EnumAttribute(object):
175  name = ''
176  value = 0
177  return EnumAttribute()

184 def infer_enum_class(node):
185  """ Specific inference for enums. """
186  names = set(('Enum', 'IntEnum', 'enum.Enum', 'enum.IntEnum'))
187  for basename in node.basenames:
188  # TODO: doesn't handle subclasses yet. This implementation
189  # is a hack to support enums.
190  if basename not in names:
191  continue
192  if node.root().name == 'enum':
193  # Skip if the class is directly from enum module.
194  break
195  for local, values in node.locals.items():
196  if any(not isinstance(value, nodes.AssignName)
197  for value in values):
198  continue
199 
200  stmt = values[0].statement()
201  if isinstance(stmt.targets[0], nodes.Tuple):
202  targets = stmt.targets[0].itered()
203  else:
204  targets = stmt.targets
205 
206  new_targets = []
207  for target in targets:
208  # Replace all the assignments with our mocked class.
209  classdef = dedent('''
210  class %(name)s(%(types)s):
211  @property
212  def value(self):
213  # Not the best return.
214  return None
215  @property
216  def name(self):
217  return %(name)r
218  ''' % {'name': target.name, 'types': ', '.join(node.basenames)})
219  fake = AstroidBuilder(MANAGER).string_build(classdef)[target.name]
220  fake.parent = target.parent
221  for method in node.mymethods():
222  fake.locals[method.name] = [method]
223  new_targets.append(fake.instantiate_class())
224  node.locals[local] = new_targets
225  break
226  return node
