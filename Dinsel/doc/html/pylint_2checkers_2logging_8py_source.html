65 def is_method_call(func, types=(), methods=()):
66  """Determines if a BoundMethod node represents a method call.
67 
68  Args:
69  func (astroid.BoundMethod): The BoundMethod AST node to check.
70  types (Optional[String]): Optional sequence of caller type names to restrict check.
71  methods (Optional[String]): Optional sequence of method names to restrict check.
72 
73  Returns:
74  bool: true if the node represents a method call for the given type and
75  method names, False otherwise.
76  """
77  return (isinstance(func, astroid.BoundMethod)
78  and isinstance(func.bound, astroid.Instance)
79  and (func.bound.name in types if types else True)
80  and (func.name in methods if methods else True))

100  """Clears any state left in this checker from last module checked."""

160  def _check_log_method(self, node, name):
161  """Checks calls to logging.log(level, format, *format_args)."""
162  if name == 'log':
163  if node.starargs or node.kwargs or len(node.args) < 2:
164  # Either a malformed call, star args, or double-star args. Beyond
165  # the scope of this checker.
166  return
167  format_pos = 1
168  elif name in CHECKED_CONVENIENCE_FUNCTIONS:
169  if node.starargs or node.kwargs or not node.args:
170  # Either no args, star args, or double-star args. Beyond the
171  # scope of this checker.
172  return
173  format_pos = 0
174  else:
175  return
176 
177  if isinstance(node.args[format_pos], astroid.BinOp) and node.args[format_pos].op == '%':
178  self.add_message('logging-not-lazy', node=node)
179  elif isinstance(node.args[format_pos], astroid.Call):
180  self._check_call_func(node.args[format_pos])
181  elif isinstance(node.args[format_pos], astroid.Const):

184  def _check_call_func(self, node):
185  """Checks that function call is not format_string.format().
186 
187  Args:
188  node (astroid.node_classes.CallFunc):
189  CallFunc AST node to be checked.
190  """

192  types = ('str', 'unicode')

197  def _check_format_string(self, node, format_arg):
198  """Checks that format string tokens match the supplied arguments.
199 
200  Args:
201  node (astroid.node_classes.NodeNG): AST node to be checked.
202  format_arg (int): Index of the format string in the node arguments.
203  """
204  num_args = _count_supplied_tokens(node.args[format_arg + 1:])
205  if not num_args:
206  # If no args were supplied, then all format strings are valid -
207  # don't check any further.
208  return
209  format_string = node.args[format_arg].value
210  if not isinstance(format_string, six.string_types):
211  # If the log format is constant non-string (e.g. logging.debug(5)),
212  # ensure there are no arguments.
213  required_num_args = 0
214  else:
215  try:
216  keyword_args, required_num_args = \
217  utils.parse_format_string(format_string)
218  if keyword_args:
219  # Keyword checking on logging strings is complicated by
220  # special keywords - out of scope.
221  return
222  except utils.UnsupportedFormatCharacter as ex:
223  char = format_string[ex.index]
224  self.add_message('logging-unsupported-format', node=node,
225  args=(char, ord(char), ex.index))
226  return
227  except utils.IncompleteFormatString:
228  self.add_message('logging-format-truncated', node=node)
229  return
230  if num_args > required_num_args:
231  self.add_message('logging-too-many-args', node=node)
232  elif num_args < required_num_args:
233  self.add_message('logging-too-few-args', node=node)

236 def is_complex_format_str(node):
237  """Checks if node represents a string with complex formatting specs.
238 
239  Args:
240  node (astroid.node_classes.NodeNG): AST node to check
241  Returns:
242  bool: True if inferred string uses complex formatting, False otherwise
243  """
244  inferred = utils.safe_infer(node)
245  if inferred is None or not isinstance(inferred.value, six.string_types):
246  return True
247  for _, _, format_spec, _ in string.Formatter().parse(inferred.value):
248  if format_spec:
249  return True
250  return False

253 def _count_supplied_tokens(args):
254  """Counts the number of tokens in an args list.
255 
256  The Python log functions allow for special keyword arguments: func,
257  exc_info and extra. To handle these cases correctly, we only count
258  arguments that aren't keywords.
259 
260  Args:
261  args (list): AST nodes that are arguments for a log format string.
262 
263  Returns:
264  int: Number of AST nodes that aren't keywords.
265  """
266  return sum(1 for arg in args if not isinstance(arg, astroid.Keyword))
