50 def _eigvalsh_to_eps(spectrum, cond=None, rcond=None):
51  """
52  Determine which eigenvalues are "small" given the spectrum.
53 
54  This is for compatibility across various linear algebra functions
55  that should agree about whether or not a Hermitian matrix is numerically
56  singular and what is its numerical matrix rank.
57  This is designed to be compatible with scipy.linalg.pinvh.
58 
59  Parameters
60  ----------
61  spectrum : 1d ndarray
62  Array of eigenvalues of a Hermitian matrix.
63  cond, rcond : float, optional
64  Cutoff for small eigenvalues.
65  Singular values smaller than rcond * largest_eigenvalue are
66  considered zero.
67  If None or -1, suitable machine precision is used.
68 
69  Returns
70  -------
71  eps : float
72  Magnitude cutoff for numerical negligibility.
73 
74  """
75  if rcond is not None:
76  cond = rcond
77  if cond in [None, -1]:
78  t = spectrum.dtype.char.lower()
79  factor = {'f': 1E3, 'd': 1E6}
80  cond = factor[t] * np.finfo(t).eps
81  eps = cond * np.max(abs(spectrum))
82  return eps

105 class _PSD(object):
106  """
107  Compute coordinated functions of a symmetric positive semidefinite matrix.
108 
109  This class addresses two issues. Firstly it allows the pseudoinverse,
110  the logarithm of the pseudo-determinant, and the rank of the matrix
111  to be computed using one call to eigh instead of three.
112  Secondly it allows these functions to be computed in a way
113  that gives mutually compatible results.
114  All of the functions are computed with a common understanding as to
115  which of the eigenvalues are to be considered negligibly small.
116  The functions are designed to coordinate with scipy.linalg.pinvh()
117  but not necessarily with np.linalg.det() or with np.linalg.matrix_rank().
118 
119  Parameters
120  ----------
121  M : array_like
122  Symmetric positive semidefinite matrix (2-D).
123  cond, rcond : float, optional
124  Cutoff for small eigenvalues.
125  Singular values smaller than rcond * largest_eigenvalue are
126  considered zero.
127  If None or -1, suitable machine precision is used.
128  lower : bool, optional
129  Whether the pertinent array data is taken from the lower
130  or upper triangle of M. (Default: lower)
131  check_finite : bool, optional
132  Whether to check that the input matrices contain only finite
133  numbers. Disabling may give a performance gain, but may result
134  in problems (crashes, non-termination) if the inputs do contain
135  infinities or NaNs.
136  allow_singular : bool, optional
137  Whether to allow a singular matrix. (Default: True)
138 
139  Notes
140  -----
141  The arguments are similar to those of scipy.linalg.pinvh().
142 
143  """
144 
145  def __init__(self, M, cond=None, rcond=None, lower=True,
146  check_finite=True, allow_singular=True):
147  # Compute the symmetric eigendecomposition.
148  # Note that eigh takes care of array conversion, chkfinite,
149  # and assertion that the matrix is square.
150  s, u = scipy.linalg.eigh(M, lower=lower, check_finite=check_finite)
151 
152  eps = _eigvalsh_to_eps(s, cond, rcond)
153  if np.min(s) < -eps:
154  raise ValueError('the input matrix must be positive semidefinite')
155  d = s[s > eps]
156  if len(d) < len(s) and not allow_singular:
157  raise np.linalg.LinAlgError('singular matrix')
158  s_pinv = _pinv_1d(s, eps)
159  U = np.multiply(u, np.sqrt(s_pinv))
160 
161  # Initialize the eagerly precomputed attributes.
162  self.rank = len(d)
163  self.U = U
164  self.log_pdet = np.sum(np.log(d))
165 
166  # Initialize an attribute to be lazily computed.
167  self._pinv = None
168 
169  @property
170  def pinv(self):
171  if self._pinv is None:
172  self._pinv = np.dot(self.U, self.U.T)
173  return self._pinv

302

305 
306  where :math:`\mu` is the mean, :math:`\Sigma` the covariance matrix,
307  and :math:`k` is the dimension of the space where :math:`x` takes values.
308

310

313  >>> import matplotlib.pyplot as plt
314  >>> from scipy.stats import multivariate_normal

317  >>> y = multivariate_normal.pdf(x, mean=2.5, cov=0.5); y
318  array([ 0.00108914, 0.01033349, 0.05946514, 0.20755375, 0.43939129,

353  def _process_parameters(self, dim, mean, cov):
354  """
355  Infer dimensionality from mean or covariance matrix, ensure that
356  mean and covariance are full vector resp. matrix.
357 
358  """
359 
360  # Try to infer dimensionality
361  if dim is None:
362  if mean is None:
363  if cov is None:
364  dim = 1
365  else:
366  cov = np.asarray(cov, dtype=float)
367  if cov.ndim < 2:
368  dim = 1
369  else:
370  dim = cov.shape[0]
371  else:
372  mean = np.asarray(mean, dtype=float)
373  dim = mean.size
374  else:
375  if not np.isscalar(dim):
376  raise ValueError("Dimension of random variable must be a scalar.")
377 
378  # Check input sizes and return full arrays for mean and cov if necessary
379  if mean is None:
380  mean = np.zeros(dim)
381  mean = np.asarray(mean, dtype=float)
382 
383  if cov is None:
384  cov = 1.0
385  cov = np.asarray(cov, dtype=float)
386 
387  if dim == 1:
388  mean.shape = (1,)
389  cov.shape = (1, 1)
390 
391  if mean.ndim != 1 or mean.shape[0] != dim:
392  raise ValueError("Array 'mean' must be a vector of length %d." % dim)
393  if cov.ndim == 0:
394  cov = cov * np.eye(dim)
395  elif cov.ndim == 1:
396  cov = np.diag(cov)
397  elif cov.ndim == 2 and cov.shape != (dim, dim):
398  rows, cols = cov.shape
399  if rows != cols:
400  msg = ("Array 'cov' must be square if it is two dimensional,"
401  " but cov.shape = %s." % str(cov.shape))
402  else:
403  msg = ("Dimension mismatch: array 'cov' is of shape %s,"
404  " but 'mean' is a vector of length %d.")
405  msg = msg % (str(cov.shape), len(mean))
406  raise ValueError(msg)
407  elif cov.ndim > 2:
408  raise ValueError("Array 'cov' must be at most two-dimensional,"
409  " but cov.ndim = %d" % cov.ndim)

431  def _logpdf(self, x, mean, prec_U, log_det_cov, rank):
432  """
433  Parameters
434  ----------
435  x : ndarray
436  Points at which to evaluate the log of the probability
437  density function
438  mean : ndarray
439  Mean of the distribution
440  prec_U : ndarray
441  A decomposition such that np.dot(prec_U, prec_U.T)
442  is the precision matrix, i.e. inverse of the covariance matrix.
443  log_det_cov : float
444  Logarithm of the determinant of the covariance matrix
445  rank : int
446  Rank of the covariance matrix.
447 
448  Notes
449  -----
450  As this function does no argument checking, it should not be
451  called directly; use 'logpdf' instead.
452 
453  """
454  dev = x - mean
455  maha = np.sum(np.square(np.dot(dev, prec_U)), axis=-1)
456  return -0.5 * (rank * _LOG_2PI + log_det_cov + maha)

713  `colcov` must be :math:`n \times n`. `mean` must be the same shape as `X`.
714

716 
717  .. math::
718 
719  f(X) = (2 \pi)^{-\frac{mn}{2}}|U|^{-\frac{n}{2}} |V|^{-\frac{m}{2}}
720  \exp\left( -\frac{1}{2} \mathrm{Tr}\left[ U^{-1} (X-M) V^{-1}
721  (X-M)^T \right] \right),
722

724  :math:`V` the among-column covariance matrix.
725

727  distribution is not currently supported. Covariance matrices must be
728  full rank.

731  `multivariate_normal` distribution. Specifically, :math:`\mathrm{Vec}(X)`
732  (the vector formed by concatenating the columns of :math:`X`) has a
733  multivariate normal distribution with mean :math:`\mathrm{Vec}(M)`
734  and covariance :math:`V \otimes U` (where :math:`\otimes` is the Kronecker
735  product). Sampling and pdf evaluation are
736  :math:`\mathcal{O}(m^3 + n^3 + m^2 n + m n^2)` for the matrix normal, but
737  :math:`\mathcal{O}(m^3 n^3)` for the equivalent multivariate normal,
738  making this equivalent form algorithmically inefficient.

787  def _process_parameters(self, mean, rowcov, colcov):
788  """
789  Infer dimensionality from mean or covariance matrices. Handle
790  defaults. Ensure compatible dimensions.
791 
792  """
793 
794  # Process mean
795  if mean is not None:
796  mean = np.asarray(mean, dtype=float)
797  meanshape = mean.shape
798  if len(meanshape) != 2:
799  raise ValueError("Array `mean` must be two dimensional.")
800  if np.any(meanshape == 0):
801  raise ValueError("Array `mean` has invalid shape.")
802 
803  # Process among-row covariance
804  rowcov = np.asarray(rowcov, dtype=float)
805  if rowcov.ndim == 0:
806  if mean is not None:
807  rowcov = rowcov * np.identity(meanshape[0])
808  else:
809  rowcov = rowcov * np.identity(1)
810  elif rowcov.ndim == 1:
811  rowcov = np.diag(rowcov)
812  rowshape = rowcov.shape
813  if len(rowshape) != 2:
814  raise ValueError("`rowcov` must be a scalar or a 2D array.")
815  if rowshape[0] != rowshape[1]:
816  raise ValueError("Array `rowcov` must be square.")
817  if rowshape[0] == 0:
818  raise ValueError("Array `rowcov` has invalid shape.")
819  numrows = rowshape[0]
820 
821  # Process among-column covariance
822  colcov = np.asarray(colcov, dtype=float)
823  if colcov.ndim == 0:
824  if mean is not None:
825  colcov = colcov * np.identity(meanshape[1])
826  else:
827  colcov = colcov * np.identity(1)
828  elif colcov.ndim == 1:
829  colcov = np.diag(colcov)
830  colshape = colcov.shape
831  if len(colshape) != 2:
832  raise ValueError("`colcov` must be a scalar or a 2D array.")
833  if colshape[0] != colshape[1]:
834  raise ValueError("Array `colcov` must be square.")
835  if colshape[0] == 0:
836  raise ValueError("Array `colcov` has invalid shape.")
837  numcols = colshape[0]
838 
839  # Ensure mean and covariances compatible
840  if mean is not None:
841  if meanshape[0] != numrows:
842  raise ValueError("Arrays `mean` and `rowcov` must have the"
843  "same number of rows.")
844  if meanshape[1] != numcols:
845  raise ValueError("Arrays `mean` and `colcov` must have the"
846  "same number of columns.")
847  else:

868  def _logpdf(self, dims, X, mean, row_prec_rt, log_det_rowcov,
869  col_prec_rt, log_det_colcov):
870  """
871  Parameters
872  ----------
873  dims : tuple
874  Dimensions of the matrix variates
875  X : ndarray
876  Points at which to evaluate the log of the probability
877  density function
878  mean : ndarray
879  Mean of the distribution

881  A decomposition such that np.dot(row_prec_rt, row_prec_rt.T)

899  scale_dev = np.tensordot(col_prec_rt.T,
900  np.dot(roll_dev, row_prec_rt), 1)
901  maha = np.sum(np.sum(np.square(scale_dev), axis=-1), axis=0)
902  return -0.5 * (numrows*numcols*_LOG_2PI + numcols*log_det_rowcov
903  + numrows*log_det_colcov + maha)

1427 class wishart_gen(multi_rv_generic):
1428  r"""
1429  A Wishart random variable.
1430 
1431  The `df` keyword specifies the degrees of freedom. The `scale` keyword
1432  specifies the scale matrix, which must be symmetric and positive definite.
1433  In this context, the scale matrix is often interpreted in terms of a
1434  multivariate normal precision matrix (the inverse of the covariance
1435  matrix).
1436 
1437  Methods
1438  -------
1439  ``pdf(x, df, scale)``
1440  Probability density function.
1441  ``logpdf(x, df, scale)``
1442  Log of the probability density function.
1443  ``rvs(df, scale, size=1, random_state=None)``
1444  Draw random samples from a Wishart distribution.
1445  ``entropy()``
1446  Compute the differential entropy of the Wishart distribution.
1447 
1448  Parameters
1449  ----------
1450  x : array_like
1451  Quantiles, with the last axis of `x` denoting the components.
1452  %(_doc_default_callparams)s
1453  %(_doc_random_state)s
1454 
1455  Alternatively, the object may be called (as a function) to fix the degrees
1456  of freedom and scale parameters, returning a "frozen" Wishart random
1457  variable:
1458 
1459  rv = wishart(df=1, scale=1)
1460  - Frozen object with the same methods but holding the given
1461  degrees of freedom and scale fixed.
1462 
1463  See Also
1464  --------
1465  invwishart, chi2
1466 
1467  Notes
1468  -----
1469  %(_doc_callparams_note)s
1470 
1471  The scale matrix `scale` must be a symmetric positive definite
1472  matrix. Singular matrices, including the symmetric positive semi-definite
1473  case, are not supported.
1474 
1475  The Wishart distribution is often denoted
1476 
1477  .. math::
1478 
1479  W_p(\nu, \Sigma)
1480 
1481  where :math:`\nu` is the degrees of freedom and :math:`\Sigma` is the
1482  :math:`p \times p` scale matrix.
1483 
1484  The probability density function for `wishart` has support over positive
1485  definite matrices :math:`S`; if :math:`S \sim W_p(\nu, \Sigma)`, then
1486  its PDF is given by:
1487 
1488  .. math::
1489 
1490  f(S) = \frac{|S|^{\frac{\nu - p - 1}{2}}}{2^{ \frac{\nu p}{2} }
1491  |\Sigma|^\frac{\nu}{2} \Gamma_p \left ( \frac{\nu}{2} \right )}
1492  \exp\left( -tr(\Sigma^{-1} S) / 2 \right)
1493 
1494  If :math:`S \sim W_p(\nu, \Sigma)` (Wishart) then
1495  :math:`S^{-1} \sim W_p^{-1}(\nu, \Sigma^{-1})` (inverse Wishart).
1496 
1497  If the scale matrix is 1-dimensional and equal to one, then the Wishart
1498  distribution :math:`W_1(\nu, 1)` collapses to the :math:`\chi^2(\nu)`
1499  distribution.

1506  Wiley, 1983.

1828  """
1829  Parameters
1830  ----------
1831  n : integer
1832  Number of variates to generate
1833  shape : iterable
1834  Shape of the variates to generate
1835  dim : int
1836  Dimension of the scale matrix
1837  df : int
1838  Degrees of freedom

1840  RandomState used for drawing the random variates.

1842  Notes
1843  -----

1845  called directly; use 'rvs' instead.
1846

1848  # Random normal variates for off-diagonal elements
1849  n_tril = dim * (dim-1) // 2
1850  covariances = random_state.normal(
1851  size=n*n_tril).reshape(shape+(n_tril,))
1852 
1853  # Random chi-square variates for diagonal elements
1854  variances = np.r_[[random_state.chisquare(df-(i+1)+1, size=n)**0.5
1855  for i in range(dim)]].reshape((dim,) + shape[::-1]).T
1856

1858  A = np.zeros(shape + (dim, dim))
1859 
1860  # Input the covariances
1861  size_idx = tuple([slice(None,None,None)]*len(shape))
1862  tril_idx = np.tril_indices(dim, k=-1)
1863  A[size_idx + tril_idx] = covariances

1865  # Input the variances
1866  diag_idx = np.diag_indices(dim)
1867  A[size_idx + diag_idx] = variances

1872  """
1873  Parameters
1874  ----------
1875  n : integer
1876  Number of variates to generate
1877  shape : iterable

1897  # Cholesky factorizations of a matrix B such that B ~ W(df, I)
1898  A = self._standard_rvs(n, shape, dim, df, random_state)
1899 
1900  # Calculate SA = C A A' C', where SA ~ W(df, scale)
1901  # Note: this is the product of a (lower) (lower) (lower)' (lower)'
1902  # or, denoting B = AA', it is C B C' where C is the lower
1903  # triangular Cholesky factorization of the scale matrix.
1904  # this appears to conflict with the instructions in [1]_, which
1905  # suggest that it should be D' B D where D is the lower
1906  # triangular factorization of the scale matrix. However, it is
1907  # meant to refer to the Bartlett (1933) representation of a
1908  # Wishart random variate as L A A' L' where L is lower triangular
1909  # so it appears that understanding D' to be upper triangular
1910  # is either a typo in or misreading of [1]_.
1911  for index in np.ndindex(shape):
1912  CA = np.dot(C, A[index])
1913  A[index] = np.dot(CA, CA.T)

3409  def _givens_to_1(self, aii, ajj, aij):
3410  """Computes a 2x2 Givens matrix to put 1's on the diagonal for the input matrix.
3411 
3412  The input matrix is a 2x2 symmetric matrix M = [ aii aij ; aij ajj ].
3413 
3414  The output matrix g is a 2x2 anti-symmetric matrix of the form [ c s ; -s c ];
3415  the elements c and s are returned.

3418  results in a matrix with bii=1, provided tr(M) - det(M) >= 1
3419  and floating point issues do not occur. Otherwise, some other
3420  valid rotation is returned. When tr(M)==2, also bjj=1.

3424  ajjd = ajj - 1.
