419 def _cubicmin(a, fa, fpa, b, fb, c, fc):
420  """
421  Finds the minimizer for a cubic polynomial that goes through the
422  points (a,fa), (b,fb), and (c,fc) with derivative at a of fpa.
423 
424  If no minimizer can be found return None
425 
426  """
427  # f(x) = A *(x-a)^3 + B*(x-a)^2 + C*(x-a) + D
428 
429  with np.errstate(divide='raise', over='raise', invalid='raise'):
430  try:
431  C = fpa
432  db = b - a
433  dc = c - a
434  denom = (db * dc) ** 2 * (db - dc)
435  d1 = np.empty((2, 2))
436  d1[0, 0] = dc ** 2
437  d1[0, 1] = -db ** 2
438  d1[1, 0] = -dc ** 3
439  d1[1, 1] = db ** 3
440  [A, B] = np.dot(d1, np.asarray([fb - fa - C * db,
441  fc - fa - C * dc]).flatten())
442  A /= denom
443  B /= denom
444  radical = B * B - 3 * A * C
445  xmin = a + (-B + np.sqrt(radical)) / (3 * A)
446  except ArithmeticError:
447  return None
448  if not np.isfinite(xmin):
449  return None
450  return xmin

453 def _quadmin(a, fa, fpa, b, fb):
454  """
455  Finds the minimizer for a quadratic polynomial that goes through
456  the points (a,fa), (b,fb) with derivative at a of fpa,
457 
458  """
459  # f(x) = B*(x-a)^2 + C*(x-a) + D
460  with np.errstate(divide='raise', over='raise', invalid='raise'):
461  try:
462  D = fa
463  C = fpa
464  db = b - a * 1.0
465  B = (fb - D - C * db) / (db * db)
466  xmin = a - C / (2.0 * B)
467  except ArithmeticError:
468  return None
469  if not np.isfinite(xmin):
470  return None
471  return xmin

474 def _zoom(a_lo, a_hi, phi_lo, phi_hi, derphi_lo,
475  phi, derphi, phi0, derphi0, c1, c2):
476  """
477  Part of the optimization algorithm in `scalar_search_wolfe2`.
478  """
479 
480  maxiter = 10
481  i = 0
482  delta1 = 0.2 # cubic interpolant check
483  delta2 = 0.1 # quadratic interpolant check
484  phi_rec = phi0
485  a_rec = 0
486  while True:
487  # interpolate to find a trial step length between a_lo and
488  # a_hi Need to choose interpolation here. Use cubic
489  # interpolation and then if the result is within delta *
490  # dalpha or outside of the interval bounded by a_lo or a_hi
491  # then use quadratic interpolation, if the result is still too
492  # close, then use bisection
493 
494  dalpha = a_hi - a_lo
495  if dalpha < 0:
496  a, b = a_hi, a_lo
497  else:
498  a, b = a_lo, a_hi
499 
500  # minimizer of cubic interpolant
501  # (uses phi_lo, derphi_lo, phi_hi, and the most recent value of phi)
502  #
503  # if the result is too close to the end points (or out of the
504  # interval) then use quadratic interpolation with phi_lo,
505  # derphi_lo and phi_hi if the result is stil too close to the
506  # end points (or out of the interval) then use bisection
507 
508  if (i > 0):
509  cchk = delta1 * dalpha
510  a_j = _cubicmin(a_lo, phi_lo, derphi_lo, a_hi, phi_hi,
511  a_rec, phi_rec)
512  if (i == 0) or (a_j is None) or (a_j > b - cchk) or (a_j < a + cchk):
513  qchk = delta2 * dalpha
514  a_j = _quadmin(a_lo, phi_lo, derphi_lo, a_hi, phi_hi)
515  if (a_j is None) or (a_j > b-qchk) or (a_j < a+qchk):
516  a_j = a_lo + 0.5*dalpha
517 
518  # Check new value of a_j
519 
520  phi_aj = phi(a_j)
521  if (phi_aj > phi0 + c1*a_j*derphi0) or (phi_aj >= phi_lo):
522  phi_rec = phi_hi
523  a_rec = a_hi
524  a_hi = a_j
525  phi_hi = phi_aj
526  else:
527  derphi_aj = derphi(a_j)
528  if abs(derphi_aj) <= -c2*derphi0:
529  a_star = a_j
530  val_star = phi_aj
531  valprime_star = derphi_aj
532  break
533  if derphi_aj*(a_hi - a_lo) >= 0:
534  phi_rec = phi_hi
535  a_rec = a_hi
536  a_hi = a_lo
537  phi_hi = phi_lo
538  else:
539  phi_rec = phi_lo
540  a_rec = a_lo
541  a_lo = a_j
542  phi_lo = phi_aj
543  derphi_lo = derphi_aj
544  i += 1
545  if (i > maxiter):
546  # Failed to find a conforming step size
547  a_star = None
548  val_star = None
549  valprime_star = None
550  break
551  return a_star, val_star, valprime_star

619 def scalar_search_armijo(phi, phi0, derphi0, c1=1e-4, alpha0=1, amin=0):
620  """Minimize over alpha, the function ``phi(alpha)``.
621 
622  Uses the interpolation algorithm (Armijo backtracking) as suggested by
623  Wright and Nocedal in 'Numerical Optimization', 1999, pg. 56-57
624 
625  alpha > 0 is assumed to be a descent direction.
626 
627  Returns
628  -------
629  alpha
630  phi1
631 
632  """
633  phi_a0 = phi(alpha0)
634  if phi_a0 <= phi0 + c1*alpha0*derphi0:
635  return alpha0, phi_a0
636 
637  # Otherwise compute the minimizer of a quadratic interpolant:
638 
639  alpha1 = -(derphi0) * alpha0**2 / 2.0 / (phi_a0 - phi0 - derphi0 * alpha0)
640  phi_a1 = phi(alpha1)
641 
642  if (phi_a1 <= phi0 + c1*alpha1*derphi0):
643  return alpha1, phi_a1
644 
645  # Otherwise loop with cubic interpolation until we find an alpha which
646  # satifies the first Wolfe condition (since we are backtracking, we will
647  # assume that the value of alpha is not too small and satisfies the second
648  # condition.
649 
650  while alpha1 > amin: # we are assuming alpha>0 is a descent direction
651  factor = alpha0**2 * alpha1**2 * (alpha1-alpha0)
652  a = alpha0**2 * (phi_a1 - phi0 - derphi0*alpha1) - \
653  alpha1**2 * (phi_a0 - phi0 - derphi0*alpha0)
654  a = a / factor
655  b = -alpha0**3 * (phi_a1 - phi0 - derphi0*alpha1) + \
656  alpha1**3 * (phi_a0 - phi0 - derphi0*alpha0)
657  b = b / factor
658 
659  alpha2 = (-b + np.sqrt(abs(b**2 - 3 * a * derphi0))) / (3.0*a)
660  phi_a2 = phi(alpha2)
661 
662  if (phi_a2 <= phi0 + c1*alpha2*derphi0):
663  return alpha2, phi_a2
664 
665  if (alpha1 - alpha2) > alpha1 / 2.0 or (1 - alpha2/alpha1) < 0.96:
666  alpha2 = alpha1 / 2.0
667 
668  alpha0 = alpha1
669  alpha1 = alpha2
670  phi_a0 = phi_a1
671  phi_a1 = phi_a2
672 
673  # Failed to find a suitable step length
674  return None, phi_a1

685

688  f : callable

701  Search parameters, see [1]_
702

704  -------
705  alpha : float
706  Step length
707  xp : ndarray
708  Next position
709  fp : float
710  Merit function value at next position
711  Fp : ndarray
712  Residual at next position
713 
714  References
715  ----------
716  [1] "Spectral residual method without gradient information for solving
717  large-scale nonlinear systems of equations." W. La Cruz,
718  J.M. Martinez, M. Raydan. Math. Comp. **75**, 1429 (2006).
719 
720  """

723

726  alpha = 1
727

733  alpha = alpha_p
734  break
735

742  alpha = -alpha_m
743  break

748  alpha_m = np.clip(alpha_tm, tau_min * alpha_m, tau_max * alpha_m)
