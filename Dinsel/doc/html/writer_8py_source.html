35  def write_packages(self, diagram):
36  """write a package diagram"""
37  # sorted to get predictable (hence testable) results
38  for i, obj in enumerate(sorted(diagram.modules(), key=lambda x: x.title)):
39  self.printer.emit_node(i, label=self.get_title(obj), shape='box')
40  obj.fig_id = i
41  # package dependencies
42  for rel in diagram.get_relationships('depends'):
43  self.printer.emit_edge(rel.from_object.fig_id, rel.to_object.fig_id,

46  def write_classes(self, diagram):
47  """write a class diagram"""
48  # sorted to get predictable (hence testable) results
49  for i, obj in enumerate(sorted(diagram.objects, key=lambda x: x.title)):
50  self.printer.emit_node(i, **self.get_values(obj))
51  obj.fig_id = i
52  # inheritance links
53  for rel in diagram.get_relationships('specialization'):
54  self.printer.emit_edge(rel.from_object.fig_id, rel.to_object.fig_id,
55  **self.inh_edges)
56  # implementation links
57  for rel in diagram.get_relationships('implements'):
58  self.printer.emit_edge(rel.from_object.fig_id, rel.to_object.fig_id,
59  **self.imp_edges)
60  # generate associations
61  for rel in diagram.get_relationships('association'):
62  self.printer.emit_edge(rel.from_object.fig_id, rel.to_object.fig_id,
63  label=rel.name, **self.ass_edges)

157  def get_values(self, obj):
158  """get label and shape for classes.
159 
160  The label contains all attributes and methods
161  """
162  if is_exception(obj.node):
163  label = r'\fb\f09%s\fn' % obj.title
164  else:
165  label = r'\fb%s\fn' % obj.title
166  if obj.shape == 'interface':
167  shape = 'ellipse'
168  else:
169  shape = 'box'
170  if not self.config.only_classnames:
171  attrs = obj.attrs
172  methods = [func.name for func in obj.methods]
173  # box width for UML like diagram
174  maxlen = max(len(name) for name in [obj.title] + methods + attrs)
175  line = '_' * (maxlen + 2)
176  label = r'%s\n\f%s' % (label, line)
177  for attr in attrs:
178  label = r'%s\n\f08%s' % (label, attr)
179  if attrs:
180  label = r'%s\n\f%s' % (label, line)
181  for func in methods:
182  label = r'%s\n\f10%s()' % (label, func)
183  return dict(label=label, shape=shape)
