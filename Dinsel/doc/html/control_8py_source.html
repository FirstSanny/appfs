63  """Programmatic access to coverage.py.
64 
65  To use::
66

68

76  def __init__(

78  auto_data=False, timid=None, branch=None, config_file=True,
79  source=None, omit=None, include=None, debug=None,
80  concurrency=None,
81  ):
82  """
83  `data_file` is the base name of the data file to use, defaulting to
84  ".coverage". `data_suffix` is appended (with a dot) to `data_file` to
85  create the final file name. If `data_suffix` is simply True, then a
86  suffix is created with the machine and process identity included.
87 
88  `cover_pylib` is a boolean determining whether Python code installed
89  with the Python interpreter is measured. This includes the Python
90  standard library and any packages installed with the interpreter.
91 
92  If `auto_data` is true, then any existing data file will be read when
93  coverage measurement starts, and data will be saved automatically when
94  measurement stops.
95 
96  If `timid` is true, then a slower and simpler trace function will be
97  used. This is important for some environments where manipulation of
98  tracing functions breaks the faster trace function.
99 
100  If `branch` is true, then branch coverage will be measured in addition
101  to the usual statement coverage.
102 
103  `config_file` determines what configuration file to read:
104 
105  * If it is ".coveragerc", it is interpreted as if it were True,
106  for backward compatibility.
107 
108  * If it is a string, it is the name of the file to read. If the
109  file can't be read, it is an error.
110 
111  * If it is True, then a few standard files names are tried
112  (".coveragerc", "setup.cfg", "tox.ini"). It is not an error for
113  these files to not be found.
114 
115  * If it is False, then no configuration file is read.
116 
117  `source` is a list of file paths or package names. Only code located
118  in the trees indicated by the file paths or package names will be
119  measured.
120 
121  `include` and `omit` are lists of file name patterns. Files that match
122  `include` will be measured, files that match `omit` will not. Each
123  will also accept a single string argument.
124 
125  `debug` is a list of strings indicating what debugging information is
126  desired.
127 
128  `concurrency` is a string indicating the concurrency library being used
129  in the measured code. Without this, coverage.py will get incorrect
130  results if these libraries are in use. Valid strings are "greenlet",
131  "eventlet", "gevent", "multiprocessing", or "thread" (the default).
132  This can also be a list of these strings.
133 
134  .. versionadded:: 4.0
135  The `concurrency` parameter.
136 
137  .. versionadded:: 4.2
138  The `concurrency` parameter can now be a list of strings.

389  def _name_for_module(self, module_globals, filename):
390  """Get the name of the module for a set of globals and file name.
391 
392  For configurability's sake, we allow __main__ modules to be matched by
393  their importable name.
394 
395  If loaded via runpy (aka -m), we can usually recover the "original"
396  full dotted module name, otherwise, we resort to interpreting the
397  file name to get the module's name. In the case that the module name
398  can't be determined, None is returned.
399 
400  """
401  if module_globals is None: # pragma: only ironpython
402  # IronPython doesn't provide globals: https://github.com/IronLanguages/main/issues/1296
403  module_globals = {}
404 
405  dunder_name = module_globals.get('__name__', None)
406 
407  if isinstance(dunder_name, str) and dunder_name != '__main__':
408  # This is the usual case: an imported module.
409  return dunder_name
410 
411  loader = module_globals.get('__loader__', None)
412  for attrname in ('fullname', 'name'): # attribute renamed in py3.2
413  if hasattr(loader, attrname):
414  fullname = getattr(loader, attrname)
415  else:
416  continue
417 
418  if isinstance(fullname, str) and fullname != '__main__':
419  # Module loaded via: runpy -m
420  return fullname
421 
422  # Script as first argument to Python command line.
423  inspectedname = inspect.getmodulename(filename)
424  if inspectedname is not None:
425  return inspectedname
426  else:
427  return dunder_name

429  def _should_trace_internal(self, filename, frame):
430  """Decide whether to trace execution in `filename`, with a reason.
431 
432  This function is called from the trace function. As each new file name
433  is encountered, this function determines whether it is traced or not.
434 
435  Returns a FileDisposition object.
436 
437  """
438  original_filename = filename
439  disp = _disposition_init(self.collector.file_disposition_class, filename)
440 
441  def nope(disp, reason):
442  """Simple helper to make it easy to return NO."""
443  disp.trace = False
444  disp.reason = reason
445  return disp
446 
447  # Compiled Python files have two file names: frame.f_code.co_filename is
448  # the file name at the time the .pyc was compiled. The second name is
449  # __file__, which is where the .pyc was actually loaded from. Since
450  # .pyc files can be moved after compilation (for example, by being
451  # installed), we look for __file__ in the frame and prefer it to the
452  # co_filename value.
453  dunder_file = frame.f_globals and frame.f_globals.get('__file__')
454  if dunder_file:
455  filename = source_for_file(dunder_file)
456  if original_filename and not original_filename.startswith('<'):
457  orig = os.path.basename(original_filename)
458  if orig != os.path.basename(filename):
459  # Files shouldn't be renamed when moved. This happens when
460  # exec'ing code. If it seems like something is wrong with
461  # the frame's file name, then just use the original.
462  filename = original_filename
463 
464  if not filename:
465  # Empty string is pretty useless.
466  return nope(disp, "empty string isn't a file name")
467 
468  if filename.startswith('memory:'):
469  return nope(disp, "memory isn't traceable")
470 
471  if filename.startswith('<'):
472  # Lots of non-file execution is represented with artificial
473  # file names like "<string>", "<doctest readme.txt[0]>", or
474  # "<exec_function>". Don't ever trace these executions, since we
475  # can't do anything with the data later anyway.
476  return nope(disp, "not a real file name")
477 
478  # pyexpat does a dumb thing, calling the trace function explicitly from
479  # C code with a C file name.
480  if re.search(r"[/\\]Modules[/\\]pyexpat.c", filename):
481  return nope(disp, "pyexpat lies about itself")
482 
483  # Jython reports the .class file to the tracer, use the source file.
484  if filename.endswith("$py.class"):
485  filename = filename[:-9] + ".py"
486 
487  canonical = canonical_filename(filename)
488  disp.canonical_filename = canonical
489 
490  # Try the plugins, see if they have an opinion about the file.
491  plugin = None
492  for plugin in self.plugins.file_tracers:
493  if not plugin._coverage_enabled:
494  continue
495 
496  try:
497  file_tracer = plugin.file_tracer(canonical)
498  if file_tracer is not None:
499  file_tracer._coverage_plugin = plugin
500  disp.trace = True
501  disp.file_tracer = file_tracer
502  if file_tracer.has_dynamic_source_filename():
503  disp.has_dynamic_filename = True
504  else:
505  disp.source_filename = canonical_filename(
506  file_tracer.source_filename()
507  )
508  break
509  except Exception:
510  self._warn(
511  "Disabling plugin %r due to an exception:" % (
512  plugin._coverage_plugin_name
513  )
514  )
515  traceback.print_exc()
516  plugin._coverage_enabled = False
517  continue
518  else:
519  # No plugin wanted it: it's Python.
520  disp.trace = True
521  disp.source_filename = canonical
522 
523  if not disp.has_dynamic_filename:
524  if not disp.source_filename:
525  raise CoverageException(
526  "Plugin %r didn't set source_filename for %r" %
527  (plugin, disp.original_filename)
528  )
529  reason = self._check_include_omit_etc_internal(
530  disp.source_filename, frame,
531  )
532  if reason:
533  nope(disp, reason)
534 
535  return disp

537  def _check_include_omit_etc_internal(self, filename, frame):
538  """Check a file name against the include, omit, etc, rules.
539 
540  Returns a string or None. String means, don't trace, and is the reason
541  why. None means no reason found to not trace.
542 
543  """
544  modulename = self._name_for_module(frame.f_globals, filename)
545 
546  # If the user specified source or include, then that's authoritative
547  # about the outer bound of what to measure and we don't have to apply
548  # any canned exclusions. If they didn't, then we have to exclude the
549  # stdlib and coverage.py directories.
550  if self.source_match:
551  if self.source_pkgs_match.match(modulename):
552  if modulename in self.source_pkgs_unmatched:
553  self.source_pkgs_unmatched.remove(modulename)
554  return None # There's no reason to skip this file.
555 
556  if not self.source_match.match(filename):
557  return "falls outside the --source trees"
558  elif self.include_match:
559  if not self.include_match.match(filename):
560  return "falls outside the --include trees"
561  else:
562  # If we aren't supposed to trace installed code, then check if this
563  # is near the Python standard library and skip it if so.
564  if self.pylib_match and self.pylib_match.match(filename):
565  return "is in the stdlib"
566 
567  # We exclude the coverage.py code itself, since a little of it
568  # will be measured otherwise.
569  if self.cover_match and self.cover_match.match(filename):
570  return "is part of coverage.py"
571 
572  # Check the file against the omit pattern.
573  if self.omit_match and self.omit_match.match(filename):
574  return "is inside an --omit pattern"
575 
576  # No reason found to skip this file.
577  return None

590  def _check_include_omit_etc(self, filename, frame):
591  """Check a file name against the include/omit/etc, rules, verbosely.
592 
593  Returns a boolean: True if the file should be traced, False if not.
594 
595  """
596  reason = self._check_include_omit_etc_internal(filename, frame)
597  if self.debug.should('trace'):
598  if not reason:

600  else:
601  msg = "Not including %r: %s" % (filename, reason)
602  self.debug.write(msg)
603 
604  return not reason

636  def set_option(self, option_name, value):
637  """Set an option in the configuration.
