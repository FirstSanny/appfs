284 def time_limited(timeout=0.5, return_val=np.nan, use_sigalrm=True):
285  """
286  Decorator for setting a timeout for pure-Python functions.
287 
288  If the function does not return within `timeout` seconds, the
289  value `return_val` is returned instead.
290 
291  On POSIX this uses SIGALRM by default. On non-POSIX, settrace is
292  used. Do not use this with threads: the SIGALRM implementation
293  does probably not work well. The settrace implementation only
294  traces the current thread.
295 
296  The settrace implementation slows down execution speed. Slowdown
297  by a factor around 10 is probably typical.
298  """
299  if POSIX and use_sigalrm:
300  def sigalrm_handler(signum, frame):
301  raise TimeoutError()
302 
303  def deco(func):
304  def wrap(*a, **kw):
305  old_handler = signal.signal(signal.SIGALRM, sigalrm_handler)
306  signal.setitimer(signal.ITIMER_REAL, timeout)
307  try:
308  return func(*a, **kw)
309  except TimeoutError:
310  return return_val
311  finally:
312  signal.setitimer(signal.ITIMER_REAL, 0)
313  signal.signal(signal.SIGALRM, old_handler)
314  return wrap
315  else:
316  def deco(func):
317  def wrap(*a, **kw):
318  start_time = time.time()
319 
320  def trace(frame, event, arg):
321  if time.time() - start_time > timeout:
322  raise TimeoutError()
323  return None # turn off tracing except at function calls
324  sys.settrace(trace)
325  try:
326  return func(*a, **kw)
327  except TimeoutError:
328  sys.settrace(None)
329  return return_val
330  finally:
331  sys.settrace(None)
332  return wrap
333  return deco

356 def mp_assert_allclose(res, std, atol=0, rtol=1e-17):
357  """
358  Compare lists of mpmath.mpf's or mpmath.mpc's directly so that it
359  can be done to higher precision than double.
360 
361  """
362  try:
363  len(res)
364  except TypeError:
365  res = list(res)
366 
367  n = len(std)
368  if len(res) != n:
369  raise AssertionError("Lengths of inputs not equal.")
370 
371  failures = []
372  for k in range(n):
373  try:
374  assert_(mpmath.fabs(res[k] - std[k]) <= atol + rtol*mpmath.fabs(std[k]))
375  except AssertionError:
376  failures.append(k)
377 
378  ndigits = int(abs(np.log10(rtol)))
379  msg = [""]
380  msg.append("Bad results ({} out of {}) for the following points:"
381  .format(len(failures), n))
382  for k in failures:
383  resrep = mpmath.nstr(res[k], ndigits, min_fixed=0, max_fixed=0)
384  stdrep = mpmath.nstr(std[k], ndigits, min_fixed=0, max_fixed=0)
385  if std[k] == 0:
386  rdiff = "inf"
387  else:
388  rdiff = mpmath.fabs((res[k] - std[k])/std[k])
389  rdiff = mpmath.nstr(rdiff, 3)
390  msg.append("{}: {} != {} (rdiff {})".format(k, resrep, stdrep, rdiff))
391  if failures:
392  assert_(False, "\n".join(msg))
