74  def get_boundaries_intersections(self, z, d, trust_radius):
75  """
76  Solve the scalar quadratic equation ||z + t d|| == trust_radius.
77  This is like a line-sphere intersection.
78  Return the two values of t, sorted from low to high.
79  """
80  a = np.dot(d, d)
81  b = 2 * np.dot(z, d)
82  c = np.dot(z, z) - trust_radius**2
83  sqrt_discriminant = math.sqrt(b*b - 4*a*c)
84  ta = (-b - sqrt_discriminant) / (2*a)
85  tb = (-b + sqrt_discriminant) / (2*a)
86  return ta, tb

93 def _minimize_trust_region(fun, x0, args=(), jac=None, hess=None, hessp=None,
94  subproblem=None, initial_trust_radius=1.0,
95  max_trust_radius=1000.0, eta=0.15, gtol=1e-4,
96  maxiter=None, disp=False, return_all=False,
97  callback=None, **unknown_options):
98  """
99  Minimization of scalar function of one or more variables using a
100  trust-region algorithm.
101 
102  Options for the trust-region algorithm are:
103  initial_trust_radius : float
104  Initial trust radius.
105  max_trust_radius : float
106  Never propose steps that are longer than this value.
107  eta : float
108  Trust region related acceptance stringency for proposed steps.
109  gtol : float
110  Gradient norm must be less than `gtol`
111  before successful termination.
112  maxiter : int
113  Maximum number of iterations to perform.
114  disp : bool
115  If True, print convergence message.
116 
117  This function is called by the `minimize` function.
118  It is not supposed to be called directly.
119  """
120  _check_unknown_options(unknown_options)
121  if jac is None:
122  raise ValueError('Jacobian is currently required for trust-region '
123  'methods')
124  if hess is None and hessp is None:
125  raise ValueError('Either the Hessian or the Hessian-vector product '
126  'is currently required for trust-region methods')
127  if subproblem is None:
128  raise ValueError('A subproblem solving strategy is required for '
129  'trust-region methods')
130  if not (0 <= eta < 0.25):
131  raise Exception('invalid acceptance stringency')
132  if max_trust_radius <= 0:
133  raise Exception('the max trust radius must be positive')
134  if initial_trust_radius <= 0:
135  raise ValueError('the initial trust radius must be positive')
136  if initial_trust_radius >= max_trust_radius:
137  raise ValueError('the initial trust radius must be less than the '
138  'max trust radius')
139 
140  # force the initial guess into a nice format
141  x0 = np.asarray(x0).flatten()
142 
143  # Wrap the functions, for a couple reasons.
144  # This tracks how many times they have been called
145  # and it automatically passes the args.
146  nfun, fun = wrap_function(fun, args)
147  njac, jac = wrap_function(jac, args)
148  nhess, hess = wrap_function(hess, args)
149  nhessp, hessp = wrap_function(hessp, args)
150 
151  # limit the number of iterations
152  if maxiter is None:
153  maxiter = len(x0)*200
154 
155  # init the search status
156  warnflag = 0
157 
158  # initialize the search
159  trust_radius = initial_trust_radius
160  x = x0
161  if return_all:
162  allvecs = [x]
163  m = subproblem(x, fun, jac, hess, hessp)
164  k = 0
165 
166  # search for the function min
167  while True:
168 
169  # Solve the sub-problem.
170  # This gives us the proposed step relative to the current position
171  # and it tells us whether the proposed step
172  # has reached the trust region boundary or not.
173  try:
174  p, hits_boundary = m.solve(trust_radius)
175  except np.linalg.linalg.LinAlgError as e:
176  warnflag = 3
177  break
178 
179  # calculate the predicted value at the proposed point
180  predicted_value = m(p)
181 
182  # define the local approximation at the proposed point
183  x_proposed = x + p
184  m_proposed = subproblem(x_proposed, fun, jac, hess, hessp)
185 
186  # evaluate the ratio defined in equation (4.4)
187  actual_reduction = m.fun - m_proposed.fun
188  predicted_reduction = m.fun - predicted_value
189  if predicted_reduction <= 0:
190  warnflag = 2
191  break
192  rho = actual_reduction / predicted_reduction
193 
194  # update the trust radius according to the actual/predicted ratio
195  if rho < 0.25:
196  trust_radius *= 0.25
197  elif rho > 0.75 and hits_boundary:
198  trust_radius = min(2*trust_radius, max_trust_radius)
199 
200  # if the ratio is high enough then accept the proposed step
201  if rho > eta:
202  x = x_proposed
203  m = m_proposed
204 
205  # append the best guess, call back, increment the iteration count
206  if return_all:
207  allvecs.append(x)
208  if callback is not None:
209  callback(x)
210  k += 1
211 
212  # check if the gradient is small enough to stop
213  if m.jac_mag < gtol:
214  warnflag = 0

216 
217  # check if we have looked at enough iterations
218  if k >= maxiter:
219  warnflag = 1
220  break
221 
222  # print some stuff if requested
223  status_messages = (
224  _status_message['success'],
225  _status_message['maxiter'],
226  'A bad approximation caused failure to predict improvement.',
227  'A linalg error occurred, such as a non-psd Hessian.',
228  )
229  if disp:

231  print(status_messages[warnflag])
232  else:
233  print('Warning: ' + status_messages[warnflag])
234  print(" Current function value: %f" % m.fun)
235  print(" Iterations: %d" % k)
236  print(" Function evaluations: %d" % nfun[0])
237  print(" Gradient evaluations: %d" % njac[0])
