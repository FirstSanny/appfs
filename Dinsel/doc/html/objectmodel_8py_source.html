4 """
5 Data object model, as per https://docs.python.org/3/reference/datamodel.html.
6 
7 This module describes, at least partially, a data object model for some
8 of astroid's nodes. The model contains special attributes that nodes such
9 as functions, classes, modules etc have, such as __doc__, __class__,
10 __module__ etc, being used when doing attribute lookups over nodes.
11 
12 For instance, inferring `obj.__class__` will first trigger an inference
13 of the `obj` variable. If it was succesfully inferred, then an attribute
14 `__class__ will be looked for in the inferred object. This is the part
15 where the data model occurs. The model is attached to those nodes
16 and the lookup mechanism will try to see if attributes such as
17 `__class__` are defined by the model or not. If they are defined,
18 the model will be requested to return the corresponding value of that
19 attribute. Thus the model can be viewed as a special part of the lookup
20 mechanism.
21 """

41 def _dunder_dict(instance, attributes):
42  obj = node_classes.Dict(parent=instance)

44  # Convert the keys to node strings

49  # but that is not useful for retrieving the special attribute's value.

295  class DescriptorBoundMethod(bases.BoundMethod):
296  """Bound method which knows how to understand calling descriptor binding."""
297  def infer_call_result(self, caller, context=None):
298  if len(caller.args) != 2:
299  raise exceptions.InferenceError(
300  "Invalid arguments for descriptor binding",
301  target=self, context=context)
302 
303  context = contextmod.copy_context(context)
304  cls = next(caller.args[0].infer(context=context))
305 
306  # Rebuild the original value, but with the parent set as the
307  # class where it will be bound.
308  new_func = func.__class__(name=func.name, doc=func.doc,
309  lineno=func.lineno, col_offset=func.col_offset,
310  parent=cls)
311  # pylint: disable=no-member
312  new_func.postinit(func.args, func.body,
313  func.decorators, func.returns)
314 
315  # Build a proper bound method that points to our newly built function.
316  proxy = bases.UnboundMethod(new_func)
317  yield bases.BoundMethod(proxy=proxy, bound=cls)
318 
319  return DescriptorBoundMethod(proxy=self._instance, bound=self._instance)
