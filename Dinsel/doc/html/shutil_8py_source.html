167 def copystat(src, dst, *, follow_symlinks=True):
168  """Copy all stat info (mode bits, atime, mtime, flags) from src to dst.
169 
170  If the optional flag `follow_symlinks` is not set, symlinks aren't followed if and
171  only if both `src` and `dst` are symlinks.
172 
173  """
174  def _nop(*args, ns=None, follow_symlinks=None):
175  pass
176 
177  # follow symlinks (aka don't not follow symlinks)
178  follow = follow_symlinks or not (os.path.islink(src) and os.path.islink(dst))
179  if follow:
180  # use the real function if it exists
181  def lookup(name):
182  return getattr(os, name, _nop)
183  else:
184  # use the real function only if it exists
185  # *and* it supports follow_symlinks
186  def lookup(name):
187  fn = getattr(os, name, _nop)
188  if fn in os.supports_follow_symlinks:
189  return fn
190  return _nop
191 
192  st = lookup("stat")(src, follow_symlinks=follow)
193  mode = stat.S_IMODE(st.st_mode)
194  lookup("utime")(dst, ns=(st.st_atime_ns, st.st_mtime_ns),
195  follow_symlinks=follow)
196  try:
197  lookup("chmod")(dst, mode, follow_symlinks=follow)
198  except NotImplementedError:
199  # if we got a NotImplementedError, it's because
200  # * follow_symlinks=False,
201  # * lchown() is unavailable, and
202  # * either
203  # * fchownat() is unavailable or
204  # * fchownat() doesn't implement AT_SYMLINK_NOFOLLOW.
205  # (it returned ENOSUP.)
206  # therefore we're out of options--we simply cannot chown the
207  # symlink. give up, suppress the error.
208  # (which is what shutil always did in this circumstance.)
209  pass
210  if hasattr(st, 'st_flags'):
211  try:
212  lookup("chflags")(dst, st.st_flags, follow_symlinks=follow)
213  except OSError as why:
214  for err in 'EOPNOTSUPP', 'ENOTSUP':

267 def copytree(src, dst, symlinks=False, ignore=None, copy_function=copy2,
268  ignore_dangling_symlinks=False):
269  """Recursively copy a directory tree.
270 
271  The destination directory must not already exist.
272  If exception(s) occur, an Error is raised with a list of reasons.
273 
274  If the optional symlinks flag is true, symbolic links in the
275  source tree result in symbolic links in the destination tree; if
276  it is false, the contents of the files pointed to by symbolic
277  links are copied. If the file pointed by the symlink doesn't
278  exist, an exception will be added in the list of errors raised in
279  an Error exception at the end of the copy process.
280 
281  You can set the optional ignore_dangling_symlinks flag to true if you
282  want to silence this exception. Notice that this has no effect on
283  platforms that don't support os.symlink.
284 
285  The optional ignore argument is a callable. If given, it
286  is called with the `src` parameter, which is the directory
287  being visited by copytree(), and `names` which is the list of
288  `src` contents, as returned by os.listdir():
289 
290  callable(src, names) -> ignored_names
291 
292  Since copytree() is called recursively, the callable will be
293  called once for each directory that is copied. It returns a
294  list of names relative to the `src` directory that should
295  not be copied.
296 
297  The optional copy_function argument is a callable that will be used
298  to copy each file. It will be called with the source path and the
299  destination path as arguments. By default, copy2() is used, but any
300  function that supports the same signature (like copy()) can be used.
301 
302  """
303  names = os.listdir(src)
304  if ignore is not None:
305  ignored_names = ignore(src, names)
306  else:
307  ignored_names = set()
308 
309  os.makedirs(dst)
310  errors = []
311  for name in names:
312  if name in ignored_names:
313  continue
314  srcname = os.path.join(src, name)
315  dstname = os.path.join(dst, name)
316  try:
317  if os.path.islink(srcname):
318  linkto = os.readlink(srcname)
319  if symlinks:
320  # We can't just leave it to `copy_function` because legacy
321  # code with a custom `copy_function` may rely on copytree
322  # doing the right thing.
323  os.symlink(linkto, dstname)
324  copystat(srcname, dstname, follow_symlinks=not symlinks)
325  else:
326  # ignore dangling symlink if the flag is on
327  if not os.path.exists(linkto) and ignore_dangling_symlinks:
328  continue
329  # otherwise let the copy occurs. copy2 will raise an error
330  if os.path.isdir(srcname):
331  copytree(srcname, dstname, symlinks, ignore,
332  copy_function)
333  else:
334  copy_function(srcname, dstname)
335  elif os.path.isdir(srcname):
336  copytree(srcname, dstname, symlinks, ignore, copy_function)
337  else:
338  # Will raise a SpecialFileError for unsupported file types
339  copy_function(srcname, dstname)
340  # catch the Error from the recursive copytree so that we can
341  # continue with other files
342  except Error as err:
343  errors.extend(err.args[0])
344  except OSError as why:
345  errors.append((srcname, dstname, str(why)))
346  try:
347  copystat(src, dst)
348  except OSError as why:
349  # Copying file access times may fail on Windows
350  if getattr(why, 'winerror', None) is None:
351  errors.append((src, dst, str(why)))
352  if errors:
353  raise Error(errors)
354  return dst

439 def rmtree(path, ignore_errors=False, onerror=None):
440  """Recursively delete a directory tree.
441 
442  If ignore_errors is set, errors are ignored; otherwise, if onerror
443  is set, it is called to handle the error with arguments (func,
444  path, exc_info) where func is platform and implementation dependent;
445  path is the argument to that function that caused it to fail; and
446  exc_info is a tuple returned by sys.exc_info(). If ignore_errors
447  is false and onerror is None, an exception is raised.
448 
449  """
450  if ignore_errors:
451  def onerror(*args):
452  pass
453  elif onerror is None:
454  def onerror(*args):
455  raise
456  if _use_fd_functions:
457  # While the unsafe rmtree works fine on bytes, the fd based does not.
458  if isinstance(path, bytes):
459  path = os.fsdecode(path)
460  # Note: To guard against symlink races, we use the standard
461  # lstat()/open()/fstat() trick.
462  try:
463  orig_st = os.lstat(path)
464  except Exception:
465  onerror(os.lstat, path, sys.exc_info())
466  return
467  try:
468  fd = os.open(path, os.O_RDONLY)
469  except Exception:
470  onerror(os.lstat, path, sys.exc_info())
471  return
472  try:
473  if os.path.samestat(orig_st, os.fstat(fd)):
474  _rmtree_safe_fd(fd, path, onerror)
475  try:
476  os.rmdir(path)
477  except OSError:
478  onerror(os.rmdir, path, sys.exc_info())
479  else:
480  try:
481  # symlinks to directories are forbidden, see bug #1669
482  raise OSError("Cannot call rmtree on a symbolic link")
483  except OSError:
484  onerror(os.path.islink, path, sys.exc_info())
485  finally:
486  os.close(fd)
487  else:
488  return _rmtree_unsafe(path, onerror)

500 def move(src, dst, copy_function=copy2):
501  """Recursively move a file or directory to another location. This is
502  similar to the Unix "mv" command. Return the file or directory's
503  destination.
504 
505  If the destination is a directory or a symlink to a directory, the source
506  is moved inside the directory. The destination path must not already
507  exist.
508 
509  If the destination already exists but is not a directory, it may be
510  overwritten depending on os.rename() semantics.
511 
512  If the destination is on our current filesystem, then rename() is used.
513  Otherwise, src is copied to the destination and then removed. Symlinks are
514  recreated under the new name if os.rename() fails because of cross
515  filesystem renames.
516 
517  The optional `copy_function` argument is a callable that will be used
518  to copy the source or it will be delegated to `copytree`.
519  By default, copy2() is used, but any function that supports the same
520  signature (like copy()) can be used.
521 
522  A lot more could be done here... A look at a mv.c shows a lot of
523  the issues this implementation glosses over.
524 
525  """
526  real_dst = dst
527  if os.path.isdir(dst):
528  if _samefile(src, dst):
529  # We might be on a case insensitive filesystem,
530  # perform the rename anyway.
531  os.rename(src, dst)
532  return
533 
534  real_dst = os.path.join(dst, _basename(src))
535  if os.path.exists(real_dst):
536  raise Error("Destination path '%s' already exists" % real_dst)
537  try:
538  os.rename(src, real_dst)
539  except OSError:
540  if os.path.islink(src):
541  linkto = os.readlink(src)
542  os.symlink(linkto, real_dst)
543  os.unlink(src)
544  elif os.path.isdir(src):
545  if _destinsrc(src, dst):
546  raise Error("Cannot move a directory '%s' into itself"
547  " '%s'." % (src, dst))
548  copytree(src, real_dst, copy_function=copy_function,
549  symlinks=True)
550  rmtree(src)
551  else:
552  copy_function(src, real_dst)
553  os.unlink(src)
554  return real_dst

589 def _make_tarball(base_name, base_dir, compress="gzip", verbose=0, dry_run=0,
590  owner=None, group=None, logger=None):
591  """Create a (possibly compressed) tar file from all the files under
592  'base_dir'.
593 
594  'compress' must be "gzip" (the default), "bzip2", "xz", or None.
595 
596  'owner' and 'group' can be used to define an owner and a group for the
597  archive that is being built. If not provided, the current owner and group
598  will be used.
599 
600  The output tar file will be named 'base_name' + ".tar", possibly plus
601  the appropriate compression extension (".gz", ".bz2", or ".xz").
602 
603  Returns the output filename.
604  """
605  tar_compression = {'gzip': 'gz', None: ''}
606  compress_ext = {'gzip': '.gz'}
607 
608  if _BZ2_SUPPORTED:
609  tar_compression['bzip2'] = 'bz2'
610  compress_ext['bzip2'] = '.bz2'
611 
612  if _LZMA_SUPPORTED:
613  tar_compression['xz'] = 'xz'
614  compress_ext['xz'] = '.xz'
615 
616  # flags for compression program, each element of list will be an argument
617  if compress is not None and compress not in compress_ext:
618  raise ValueError("bad value for 'compress', or compression format not "
619  "supported : {0}".format(compress))
620 
621  archive_name = base_name + '.tar' + compress_ext.get(compress, '')
622  archive_dir = os.path.dirname(archive_name)
623 
624  if archive_dir and not os.path.exists(archive_dir):
625  if logger is not None:
626  logger.info("creating %s", archive_dir)
627  if not dry_run:
628  os.makedirs(archive_dir)
629 
630  # creating the tarball
631  if logger is not None:
632  logger.info('Creating tar archive')
633 
634  uid = _get_uid(owner)
635  gid = _get_gid(group)
636 
637  def _set_uid_gid(tarinfo):
638  if gid is not None:
639  tarinfo.gid = gid
640  tarinfo.gname = group
641  if uid is not None:
642  tarinfo.uid = uid
643  tarinfo.uname = owner
644  return tarinfo
645 
646  if not dry_run:
647  tar = tarfile.open(archive_name, 'w|%s' % tar_compression[compress])
648  try:
649  tar.add(base_dir, filter=_set_uid_gid)
650  finally:
651  tar.close()
652 
653  return archive_name

1104  if _access_check(cmd, mode):
1105  return cmd
1106  return None
1107 
1108  if path is None:
1109  path = os.environ.get("PATH", os.defpath)
1110  if not path:
1111  return None

1115  # The current directory takes precedence on Windows.
1116  if not os.curdir in path:
1117  path.insert(0, os.curdir)
