137 def collect_string_fields(format_string):
138  """ Given a format string, return an iterator
139  of all the valid format fields. It handles nested fields
140  as well.
141  """
142 
143  formatter = string.Formatter()
144  try:
145  parseiterator = formatter.parse(format_string)
146  for result in parseiterator:
147  if all(item is None for item in result[1:]):
148  # not a replacement format
149  continue
150  name = result[1]
151  nested = result[2]
152  yield name
153  if nested:
154  for field in collect_string_fields(nested):
155  yield field
156  except ValueError as exc:
157  # Probably the format string is invalid.
158  if exc.args[0].startswith("cannot switch from manual"):
159  # On Jython, parsing a string with both manual
160  # and automatic positions will fail with a ValueError,
161  # while on CPython it will simply return the fields,
162  # the validation being done in the interpreter (?).
163  # We're just returning two mixed fields in order
164  # to trigger the format-combined-specification check.
165  yield ""
166  yield "1"
167  return
168  raise utils.IncompleteFormatString(format_string)

170 def parse_format_method_string(format_string):
171  """
172  Parses a PEP 3101 format string, returning a tuple of
173  (keys, num_args, manual_pos_arg),
174  where keys is the set of mapping keys in the format string, num_args
175  is the number of arguments required by the format string and
176  manual_pos_arg is the number of arguments passed with the position.
177  """
178  keys = []
179  num_args = 0
180  manual_pos_arg = set()
181  for name in collect_string_fields(format_string):
182  if name and str(name).isdigit():
183  manual_pos_arg.add(str(name))
184  elif name:
185  keyname, fielditerator = split_format_field_names(name)
186  if isinstance(keyname, numbers.Number):
187  # In Python 2 it will return long which will lead
188  # to different output between 2 and 3
189  manual_pos_arg.add(str(keyname))
190  keyname = int(keyname)
191  try:
192  keys.append((keyname, list(fielditerator)))
193  except ValueError:
194  raise utils.IncompleteFormatString()
195  else:
196  num_args += 1
197  return keys, num_args, len(manual_pos_arg)

214 def get_access_path(key, parts):
215  """ Given a list of format specifiers, returns
216  the final access path (e.g. a.b.c[0][1]).
217  """
218  path = []

237  def visit_binop(self, node):
238  if node.op != '%':
239  return

241  args = node.right
242 
243  if not (isinstance(left, astroid.Const)
244  and isinstance(left.value, six.string_types)):

253  node=node, args=(c, ord(c), e.index))

255  except utils.IncompleteFormatString:
256  self.add_message('truncated-format-string', node=node)
257  return

259  # The format string uses both named and unnamed format
260  # specifiers.
261  self.add_message('mixed-format-string', node=node)
262  elif required_keys:
263  # The format string uses only named format specifiers.
264  # Check that the RHS of the % operator is a mapping object
265  # that contains precisely the set of keys required by the
266  # format string.
267  if isinstance(args, astroid.Dict):
268  keys = set()
269  unknown_keys = False
270  for k, _ in args.items:
271  if isinstance(k, astroid.Const):
272  key = k.value
273  if isinstance(key, six.string_types):
274  keys.add(key)
275  else:
276  self.add_message('bad-format-string-key',
277  node=node, args=key)
278  else:
279  # One of the keys was something other than a
280  # constant. Since we can't tell what it is,
281  # suppress checks for missing keys in the
282  # dictionary.
283  unknown_keys = True
284  if not unknown_keys:
285  for key in required_keys:
286  if key not in keys:
287  self.add_message('missing-format-string-key',
288  node=node, args=key)
289  for key in keys:
290  if key not in required_keys:
291  self.add_message('unused-format-string-key',
292  node=node, args=key)
293  elif isinstance(args, OTHER_NODES + (astroid.Tuple,)):
294  type_name = type(args).__name__
295  self.add_message('format-needs-mapping',
296  node=node, args=type_name)
297  # else:
298  # The RHS of the format specifier is a name or
299  # expression. It may be a mapping object, so
300  # there's nothing we can check.
301  else:
302  # The format string uses only unnamed format specifiers.
303  # Check that the number of arguments passed to the RHS of
304  # the % operator matches the number required by the format
305  # string.
306  if isinstance(args, astroid.Tuple):
307  rhs_tuple = utils.safe_infer(args)
308  num_args = None
309  if rhs_tuple not in (None, astroid.Uninferable):
310  num_args = len(rhs_tuple.elts)
311  elif isinstance(args, OTHER_NODES + (astroid.Dict, astroid.DictComp)):
312  num_args = 1
313  else:
314  # The RHS of the format specifier is a name or
315  # expression. It could be a tuple of unknown size, so
316  # there's nothing we can check.
317  num_args = None
318  if num_args is not None:
319  if num_args > required_num_args:
320  self.add_message('too-many-format-args', node=node)
321  elif num_args < required_num_args:
322  self.add_message('too-few-format-args', node=node)

342  def _check_new_format(self, node, func):
343  """ Check the new string formatting. """
344  # TODO: skip (for now) format nodes which don't have
345  # an explicit string on the left side of the format operation.
346  # We do this because our inference engine can't properly handle
347  # redefinitions of the original string.
348  # For more details, see issue 287.
349  #
350  # Note that there may not be any left side at all, if the format method
351  # has been assigned to another variable. See issue 351. For example:
352  #
353  # fmt = 'some string {}'.format
354  # fmt('arg')
355  if (isinstance(node.func, astroid.Attribute)
356  and not isinstance(node.func.expr, astroid.Const)):
357  return
358  try:
359  strnode = next(func.bound.infer())
360  except astroid.InferenceError:
361  return
362  if not isinstance(strnode, astroid.Const):
363  return
364  if not isinstance(strnode.value, six.string_types):
365  return
366 
367  if node.starargs or node.kwargs:
368  return
369  try:
370  positional, named = get_args(node)
371  except astroid.InferenceError:
372  return
373  try:
374  fields, num_args, manual_pos = parse_format_method_string(strnode.value)
375  except utils.IncompleteFormatString:
376  self.add_message('bad-format-string', node=node)
377  return
378 
379  named_fields = set(field[0] for field in fields
380  if isinstance(field[0], six.string_types))
381  if num_args and manual_pos:
382  self.add_message('format-combined-specification',
383  node=node)
384  return
385 
386  check_args = False
387  # Consider "{[0]} {[1]}" as num_args.
388  num_args += sum(1 for field in named_fields
389  if field == '')
390  if named_fields:
391  for field in named_fields:
392  if field not in named and field:
393  self.add_message('missing-format-argument-key',
394  node=node,
395  args=(field, ))
396  for field in named:
397  if field not in named_fields:
398  self.add_message('unused-format-string-argument',
399  node=node,
400  args=(field, ))
401  # num_args can be 0 if manual_pos is not.
402  num_args = num_args or manual_pos
403  if positional or num_args:
404  empty = any(True for field in named_fields
405  if field == '')
406  if named or empty:
407  # Verify the required number of positional arguments
408  # only if the .format got at least one keyword argument.
409  # This means that the format strings accepts both
410  # positional and named fields and we should warn
411  # when one of the them is missing or is extra.
412  check_args = True
413  else:
414  check_args = True
415  if check_args:
416  # num_args can be 0 if manual_pos is not.
417  num_args = num_args or manual_pos
418  if positional > num_args:
419  self.add_message('too-many-format-args', node=node)
420  elif positional < num_args:
421  self.add_message('too-few-format-args', node=node)

425  def _check_new_format_specifiers(self, node, fields, named):
426  """
427  Check attribute and index access in the format
428  string ("{0.a}" and "{0[a]}").
429  """
430  for key, specifiers in fields:
431  # Obtain the argument. If it can't be obtained
432  # or infered, skip this check.
433  if key == '':
434  # {[0]} will have an unnamed argument, defaulting
435  # to 0. It will not be present in `named`, so use the value
436  # 0 for it.
437  key = 0
438  if isinstance(key, numbers.Number):
439  try:
440  argname = utils.get_argument_from_call(node, key)
441  except utils.NoSuchArgumentError:
442  continue
443  else:
444  if key not in named:
445  continue
446  argname = named[key]
447  if argname in (astroid.YES, None):
448  continue
449  try:
450  argument = next(argname.infer())
451  except astroid.InferenceError:
452  continue
453  if not specifiers or argument is astroid.YES:
454  # No need to check this key if it doesn't
455  # use attribute / item access
456  continue
457  if argument.parent and isinstance(argument.parent, astroid.Arguments):
458  # Ignore any object coming from an argument,
459  # because we can't infer its value properly.
460  continue
461  previous = argument
462  parsed = []
463  for is_attribute, specifier in specifiers:
464  if previous is astroid.YES:
465  break
466  parsed.append((is_attribute, specifier))
467  if is_attribute:
468  try:
469  previous = previous.getattr(specifier)[0]
470  except astroid.NotFoundError:
471  if (hasattr(previous, 'has_dynamic_getattr') and
472  previous.has_dynamic_getattr()):
473  # Don't warn if the object has a custom __getattr__
474  break
475  path = get_access_path(key, parsed)
476  self.add_message('missing-format-attribute',
477  args=(specifier, path),
478  node=node)
479  break
480  else:
481  warn_error = False
482  if hasattr(previous, 'getitem'):
483  try:
484  previous = previous.getitem(astroid.Const(specifier))
485  except (astroid.AstroidIndexError,
486  astroid.AstroidTypeError,
487  astroid.AttributeInferenceError):
488  warn_error = True
489  except astroid.InferenceError:
490  break
491  if previous is astroid.Uninferable:
492  break
493  else:
494  try:
495  # Lookup __getitem__ in the current node,
496  # but skip further checks, because we can't
497  # retrieve the looked object
498  previous.getattr('__getitem__')
499  break
500  except astroid.NotFoundError:
501  warn_error = True
502  if warn_error:
503  path = get_access_path(key, parsed)
504  self.add_message('invalid-format-index',
505  args=(specifier, path),
506  node=node)
507  break
508 
509  try:
510  previous = next(previous.infer())
511  except astroid.InferenceError:
512  # can't check further if we can't infer it
513  break

554  def process_string_token(self, token, start_row):
555  for i, c in enumerate(token):
556  if c in '\'\"':

570  def process_non_raw_string_token(self, prefix, string_body, start_row):
571  """check for bad escapes in a non-raw string.
572 
573  prefix: lowercase string of eg 'ur' string prefix markers.
574  string_body: the un-parsed body of the string, not including the quote
575  marks.
576  start_row: integer line number in the source.
577  """
578  # Walk through the string; if we see a backslash then escape the next
579  # character, and skip over it. If we see a non-escaped character,
580  # alert, and continue.
581  #
582  # Accept a backslash when it escapes a backslash, or a quote, or
583  # end-of-line, or one of the letters that introduce a special escape
584  # sequence <http://docs.python.org/reference/lexical_analysis.html>
585  #
586  # TODO(mbp): Maybe give a separate warning about the rarely-used
587  # \a \b \v \f?
588  #
589  # TODO(mbp): We could give the column of the problem character, but
590  # add_message doesn't seem to have a way to pass it through at present.
591  i = 0
592  while True:
593  i = string_body.find('\\', i)
594  if i == -1:
595  break
596  # There must be a next character; having a backslash at the end
597  # of the string would be a SyntaxError.
598  next_char = string_body[i+1]
599  match = string_body[i:i+2]
600  if next_char in self.UNICODE_ESCAPE_CHARACTERS:
601  if 'u' in prefix:
602  pass
603  elif (_PY3K or self._unicode_literals) and 'b' not in prefix:
604  pass # unicode by default
605  else:
606  self.add_message('anomalous-unicode-escape-in-string',
607  line=start_row, args=(match, ))
608  elif next_char not in self.ESCAPE_CHARACTERS:
609  self.add_message('anomalous-backslash-in-string',
610  line=start_row, args=(match, ))
611  # Whether it was a valid escape or not, backslash followed by
612  # another character can always be consumed whole: the second
613  # character can never be the start of a new backslash escape.
614  i += 2
