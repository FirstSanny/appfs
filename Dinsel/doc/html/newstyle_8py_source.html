53 class NewStyleConflictChecker(BaseChecker):
54  """checks for usage of new style capabilities on old style classes and
55  other new/old styles conflicts problems
56  * use of property, __slots__, super
57  * "super" usage
58  """
59 
60  __implements__ = (IAstroidChecker,)
61 
62  # configuration section name
63  name = 'newstyle'
64  # messages
65  msgs = MSGS
66  priority = -2
67  # configuration options
68  options = ()

70  @check_messages('slots-on-old-class', 'old-style-class')
71  def visit_classdef(self, node):
72  """ Check __slots__ in old style classes and old
73  style class definition.
74  """
75  if '__slots__' in node and not node.newstyle:
76  confidence = (INFERENCE if has_known_bases(node)
77  else INFERENCE_FAILURE)
78  self.add_message('slots-on-old-class', node=node,
79  confidence=confidence)
80  # The node type could be class, exception, metaclass, or
81  # interface. Presumably, the non-class-type nodes would always
82  # have an explicit base class anyway.
83  if not node.bases and node.type == 'class' and not node.metaclass():
84  # We use confidence HIGH here because this message should only ever
85  # be emitted for classes at the root of the inheritance hierarchyself.
86  self.add_message('old-style-class', node=node, confidence=HIGH)

88  @check_messages('property-on-old-class')
89  def visit_call(self, node):
90  """check property usage"""
91  parent = node.parent.frame()
92  if (isinstance(parent, astroid.ClassDef) and
93  not parent.newstyle and
94  isinstance(node.func, astroid.Name)):
95  confidence = (INFERENCE if has_known_bases(parent)
96  else INFERENCE_FAILURE)
97  name = node.func.name
98  if name == 'property':
99  self.add_message('property-on-old-class', node=node,
100  confidence=confidence)

102  @check_messages('super-on-old-class', 'bad-super-call', 'missing-super-argument')
103  def visit_functiondef(self, node):
104  """check use of super"""
105  # ignore actual functions or method within a new style class
106  if not node.is_method():
107  return
108  klass = node.parent.frame()
109  for stmt in node.nodes_of_class(astroid.Call):
110  if node_frame_class(stmt) != node_frame_class(node):
111  # Don't look down in other scopes.
112  continue
113 
114  expr = stmt.func
115  if not isinstance(expr, astroid.Attribute):
116  continue
117 
118  call = expr.expr
119  # skip the test if using super
120  if not (isinstance(call, astroid.Call) and
121  isinstance(call.func, astroid.Name) and
122  call.func.name == 'super'):
123  continue
124 
125  if not klass.newstyle and has_known_bases(klass):
126  # super should not be used on an old style class
127  self.add_message('super-on-old-class', node=node)
128  else:
129  # super first arg should be the class
130  if not call.args:
131  if sys.version_info[0] == 3:
132  # unless Python 3
133  continue
134  else:
135  self.add_message('missing-super-argument', node=call)
136  continue
137 
138  # calling super(type(self), self) can lead to recursion loop
139  # in derived classes
140  arg0 = call.args[0]
141  if isinstance(arg0, astroid.Call) and \
142  isinstance(arg0.func, astroid.Name) and \
143  arg0.func.name == 'type':
144  self.add_message('bad-super-call', node=call, args=('type', ))
145  continue
146 
147  # calling super(self.__class__, self) can lead to recursion loop
148  # in derived classes
149  if len(call.args) >= 2 and \
150  isinstance(call.args[1], astroid.Name) and \
151  call.args[1].name == 'self' and \
152  isinstance(arg0, astroid.Attribute) and \
153  arg0.attrname == '__class__':
154  self.add_message('bad-super-call', node=call, args=('self.__class__', ))
155  continue
156 
157  try:
158  supcls = call.args and next(call.args[0].infer(), None)
159  except astroid.InferenceError:
160  continue
161 
162  if klass is not supcls:
163  name = None
164  # if supcls is not YES, then supcls was infered
165  # and use its name. Otherwise, try to look
166  # for call.args[0].name
167  if supcls:
168  name = supcls.name
169  elif call.args and hasattr(call.args[0], 'name'):
170  name = call.args[0].name
171  if name:
172  self.add_message('bad-super-call', node=call, args=(name, ))
173
