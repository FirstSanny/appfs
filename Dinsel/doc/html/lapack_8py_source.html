469 def get_lapack_funcs(names, arrays=(), dtype=None):
470  """Return available LAPACK function objects from names.
471 
472  Arrays are used to determine the optimal prefix of LAPACK routines.
473 
474  Parameters
475  ----------
476  names : str or sequence of str
477  Name(s) of LAPACK functions without type prefix.
478 
479  arrays : sequence of ndarrays, optional
480  Arrays can be given to determine optimal prefix of LAPACK
481  routines. If not given, double-precision routines will be
482  used, otherwise the most generic type in arrays will be used.
483 
484  dtype : str or dtype, optional
485  Data-type specifier. Not used if `arrays` is non-empty.
486 
487 
488  Returns
489  -------
490  funcs : list
491  List containing the found function(s).
492 
493 
494  Notes
495  -----
496  This routine automatically chooses between Fortran/C
497  interfaces. Fortran code is used whenever possible for arrays with
498  column major order. In all other cases, C code is preferred.
499 
500  In LAPACK, the naming convention is that all functions start with a
501  type prefix, which depends on the type of the principal
502  matrix. These can be one of {'s', 'd', 'c', 'z'} for the numpy
503  types {float32, float64, complex64, complex128} respectevely, and
504  are stored in attribute `typecode` of the returned functions.
505  """
506  return _get_funcs(names, arrays, dtype,
507  "LAPACK", _flapack, _clapack,
508  "flapack", "clapack", _lapack_alias)

511 def _compute_lwork(routine, *args, **kwargs):
512  """
513  Round floating-point lwork returned by lapack to integer.
514 
515  Several LAPACK routines compute optimal values for LWORK, which
516  they return in a floating-point variable. However, for large
517  values of LWORK, single-precision floating point is not sufficient
518  to hold the exact value --- some LAPACK versions (<= 3.5.0 at
519  least) truncate the returned integer to single precision and in
520  some cases this can be smaller than the required value.
521  """
522  wi = routine(*args, **kwargs)
523  if len(wi) < 2:
524  raise ValueError('')
525  info = wi[-1]
526  if info != 0:
527  raise ValueError("Internal work array size computation failed: "
528  "%d" % (info,))
529 
530  lwork = [w.real for w in wi[:-1]]
531 
532  dtype = getattr(routine, 'dtype', None)
533  if dtype == _np.float32 or dtype == _np.complex64:
534  # Single-precision routine -- take next fp value to work
535  # around possible truncation in LAPACK code
536  lwork = _np.nextafter(lwork, _np.inf, dtype=_np.float32)
537 
538  lwork = _np.array(lwork, _np.int64)
539  if _np.any(_np.logical_or(lwork < 0, lwork > _np.iinfo(_np.int32).max)):
540  raise ValueError("Too large work array required -- computation cannot "
541  "be performed with standard 32-bit LAPACK.")
542  lwork = lwork.astype(_np.int32)
543  if lwork.size == 1:
544  return lwork[0]
545  return lwork
