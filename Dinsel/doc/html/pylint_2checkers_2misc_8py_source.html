52  def _check_note(self, notes, lineno, line):
53  # First, simply check if the notes are in the line at all. This is an
54  # optimisation to prevent using the regular expression on every line,
55  # but rather only on lines which may actually contain one of the notes.
56  # This prevents a pathological problem with lines that are hundreds
57  # of thousands of characters long.
58  for note in self.config.notes:
59  if note in line:
60  break
61  else:
62  return
63 
64  match = notes.search(line)
65  if not match:
66  return
67  self.add_message('fixme', args=line[match.start(1):].rstrip(), line=lineno)
68 
69  def _check_encoding(self, lineno, line, file_encoding):
70  try:
71  return six.text_type(line, file_encoding)
72  except UnicodeDecodeError as ex:
73  self.add_message('invalid-encoded-data', line=lineno,
74  args=(file_encoding, ex.args[2]))

76  def process_module(self, module):
77  """inspect the source file to find encoding problem or fixmes like
78  notes
79  """
80  if self.config.notes:
81  notes = re.compile(
82  r'.*?#\s*(%s)(:*\s*.*)' % "|".join(self.config.notes))
83  else:
84  notes = None
85  if module.file_encoding:
86  encoding = module.file_encoding
87  else:
88  encoding = 'ascii'
89 
90  with module.stream() as stream:
91  for lineno, line in enumerate(stream):
92  line = self._check_encoding(lineno + 1, line, encoding)
93  if line is not None and notes:
94  self._check_note(notes, lineno + 1, line)
