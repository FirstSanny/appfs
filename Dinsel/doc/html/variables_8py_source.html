94 def _detect_global_scope(node, frame, defframe):
95  """ Detect that the given frames shares a global
96  scope.
97 
98  Two frames shares a global scope when neither
99  of them are hidden under a function scope, as well
100  as any of parent scope of them, until the root scope.
101  In this case, depending from something defined later on
102  will not work, because it is still undefined.
103 
104  Example:
105  class A:
106  # B has the same global scope as `C`, leading to a NameError.
107  class B(C): ...
108  class C: ...
109 
110  """
111  def_scope = scope = None
112  if frame and frame.parent:
113  scope = frame.parent.scope()
114  if defframe and defframe.parent:
115  def_scope = defframe.parent.scope()
116  if isinstance(frame, astroid.FunctionDef):
117  # If the parent of the current node is a
118  # function, then it can be under its scope
119  # (defined in, which doesn't concern us) or
120  # the `->` part of annotations. The same goes
121  # for annotations of function arguments, they'll have
122  # their parent the Arguments node.
123  if not isinstance(node.parent,
124  (astroid.FunctionDef, astroid.Arguments)):
125  return False
126  elif any(not isinstance(f, (astroid.ClassDef, astroid.Module))
127  for f in (frame, defframe)):
128  # Not interested in other frames, since they are already
129  # not in a global scope.
130  return False
131 
132  break_scopes = []
133  for s in (scope, def_scope):
134  # Look for parent scopes. If there is anything different
135  # than a module or a class scope, then they frames don't
136  # share a global scope.
137  parent_scope = s
138  while parent_scope:
139  if not isinstance(parent_scope, (astroid.ClassDef, astroid.Module)):
140  break_scopes.append(parent_scope)
141  break
142  if parent_scope.parent:
143  parent_scope = parent_scope.parent.scope()
144  else:
145  break
146  if break_scopes and len(set(break_scopes)) != 1:
147  # Store different scopes than expected.
148  # If the stored scopes are, in fact, the very same, then it means
149  # that the two frames (frame and defframe) shares the same scope,
150  # and we could apply our lineno analysis over them.
151  # For instance, this works when they are inside a function, the node
152  # that uses a definition and the definition itself.
153  return False
154  # At this point, we are certain that frame and defframe shares a scope
155  # and the definition of the first depends on the second.
156  return frame.lineno < defframe.lineno

158 def _fix_dot_imports(not_consumed):
159  """ Try to fix imports with multiple dots, by returning a dictionary
160  with the import names expanded. The function unflattens root imports,
161  like 'xml' (when we have both 'xml.etree' and 'xml.sax'), to 'xml.etree'
162  and 'xml.sax' respectively.
163  """
164  # TODO: this should be improved in issue astroid #46
165  names = {}
166  for name, stmts in six.iteritems(not_consumed):
167  if any(isinstance(stmt, astroid.AssignName)
168  and isinstance(stmt.assign_type(), astroid.AugAssign)
169  for stmt in stmts):
170  continue
171  for stmt in stmts:
172  if not isinstance(stmt, (astroid.ImportFrom, astroid.Import)):
173  continue
174  for imports in stmt.names:
175  second_name = None
176  if imports[0] == "*":
177  # In case of wildcard imports,
178  # pick the name from inside the imported module.
179  second_name = name
180  else:
181  if imports[0].find(".") > -1 or name in imports:
182  # Most likely something like 'xml.etree',
183  # which will appear in the .locals as 'xml'.
184  # Only pick the name if it wasn't consumed.
185  second_name = imports[0]
186  if second_name and second_name not in names:
187  names[second_name] = stmt
188  return sorted(names.items(), key=lambda a: a[1].fromlineno)

190 def _find_frame_imports(name, frame):
191  """
192  Detect imports in the frame, with the required
193  *name*. Such imports can be considered assignments.
194  Returns True if an import for the given name was found.
195  """
196  imports = frame.nodes_of_class((astroid.Import, astroid.ImportFrom))
197  for import_node in imports:
198  for import_name, import_alias in import_node.names:
199  # If the import uses an alias, check only that.
200  # Otherwise, check only the import name.
201  if import_alias:
202  if import_alias == name:
203  return True
204  elif import_name and import_name == name:
205  return True

453  def _check_all(self, node, not_consumed):
454  assigned = next(node.igetattr('__all__'))
455  if assigned is astroid.YES:
456  return
457 
458  for elt in getattr(assigned, 'elts', ()):
459  try:
460  elt_name = next(elt.infer())
461  except astroid.InferenceError:
462  continue
463  if elt_name is astroid.YES:
464  continue
465  if not elt_name.parent:
466  continue
467 
468  if (not isinstance(elt_name, astroid.Const)
469  or not isinstance(elt_name.value, six.string_types)):
470  self.add_message('invalid-all-object',
471  args=elt.as_string(), node=elt)
472  continue
473 
474  elt_name = elt_name.value
475  # If elt is in not_consumed, remove it from not_consumed
476  if elt_name in not_consumed:
477  del not_consumed[elt_name]
478  continue
479

481  if not node.package:
482  self.add_message('undefined-all-variable',

484  node=elt)

486  basename = os.path.splitext(node.file)[0]
487  if os.path.basename(basename) == '__init__':

494  node=elt)
495  except SyntaxError:
496  # don't yield an syntax-error warning,

508  def _check_imports(self, not_consumed):
509  local_names = _fix_dot_imports(not_consumed)
510  checked = set()
511  for name, stmt in local_names:
512  for imports in stmt.names:
513  real_name = imported_name = imports[0]
514  if imported_name == "*":
515  real_name = name
516  as_name = imports[1]
517  if real_name in checked:
518  continue
519  if name not in (real_name, as_name):
520  continue
521  checked.add(real_name)
522 
523  if (isinstance(stmt, astroid.Import) or
524  (isinstance(stmt, astroid.ImportFrom) and
525  not stmt.modname)):
526  if (isinstance(stmt, astroid.ImportFrom) and
527  SPECIAL_OBJ.search(imported_name)):
528  # Filter special objects (__doc__, __all__) etc.,
529  # because they can be imported for exporting.
530  continue
531  if as_name == "_":
532  continue
533  if as_name is None:
534  msg = "import %s" % imported_name
535  else:
536  msg = "%s imported as %s" % (imported_name, as_name)
537  self.add_message('unused-import', args=msg, node=stmt)
538  elif (isinstance(stmt, astroid.ImportFrom)
539  and stmt.modname != FUTURE):
540 
541  if SPECIAL_OBJ.search(imported_name):
542  # Filter special objects (__doc__, __all__) etc.,
543  # because they can be imported for exporting.
544  continue
545 
546  if _is_from_future_import(stmt, name):
547  # Check if the name is in fact loaded from a
548  # __future__ import in another module.
549  continue
550 
551  if imported_name == '*':
552  self.add_message('unused-wildcard-import',
553  args=name, node=stmt)
554  else:
555  if as_name is None:
556  msg = "%s imported from %s" % (imported_name, stmt.modname)
557  else:
558  fields = (imported_name, stmt.modname, as_name)
559  msg = "%s imported from %s as %s" % fields
560  self.add_message('unused-import', args=msg, node=stmt)
