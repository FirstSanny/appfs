55 def autocomplete():
56  """Command and option completion for the main option parser (and options)
57  and its subcommands (and options).
58 
59  Enable by sourcing one of the completion shell scripts (bash, zsh or fish).
60  """
61  # Don't complete if user hasn't sourced bash_completion file.
62  if 'PIP_AUTO_COMPLETE' not in os.environ:
63  return
64  cwords = os.environ['COMP_WORDS'].split()[1:]
65  cword = int(os.environ['COMP_CWORD'])
66  try:
67  current = cwords[cword - 1]
68  except IndexError:
69  current = ''
70 
71  subcommands = [cmd for cmd, summary in get_summaries()]
72  options = []
73  # subcommand
74  try:
75  subcommand_name = [w for w in cwords if w in subcommands][0]
76  except IndexError:
77  subcommand_name = None
78 
79  parser = create_main_parser()
80  # subcommand options
81  if subcommand_name:
82  # special case: 'help' subcommand has no options
83  if subcommand_name == 'help':
84  sys.exit(1)
85  # special case: list locally installed dists for uninstall command
86  if subcommand_name == 'uninstall' and not current.startswith('-'):
87  installed = []
88  lc = current.lower()
89  for dist in get_installed_distributions(local_only=True):
90  if dist.key.startswith(lc) and dist.key not in cwords[1:]:
91  installed.append(dist.key)
92  # if there are no dists installed, fall back to option completion
93  if installed:
94  for dist in installed:
95  print(dist)
96  sys.exit(1)
97 
98  subcommand = commands_dict[subcommand_name]()
99  options += [(opt.get_opt_string(), opt.nargs)
100  for opt in subcommand.parser.option_list_all
101  if opt.help != optparse.SUPPRESS_HELP]
102 
103  # filter out previously specified options from available options
104  prev_opts = [x.split('=')[0] for x in cwords[1:cword - 1]]
105  options = [(x, v) for (x, v) in options if x not in prev_opts]
106  # filter options by current input
107  options = [(k, v) for k, v in options if k.startswith(current)]
108  for option in options:
109  opt_label = option[0]
110  # append '=' to options which require args
111  if option[1]:
112  opt_label += '='
113  print(opt_label)
114  else:
115  # show main parser options only when necessary
116  if current.startswith('-') or current.startswith('--'):
117  opts = [i.option_list for i in parser.option_groups]
118  opts.append(parser.option_list)
119  opts = (o for it in opts for o in it)
120 
121  subcommands += [i.get_opt_string() for i in opts
122  if i.help != optparse.SUPPRESS_HELP]
123 
124  print(' '.join([x for x in subcommands if x.startswith(current)]))
125  sys.exit(1)

158 def parseopts(args):
159  parser = create_main_parser()
160 
161  # Note: parser calls disable_interspersed_args(), so the result of this
162  # call is to split the initial args into the general options before the
163  # subcommand and everything else.
164  # For example:
165  # args: ['--timeout=5', 'install', '--user', 'INITools']
166  # general_options: ['--timeout==5']
167  # args_else: ['install', '--user', 'INITools']
168  general_options, args_else = parser.parse_args(args)
169 
170  # --version
171  if general_options.version:
172  sys.stdout.write(parser.version)
173  sys.stdout.write(os.linesep)
174  sys.exit()
175 
176  # pip || pip help -> print_help()
177  if not args_else or (args_else[0] == 'help' and len(args_else) == 1):
178  parser.print_help()
179  sys.exit()
180 
181  # the subcommand name
182  cmd_name = args_else[0]
183 
184  if cmd_name not in commands_dict:
185  guess = get_similar_commands(cmd_name)
186 
187  msg = ['unknown command "%s"' % cmd_name]
188  if guess:
189  msg.append('maybe you meant "%s"' % guess)
190 
191  raise CommandError(' - '.join(msg))
192 
193  # all the args without the subcommand
194  cmd_args = args[:]
195  cmd_args.remove(cmd_name)
