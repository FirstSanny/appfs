62 def spsolve(A, b, permc_spec=None, use_umfpack=True):
63  """Solve the sparse linear system Ax=b, where b may be a vector or a matrix.
64 
65  Parameters
66  ----------
67  A : ndarray or sparse matrix
68  The square matrix A will be converted into CSC or CSR form
69  b : ndarray or sparse matrix
70  The matrix or vector representing the right hand side of the equation.
71  If a vector, b.shape must be (n,) or (n, 1).
72  permc_spec : str, optional
73  How to permute the columns of the matrix for sparsity preservation.
74  (default: 'COLAMD')
75 
76  - ``NATURAL``: natural ordering.
77  - ``MMD_ATA``: minimum degree ordering on the structure of A^T A.
78  - ``MMD_AT_PLUS_A``: minimum degree ordering on the structure of A^T+A.
79  - ``COLAMD``: approximate minimum degree column ordering
80  use_umfpack : bool, optional
81  if True (default) then use umfpack for the solution. This is
82  only referenced if b is a vector and ``scikit-umfpack`` is installed.
83 
84  Returns
85  -------
86  x : ndarray or sparse matrix
87  the solution of the sparse linear equation.
88  If b is a vector, then x is a vector of size A.shape[1]
89  If b is a matrix, then x is a matrix of size (A.shape[1], b.shape[1])
90 
91  Notes
92  -----
93  For solving the matrix expression AX = B, this solver assumes the resulting
94  matrix X is sparse, as is often the case for very sparse inputs. If the
95  resulting X is dense, the construction of this sparse result will be
96  relatively expensive. In that case, consider converting A to a dense
97  matrix and using scipy.linalg.solve or its variants.
98  """
99  if not (isspmatrix_csc(A) or isspmatrix_csr(A)):
100  A = csc_matrix(A)
101  warn('spsolve requires A be CSC or CSR matrix format',
102  SparseEfficiencyWarning)
103 
104  # b is a vector only if b have shape (n,) or (n, 1)
105  b_is_sparse = isspmatrix(b)
106  if not b_is_sparse:
107  b = asarray(b)
108  b_is_vector = ((b.ndim == 1) or (b.ndim == 2 and b.shape[1] == 1))
109 
110  A.sort_indices()
111  A = A.asfptype() # upcast to a floating point format
112  result_dtype = np.promote_types(A.dtype, b.dtype)
113  if A.dtype != result_dtype:
114  A = A.astype(result_dtype)
115  if b.dtype != result_dtype:
116  b = b.astype(result_dtype)
117 
118  # validate input shapes
119  M, N = A.shape
120  if (M != N):
121  raise ValueError("matrix must be square (has shape %s)" % ((M, N),))
122 
123  if M != b.shape[0]:
124  raise ValueError("matrix - rhs dimension mismatch (%s - %s)"
125  % (A.shape, b.shape[0]))
126 
127  use_umfpack = use_umfpack and useUmfpack
128 
129  if b_is_vector and use_umfpack:
130  if b_is_sparse:
131  b_vec = b.toarray()
132  else:
133  b_vec = b
134  b_vec = asarray(b_vec, dtype=A.dtype).ravel()
135 
136  if noScikit:
137  raise RuntimeError('Scikits.umfpack not installed.')
138 
139  if A.dtype.char not in 'dD':
140  raise ValueError("convert matrix data to double, please, using"
141  " .astype(), or set linsolve.useUmfpack = False")
142 
143  umf = umfpack.UmfpackContext(_get_umf_family(A))
144  x = umf.linsolve(umfpack.UMFPACK_A, A, b_vec,
145  autoTranspose=True)
146  else:
147  if b_is_vector and b_is_sparse:
148  b = b.toarray()
149  b_is_sparse = False
150 
151  if not b_is_sparse:
152  if isspmatrix_csc(A):
153  flag = 1 # CSC format
154  else:
155  flag = 0 # CSR format
156 
157  options = dict(ColPerm=permc_spec)
158  x, info = _superlu.gssv(N, A.nnz, A.data, A.indices, A.indptr,
159  b, flag, options=options)
160  if info != 0:
161  warn("Matrix is exactly singular", MatrixRankWarning)
162  x.fill(np.nan)
163  if b_is_vector:
164  x = x.ravel()
165  else:
166  # b is sparse
167  Afactsolve = factorized(A)
168 
169  if not isspmatrix_csc(b):
170  warn('spsolve is more efficient when sparse b '
171  'is in the CSC matrix format', SparseEfficiencyWarning)
172  b = csc_matrix(b)
173 
174  # Create a sparse output matrix by repeatedly applying
175  # the sparse factorization to solve columns of b.
176  data_segs = []
177  row_segs = []
178  col_segs = []
179  for j in range(b.shape[1]):
180  bj = b[:, j].A.ravel()
181  xj = Afactsolve(bj)
182  w = np.flatnonzero(xj)
183  segment_length = w.shape[0]
184  row_segs.append(w)
185  col_segs.append(np.ones(segment_length, dtype=int)*j)
186  data_segs.append(np.asarray(xj[w], dtype=A.dtype))
187  sparse_data = np.concatenate(data_segs)
188  sparse_row = np.concatenate(row_segs)
189  sparse_col = np.concatenate(col_segs)
190  x = A.__class__((sparse_data, (sparse_row, sparse_col)),
191  shape=b.shape, dtype=A.dtype)
192 
193  return x

389 def spsolve_triangular(A, b, lower=True, overwrite_A=False, overwrite_b=False):
390  """
391  Solve the equation `A x = b` for `x`, assuming A is a triangular matrix.
392 
393  Parameters
394  ----------
395  A : (M, M) sparse matrix
396  A sparse square triangular matrix. Should be in CSR format.
397  b : (M,) or (M, N) array_like
398  Right-hand side matrix in `A x = b`
399  lower : bool, optional
400  Whether `A` is a lower or upper triangular matrix.
401  Default is lower triangular matrix.
402  overwrite_A : bool, optional
403  Allow changing `A`. The indices of `A` are going to be sorted and zero
404  entries are going to be removed.
405  Enabling gives a performance gain. Default is False.
406  overwrite_b : bool, optional
407  Allow overwriting data in `b`.
408  Enabling gives a performance gain. Default is False.
409 
410  Returns
411  -------
412  x : (M,) or (M, N) ndarray
413  Solution to the system `A x = b`. Shape of return matches shape of `b`.
414 
415  Raises
416  ------
417  LinAlgError
418  If `A` is singular or not triangular.
419  ValueError
420  If shape of `A` or shape of `b` do not match the requirements.
421 
422  Notes
423  -----
424  .. versionadded:: 0.19.0
425  """
426 
427  # Check the input for correct type and format.
428  if not isspmatrix_csr(A):
429  warn('CSR matrix format is required. Converting to CSR matrix.',
430  SparseEfficiencyWarning)
431  A = csr_matrix(A)
432  elif not overwrite_A:
433  A = A.copy()
434 
435  if A.shape[0] != A.shape[1]:
436  raise ValueError('A must be a square matrix but its shape is {}.'.format(A.shape))
437 
438  A.eliminate_zeros()
439  A.sort_indices()
440 
441  b = np.asanyarray(b)
442 
443  if b.ndim not in [1, 2]:
444  raise ValueError('b must have 1 or 2 dims but its shape is {}.'.format(b.shape))
445  if A.shape[0] != b.shape[0]:
446  raise ValueError('The size of the dimensions of A must be equal to '
447  'the size of the first dimension of b but the shape of A is '
448  '{} and the shape of b is {}.'.format(A.shape, b.shape))
449 
450  # Init x as copy of b.
451  if overwrite_b:
452  x = b
453  else:
454  x = b.copy()
455 
456  # Choose forward or backward order.
457  if lower:
458  row_indices = range(len(b))
459  else:
460  row_indices = range(len(b)-1, -1, -1)
461 
462  # Fill x iteratively.
463  for i in row_indices:
464 
465  # Get indices for i-th row.
466  indptr_start = A.indptr[i]
467  indptr_stop = A.indptr[i+1]
468  if lower:
469  A_diagonal_index_row_i = indptr_stop-1
470  A_off_diagonal_indices_row_i = slice(indptr_start,indptr_stop-1)
471  else:
472  A_diagonal_index_row_i = indptr_start
473  A_off_diagonal_indices_row_i = slice(indptr_start+1,indptr_stop)
474 
475  # Check regularity and triangularity of A.
476  if indptr_stop <= indptr_start or A.indices[A_diagonal_index_row_i] < i:
477  raise LinAlgError('A is singular: '
478  '{}th diagonal is zero!'.format(i))
479  if A.indices[A_diagonal_index_row_i] > i:
480  raise LinAlgError('A is no triangular matrix: entry '
481  '[{},{}] is not zero!'.format(i, A.indices[A_diagonal_index_row_i]))
482 
483  # Incorporate off-diagonal entries.
484  A_column_indices_in_row_i = A.indices[A_off_diagonal_indices_row_i]
485  A_values_in_row_i = A.data[A_off_diagonal_indices_row_i]
486  x[i] -= np.dot(x[A_column_indices_in_row_i].T, A_values_in_row_i)
487 
488  # Compute i-th entry of x.
489  x[i] /= A.data[A_diagonal_index_row_i]
490 
491  return x
