108  def visit_classdef(self, node):
109  """return an astroid.ClassDef node as string"""
110  decorate = node.decorators.accept(self) if node.decorators else ''
111  bases = ', '.join([n.accept(self) for n in node.bases])
112  if sys.version_info[0] == 2:
113  bases = '(%s)' % bases if bases else ''
114  else:
115  metaclass = node.metaclass()
116  if metaclass and not node.has_metaclass_hack():
117  if bases:

119  else:

121  else:

123  docs = '\n%s"""%s"""' % (self.indent, node.doc) if node.doc else ''
124  return '\n\n%sclass %s%s:%s\n%s\n' % (decorate, node.name, bases, docs,
