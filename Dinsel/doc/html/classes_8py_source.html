57 def _signature_from_call(call):
58  kws = {}
59  args = []
60  starred_kws = []
61  starred_args = []
62  for keyword in call.keywords or []:
63  arg, value = keyword.arg, keyword.value
64  if arg is None and isinstance(value, astroid.Name):
65  # Starred node and we are interested only in names,
66  # otherwise some transformation might occur for the parameter.
67  starred_kws.append(value.name)
68  elif isinstance(value, astroid.Name):
69  kws[arg] = value.name
70  else:
71  kws[arg] = None
72 
73  for arg in call.args:
74  if isinstance(arg, astroid.Starred) and isinstance(arg.value, astroid.Name):
75  # Positional variadic and a name, otherwise some transformation
76  # might have occurred.
77  starred_args.append(arg.value.name)
78  elif isinstance(arg, astroid.Name):
79  args.append(arg.name)
80  else:
81  args.append(None)
82 
83  return _CallSignature(args, kws, starred_args, starred_kws)

94 def _definition_equivalent_to_call(definition, call):
95  '''Check if a definition signature is equivalent to a call.'''
96  if definition.kwargs:
97  same_kw_variadics = definition.kwargs in call.starred_kws
98  else:
99  same_kw_variadics = not call.starred_kws
100  if definition.varargs:
101  same_args_variadics = definition.varargs in call.starred_args
102  else:
103  same_args_variadics = not call.starred_args
104  same_kwonlyargs = all(kw in call.kws for kw in definition.kwonlyargs)
105  same_args = definition.args == call.args
106 
107  no_additional_kwarg_arguments = True
108  if call.kws:
109  for keyword in call.kws:
110  is_arg = keyword in call.args
111  is_kwonly = keyword in definition.kwonlyargs
112  if not is_arg and not is_kwonly:
113  # Maybe this argument goes into **kwargs,
114  # or it is an extraneous argument.
115  # In any case, the signature is different than
116  # the call site, which stops our search.
117  no_additional_kwarg_arguments = False
118  break
119 
120  return all((
121  same_args,
122  same_kwonlyargs,
123  same_args_variadics,
124  same_kw_variadics,
125  no_additional_kwarg_arguments,
126  ))

137 def _has_different_parameters(original, overridden, dummy_parameter_regex):
138  zipped = six.moves.zip_longest(original, overridden)
139  for original_param, overridden_param in zipped:

152 def _different_parameters(original, overridden, dummy_parameter_regex):
153  """Determine if the two methods have different parameters
154 
155  They are considered to have different parameters if:
156 
157  * they have different positional parameters, including different names
158 
159  * one of the methods is having variadics, while the other is not
160 
161  * they have different keyword only parameters.
162 
163  """
164  original_parameters = _positional_parameters(original)
165  overridden_parameters = _positional_parameters(overridden)
166 
167  different_positional = _has_different_parameters(
168  original_parameters,
169  overridden_parameters,
170  dummy_parameter_regex)
171  different_kwonly = _has_different_parameters(
172  original.args.kwonlyargs,
173  overridden.args.kwonlyargs,
174  dummy_parameter_regex)
175  if original.name in PYMETHODS:
176  # Ignore the difference for special methods. If the parameter
177  # numbers are different, then that is going to be caught by
178  # unexpected-special-method-signature.
179  # If the names are different, it doesn't matter, since they can't
180  # be used as keyword arguments anyway.
181  different_positional = different_kwonly = False
182 
183  # Both or none should have extra variadics, otherwise the method
184  # loses or gains capabilities that are not reflected into the parent method,
185  # leading to potential inconsistencies in the code.
186  different_kwarg = sum(
187  1 for param in (original.args.kwarg, overridden.args.kwarg)
188  if not param) == 1
189  different_vararg = sum(
190  1 for param in (original.args.vararg, overridden.args.vararg)
191  if not param) == 1
192 
193  return any((
194  different_positional,
195  different_kwarg,
196  different_vararg,
197  different_kwonly
198  ))

224 def _called_in_methods(func, klass, methods):
225  """ Check if the func was called in any of the given methods,
226  belonging to the *klass*. Returns True if so, False otherwise.
227  """
228  if not isinstance(func, astroid.FunctionDef):
229  return False
230  for method in methods:
231  try:
232  infered = klass.getattr(method)
233  except astroid.NotFoundError:
234  continue
235  for infer_method in infered:
236  for callfunc in infer_method.nodes_of_class(astroid.Call):
237  try:
238  bound = next(callfunc.func.infer())
239  except (astroid.InferenceError, StopIteration):
240  continue
241  if not isinstance(bound, astroid.BoundMethod):
242  continue
243  func_obj = bound._proxied
244  if isinstance(func_obj, astroid.UnboundMethod):
245  func_obj = func_obj._proxied
246  if func_obj.name == func.name:
247  return True
248  return False

251 def _is_attribute_property(name, klass):
252  """ Check if the given attribute *name* is a property
253  in the given *klass*.
254 
255  It will look for `property` calls or for functions
256  with the given name, decorated by `property` or `property`
257  subclasses.
258  Returns ``True`` if the name is a property in the given klass,
259  ``False`` otherwise.
260  """
261 
262  try:
263  attributes = klass.getattr(name)
264  except astroid.NotFoundError:
265  return False

273  decorated_with_property(infered)):
274  return True

290 def _safe_infer_call_result(node, caller, context=None):
291  """
292  Safely infer the return value of a function.
293 
294  Returns None if inference failed or if there is some ambiguity (more than
295  one node has been inferred). Otherwise returns infered value.
296  """

298  inferit = node.infer_call_result(caller, context=context)
299  value = next(inferit)

301  return # inference failed
302  except StopIteration:

577  def leave_classdef(self, cnode):
578  """close a class node:
579  check that instance attributes are defined in __init__ and check
580  access to existent members
581  """
582  # check access to existent members on non metaclass classes
583  if self._ignore_mixin and cnode.name[-5:].lower() == 'mixin':
584  # We are in a mixin class. No need to try to figure out if
585  # something is missing, since it is most likely that it will
586  # miss.
587  return
588 
589  accessed = self._accessed.accessed(cnode)
590  if cnode.type != 'metaclass':
591  self._check_accessed_members(cnode, accessed)
592  # checks attributes are defined in an allowed method such as __init__
593  if not self.linter.is_message_enabled('attribute-defined-outside-init'):
594  return
595  defining_methods = self.config.defining_attr_methods
596  current_module = cnode.root()
597  for attr, nodes in six.iteritems(cnode.instance_attrs):
598  # skip nodes which are not in the current module and it may screw up
599  # the output, while it's not worth it
600  nodes = [n for n in nodes if not
601  isinstance(n.statement(), (astroid.Delete, astroid.AugAssign))
602  and n.root() is current_module]
603  if not nodes:
604  continue # error detected by typechecking
605  # check if any method attr is defined in is a defining method
606  if any(node.frame().name in defining_methods
607  for node in nodes):
608  continue
609 
610  # check attribute is defined in a parent's __init__
611  for parent in cnode.instance_attr_ancestors(attr):
612  attr_defined = False
613  # check if any parent method attr is defined in is a defining method
614  for node in parent.instance_attrs[attr]:
615  if node.frame().name in defining_methods:
616  attr_defined = True
617  if attr_defined:
618  # we're done :)
619  break
620  else:
621  # check attribute is defined as a class attribute
622  try:
623  cnode.local_attr(attr)
624  except astroid.NotFoundError:
625  for node in nodes:
626  if node.frame().name not in defining_methods:
627  # If the attribute was set by a callfunc in any
628  # of the defining methods, then don't emit
629  # the warning.
630  if _called_in_methods(node.frame(), cnode,
631  defining_methods):
632  continue
633  self.add_message('attribute-defined-outside-init',
634  args=attr, node=node)

691  def _check_useless_super_delegation(self, function):

708  return
