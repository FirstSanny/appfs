37 def cholesky(a, lower=False, overwrite_a=False, check_finite=True):
38  """
39  Compute the Cholesky decomposition of a matrix.
40 
41  Returns the Cholesky decomposition, :math:`A = L L^*` or
42  :math:`A = U^* U` of a Hermitian positive-definite matrix A.
43 
44  Parameters
45  ----------
46  a : (M, M) array_like
47  Matrix to be decomposed
48  lower : bool, optional
49  Whether to compute the upper or lower triangular Cholesky
50  factorization. Default is upper-triangular.
51  overwrite_a : bool, optional
52  Whether to overwrite data in `a` (may improve performance).
53  check_finite : bool, optional
54  Whether to check that the input matrix contains only finite numbers.
55  Disabling may give a performance gain, but may result in problems
56  (crashes, non-termination) if the inputs do contain infinities or NaNs.
57 
58  Returns
59  -------
60  c : (M, M) ndarray
61  Upper- or lower-triangular Cholesky factor of `a`.
62 
63  Raises
64  ------
65  LinAlgError : if decomposition fails.
66 
67  Examples
68  --------
69  >>> from scipy import array, linalg, dot
70  >>> a = array([[1,-2j],[2j,5]])
71  >>> L = linalg.cholesky(a, lower=True)
72  >>> L
73  array([[ 1.+0.j, 0.+0.j],
74  [ 0.+2.j, 1.+0.j]])
75  >>> dot(L, L.T.conj())
76  array([[ 1.+0.j, 0.-2.j],
77  [ 0.+2.j, 5.+0.j]])
78 
79  """
80  c, lower = _cholesky(a, lower=lower, overwrite_a=overwrite_a, clean=True,
81  check_finite=check_finite)
82  return c

137  """Solve the linear equations A x = b, given the Cholesky factorization of A.

151

157  See also

184 def cholesky_banded(ab, overwrite_ab=False, lower=False, check_finite=True):
185  """
186  Cholesky decompose a banded Hermitian positive-definite matrix
187 
188  The matrix a is stored in ab either in lower diagonal or upper
189  diagonal ordered form::
190 
191  ab[u + i - j, j] == a[i,j] (if upper form; i <= j)
192  ab[ i - j, j] == a[i,j] (if lower form; i >= j)
193 
194  Example of ab (shape of a is (6,6), u=2)::
195 
196  upper form:
197  * * a02 a13 a24 a35
198  * a01 a12 a23 a34 a45
199  a00 a11 a22 a33 a44 a55
200 
201  lower form:
202  a00 a11 a22 a33 a44 a55
203  a10 a21 a32 a43 a54 *
204  a20 a31 a42 a53 * *
205 
206  Parameters
207  ----------
208  ab : (u + 1, M) array_like
209  Banded matrix
210  overwrite_ab : bool, optional
211  Discard data in ab (may enhance performance)
212  lower : bool, optional
213  Is the matrix in the lower form. (Default is upper form)
214  check_finite : bool, optional
215  Whether to check that the input matrix contains only finite numbers.
216  Disabling may give a performance gain, but may result in problems
217  (crashes, non-termination) if the inputs do contain infinities or NaNs.
218 
219  Returns
220  -------
221  c : (u + 1, M) ndarray
222  Cholesky factorization of a, in the same banded format as ab
223 
224  """
225  if check_finite:
226  ab = asarray_chkfinite(ab)
227  else:
228  ab = asarray(ab)
229 
230  pbtrf, = get_lapack_funcs(('pbtrf',), (ab,))
231  c, info = pbtrf(ab, lower=lower, overwrite_ab=overwrite_ab)
232  if info > 0:
233  raise LinAlgError("%d-th leading minor not positive definite" % info)
234  if info < 0:
235  raise ValueError('illegal value in %d-th argument of internal pbtrf'
236  % -info)
237  return c
