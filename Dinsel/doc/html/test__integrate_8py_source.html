387 class SimpleOscillator(ODE):
388  r"""
389  Free vibration of a simple oscillator::
390  m \ddot{u} + k u = 0, u(0) = u_0 \dot{u}(0) \dot{u}_0
391  Solution::
392  u(t) = u_0*cos(sqrt(k/m)*t)+\dot{u}_0*sin(sqrt(k/m)*t)/sqrt(k/m)
393  """
394  stop_t = 1 + 0.09
395  z0 = array([1.0, 0.1], float)
396 
397  k = 4.0
398  m = 1.0
399 
400  def f(self, z, t):
401  tmp = zeros((2, 2), float)
402  tmp[0, 1] = 1.0
403  tmp[1, 0] = -self.k / self.m
404  return dot(tmp, z)
405 
406  def verify(self, zs, t):
407  omega = sqrt(self.k / self.m)
408  u = self.z0[0]*cos(omega*t) + self.z0[1]*sin(omega*t)/omega
409  return allclose(u, zs[:, 0], atol=self.atol, rtol=self.rtol)

443 class CoupledDecay(ODE):
444  r"""
445  3 coupled decays suited for banded treatment
446  (banded mode makes it necessary when N>>3)
447  """
448 
449  stiff = True
450  stop_t = 0.5
451  z0 = [5.0, 7.0, 13.0]
452  lband = 1
453  uband = 0
454 
455  lmbd = [0.17, 0.23, 0.29] # fictious decay constants
456 
457  def f(self, z, t):
458  lmbd = self.lmbd
459  return np.array([-lmbd[0]*z[0],
460  -lmbd[1]*z[1] + lmbd[0]*z[0],
461  -lmbd[2]*z[2] + lmbd[1]*z[1]])
462 
463  def jac(self, z, t):
464  # The full Jacobian is
465  #
466  # [-lmbd[0] 0 0 ]
467  # [ lmbd[0] -lmbd[1] 0 ]
468  # [ 0 lmbd[1] -lmbd[2]]
469  #
470  # The lower and upper bandwidths are lband=1 and uband=0, resp.
471  # The representation of this array in packed format is
472  #
473  # [-lmbd[0] -lmbd[1] -lmbd[2]]
474  # [ lmbd[0] lmbd[1] 0 ]
475 
476  lmbd = self.lmbd
477  j = np.zeros((self.lband + self.uband + 1, 3), order='F')
478 
479  def set_j(ri, ci, val):
480  j[self.uband + ri - ci, ci] = val
481  set_j(0, 0, -lmbd[0])
482  set_j(1, 0, lmbd[0])
483  set_j(1, 1, -lmbd[1])
484  set_j(2, 1, lmbd[1])
485  set_j(2, 2, -lmbd[2])
486  return j
487 
488  def verify(self, zs, t):
489  # Formulae derived by hand
490  lmbd = np.array(self.lmbd)
491  d10 = lmbd[1] - lmbd[0]
492  d21 = lmbd[2] - lmbd[1]
493  d20 = lmbd[2] - lmbd[0]
494  e0 = np.exp(-lmbd[0] * t)
495  e1 = np.exp(-lmbd[1] * t)
496  e2 = np.exp(-lmbd[2] * t)
497  u = np.vstack((
498  self.z0[0] * e0,
499  self.z0[1] * e1 + self.z0[0] * lmbd[0] / d10 * (e0 - e1),
500  self.z0[2] * e2 + self.z0[1] * lmbd[1] / d21 * (e1 - e2) +
501  lmbd[1] * lmbd[0] * self.z0[0] / d10 *
502  (1 / d20 * (e0 - e2) - 1 / d21 * (e1 - e2)))).transpose()
503  return allclose(u, zs, atol=self.atol, rtol=self.rtol)
