85  def _runtest(self, args, reporter=None, out=None, code=28):
86  if out is None:
87  out = six.StringIO()
88  pylint_code = self._run_pylint(args, reporter=reporter, out=out)
89  if reporter:
90  output = reporter.out.getvalue()
91  elif hasattr(out, 'getvalue'):
92  output = out.getvalue()
93  else:
94  output = None
95  msg = 'expected output status %s, got %s' % (code, pylint_code)
96  if output is not None:
97  msg = '%s. Below pylint output: \n%s' % (msg, output)
98  assert pylint_code == code, msg

100  def _run_pylint(self, args, out, reporter=None):
101  args = args + ['--persistent=no']
102  with _patch_streams(out):
103  with pytest.raises(SystemExit) as cm:
104  with warnings.catch_warnings():
105  warnings.simplefilter("ignore")
106  Run(args, reporter=reporter)
107  return cm.value.code
