16 def svd(a, full_matrices=True, compute_uv=True, overwrite_a=False,
17  check_finite=True, lapack_driver='gesdd'):
18  """
19  Singular Value Decomposition.
20 
21  Factorizes the matrix a into two unitary matrices U and Vh, and
22  a 1-D array s of singular values (real, non-negative) such that
23  ``a == U*S*Vh``, where S is a suitably shaped matrix of zeros with
24  main diagonal s.
25 
26  Parameters
27  ----------
28  a : (M, N) array_like
29  Matrix to decompose.
30  full_matrices : bool, optional
31  If True, `U` and `Vh` are of shape ``(M,M)``, ``(N,N)``.
32  If False, the shapes are ``(M,K)`` and ``(K,N)``, where
33  ``K = min(M,N)``.
34  compute_uv : bool, optional
35  Whether to compute also `U` and `Vh` in addition to `s`.
36  Default is True.
37  overwrite_a : bool, optional
38  Whether to overwrite `a`; may improve performance.
39  Default is False.
40  check_finite : bool, optional
41  Whether to check that the input matrix contains only finite numbers.
42  Disabling may give a performance gain, but may result in problems
43  (crashes, non-termination) if the inputs do contain infinities or NaNs.
44  lapack_driver : {'gesdd', 'gesvd'}, optional
45  Whether to use the more efficient divide-and-conquer approach
46  (``'gesdd'``) or general rectangular approach (``'gesvd'``)
47  to compute the SVD. MATLAB and Octave use the ``'gesvd'`` approach.
48  Default is ``'gesdd'``.
49 
50  .. versionadded:: 0.18
51 
52  Returns
53  -------
54  U : ndarray
55  Unitary matrix having left singular vectors as columns.
56  Of shape ``(M,M)`` or ``(M,K)``, depending on `full_matrices`.
57  s : ndarray
58  The singular values, sorted in non-increasing order.
59  Of shape (K,), with ``K = min(M, N)``.
60  Vh : ndarray
61  Unitary matrix having right singular vectors as rows.
62  Of shape ``(N,N)`` or ``(K,N)`` depending on `full_matrices`.
63 
64  For ``compute_uv=False``, only `s` is returned.
65 
66  Raises
67  ------
68  LinAlgError
69  If SVD computation does not converge.
70 
71  See also
72  --------
73  svdvals : Compute singular values of a matrix.
74  diagsvd : Construct the Sigma matrix, given the vector s.
75 
76  Examples
77  --------
78  >>> from scipy import linalg
79  >>> a = np.random.randn(9, 6) + 1.j*np.random.randn(9, 6)
80  >>> U, s, Vh = linalg.svd(a)
81  >>> U.shape, Vh.shape, s.shape
82  ((9, 9), (6, 6), (6,))
83 
84  >>> U, s, Vh = linalg.svd(a, full_matrices=False)
85  >>> U.shape, Vh.shape, s.shape
86  ((9, 6), (6, 6), (6,))
87  >>> S = linalg.diagsvd(s, 6, 6)
88  >>> np.allclose(a, np.dot(U, np.dot(S, Vh)))
89  True
90 
91  >>> s2 = linalg.svd(a, compute_uv=False)
92  >>> np.allclose(s, s2)
93  True
94 
95  """
96  a1 = _asarray_validated(a, check_finite=check_finite)
97  if len(a1.shape) != 2:
98  raise ValueError('expected matrix')
99  m, n = a1.shape
100  overwrite_a = overwrite_a or (_datacopied(a1, a))
101 
102  if not isinstance(lapack_driver, string_types):
103  raise TypeError('lapack_driver must be a string')
104  if lapack_driver not in ('gesdd', 'gesvd'):
105  raise ValueError('lapack_driver must be "gesdd" or "gesvd", not "%s"'
106  % (lapack_driver,))
107  funcs = (lapack_driver, lapack_driver + '_lwork')
108  gesXd, gesXd_lwork = get_lapack_funcs(funcs, (a1,))
109 
110  # compute optimal lwork
111  lwork = _compute_lwork(gesXd_lwork, a1.shape[0], a1.shape[1],
112  compute_uv=compute_uv, full_matrices=full_matrices)
113 
114  # perform decomposition
115  u, s, v, info = gesXd(a1, compute_uv=compute_uv, lwork=lwork,
116  full_matrices=full_matrices, overwrite_a=overwrite_a)
117 
118  if info > 0:
119  raise LinAlgError("SVD did not converge")
120  if info < 0:
121  raise ValueError('illegal value in %d-th argument of internal gesdd'
122  % -info)
123  if compute_uv:
124  return u, s, v
125  else:
126  return s

129 def svdvals(a, overwrite_a=False, check_finite=True):
130  """
131  Compute singular values of a matrix.
132 
133  Parameters
134  ----------
135  a : (M, N) array_like
136  Matrix to decompose.
137  overwrite_a : bool, optional
138  Whether to overwrite `a`; may improve performance.
139  Default is False.
140  check_finite : bool, optional
141  Whether to check that the input matrix contains only finite numbers.
142  Disabling may give a performance gain, but may result in problems
143  (crashes, non-termination) if the inputs do contain infinities or NaNs.
144 
145  Returns
146  -------
147  s : (min(M, N),) ndarray
148  The singular values, sorted in decreasing order.
149 
150  Raises
151  ------
152  LinAlgError
153  If SVD computation does not converge.
154 
155  Notes
156  -----
157  ``svdvals(a)`` only differs from ``svd(a, compute_uv=False)`` by its
158  handling of the edge case of empty ``a``, where it returns an
159  empty sequence:
160 
161  >>> a = np.empty((0, 2))
162  >>> from scipy.linalg import svdvals
163  >>> svdvals(a)
164  array([], dtype=float64)
165 
166  See also
167  --------
168  svd : Compute the full singular value decomposition of a matrix.
169  diagsvd : Construct the Sigma matrix, given the vector s.
170 
171  """
172  a = _asarray_validated(a, check_finite=check_finite)
173  if a.size:
174  return svd(a, compute_uv=0, overwrite_a=overwrite_a,
175  check_finite=False)
176  elif len(a.shape) != 2:
177  raise ValueError('expected matrix')
178  else:
179  return numpy.empty(0)

182 def diagsvd(s, M, N):
183  """
184  Construct the sigma matrix in SVD from singular values and size M, N.
185 
186  Parameters
187  ----------
188  s : (M,) or (N,) array_like
189  Singular values
190  M : int
191  Size of the matrix whose singular values are `s`.
192  N : int
193  Size of the matrix whose singular values are `s`.
194 
195  Returns
196  -------
197  S : (M, N) ndarray
198  The S-matrix in the singular value decomposition
199 
200  """
201  part = diag(s)
202  typ = part.dtype.char
203  MorN = len(s)
204  if MorN == M:
205  return r_['-1', part, zeros((M, N-M), typ)]
206  elif MorN == N:
207  return r_[part, zeros((M-N, N), typ)]
208  else:
209  raise ValueError("Length of s must be M or N.")

214 def orth(A):
215  """
216  Construct an orthonormal basis for the range of A using SVD
217 
218  Parameters
219  ----------
220  A : (M, N) array_like
221  Input array
222 
223  Returns
224  -------
225  Q : (M, K) ndarray
226  Orthonormal basis for the range of A.
227  K = effective rank of A, as determined by automatic cutoff
228 
229  See also
230  --------
231  svd : Singular value decomposition of a matrix
232 
233  """
234  u, s, vh = svd(A, full_matrices=False)
235  M, N = A.shape
236  eps = numpy.finfo(float).eps
237  tol = max(M, N) * numpy.amax(s) * eps
238  num = numpy.sum(s > tol, dtype=int)
239  Q = u[:, :num]
240  return Q
