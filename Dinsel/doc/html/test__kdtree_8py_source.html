18 def distance_box(a, b, p, boxsize):
19  diff = a - b
20  diff[diff > 0.5 * boxsize] -= boxsize
21  diff[diff < -0.5 * boxsize] += boxsize
22  d = minkowski_distance(diff, 0, p)
23  return d

1078 def simulate_periodic_box(kdtree, data, k, boxsize):
1079  dd = []
1080  ii = []
1081  x = np.arange(3 ** data.shape[1])
1082  nn = np.array(np.unravel_index(x, [3] * data.shape[1])).T
1083  nn = nn - 1.0
1084  for n in nn:
1085  image = data + n * 1.0 * boxsize
1086  dd2, ii2 = kdtree.query(image, k)
1087  dd2 = dd2.reshape(-1, k)
1088  ii2 = ii2.reshape(-1, k)
1089  dd.append(dd2)
1090  ii.append(ii2)
1091  dd = np.concatenate(dd, axis=-1)
1092  ii = np.concatenate(ii, axis=-1)
1093 
1094  result = np.empty([len(data), len(nn) * k], dtype=[
1095  ('ii', 'i8'),
1096  ('dd', 'f8')])
1097  result['ii'][:] = ii
1098  result['dd'][:] = dd
1099  result.sort(order='dd')
1100  return result['dd'][:, :k], result['ii'][:,:k]
