47 class Super(node_classes.NodeNG):
48  """Proxy class over a super call.
49 
50  This class offers almost the same behaviour as Python's super,
51  which is MRO lookups for retrieving attributes from the parents.
52 
53  The *mro_pointer* is the place in the MRO from where we should
54  start looking, not counting it. *mro_type* is the object which
55  provides the MRO, it can be both a type or an instance.
56  *self_class* is the class where the super call is, while
57  *scope* is the function where the super call is.
58  """
59  # pylint: disable=unnecessary-lambda
60  special_attributes = util.lazy_descriptor(lambda: objectmodel.SuperModel())

73  def super_mro(self):
74  """Get the MRO which will be used to lookup attributes in this super."""
75  if not isinstance(self.mro_pointer, scoped_nodes.ClassDef):
76  raise exceptions.SuperError(
77  "The first argument to super must be a subtype of "
78  "type, not {mro_pointer}.", super_=self)
79 
80  if isinstance(self.type, scoped_nodes.ClassDef):
81  # `super(type, type)`, most likely in a class method.
82  self._class_based = True
83  mro_type = self.type
84  else:
85  mro_type = getattr(self.type, '_proxied', None)
86  if not isinstance(mro_type, (bases.Instance, scoped_nodes.ClassDef)):
87  raise exceptions.SuperError(
88  "The second argument to super must be an "
89  "instance or subtype of type, not {type}.",
90  super_=self)
91 
92  if not mro_type.newstyle:
93  raise exceptions.SuperError("Unable to call super on old-style classes.", super_=self)
94 
95  mro = mro_type.mro()
96  if self.mro_pointer not in mro:
97  raise exceptions.SuperError(
98  "The second argument to super must be an "
99  "instance or subtype of type, not {type}.",
100  super_=self)
101 
102  index = mro.index(self.mro_pointer)
103  return mro[index + 1:]

121  def igetattr(self, name, context=None):
122  """Retrieve the inferred values of the given attribute name."""
123 
124  if name in self.special_attributes:
125  yield self.special_attributes.lookup(name)
126  return
127 
128  try:
129  mro = self.super_mro()
130  # Don't let invalid MROs or invalid super calls
131  # leak out as is from this function.
132  except exceptions.SuperError as exc:
133  util.reraise(exceptions.AttributeInferenceError(
134  ('Lookup for {name} on {target!r} because super call {super!r} '
135  'is invalid.'),
136  target=self, attribute=name, context=context, super_=exc.super_))
137  except exceptions.MroError as exc:
138  util.reraise(exceptions.AttributeInferenceError(
139  ('Lookup for {name} on {target!r} failed because {cls!r} has an '
140  'invalid MRO.'),
141  target=self, attribute=name, context=context, mros=exc.mros,
142  cls=exc.cls))
143  found = False
144  for cls in mro:
145  if name not in cls.locals:
146  continue
147 
148  found = True
149  for inferred in bases._infer_stmts([cls[name]], context, frame=self):
150  if not isinstance(inferred, scoped_nodes.FunctionDef):
151  yield inferred
152  continue
153 
154  # We can obtain different descriptors from a super depending
155  # on what we are accessing and where the super call is.
156  if inferred.type == 'classmethod':
157  yield bases.BoundMethod(inferred, cls)
158  elif self._scope.type == 'classmethod' and inferred.type == 'method':
159  yield inferred
160  elif self._class_based or inferred.type == 'staticmethod':
161  yield inferred
162  elif bases._is_property(inferred):
163  # TODO: support other descriptors as well.
164  for value in inferred.infer_call_result(self, context):
165  yield value
166  else:
167  yield bases.BoundMethod(inferred, cls)
168 
169  if not found:
170  raise exceptions.AttributeInferenceError(target=self,
171  attribute=name,
172  context=context)
173 
174  def getattr(self, name, context=None):
175  return list(self.igetattr(name, context=context))
