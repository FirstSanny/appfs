129 def _underline_token(token):
130  length = token[3][1] - token[2][1]
131  offset = token[2][1]
132  referenced_line = token[4]
133  # If the referenced line does not end with a newline char, fix it
134  if referenced_line[-1] != '\n':
135  referenced_line += '\n'

138 def _column_distance(token1, token2):
139  if token1 == token2:

173 def _get_indent_hint_line(bar_positions, bad_position):
174  """Return a line with |s for each of the positions in the given lists."""
175  if not bar_positions:
176  return ('', '')
177  delta_message = ''
178  markers = [(pos, '|') for pos in bar_positions]
179  if len(markers) == 1:
180  # if we have only one marker we'll provide an extra hint on how to fix
181  expected_position = markers[0][0]
182  delta = abs(expected_position - bad_position)
183  direction = 'add' if expected_position > bad_position else 'remove'
184  delta_message = _CONTINUATION_HINT_MESSAGE % (
185  direction, delta, 's' if delta > 1 else '')
186  markers.append((bad_position, '^'))
187  markers.sort()
188  line = [' '] * (markers[-1][0] + 1)
189  for position, marker in markers:
190  line[position] = marker
191  return (''.join(line), delta_message)

346  def _hanging_indent_after_bracket(self, bracket, position):
347  """Extracts indentation information for a hanging indent."""
348  indentation = _get_indent_length(self._tokens.line(position))
349  if self._is_block_opener and self._continuation_size == self._block_indent_size:
350  return _ContinuedIndent(
351  HANGING_BLOCK,

355  _BeforeBlockOffsets(indentation + self._continuation_size,
356  indentation + self._continuation_size * 2))

359  # correct indent should be relative to the key instead of the

362  next_align = self._cont_stack[-1].valid_continuation_offsets.copy()

364  next_align[next_align_keys[0] + self._continuation_size] = True

368  # 'c'

371  return _ContinuedIndent(HANGING_DICT_VALUE, bracket, position, paren_align, next_align)

516  def _check_keyword_parentheses(self, tokens, start):
517  """Check that there are not unnecessary parens after a keyword.
518 
519  Parens are unnecessary if there is exactly one balanced outer pair on a
520  line, and it is followed by a colon, and contains no commas (i.e. is not a
521  tuple).
522 
523  Args:
524  tokens: list of Tokens; the entire list of Tokens.
525  start: int; the position of the keyword in the token list.
526  """
527  # If the next token is not a paren, we're fine.
528  if self._inside_brackets(':') and tokens[start][1] == 'for':
529  self._pop_token()
530  if tokens[start+1][1] != '(':
531  return
532 
533  found_and_or = False
534  depth = 0
535  keyword_token = tokens[start][1]
536  line_num = tokens[start][2][0]
537 
538  for i in range(start, len(tokens) - 1):
539  token = tokens[i]
540 
541  # If we hit a newline, then assume any parens were for continuation.
542  if token[0] == tokenize.NL:
543  return
544 
545  if token[1] == '(':
546  depth += 1
547  elif token[1] == ')':
548  depth -= 1
549  if depth:
550  continue
551  # ')' can't happen after if (foo), since it would be a syntax error.
552  if (tokens[i+1][1] in (':', ')', ']', '}', 'in') or
553  tokens[i+1][0] in (tokenize.NEWLINE,
554  tokenize.ENDMARKER,
555  tokenize.COMMENT)):
556  # The empty tuple () is always accepted.
557  if i == start + 2:
558  return
559  if keyword_token == 'not':
560  if not found_and_or:
561  self.add_message('superfluous-parens', line=line_num,
562  args=keyword_token)
563  elif keyword_token in ('return', 'yield'):
564  self.add_message('superfluous-parens', line=line_num,
565  args=keyword_token)
566  elif keyword_token not in self._keywords_with_parens:
567  if not (tokens[i+1][1] == 'in' and found_and_or):
568  self.add_message('superfluous-parens', line=line_num,
569  args=keyword_token)
570  return
571  elif depth == 1:
572  # This is a tuple, which is always acceptable.
573  if token[1] == ',':
574  return
575  # 'and' and 'or' are the only boolean operators with lower precedence
576  # than 'not', so parens are only required when they are found.
577  elif token[1] in ('and', 'or'):
578  found_and_or = True
579  # A yield inside an expression must always be in parentheses,
580  # quit early without error.
581  elif token[1] == 'yield':
582  return
583  # A generator expression always has a 'for' token in it, and
584  # the 'for' token is only legal inside parens when it is in a
585  # generator expression. The parens are necessary here, so bail
586  # without an error.
587  elif token[1] == 'for':
588  return
