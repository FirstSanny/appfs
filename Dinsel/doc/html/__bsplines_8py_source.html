32 class BSpline(object):
33  r"""Univariate spline in the B-spline basis.
34 
35  .. math::
36 
37  S(x) = \sum_{j=0}^{n-1} c_j B_{j, k; t}(x)
38 
39  where :math:`B_{j, k; t}` are B-spline basis functions of degree `k`
40  and knots `t`.
41 
42  Parameters
43  ----------
44  t : ndarray, shape (n+k+1,)
45  knots
46  c : ndarray, shape (>=n, ...)
47  spline coefficients
48  k : int
49  B-spline order
50  extrapolate : bool, optional
51  whether to extrapolate beyond the base interval, ``t[k] .. t[n]``,
52  or to return nans.
53  If True, extrapolates the first and last polynomial pieces of b-spline
54  functions active on the base interval.
55  Default is True.
56  axis : int, optional
57  Interpolation axis. Default is zero.
58 
59  Attributes
60  ----------
61  t : ndarray
62  knot vector
63  c : ndarray
64  spline coefficients
65  k : int
66  spline degree
67  extrapolate : bool
68  If True, extrapolates the first and last polynomial pieces of b-spline
69  functions active on the base interval.
70  axis : int
71  Interpolation axis.
72  tck : tuple
73  A read-only equivalent of ``(self.t, self.c, self.k)``
74 
75  Methods
76  -------
77  __call__
78  basis_element
79  derivative
80  antiderivative
81  integrate
82  construct_fast
83 
84  Notes
85  -----
86  B-spline basis elements are defined via
87 
88  .. math::
89 
90  B_{i, 0}(x) = 1, \textrm{if $t_i \le x < t_{i+1}$, otherwise $0$,}
91 
92  B_{i, k}(x) = \frac{x - t_i}{t_{i+k} - t_i} B_{i, k-1}(x)
93  + \frac{t_{i+k+1} - x}{t_{i+k+1} - t_{i+1}} B_{i+1, k-1}(x)
94 
95  **Implementation details**
96 
97  - At least ``k+1`` coefficients are required for a spline of degree `k`,
98  so that ``n >= k+1``. Additional coefficients, ``c[j]`` with
99  ``j > n``, are ignored.
100 
101  - B-spline basis elements of degree `k` form a partition of unity on the
102  *base interval*, ``t[k] <= x <= t[n]``.
103 
104 
105  Examples
106  --------
107 
108  Translating the recursive definition of B-splines into Python code, we have:
109 
110  >>> def B(x, k, i, t):
111  ... if k == 0:
112  ... return 1.0 if t[i] <= x < t[i+1] else 0.0
113  ... if t[i+k] == t[i]:
114  ... c1 = 0.0
115  ... else:
116  ... c1 = (x - t[i])/(t[i+k] - t[i]) * B(x, k-1, i, t)
117  ... if t[i+k+1] == t[i+1]:
118  ... c2 = 0.0
119  ... else:
120  ... c2 = (t[i+k+1] - x)/(t[i+k+1] - t[i+1]) * B(x, k-1, i+1, t)
121  ... return c1 + c2
122 
123  >>> def bspline(x, t, c, k):
124  ... n = len(t) - k - 1
125  ... assert (n >= k+1) and (len(c) >= n)
126  ... return sum(c[i] * B(x, k, i, t) for i in range(n))
127 
128  Note that this is an inefficient (if straightforward) way to
129  evaluate B-splines --- this spline class does it in an equivalent,
130  but much more efficient way.
131 
132  Here we construct a quadratic spline function on the base interval
133  ``2 <= x <= 4`` and compare with the naive way of evaluating the spline:
134 
135  >>> from scipy.interpolate import BSpline
136  >>> k = 2
137  >>> t = [0, 1, 2, 3, 4, 5, 6]
138  >>> c = [-1, 2, 0, -1]
139  >>> spl = BSpline(t, c, k)
140  >>> spl(2.5)
141  array(1.375)
142  >>> bspline(2.5, t, c, k)
143  1.375
144 
145  Note that outside of the base interval results differ. This is because
146  `BSpline` extrapolates the first and last polynomial pieces of b-spline
147  functions active on the base interval.
148 
149  >>> import matplotlib.pyplot as plt
150  >>> fig, ax = plt.subplots()
151  >>> xx = np.linspace(1.5, 4.5, 50)
152  >>> ax.plot(xx, [bspline(x, t, c ,k) for x in xx], 'r-', lw=3, label='naive')
153  >>> ax.plot(xx, spl(xx), 'b-', lw=4, alpha=0.7, label='BSpline')
154  >>> ax.grid(True)
155  >>> ax.legend(loc='best')
156  >>> plt.show()

230  @classmethod
231  def basis_element(cls, t, extrapolate=True):
232  """Return a B-spline basis element ``B(x | t[0], ..., t[k+1])``.
233 
234  Parameters
235  ----------
236  t : ndarray, shape (k+1,)
237  internal knots
238  extrapolate : bool, optional
239  whether to extrapolate beyond the base interval, ``t[0] .. t[k+1]``,
240  or to return nans. Default is True.
241 
242  Returns
243  -------
244  basis_element : callable
245  A callable representing a B-spline basis element for the knot
246  vector `t`.
247 
248  Notes
249  -----
250  The order of the b-spline, `k`, is inferred from the length of `t` as
251  ``len(t)-2``. The knot vector is constructed by appending and prepending
252  ``k+1`` elements to internal knots `t`.
253 
254  Examples
255  --------
256 
257  Construct a cubic b-spline:
258 
259  >>> from scipy.interpolate import BSpline
260  >>> b = BSpline.basis_element([0, 1, 2, 3, 4])
261  >>> k = b.k
262  >>> b.t[k:-k]
263  array([ 0., 1., 2., 3., 4.])
264  >>> k
265  3
266 
267  Construct a second order b-spline on ``[0, 1, 1, 2]``, and compare
268  to its explicit form:
269 
270  >>> t = [-1, 0, 1, 1, 2]
271  >>> b = BSpline.basis_element(t[1:])
272  >>> def f(x):
273  ... return np.where(x < 1, x*x, (2. - x)**2)
274 
275  >>> import matplotlib.pyplot as plt
276  >>> fig, ax = plt.subplots()
277  >>> x = np.linspace(0, 2, 51)
278  >>> ax.plot(x, b(x), 'g', lw=3)
279  >>> ax.plot(x, f(x), 'r', lw=8, alpha=0.4)
280  >>> ax.grid(True)
281  >>> plt.show()
282 
283  """
284  k = len(t) - 2
285  t = _as_float_array(t)
286  t = np.r_[(t[0]-1,) * k, t, (t[-1]+1,) * k]
287  c = np.zeros_like(t)
288  c[k] = 1.
289  return cls.construct_fast(t, c, k, extrapolate)

291  def __call__(self, x, nu=0, extrapolate=None):
292  """
293  Evaluate a spline function.
294 
295  Parameters
296  ----------
297  x : array_like
298  points to evaluate the spline at.
299  nu: int, optional
300  derivative to evaluate (default is 0).
301  extrapolate : bool, optional
302  whether to extrapolate based on the first and last intervals
303  or return nans. Default is `self.extrapolate`.
304 
305  Returns
306  -------
307  y : array_like
308  Shape is determined by replacing the interpolation axis
309  in the coefficient array with the shape of `x`.
310 
311  """
312  if extrapolate is None:
313  extrapolate = self.extrapolate
314  x = np.asarray(x)
315  x_shape, x_ndim = x.shape, x.ndim
316  x = np.ascontiguousarray(x.ravel(), dtype=np.float_)
317  out = np.empty((len(x), prod(self.c.shape[1:])), dtype=self.c.dtype)
318  self._ensure_c_contiguous()
319  self._evaluate(x, nu, extrapolate, out)
320  out = out.reshape(x_shape + self.c.shape[1:])
321  if self.axis != 0:
322  # transpose to move the calculated values to the interpolation axis
323  l = list(range(out.ndim))
324  l = l[x_ndim:x_ndim+self.axis] + l[:x_ndim] + l[x_ndim+self.axis:]
325  out = out.transpose(l)
326  return out

398  def integrate(self, a, b, extrapolate=None):
399  """Compute a definite integral of the spline.
400 
401  Parameters
402  ----------
403  a : float
404  Lower limit of integration.
405  b : float
406  Upper limit of integration.
407  extrapolate : bool, optional
408  whether to extrapolate beyond the base interval, ``t[k] .. t[-k-1]``,
409  or take the spline to be zero outside of the base interval.
410  Default is True.
411 
412  Returns
413  -------
414  I : array_like
415  Definite integral of the spline over the interval ``[a, b]``.
416 
417  Examples
418  --------
419  Construct the linear spline ``x if x < 1 else 2 - x`` on the base
420  interval :math:`[0, 2]`, and integrate it
421 
422  >>> from scipy.interpolate import BSpline
423  >>> b = BSpline.basis_element([0, 1, 2])
424  >>> b.integrate(0, 1)
425  array(0.5)
426 
427  If the integration limits are outside of the base interval, the result
428  is controlled by the `extrapolate` parameter
429 
430  >>> b.integrate(-1, 1)
431  array(0.0)
432  >>> b.integrate(-1, 1, extrapolate=False)
433  array(0.5)
434 
435  >>> import matplotlib.pyplot as plt
436  >>> fig, ax = plt.subplots()
437  >>> ax.grid(True)
438  >>> ax.axvline(0, c='r', lw=5, alpha=0.5) # base interval
439  >>> ax.axvline(2, c='r', lw=5, alpha=0.5)
440  >>> xx = [-1, 1, 2]
441  >>> ax.plot(xx, b(xx))
442  >>> plt.show()
443 
444  """
445  if extrapolate is None:
446  extrapolate = self.extrapolate
447 
448  if not extrapolate:
449  # shrink the integration interval, if needed
450  a = max(a, self.t[self.k])
451  b = min(b, self.t[-self.k - 1])
452 
453  if self.c.ndim == 1:
454  # fast path: use FITPACK's routine (cf _fitpack_impl.splint)
455  t, c, k = self.tck
456  aint, wrk = _dierckx._splint(t, c, k, a, b)
457  return aint
458 
459  # prepare t & c
460  self._ensure_c_contiguous()
461 
462  # compute the antiderivative
463  c = self.c
464  ct = len(self.t) - len(c)
465  if ct > 0:
466  c = np.r_[c, np.zeros((ct,) + c.shape[1:])]
467  t, c, k = _fitpack_impl.splantider((self.t, c, self.k), 1)
468 
469  # evaluate the diff of antiderivatives
470  x = np.asarray([a, b], dtype=np.float_)
471  out = np.empty((2, prod(c.shape[1:])), dtype=c.dtype)
472  _bspl.evaluate_spline(t, c.reshape(c.shape[0], -1),
473  k, x, 0, extrapolate, out)
474  out = out[1] - out[0]
475  return out.reshape(c.shape[1:])

510 def make_interp_spline(x, y, k=3, t=None, bc_type=None, axis=0,
511  check_finite=True):
512  """Compute the (coefficients of) interpolating B-spline.
513 
514  Parameters
515  ----------
516  x : array_like, shape (n,)
517  Abscissas.
518  y : array_like, shape (n, ...)
519  Ordinates.
520  k : int, optional
521  B-spline degree. Default is cubic, k=3.
522  t : array_like, shape (nt + k + 1,), optional.
523  Knots.
524  The number of knots needs to agree with the number of datapoints and
525  the number of derivatives at the edges. Specifically, ``nt - n`` must
526  equal ``len(deriv_l) + len(deriv_r)``.
527  bc_type : 2-tuple or None
528  Boundary conditions.
529  Default is None, which means choosing the boundary conditions
530  automatically. Otherwise, it must be a length-two tuple where the first
531  element sets the boundary conditions at ``x[0]`` and the second
532  element sets the boundary conditions at ``x[-1]``. Each of these must
533  be an iterable of pairs ``(order, value)`` which gives the values of
534  derivatives of specified orders at the given edge of the interpolation
535  interval.
536  axis : int, optional
537  Interpolation axis. Default is 0.
538  check_finite : bool, optional
539  Whether to check that the input arrays contain only finite numbers.
540  Disabling may give a performance gain, but may result in problems
541  (crashes, non-termination) if the inputs do contain infinities or NaNs.
542  Default is True.
543 
544  Returns
545  -------
546  b : a BSpline object of the degree ``k`` and with knots ``t``.
547 
548  Examples
549  --------
550 
551  Use cubic interpolation on Chebyshev nodes:
552 
553  >>> def cheb_nodes(N):
554  ... jj = 2.*np.arange(N) + 1
555  ... x = np.cos(np.pi * jj / 2 / N)[::-1]
556  ... return x
557 
558  >>> x = cheb_nodes(20)
559  >>> y = np.sqrt(1 - x**2)
560 
561  >>> from scipy.interpolate import BSpline, make_interp_spline
562  >>> b = make_interp_spline(x, y)
563  >>> np.allclose(b(x), y)
564  True
565 
566  Note that the default is a cubic spline with a not-a-knot boundary condition
567 
568  >>> b.k
569  3
570 
571  Here we use a 'natural' spline, with zero 2nd derivatives at edges:
572 
573  >>> l, r = [(2, 0)], [(2, 0)]
574  >>> b_n = make_interp_spline(x, y, bc_type=(l, r))
575  >>> np.allclose(b_n(x), y)
576  True
577  >>> x0, x1 = x[0], x[-1]
578  >>> np.allclose([b_n(x0, 2), b_n(x1, 2)], [0, 0])
579  True
580 
581  Interpolation of parametric curves is also supported. As an example, we
582  compute a discretization of a snail curve in polar coordinates
583 
584  >>> phi = np.linspace(0, 2.*np.pi, 40)
585  >>> r = 0.3 + np.cos(phi)
586  >>> x, y = r*np.cos(phi), r*np.sin(phi) # convert to Cartesian coordinates
587 
588  Build an interpolating curve, parameterizing it by the angle
589 
590  >>> from scipy.interpolate import make_interp_spline
591  >>> spl = make_interp_spline(phi, np.c_[x, y])
592 
593  Evaluate the interpolant on a finer grid (note that we transpose the result
594  to unpack it into a pair of x- and y-arrays)
595 
596  >>> phi_new = np.linspace(0, 2.*np.pi, 100)
597  >>> x_new, y_new = spl(phi_new).T
598 
599  Plot the result
600 
601  >>> import matplotlib.pyplot as plt
602  >>> plt.plot(x, y, 'o')
603  >>> plt.plot(x_new, y_new, '-')
604  >>> plt.show()
605 
606  See Also
607  --------
608  BSpline : base class representing the B-spline objects
609  CubicSpline : a cubic spline in the polynomial basis
610  make_lsq_spline : a similar factory function for spline fitting
611  UnivariateSpline : a wrapper over FITPACK spline fitting routines
612  splrep : a wrapper over FITPACK spline fitting routines
613 
614  """
615  if bc_type is None:
616  bc_type = (None, None)
617  deriv_l, deriv_r = bc_type
618 
619  # special-case k=0 right away
620  if k == 0:
621  if any(_ is not None for _ in (t, deriv_l, deriv_r)):
622  raise ValueError("Too much info for k=0: t and bc_type can only "
623  "be None.")
624  x = _as_float_array(x, check_finite)
625  t = np.r_[x, x[-1]]
626  c = np.asarray(y)
627  c = np.ascontiguousarray(c, dtype=_get_dtype(c.dtype))
628  return BSpline.construct_fast(t, c, k, axis=axis)
629 
630  # special-case k=1 (e.g., Lyche and Morken, Eq.(2.16))
631  if k == 1 and t is None:
632  if not (deriv_l is None and deriv_r is None):
633  raise ValueError("Too much info for k=1: bc_type can only be None.")
634  x = _as_float_array(x, check_finite)
635  t = np.r_[x[0], x, x[-1]]
636  c = np.asarray(y)
637  c = np.ascontiguousarray(c, dtype=_get_dtype(c.dtype))

734 def make_lsq_spline(x, y, t, k=3, w=None, axis=0, check_finite=True):
735  r"""Compute the (coefficients of) an LSQ B-spline.
736 
737  The result is a linear combination
738 
739  .. math::
740 
741  S(x) = \sum_j c_j B_j(x; t)
742 
743  of the B-spline basis elements, :math:`B_j(x; t)`, which minimizes
744 
745  .. math::
746 
747  \sum_{j} \left( w_j \times (S(x_j) - y_j) \right)^2
748 
749  Parameters
750  ----------
751  x : array_like, shape (m,)
752  Abscissas.
753  y : array_like, shape (m, ...)
754  Ordinates.
755  t : array_like, shape (n + k + 1,).
756  Knots.
757  Knots and data points must satisfy Schoenberg-Whitney conditions.
758  k : int, optional
759  B-spline degree. Default is cubic, k=3.
760  w : array_like, shape (n,), optional
761  Weights for spline fitting. Must be positive. If ``None``,
762  then weights are all equal.
763  Default is ``None``.
764  axis : int, optional
765  Interpolation axis. Default is zero.
766  check_finite : bool, optional
767  Whether to check that the input arrays contain only finite numbers.
768  Disabling may give a performance gain, but may result in problems
769  (crashes, non-termination) if the inputs do contain infinities or NaNs.
770  Default is True.
771 
772  Returns
773  -------
774  b : a BSpline object of the degree `k` with knots `t`.
775 
776  Notes
777  -----
778 
779  The number of data points must be larger than the spline degree `k`.
780 
781  Knots `t` must satisfy the Schoenberg-Whitney conditions,
782  i.e., there must be a subset of data points ``x[j]`` such that
783  ``t[j] < x[j] < t[j+k+1]``, for ``j=0, 1,...,n-k-2``.
784 
785  Examples
786  --------
787  Generate some noisy data:
788 
789  >>> x = np.linspace(-3, 3, 50)
790  >>> y = np.exp(-x**2) + 0.1 * np.random.randn(50)
791 
792  Now fit a smoothing cubic spline with a pre-defined internal knots.
793  Here we make the knot vector (k+1)-regular by adding boundary knots:
794 
795  >>> from scipy.interpolate import make_lsq_spline, BSpline
796  >>> t = [-1, 0, 1]
797  >>> k = 3
798  >>> t = np.r_[(x[0],)*(k+1),
799  ... t,
800  ... (x[-1],)*(k+1)]
801  >>> spl = make_lsq_spline(x, y, t, k)
802 
803  For comparison, we also construct an interpolating spline for the same
804  set of data:
805 
806  >>> from scipy.interpolate import make_interp_spline
807  >>> spl_i = make_interp_spline(x, y)
808 
809  Plot both:
810 
811  >>> import matplotlib.pyplot as plt
812  >>> xs = np.linspace(-3, 3, 100)
813  >>> plt.plot(x, y, 'ro', ms=5)
814  >>> plt.plot(xs, spl(xs), 'g-', lw=3, label='LSQ spline')
815  >>> plt.plot(xs, spl_i(xs), 'b-', lw=3, alpha=0.7, label='interp spline')
816  >>> plt.legend(loc='best')
817  >>> plt.show()
818 
819  **NaN handling**: If the input arrays contain ``nan`` values, the result is
820  not useful since the underlying spline fitting routines cannot deal with
821  ``nan``. A workaround is to use zero weights for not-a-number data points:
822 
823  >>> y[8] = np.nan
824  >>> w = np.isnan(y)
825  >>> y[w] = 0.
826  >>> tck = make_lsq_spline(x, y, t, w=~w)
827 
828  Notice the need to replace a ``nan`` by a numerical value (precise value
829  does not matter as long as the corresponding weight is zero.)
830 
831  See Also
832  --------
833  BSpline : base class representing the B-spline objects
834  make_interp_spline : a similar factory function for interpolating splines
835  LSQUnivariateSpline : a FITPACK-based spline fitting routine
836  splrep : a FITPACK-based fitting routine
837 
838  """
839  x = _as_float_array(x, check_finite)
840  y = _as_float_array(y, check_finite)
841  t = _as_float_array(t, check_finite)
842  if w is not None:
843  w = _as_float_array(w, check_finite)
844  else:
845  w = np.ones_like(x)
846  k = int(k)
847 
848  axis = axis % y.ndim
849  y = np.rollaxis(y, axis) # now internally interp axis is zero
850 
851  if x.ndim != 1 or np.any(x[1:] - x[:-1] <= 0):
852  raise ValueError("Expect x to be a 1-D sorted array_like.")
853  if x.shape[0] < k+1:
854  raise("Need more x points.")
855  if k < 0:
856  raise ValueError("Expect non-negative k.")
857  if t.ndim != 1 or np.any(t[1:] - t[:-1] < 0):
858  raise ValueError("Expect t to be a 1-D sorted array_like.")
859  if x.size != y.shape[0]:
860  raise ValueError('x & y are incompatible.')
861  if k > 0 and np.any((x < t[k]) | (x > t[-k])):

888  return BSpline.construct_fast(t, c, k, axis=axis)
