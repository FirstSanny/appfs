71 def _get_first_import(node, context, name, base, level, alias):
72  """return the node where [base.]<name> is imported or None if not found
73  """
74  fullname = '%s.%s' % (base, name) if base else name
75 
76  first = None
77  found = False
78  for first in context.body:
79  if first is node:
80  continue
81  if first.scope() is node.scope() and first.fromlineno > node.fromlineno:
82  continue
83  if isinstance(first, astroid.Import):
84  if any(fullname == iname[0] for iname in first.names):
85  found = True
86  break
87  elif isinstance(first, astroid.ImportFrom):
88  if level == first.level:
89  for imported_name, imported_alias in first.names:
90  if fullname == '%s.%s' % (first.modname, imported_name):
91  found = True
92  break
93  if name != '*' and name == imported_name and not (alias or imported_alias):
94  found = True
95  break
96  if found:
97  break
98  if found and not are_exclusive(first, node):
99  return first

112  """get a list of 2-uple (module, list_of_files_which_import_this_module),
113  it will return a dictionary to represent this as a tree

124 def _repr_tree_defs(data, indent_str=None):
125  """return a string which represents imports as a tree"""
126  lines = []
127  nodes = data.items()
128  for i, (mod, (sub, files)) in enumerate(sorted(nodes, key=lambda x: x[0])):
129  if not files:
130  files = ''
131  else:
132  files = '(%s)' % ','.join(sorted(files))
133  if indent_str is None:
134  lines.append('%s %s' % (mod, files))
135  sub_indent_str = ' '
136  else:
137  lines.append(r'%s\-%s %s' % (indent_str, mod, files))
138  if i == len(nodes)-1:
139  sub_indent_str = '%s ' % indent_str
140  else:
141  sub_indent_str = '%s| ' % indent_str
142  if sub:
143  lines.append(_repr_tree_defs(sub, sub_indent_str))
144  return '\n'.join(lines)

147 def _dependencies_graph(filename, dep_info):
148  """write dependencies as a dot (graphviz) file
149  """
150  done = {}
151  printer = DotBackend(filename[:-4], rankdir='LR')
152  printer.emit('URL="." node[shape="box"]')
153  for modname, dependencies in sorted(six.iteritems(dep_info)):
154  done[modname] = 1
155  printer.emit_node(modname)
156  for depmodname in dependencies:
157  if depmodname not in done:
158  done[depmodname] = 1
159  printer.emit_node(depmodname)
160  for depmodname, dependencies in sorted(six.iteritems(dep_info)):
161  for modname in dependencies:
162  printer.emit_edge(modname, depmodname)
163  printer.generate(filename)

556  def _check_imports_order(self, _module_node):
557  """Checks imports of module `node` are grouped by category
558 
559  Imports must follow this order: standard, 3rd party, local
560  """
561  extern_imports = []
562  local_imports = []
563  std_imports = []
564  extern_not_nested = []
565  local_not_nested = []
566  isort_obj = isort.SortImports(
567  file_contents='', known_third_party=self.config.known_third_party,
568  known_standard_library=self.config.known_standard_library,
569  )
570  for node, modname in self._imports_stack:
571  if modname.startswith('.'):
572  package = '.' + modname.split('.')[1]
573  else:
574  package = modname.split('.')[0]
575  nested = not isinstance(node.parent, astroid.Module)
576  import_category = isort_obj.place_module(package)
577  if import_category in ('FUTURE', 'STDLIB'):
578  std_imports.append((node, package))
579  wrong_import = extern_not_nested or local_not_nested
580  if self._is_fallback_import(node, wrong_import):
581  continue
582  if wrong_import and not nested:
583  self.add_message('wrong-import-order', node=node,
584  args=('standard import "%s"' % node.as_string(),
585  '"%s"' % wrong_import[0][0].as_string()))
586  elif import_category in ('FIRSTPARTY', 'THIRDPARTY'):
587  extern_imports.append((node, package))
588  if not nested:
589  extern_not_nested.append((node, package))
590  wrong_import = local_not_nested
591  if wrong_import and not nested:
592  self.add_message('wrong-import-order', node=node,
593  args=('external import "%s"' % node.as_string(),
594  '"%s"' % wrong_import[0][0].as_string()))
595  elif import_category == 'LOCALFOLDER':
596  local_imports.append((node, package))
597  if not nested:
598  local_not_nested.append((node, package))
599  return std_imports, extern_imports, local_imports
