18 def get_package_name(filepath):
19  """
20  Given a path where a package is installed, determine its name.
21 
22  Parameters
23  ----------
24  filepath : str
25  Path to a file. If the determination fails, "numpy" is returned.
26 
27  Examples
28  --------
29  >>> np.testing.nosetester.get_package_name('nonsense')
30  'numpy'
31 
32  """
33 
34  fullpath = filepath[:]
35  pkg_name = []
36  while 'site-packages' in filepath or 'dist-packages' in filepath:
37  filepath, p2 = os.path.split(filepath)
38  if p2 in ('site-packages', 'dist-packages'):
39  break
40  pkg_name.append(p2)
41 
42  # if package name determination failed, just default to numpy/scipy
43  if not pkg_name:
44  if 'scipy' in fullpath:
45  return 'scipy'
46  else:
47  return 'numpy'
48 
49  # otherwise, reverse to get correct order and return
50  pkg_name.reverse()
51 
52  # don't include the outer egg directory
53  if pkg_name[0].endswith('.egg'):
54  pkg_name.pop(0)
55 
56  return '.'.join(pkg_name)

59 def run_module_suite(file_to_run=None, argv=None):
60  """
61  Run a test module.
62 
63  Equivalent to calling ``$ nosetests <argv> <file_to_run>`` from
64  the command line
65 
66  Parameters
67  ----------
68  file_to_run : str, optional
69  Path to test module, or None.
70  By default, run the module from which this function is called.
71  argv : list of strings
72  Arguments to be passed to the nose test runner. ``argv[0]`` is
73  ignored. All command line arguments accepted by ``nosetests``
74  will work. If it is the default value None, sys.argv is used.
75 
76  .. versionadded:: 1.9.0
77 
78  Examples
79  --------
80  Adding the following::
81 
82  if __name__ == "__main__" :
83  run_module_suite(argv=sys.argv)
84 
85  at the end of a test module will run the tests when that module is
86  called in the python interpreter.
87 
88  Alternatively, calling::
89 
90  >>> run_module_suite(file_to_run="numpy/tests/test_matlib.py")
91 
92  from an interpreter will run all the test routine in 'test_matlib.py'.
93  """
94  if file_to_run is None:
95  f = sys._getframe(1)
96  file_to_run = f.f_locals.get('__file__', None)
97  if file_to_run is None:
98  raise AssertionError
99 
100  if argv is None:
101  argv = sys.argv + [file_to_run]
102  else:
103  argv = argv + [file_to_run]
104 
105  nose = import_nose()
106  from .noseclasses import KnownFailurePlugin
107  nose.run(argv=argv, addplugins=[KnownFailurePlugin()])

110 class NoseTester(object):
111  """
112  Nose test runner.
113 
114  This class is made available as numpy.testing.Tester, and a test function
115  is typically added to a package's __init__.py like so::
116 
117  from numpy.testing import Tester
118  test = Tester().test
119 
120  Calling this test function finds and runs all tests associated with the
121  package and all its sub-packages.
122 
123  Attributes
124  ----------
125  package_path : str
126  Full path to the package to test.
127  package_name : str
128  Name of the package to test.
129 
130  Parameters
131  ----------
132  package : module, str or None, optional
133  The package to test. If a string, this should be the full path to
134  the package. If None (default), `package` is set to the module from
135  which `NoseTester` is initialized.
136  raise_warnings : None, str or sequence of warnings, optional
137  This specifies which warnings to configure as 'raise' instead
138  of being shown once during the test execution. Valid strings are:
139 
140  - "develop" : equals ``(Warning,)``
141  - "release" : equals ``()``, don't raise on any warnings.
142 
143  Default is "release".
144  depth : int, optional
145  If `package` is None, then this can be used to initialize from the
146  module of the caller of (the caller of (...)) the code that
147  initializes `NoseTester`. Default of 0 means the module of the
148  immediate caller; higher values are useful for utility routines that
149  want to initialize `NoseTester` objects on behalf of other code.
150 
151  """
152  def __init__(self, package=None, raise_warnings="release", depth=0):
153  # Back-compat: 'None' used to mean either "release" or "develop"
154  # depending on whether this was a release or develop version of
155  # numpy. Those semantics were fine for testing numpy, but not so
156  # helpful for downstream projects like scipy that use
157  # numpy.testing. (They want to set this based on whether *they* are a
158  # release or develop version, not whether numpy is.) So we continue to
159  # accept 'None' for back-compat, but it's now just an alias for the
160  # default "release".
161  if raise_warnings is None:
162  raise_warnings = "release"
163 
164  package_name = None
165  if package is None:
166  f = sys._getframe(1 + depth)
167  package_path = f.f_locals.get('__file__', None)
168  if package_path is None:
169  raise AssertionError
170  package_path = os.path.dirname(package_path)
171  package_name = f.f_locals.get('__name__', None)
172  elif isinstance(package, type(os)):
173  package_path = os.path.dirname(package.__file__)
174  package_name = getattr(package, '__name__', None)
175  else:
176  package_path = str(package)
177 
178  self.package_path = package_path
179 
180  # Find the package name under test; this name is used to limit coverage
181  # reporting (if enabled).
182  if package_name is None:
183  package_name = get_package_name(package_path)
184  self.package_name = package_name
185 
186  # Set to "release" in constructor in maintenance branches.
187  self.raise_warnings = raise_warnings

189  def _test_argv(self, label, verbose, extra_argv):
190  ''' Generate argv for nosetest command
191 
192  Parameters
193  ----------
194  label : {'fast', 'full', '', attribute identifier}, optional
195  see ``test`` docstring
196  verbose : int, optional
197  Verbosity value for test outputs, in the range 1-10. Default is 1.
198  extra_argv : list, optional
199  List with any extra arguments to pass to nosetests.
200 
201  Returns
202  -------
203  argv : list
204  command line arguments that will be passed to nose
205  '''
206  argv = [__file__, self.package_path, '-s']
207  if label and label != 'full':
208  if not isinstance(label, basestring):
209  raise TypeError('Selection label should be a string')
210  if label == 'fast':
211  label = 'not slow'
212  argv += ['-A', label]
213  argv += ['--verbosity', str(verbose)]
214 
215  # When installing with setuptools, and also in some other cases, the
216  # test_*.py files end up marked +x executable. Nose, by default, does
217  # not run files marked with +x as they might be scripts. However, in
218  # our case nose only looks for test_*.py files under the package
219  # directory, which should be safe.
220  argv += ['--exe']
221 
222  if extra_argv:
223  argv += extra_argv
224  return argv

256  def prepare_test_args(self, label='fast', verbose=1, extra_argv=None,
257  doctests=False, coverage=False):
258  """
259  Run tests for module using nose.
260 
261  This method does the heavy lifting for the `test` method. It takes all
262  the same arguments, for details see `test`.
263 
264  See Also
265  --------
266  test
267 
268  """
269  # fail with nice error message if nose is not present
270  import_nose()
271  # compile argv
272  argv = self._test_argv(label, verbose, extra_argv)
273  # our way of doing coverage
274  if coverage:
275  argv += ['--cover-package=%s' % self.package_name, '--with-coverage',
276  '--cover-tests', '--cover-erase']
277  # construct list of plugins
278  import nose.plugins.builtin
279  from .noseclasses import KnownFailurePlugin, Unplugger
280  plugins = [KnownFailurePlugin()]
281  plugins += [p() for p in nose.plugins.builtin.plugins]
282  # add doctesting if required
283  doctest_argv = '--with-doctest' in argv
284  if doctests == False and doctest_argv:
285  doctests = True
286  plug = self._get_custom_doctester()
287  if plug is None:
288  # use standard doctesting
289  if doctests and not doctest_argv:
290  argv += ['--with-doctest']
291  else: # custom doctesting
292  if doctest_argv: # in fact the unplugger would take care of this
293  argv.remove('--with-doctest')
294  plugins += [Unplugger('doctest'), plug]
295  if doctests:
296  argv += ['--with-' + plug.name]
297  return argv, plugins

299  def test(self, label='fast', verbose=1, extra_argv=None,
300  doctests=False, coverage=False, raise_warnings=None):
301  """
302  Run tests for module using nose.
303 
304  Parameters
305  ----------
306  label : {'fast', 'full', '', attribute identifier}, optional
307  Identifies the tests to run. This can be a string to pass to
308  the nosetests executable with the '-A' option, or one of several
309  special values. Special values are:
310  * 'fast' - the default - which corresponds to the ``nosetests -A``
311  option of 'not slow'.
312  * 'full' - fast (as above) and slow tests as in the
313  'no -A' option to nosetests - this is the same as ''.
314  * None or '' - run all tests.
315  attribute_identifier - string passed directly to nosetests as '-A'.
316  verbose : int, optional
317  Verbosity value for test outputs, in the range 1-10. Default is 1.
318  extra_argv : list, optional
319  List with any extra arguments to pass to nosetests.
320  doctests : bool, optional
321  If True, run doctests in module. Default is False.
322  coverage : bool, optional
323  If True, report coverage of NumPy code. Default is False.
324  (This requires the `coverage module:
325  <http://nedbatchelder.com/code/modules/coverage.html>`_).
326  raise_warnings : None, str or sequence of warnings, optional
327  This specifies which warnings to configure as 'raise' instead
328  of being shown once during the test execution. Valid strings are:
329 
330  - "develop" : equals ``(Warning,)``
331  - "release" : equals ``()``, don't raise on any warnings.
332 
333  The default is to use the class initialization value.
334 
335  Returns
336  -------
337  result : object
338  Returns the result of running the tests as a
339  ``nose.result.TextTestResult`` object.
340 
341  Notes
342  -----
343  Each NumPy module exposes `test` in its namespace to run all tests for it.
344  For example, to run all tests for numpy.lib:
345 
346  >>> np.lib.test() #doctest: +SKIP
347 
348  Examples
349  --------
350  >>> result = np.lib.test() #doctest: +SKIP
351  Running unit tests for numpy.lib
352  ...
353  Ran 976 tests in 3.933s
354 
355  OK
356 
357  >>> result.errors #doctest: +SKIP
358  []
359  >>> result.knownfail #doctest: +SKIP
360  []
361  """
362 
363  # cap verbosity at 3 because nose becomes *very* verbose beyond that
364  verbose = min(verbose, 3)
365 
366  from . import utils
367  utils.verbose = verbose
368 
369  if doctests:
370  print("Running unit tests and doctests for %s" % self.package_name)
371  else:
372  print("Running unit tests for %s" % self.package_name)
373 
374  self._show_system_info()
375 
376  # reset doctest state on every run
377  import doctest
378  doctest.master = None
379 
380  if raise_warnings is None:
381  raise_warnings = self.raise_warnings
382 
383  _warn_opts = dict(develop=(Warning,),
384  release=())
385  if isinstance(raise_warnings, basestring):
386  raise_warnings = _warn_opts[raise_warnings]
387 
388  with suppress_warnings("location") as sup:
389  # Reset the warning filters to the default state,
390  # so that running the tests is more repeatable.
391  warnings.resetwarnings()
392  # Set all warnings to 'warn', this is because the default 'once'
393  # has the bad property of possibly shadowing later warnings.
394  warnings.filterwarnings('always')
395  # Force the requested warnings to raise
396  for warningtype in raise_warnings:
397  warnings.filterwarnings('error', category=warningtype)
398  # Filter out annoying import messages.
399  sup.filter(message='Not importing directory')
400  sup.filter(message="numpy.dtype size changed")
401  sup.filter(message="numpy.ufunc size changed")
402  sup.filter(category=np.ModuleDeprecationWarning)
403  # Filter out boolean '-' deprecation messages. This allows
404  # older versions of scipy to test without a flood of messages.
405  sup.filter(message=".*boolean negative.*")
406  sup.filter(message=".*boolean subtract.*")
407  # Filter out distutils cpu warnings (could be localized to
408  # distutils tests). ASV has problems with top level import,
409  # so fetch module for suppression here.
410  with warnings.catch_warnings():
411  warnings.simplefilter("always")
412  from ..distutils import cpuinfo
413  sup.filter(category=UserWarning, module=cpuinfo)
414  # See #7949: Filter out deprecation warnings due to the -3 flag to
415  # python 2
416  if sys.version_info.major == 2 and sys.py3kwarning:
417  # This is very specific, so using the fragile module filter
418  # is fine
419  import threading
420  sup.filter(DeprecationWarning,
421  r"sys\.exc_clear\(\) not supported in 3\.x",
422  module=threading)
423  sup.filter(DeprecationWarning, message=r"in 3\.x, __setslice__")
424  sup.filter(DeprecationWarning, message=r"in 3\.x, __getslice__")
425  sup.filter(DeprecationWarning, message=r"buffer\(\) not supported in 3\.x")
426  sup.filter(DeprecationWarning, message=r"CObject type is not supported in 3\.x")
427  sup.filter(DeprecationWarning, message=r"comparing unequal types not supported in 3\.x")
428  # Filter out some deprecation warnings inside nose 1.3.7 when run
429  # on python 3.5b2. See
430  # https://github.com/nose-devs/nose/issues/929
431  # Note: it is hard to filter based on module for sup (lineno could
432  # be implemented).
433  warnings.filterwarnings("ignore", message=".*getargspec.*",
434  category=DeprecationWarning,
435  module=r"nose\.")
436 
437  from .noseclasses import NumpyTestProgram
438 
439  argv, plugins = self.prepare_test_args(
440  label, verbose, extra_argv, doctests, coverage)
441 
442  t = NumpyTestProgram(argv=argv, exit=False, plugins=plugins)
443 
444  return t.result

449 
450  Parameters

452  label : {'fast', 'full', '', attribute identifier}, optional

456  * 'fast' - the default - which corresponds to the ``nosetests -A``

466

470  Returns True if running the benchmarks works, False if an error

472

476  of "test", and can be found under the "benchmarks" sub-directory of the
477  module.
478

480  for it.
481 
482  Examples
483  --------
484  >>> success = np.lib.bench() #doctest: +SKIP
485  Running benchmarks for numpy.lib
486  ...
487  using 562341 items:
488  unique:
489  0.11
490  unique1d:
491  0.11

494  ...

498  True
499 
500  """
501 
502  print("Running benchmarks for %s" % self.package_name)
503  self._show_system_info()
504 
505  argv = self._test_argv(label, verbose, extra_argv)
506  argv += ['--match', r'(?:^|[\\b_\\.%s-])[Bb]ench' % os.sep]
507 
508  # import nose or make informative error
509  nose = import_nose()
510 
511  # get plugin to disable doctests
512  from .noseclasses import Unplugger
513  add_plugins = [Unplugger('doctest')]
