137 def _pivot_col(T, tol=1.0E-12, bland=False):
138  """
139  Given a linear programming simplex tableau, determine the column
140  of the variable to enter the basis.
141 
142  Parameters
143  ----------
144  T : 2D ndarray
145  The simplex tableau.
146  tol : float
147  Elements in the objective row larger than -tol will not be considered
148  for pivoting. Nominally this value is zero, but numerical issues
149  cause a tolerance about zero to be necessary.
150  bland : bool
151  If True, use Bland's rule for selection of the column (select the
152  first column with a negative coefficient in the objective row,
153  regardless of magnitude).
154 
155  Returns
156  -------
157  status: bool
158  True if a suitable pivot column was found, otherwise False.
159  A return of False indicates that the linear programming simplex
160  algorithm is complete.
161  col: int
162  The index of the column of the pivot element.
163  If status is False, col will be returned as nan.
164  """
165  ma = np.ma.masked_where(T[-1, :-1] >= -tol, T[-1, :-1], copy=False)
166  if ma.count() == 0:
167  return False, np.nan
168  if bland:
169  return True, np.where(ma.mask == False)[0][0]
170  return True, np.ma.where(ma == ma.min())[0][0]

173 def _pivot_row(T, pivcol, phase, tol=1.0E-12):
174  """
175  Given a linear programming simplex tableau, determine the row for the
176  pivot operation.
177 
178  Parameters
179  ----------
180  T : 2D ndarray
181  The simplex tableau.
182  pivcol : int
183  The index of the pivot column.
184  phase : int
185  The phase of the simplex algorithm (1 or 2).
186  tol : float
187  Elements in the pivot column smaller than tol will not be considered
188  for pivoting. Nominally this value is zero, but numerical issues
189  cause a tolerance about zero to be necessary.
190 
191  Returns
192  -------
193  status: bool
194  True if a suitable pivot row was found, otherwise False. A return
195  of False indicates that the linear programming problem is unbounded.
196  row: int
197  The index of the row of the pivot element. If status is False, row
198  will be returned as nan.
199  """
200  if phase == 1:
201  k = 2
202  else:
203  k = 1
204  ma = np.ma.masked_where(T[:-k, pivcol] <= tol, T[:-k, pivcol], copy=False)
205  if ma.count() == 0:
206  return False, np.nan
207  mb = np.ma.masked_where(T[:-k, pivcol] <= tol, T[:-k, -1], copy=False)
208  q = mb / ma
209  return True, np.ma.where(q == q.min())[0][0]

212 def _solve_simplex(T, n, basis, maxiter=1000, phase=2, callback=None,
213  tol=1.0E-12, nit0=0, bland=False):
214  """
215  Solve a linear programming problem in "standard maximization form" using
216  the Simplex Method.
217 
218  Minimize :math:`f = c^T x`
219 
220  subject to
221 
222  .. math::
223 
224  Ax = b
225  x_i >= 0
226  b_j >= 0
227 
228  Parameters
229  ----------
230  T : array_like
231  A 2-D array representing the simplex T corresponding to the
232  maximization problem. It should have the form:
233 
234  [[A[0, 0], A[0, 1], ..., A[0, n_total], b[0]],
235  [A[1, 0], A[1, 1], ..., A[1, n_total], b[1]],
236  .
237  .
238  .
239  [A[m, 0], A[m, 1], ..., A[m, n_total], b[m]],
240  [c[0], c[1], ..., c[n_total], 0]]
241 
242  for a Phase 2 problem, or the form:
243 
244  [[A[0, 0], A[0, 1], ..., A[0, n_total], b[0]],
245  [A[1, 0], A[1, 1], ..., A[1, n_total], b[1]],
246  .
247  .
248  .
249  [A[m, 0], A[m, 1], ..., A[m, n_total], b[m]],
250  [c[0], c[1], ..., c[n_total], 0],
251  [c'[0], c'[1], ..., c'[n_total], 0]]
252 
253  for a Phase 1 problem (a Problem in which a basic feasible solution is
254  sought prior to maximizing the actual objective. T is modified in
255  place by _solve_simplex.
256  n : int
257  The number of true variables in the problem.
258  basis : array
259  An array of the indices of the basic variables, such that basis[i]
260  contains the column corresponding to the basic variable for row i.
261  Basis is modified in place by _solve_simplex
262  maxiter : int
263  The maximum number of iterations to perform before aborting the
264  optimization.
265  phase : int
266  The phase of the optimization being executed. In phase 1 a basic
267  feasible solution is sought and the T has an additional row representing
268  an alternate objective function.
269  callback : callable, optional
270  If a callback function is provided, it will be called within each
271  iteration of the simplex algorithm. The callback must have the
272  signature `callback(xk, **kwargs)` where xk is the current solution
273  vector and kwargs is a dictionary containing the following::
274  "T" : The current Simplex algorithm T
275  "nit" : The current iteration.
276  "pivot" : The pivot (row, column) used for the next iteration.
277  "phase" : Whether the algorithm is in Phase 1 or Phase 2.
278  "basis" : The indices of the columns of the basic variables.
279  tol : float
280  The tolerance which determines when a solution is "close enough" to
281  zero in Phase 1 to be considered a basic feasible solution or close
282  enough to positive to to serve as an optimal solution.
283  nit0 : int
284  The initial iteration number used to keep an accurate iteration total
285  in a two-phase problem.
286  bland : bool
287  If True, choose pivots using Bland's rule [3]. In problems which
288  fail to converge due to cycling, using Bland's rule can provide
289  convergence at the expense of a less optimal path about the simplex.
290 
291  Returns
292  -------
293  res : OptimizeResult
294  The optimization result represented as a ``OptimizeResult`` object.
295  Important attributes are: ``x`` the solution array, ``success`` a
296  Boolean flag indicating if the optimizer exited successfully and
297  ``message`` which describes the cause of the termination. Possible
298  values for the ``status`` attribute are:
299  0 : Optimization terminated successfully
300  1 : Iteration limit reached
301  2 : Problem appears to be infeasible
302  3 : Problem appears to be unbounded
303 
304  See `OptimizeResult` for a description of other attributes.
305  """
306  nit = nit0
307  complete = False
308 
309  if phase == 1:
310  m = T.shape[0]-2
311  elif phase == 2:
312  m = T.shape[0]-1
313  else:
314  raise ValueError("Argument 'phase' to _solve_simplex must be 1 or 2")
315 
316  if phase == 2:
317  # Check if any artificial variables are still in the basis.
318  # If yes, check if any coefficients from this row and a column
319  # corresponding to one of the non-artificial variable is non-zero.
320  # If found, pivot at this term. If not, start phase 2.
321  # Do this for all artificial variables in the basis.
322  # Ref: "An Introduction to Linear Programming and Game Theory"
323  # by Paul R. Thie, Gerard E. Keough, 3rd Ed,
324  # Chapter 3.7 Redundant Systems (pag 102)
325  for pivrow in [row for row in range(basis.size)
326  if basis[row] > T.shape[1] - 2]:
327  non_zero_row = [col for col in range(T.shape[1] - 1)
328  if T[pivrow, col] != 0]
329  if len(non_zero_row) > 0:
330  pivcol = non_zero_row[0]
331  # variable represented by pivcol enters
332  # variable in basis[pivrow] leaves
333  basis[pivrow] = pivcol
334  pivval = T[pivrow][pivcol]
335  T[pivrow, :] = T[pivrow, :] / pivval
336  for irow in range(T.shape[0]):
337  if irow != pivrow:
338  T[irow, :] = T[irow, :] - T[pivrow, :]*T[irow, pivcol]
339  nit += 1
340 
341  if len(basis[:m]) == 0:
342  solution = np.zeros(T.shape[1] - 1, dtype=np.float64)
343  else:
344  solution = np.zeros(max(T.shape[1] - 1, max(basis[:m]) + 1),
345  dtype=np.float64)
346 
347  while not complete:
348  # Find the pivot column
349  pivcol_found, pivcol = _pivot_col(T, tol, bland)
350  if not pivcol_found:
351  pivcol = np.nan
352  pivrow = np.nan
353  status = 0
354  complete = True
355  else:
356  # Find the pivot row
357  pivrow_found, pivrow = _pivot_row(T, pivcol, phase, tol)
358  if not pivrow_found:
359  status = 3
360  complete = True
361 
362  if callback is not None:
363  solution[:] = 0
364  solution[basis[:m]] = T[:m, -1]
365  callback(solution[:n], **{"tableau": T,
366  "phase":phase,
367  "nit":nit,
368  "pivot":(pivrow, pivcol),
369  "basis":basis,
370  "complete": complete and phase == 2})
371 
372  if not complete:
373  if nit >= maxiter:
374  # Iteration limit exceeded
375  status = 1
376  complete = True
377  else:
378  # variable represented by pivcol enters
379  # variable in basis[pivrow] leaves
380  basis[pivrow] = pivcol
381  pivval = T[pivrow][pivcol]
382  T[pivrow, :] = T[pivrow, :] / pivval
383  for irow in range(T.shape[0]):
384  if irow != pivrow:
385  T[irow, :] = T[irow, :] - T[pivrow, :]*T[irow, pivcol]
386  nit += 1
387 
388  return nit, status

391 def _linprog_simplex(c, A_ub=None, b_ub=None, A_eq=None, b_eq=None,
392  bounds=None, maxiter=1000, disp=False, callback=None,
393  tol=1.0E-12, bland=False, **unknown_options):
394  """
395  Solve the following linear programming problem via a two-phase
396  simplex algorithm.
397 
398  minimize: c^T * x
399 
400  subject to: A_ub * x <= b_ub
401  A_eq * x == b_eq
402 
403  Parameters
404  ----------
405  c : array_like
406  Coefficients of the linear objective function to be minimized.
407  A_ub : array_like
408  2-D array which, when matrix-multiplied by x, gives the values of the
409  upper-bound inequality constraints at x.
410  b_ub : array_like
411  1-D array of values representing the upper-bound of each inequality
412  constraint (row) in A_ub.
413  A_eq : array_like
414  2-D array which, when matrix-multiplied by x, gives the values of the
415  equality constraints at x.
416  b_eq : array_like
417  1-D array of values representing the RHS of each equality constraint
418  (row) in A_eq.
419  bounds : array_like
420  The bounds for each independent variable in the solution, which can take
421  one of three forms::
422  None : The default bounds, all variables are non-negative.
423  (lb, ub) : If a 2-element sequence is provided, the same
424  lower bound (lb) and upper bound (ub) will be applied
425  to all variables.
426  [(lb_0, ub_0), (lb_1, ub_1), ...] : If an n x 2 sequence is provided,
427  each variable x_i will be bounded by lb[i] and ub[i].
428  Infinite bounds are specified using -np.inf (negative)
429  or np.inf (positive).
430  callback : callable
431  If a callback function is provide, it will be called within each
432  iteration of the simplex algorithm. The callback must have the
433  signature `callback(xk, **kwargs)` where xk is the current solution
434  vector and kwargs is a dictionary containing the following::
435  "tableau" : The current Simplex algorithm tableau
436  "nit" : The current iteration.
437  "pivot" : The pivot (row, column) used for the next iteration.
438  "phase" : Whether the algorithm is in Phase 1 or Phase 2.
439  "bv" : A structured array containing a string representation of each
440  basic variable and its current value.
441 
442  Options
443  -------
444  maxiter : int
445  The maximum number of iterations to perform.
446  disp : bool
447  If True, print exit status message to sys.stdout
448  tol : float
449  The tolerance which determines when a solution is "close enough" to zero
450  in Phase 1 to be considered a basic feasible solution or close enough
451  to positive to to serve as an optimal solution.
452  bland : bool
453  If True, use Bland's anti-cycling rule [3] to choose pivots to
454  prevent cycling. If False, choose pivots which should lead to a
455  converged solution more quickly. The latter method is subject to
456  cycling (non-convergence) in rare instances.
457 
458  Returns
459  -------
460  A scipy.optimize.OptimizeResult consisting of the following fields::
461  x : ndarray
462  The independent variable vector which optimizes the linear
463  programming problem.
464  fun : float
465  Value of the objective function.
466  slack : ndarray
467  The values of the slack variables. Each slack variable corresponds
468  to an inequality constraint. If the slack is zero, then the
469  corresponding constraint is active.
470  success : bool
471  Returns True if the algorithm succeeded in finding an optimal
472  solution.
473  status : int
474  An integer representing the exit status of the optimization::
475  0 : Optimization terminated successfully
476  1 : Iteration limit reached
477  2 : Problem appears to be infeasible
478  3 : Problem appears to be unbounded
479  nit : int
480  The number of iterations performed.
481  message : str
482  A string descriptor of the exit status of the optimization.
483 
484  Examples
485  --------
486  Consider the following problem:
487 
488  Minimize: f = -1*x[0] + 4*x[1]
489 
490  Subject to: -3*x[0] + 1*x[1] <= 6
491  1*x[0] + 2*x[1] <= 4
492  x[1] >= -3
493 
494  where: -inf <= x[0] <= inf
495 
496  This problem deviates from the standard linear programming problem. In
497  standard form, linear programming problems assume the variables x are
498  non-negative. Since the variables don't have standard bounds where
499  0 <= x <= inf, the bounds of the variables must be explicitly set.
500 
501  There are two upper-bound constraints, which can be expressed as
502 
503  dot(A_ub, x) <= b_ub
504 
505  The input for this problem is as follows:
506 
507  >>> from scipy.optimize import linprog
508  >>> c = [-1, 4]
509  >>> A = [[-3, 1], [1, 2]]
510  >>> b = [6, 4]
511  >>> x0_bnds = (None, None)
512  >>> x1_bnds = (-3, None)
513  >>> res = linprog(c, A, b, bounds=(x0_bnds, x1_bnds))
514  >>> print(res)
515  fun: -22.0
516  message: 'Optimization terminated successfully.'
517  nit: 1
518  slack: array([ 39., 0.])
519  status: 0
520  success: True
521  x: array([ 10., -3.])
522 
523  References
524  ----------
525  .. [1] Dantzig, George B., Linear programming and extensions. Rand
526  Corporation Research Study Princeton Univ. Press, Princeton, NJ, 1963
527  .. [2] Hillier, S.H. and Lieberman, G.J. (1995), "Introduction to
528  Mathematical Programming", McGraw-Hill, Chapter 4.
529  .. [3] Bland, Robert G. New finite pivoting rules for the simplex method.
530  Mathematics of Operations Research (2), 1977: pp. 103-107.
531  """
532  _check_unknown_options(unknown_options)
533 
534  status = 0
535  messages = {0: "Optimization terminated successfully.",
536  1: "Iteration limit reached.",
537  2: "Optimization failed. Unable to find a feasible"
538  " starting point.",
539  3: "Optimization failed. The problem appears to be unbounded.",
540  4: "Optimization failed. Singular matrix encountered."}
541  have_floor_variable = False
542 
543  cc = np.asarray(c)
544 
545  # The initial value of the objective function element in the tableau
546  f0 = 0
547 
548  # The number of variables as given by c
549  n = len(c)
550 
551  # Convert the input arguments to arrays (sized to zero if not provided)
552  Aeq = np.asarray(A_eq) if A_eq is not None else np.empty([0, len(cc)])
553  Aub = np.asarray(A_ub) if A_ub is not None else np.empty([0, len(cc)])
554  beq = np.ravel(np.asarray(b_eq)) if b_eq is not None else np.empty([0])
555  bub = np.ravel(np.asarray(b_ub)) if b_ub is not None else np.empty([0])
556 
557  # Analyze the bounds and determine what modifications to be made to
558  # the constraints in order to accommodate them.
559  L = np.zeros(n, dtype=np.float64)
560  U = np.ones(n, dtype=np.float64)*np.inf
561  if bounds is None or len(bounds) == 0:
562  pass
563  elif len(bounds) == 2 and not hasattr(bounds[0], '__len__'):
564  # All bounds are the same
565  a = bounds[0] if bounds[0] is not None else -np.inf
566  b = bounds[1] if bounds[1] is not None else np.inf
567  L = np.asarray(n*[a], dtype=np.float64)
568  U = np.asarray(n*[b], dtype=np.float64)
569  else:
570  if len(bounds) != n:
571  status = -1
572  message = ("Invalid input for linprog with method = 'simplex'. "
573  "Length of bounds is inconsistent with the length of c")
574  else:
575  try:
576  for i in range(n):
577  if len(bounds[i]) != 2:
578  raise IndexError()
579  L[i] = bounds[i][0] if bounds[i][0] is not None else -np.inf
580  U[i] = bounds[i][1] if bounds[i][1] is not None else np.inf
581  except IndexError:
582  status = -1
583  message = ("Invalid input for linprog with "
584  "method = 'simplex'. bounds must be a n x 2 "
585  "sequence/array where n = len(c).")
586 
587  if np.any(L == -np.inf):
588  # If any lower-bound constraint is a free variable
589  # add the first column variable as the "floor" variable which
590  # accommodates the most negative variable in the problem.
591  n = n + 1
592  L = np.concatenate([np.array([0]), L])
593  U = np.concatenate([np.array([np.inf]), U])
594  cc = np.concatenate([np.array([0]), cc])
595  Aeq = np.hstack([np.zeros([Aeq.shape[0], 1]), Aeq])
596  Aub = np.hstack([np.zeros([Aub.shape[0], 1]), Aub])
597  have_floor_variable = True
598 
599  # Now before we deal with any variables with lower bounds < 0,
600  # deal with finite bounds which can be simply added as new constraints.
601  # Also validate bounds inputs here.
602  for i in range(n):
603  if(L[i] > U[i]):
604  status = -1
605  message = ("Invalid input for linprog with method = 'simplex'. "
606  "Lower bound %d is greater than upper bound %d" % (i, i))
607 
608  if np.isinf(L[i]) and L[i] > 0:
609  status = -1
610  message = ("Invalid input for linprog with method = 'simplex'. "
611  "Lower bound may not be +infinity")
612 
613  if np.isinf(U[i]) and U[i] < 0:
614  status = -1
615  message = ("Invalid input for linprog with method = 'simplex'. "
616  "Upper bound may not be -infinity")
617 
618  if np.isfinite(L[i]) and L[i] > 0:
619  # Add a new lower-bound (negative upper-bound) constraint
620  Aub = np.vstack([Aub, np.zeros(n)])
621  Aub[-1, i] = -1
622  bub = np.concatenate([bub, np.array([-L[i]])])
623  L[i] = 0
624 
625  if np.isfinite(U[i]):
626  # Add a new upper-bound constraint
627  Aub = np.vstack([Aub, np.zeros(n)])
628  Aub[-1, i] = 1
629  bub = np.concatenate([bub, np.array([U[i]])])
630  U[i] = np.inf
631 
632  # Now find negative lower bounds (finite or infinite) which require a
633  # change of variables or free variables and handle them appropriately
634  for i in range(0, n):
635  if L[i] < 0:
636  if np.isfinite(L[i]) and L[i] < 0:
637  # Add a change of variables for x[i]
638  # For each row in the constraint matrices, we take the
639  # coefficient from column i in A,
640  # and subtract the product of that and L[i] to the RHS b
641  beq = beq - Aeq[:, i] * L[i]
642  bub = bub - Aub[:, i] * L[i]
643  # We now have a nonzero initial value for the objective
644  # function as well.
645  f0 = f0 - cc[i] * L[i]
646  else:

648  # where v is the first column in all matrices.
649  Aeq[:, 0] = Aeq[:, 0] - Aeq[:, i]
650  Aub[:, 0] = Aub[:, 0] - Aub[:, i]
651  cc[0] = cc[0] - cc[i]
652 
653  if np.isinf(U[i]):

656  message = ("Invalid input for linprog with "
657  "method = 'simplex'. Upper bound may not be -inf.")
658

660  mub = len(bub)
661 
662  # The number of equality constraints (rows in A_eq and elements in b_eq)
663  meq = len(beq)
664 
665  # The total number of constraints
666  m = mub+meq
667 
668  # The number of slack variables (one for each of the upper-bound constraints)
669  n_slack = mub
670

672  # constraint)
673  n_artificial = meq + np.count_nonzero(bub < 0)

679 
680  try:

682  except ValueError:
683  raise ValueError("Invalid input. A_eq must be two-dimensional")
684 
685  if Aeq_rows != meq:
686  status = -1
687  message = ("Invalid input for linprog with method = 'simplex'. "
688  "The number of rows in A_eq must be equal "
689  "to the number of values in b_eq")
690 
691  if Aub_rows != mub:
692  status = -1
693  message = ("Invalid input for linprog with method = 'simplex'. "
694  "The number of rows in A_ub must be equal "
695  "to the number of values in b_ub")
696 
697  if Aeq_cols > 0 and Aeq_cols != n:
698  status = -1
699  message = ("Invalid input for linprog with method = 'simplex'. "
700  "Number of columns in A_eq must be equal "
701  "to the size of c")
702 
703  if Aub_cols > 0 and Aub_cols != n:
704  status = -1
705  message = ("Invalid input for linprog with method = 'simplex'. "
706  "Number of columns in A_ub must be equal to the size of c")
707 
708  if status != 0:
709  # Invalid inputs provided
710  raise ValueError(message)
711 
712  # Create the tableau
713  T = np.zeros([m+2, n+n_slack+n_artificial+1])
714 
715  # Insert objective into tableau
716  T[-2, :n] = cc
717  T[-2, -1] = f0
718 
719  b = T[:-2, -1]
720 
721  if meq > 0:
722  # Add Aeq to the tableau
723  T[:meq, :n] = Aeq
724  # Add beq to the tableau
725  b[:meq] = beq
726  if mub > 0:
727  # Add Aub to the tableau
728  T[meq:meq+mub, :n] = Aub
729  # At bub to the tableau
730  b[meq:meq+mub] = bub
731  # Add the slack variables to the tableau
732  np.fill_diagonal(T[meq:m, n:n+n_slack], 1)
733 
734  # Further set up the tableau.
735  # If a row corresponds to an equality constraint or a negative b (a lower
736  # bound constraint), then an artificial variable is added for that row.
737  # Also, if b is negative, first flip the signs in that constraint.
738  slcount = 0
739  avcount = 0
740  basis = np.zeros(m, dtype=int)
741  r_artificial = np.zeros(n_artificial, dtype=int)
742  for i in range(m):
743  if i < meq or b[i] < 0:
744  # basic variable i is in column n+n_slack+avcount
745  basis[i] = n+n_slack+avcount
746  r_artificial[avcount] = i
747  avcount += 1
748  if b[i] < 0:
749  b[i] *= -1
750  T[i, :-1] *= -1
751  T[i, basis[i]] = 1
752  T[-1, basis[i]] = 1
753  else:
754  # basic variable i is in column n+slcount
755  basis[i] = n+slcount
756  slcount += 1
757 
758  # Make the artificial variables basic feasible variables by subtracting
759  # each row with an artificial variable from the Phase 1 objective
760  for r in r_artificial:
761  T[-1, :] = T[-1, :] - T[r, :]
762 
763  nit1, status = _solve_simplex(T, n, basis, phase=1, callback=callback,
764  maxiter=maxiter, tol=tol, bland=bland)
765 
766  # if pseudo objective is zero, remove the last row from the tableau and
767  # proceed to phase 2
768  if abs(T[-1, -1]) < tol:
769  # Remove the pseudo-objective row from the tableau
770  T = T[:-1, :]
771  # Remove the artificial variable columns from the tableau
772  T = np.delete(T, np.s_[n+n_slack:n+n_slack+n_artificial], 1)
773  else:
774  # Failure to find a feasible starting point
775  status = 2
776 
777  if status != 0:
778  message = messages[status]
779  if disp:
780  print(message)
781  return OptimizeResult(x=np.nan, fun=-T[-1, -1], nit=nit1, status=status,
782  message=message, success=False)
783 
784  # Phase 2
785  nit2, status = _solve_simplex(T, n, basis, maxiter=maxiter-nit1, phase=2,
786  callback=callback, tol=tol, nit0=nit1,
787  bland=bland)
788 
789  solution = np.zeros(n+n_slack+n_artificial)
790  solution[basis[:m]] = T[:m, -1]
791  x = solution[:n]
792  slack = solution[n:n+n_slack]
793 
794  # For those variables with finite negative lower bounds,
795  # reverse the change of variables
796  masked_L = np.ma.array(L, mask=np.isinf(L), fill_value=0.0).filled()
797  x = x + masked_L
798 
799  # For those variables with infinite negative lower bounds,
800  # take x[i] as the difference between x[i] and the floor variable.
801  if have_floor_variable:
802  for i in range(1, n):
803  if np.isinf(L[i]):
804  x[i] -= x[0]
805  x = x[1:]
806 
807  # Optimization complete at this point
808  obj = -T[-1, -1]
809 
810  if status in (0, 1):
811  if disp:
812  print(messages[status])
813  print(" Current function value: {0: <12.6f}".format(obj))
814  print(" Iterations: {0:d}".format(nit2))
815  else:
816  if disp:
817  print(messages[status])
818  print(" Iterations: {0:d}".format(nit2))
819 
820  return OptimizeResult(x=x, fun=obj, nit=int(nit2), status=status, slack=slack,
821  message=messages[status], success=(status == 0))

825  bounds=None, method='simplex', callback=None,
826  options=None):
827  """
828  Minimize a linear objective function subject to linear
829  equality and inequality constraints.
830 
831  Linear Programming is intended to solve the following problem form:
832 
833  Minimize: c^T * x
834 
835  Subject to: A_ub * x <= b_ub
836  A_eq * x == b_eq
837 
838  Parameters
839  ----------
840  c : array_like
841  Coefficients of the linear objective function to be minimized.
842  A_ub : array_like, optional
843  2-D array which, when matrix-multiplied by x, gives the values of the
844  upper-bound inequality constraints at x.
845  b_ub : array_like, optional
846  1-D array of values representing the upper-bound of each inequality
847  constraint (row) in A_ub.
848  A_eq : array_like, optional
849  2-D array which, when matrix-multiplied by x, gives the values of the
850  equality constraints at x.
851  b_eq : array_like, optional
852  1-D array of values representing the RHS of each equality constraint
853  (row) in A_eq.
854  bounds : sequence, optional
855  ``(min, max)`` pairs for each element in ``x``, defining
856  the bounds on that parameter. Use None for one of ``min`` or
857  ``max`` when there is no bound in that direction. By default
858  bounds are ``(0, None)`` (non-negative)
859  If a sequence containing a single tuple is provided, then ``min`` and
860  ``max`` will be applied to all variables in the problem.
861  method : str, optional
862  Type of solver. At this time only 'simplex' is supported
863  :ref:`(see here) <optimize.linprog-simplex>`.
864  callback : callable, optional
865  If a callback function is provide, it will be called within each
866  iteration of the simplex algorithm. The callback must have the signature
867  `callback(xk, **kwargs)` where xk is the current solution vector
868  and kwargs is a dictionary containing the following::
869 
870  "tableau" : The current Simplex algorithm tableau
871  "nit" : The current iteration.
872  "pivot" : The pivot (row, column) used for the next iteration.
873  "phase" : Whether the algorithm is in Phase 1 or Phase 2.
874  "basis" : The indices of the columns of the basic variables.
875 
876  options : dict, optional
877  A dictionary of solver options. All methods accept the following
878  generic options:
879 
880  maxiter : int
881  Maximum number of iterations to perform.
882  disp : bool
883  Set to True to print convergence messages.
884 
885  For method-specific options, see `show_options('linprog')`.
886 
887  Returns
888  -------
889  A `scipy.optimize.OptimizeResult` consisting of the following fields:
890 
891  x : ndarray
892  The independent variable vector which optimizes the linear
893  programming problem.
894  fun : float
895  Value of the objective function.
896  slack : ndarray
897  The values of the slack variables. Each slack variable corresponds
898  to an inequality constraint. If the slack is zero, then the
899  corresponding constraint is active.
900  success : bool
901  Returns True if the algorithm succeeded in finding an optimal
902  solution.
903  status : int
904  An integer representing the exit status of the optimization::
905 
906  0 : Optimization terminated successfully
907  1 : Iteration limit reached
908  2 : Problem appears to be infeasible
909  3 : Problem appears to be unbounded
910 
911  nit : int
912  The number of iterations performed.
913  message : str
914  A string descriptor of the exit status of the optimization.
915 
916  See Also
917  --------
918  show_options : Additional options accepted by the solvers
919 
920  Notes
921  -----
922  This section describes the available solvers that can be selected by the
923  'method' parameter. The default method is :ref:`Simplex <optimize.linprog-simplex>`.
924 
925  Method *Simplex* uses the Simplex algorithm (as it relates to Linear
926  Programming, NOT the Nelder-Mead Simplex) [1]_, [2]_. This algorithm
927  should be reasonably reliable and fast.

930

933  .. [1] Dantzig, George B., Linear programming and extensions. Rand

940  Examples
941  --------
942  Consider the following problem:
943 
944  Minimize: f = -1*x[0] + 4*x[1]
945 
946  Subject to: -3*x[0] + 1*x[1] <= 6
947  1*x[0] + 2*x[1] <= 4
948  x[1] >= -3
949 
950  where: -inf <= x[0] <= inf
951 
952  This problem deviates from the standard linear programming problem.
953  In standard form, linear programming problems assume the variables x are
954  non-negative. Since the variables don't have standard bounds where
955  0 <= x <= inf, the bounds of the variables must be explicitly set.
956 
957  There are two upper-bound constraints, which can be expressed as
958 
959  dot(A_ub, x) <= b_ub
960 
961  The input for this problem is as follows:
962 
963  >>> c = [-1, 4]
964  >>> A = [[-3, 1], [1, 2]]
965  >>> b = [6, 4]
966  >>> x0_bounds = (None, None)
967  >>> x1_bounds = (-3, None)
968  >>> from scipy.optimize import linprog
969  >>> res = linprog(c, A_ub=A, b_ub=b, bounds=(x0_bounds, x1_bounds),
970  ... options={"disp": True})
971  Optimization terminated successfully.
972  Current function value: -22.000000
973  Iterations: 1
974  >>> print(res)
975  fun: -22.0
976  message: 'Optimization terminated successfully.'
977  nit: 1
978  slack: array([ 39., 0.])
979  status: 0
980  success: True
981  x: array([ 10., -3.])
982 
983  Note the actual objective value is 11.428571. In this case we minimized
984  the negative of the objective function.
985
