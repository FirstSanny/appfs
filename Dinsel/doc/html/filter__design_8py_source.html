39 def findfreqs(num, den, N, kind='ba'):
40  """
41  Find array of frequencies for computing the response of an analog filter.
42 
43  Parameters
44  ----------
45  num, den : array_like, 1-D
46  The polynomial coefficients of the numerator and denominator of the
47  transfer function of the filter or LTI system, where the coefficients
48  are ordered from highest to lowest degree. Or, the roots of the
49  transfer function numerator and denominator (i.e. zeroes and poles).
50  N : int
51  The length of the array to be computed.
52  kind : str {'ba', 'zp'}, optional
53  Specifies whether the numerator and denominator are specified by their
54  polynomial coefficients ('ba'), or their roots ('zp').
55 
56  Returns
57  -------
58  w : (N,) ndarray
59  A 1-D array of frequencies, logarithmically spaced.
60 
61  Examples
62  --------
63  Find a set of nine frequencies that span the "interesting part" of the
64  frequency response for the filter with the transfer function
65 
66  H(s) = s / (s^2 + 8s + 25)
67 
68  >>> from scipy import signal
69  >>> signal.findfreqs([1, 0], [1, 8, 25], N=9)
70  array([ 1.00000000e-02, 3.16227766e-02, 1.00000000e-01,
71  3.16227766e-01, 1.00000000e+00, 3.16227766e+00,
72  1.00000000e+01, 3.16227766e+01, 1.00000000e+02])
73  """
74  if kind == 'ba':
75  ep = atleast_1d(roots(den)) + 0j
76  tz = atleast_1d(roots(num)) + 0j
77  elif kind == 'zp':
78  ep = atleast_1d(den) + 0j
79  tz = atleast_1d(num) + 0j
80  else:
81  raise ValueError("input must be one of {'ba', 'zp'}")
82 
83  if len(ep) == 0:
84  ep = atleast_1d(-1000) + 0j
85 
86  ez = r_['-1',
87  numpy.compress(ep.imag >= 0, ep, axis=-1),
88  numpy.compress((abs(tz) < 1e5) & (tz.imag >= 0), tz, axis=-1)]
89 
90  integ = abs(ez) < 1e-10
91  hfreq = numpy.around(numpy.log10(numpy.max(3 * abs(ez.real + integ) +
92  1.5 * ez.imag)) + 0.5)
93  lfreq = numpy.around(numpy.log10(0.1 * numpy.min(abs(real(ez + integ)) +
94  2 * ez.imag)) - 0.5)
95 
96  w = logspace(lfreq, hfreq, N)
97  return w

100 def freqs(b, a, worN=None, plot=None):
101  """
102  Compute frequency response of analog filter.
103 
104  Given the M-order numerator `b` and N-order denominator `a` of an analog
105  filter, compute its frequency response::
106 
107  b[0]*(jw)**M + b[1]*(jw)**(M-1) + ... + b[M]
108  H(w) = ----------------------------------------------
109  a[0]*(jw)**N + a[1]*(jw)**(N-1) + ... + a[N]
110 
111  Parameters
112  ----------
113  b : array_like
114  Numerator of a linear filter.
115  a : array_like
116  Denominator of a linear filter.
117  worN : {None, int, array_like}, optional
118  If None, then compute at 200 frequencies around the interesting parts
119  of the response curve (determined by pole-zero locations). If a single
120  integer, then compute at that many frequencies. Otherwise, compute the
121  response at the angular frequencies (e.g. rad/s) given in `worN`.
122  plot : callable, optional
123  A callable that takes two arguments. If given, the return parameters
124  `w` and `h` are passed to plot. Useful for plotting the frequency
125  response inside `freqs`.
126 
127  Returns
128  -------
129  w : ndarray
130  The angular frequencies at which `h` was computed.
131  h : ndarray
132  The frequency response.
133 
134  See Also
135  --------
136  freqz : Compute the frequency response of a digital filter.
137 
138  Notes
139  -----
140  Using Matplotlib's "plot" function as the callable for `plot` produces
141  unexpected results, this plots the real part of the complex transfer
142  function, not the magnitude. Try ``lambda w, h: plot(w, abs(h))``.
143 
144  Examples
145  --------
146  >>> from scipy.signal import freqs, iirfilter
147 
148  >>> b, a = iirfilter(4, [1, 10], 1, 60, analog=True, ftype='cheby1')
149 
150  >>> w, h = freqs(b, a, worN=np.logspace(-1, 2, 1000))
151 
152  >>> import matplotlib.pyplot as plt
153  >>> plt.semilogx(w, 20 * np.log10(abs(h)))
154  >>> plt.xlabel('Frequency')
155  >>> plt.ylabel('Amplitude response [dB]')
156  >>> plt.grid()
157  >>> plt.show()
158 
159  """
160  if worN is None:
161  w = findfreqs(b, a, 200)
162  elif isinstance(worN, int):
163  N = worN
164  w = findfreqs(b, a, N)
165  else:
166  w = worN
167  w = atleast_1d(w)
168  s = 1j * w
169  h = polyval(b, s) / polyval(a, s)
170  if plot is not None:
171  plot(w, h)
172 
173  return w, h

176 def freqs_zpk(z, p, k, worN=None):
177  """
178  Compute frequency response of analog filter.
179 
180  Given the zeros `z`, poles `p`, and gain `k` of a filter, compute its
181  frequency response::
182 
183  (jw-z[0]) * (jw-z[1]) * ... * (jw-z[-1])
184  H(w) = k * ----------------------------------------
185  (jw-p[0]) * (jw-p[1]) * ... * (jw-p[-1])
186 
187  Parameters
188  ----------
189  z : array_like
190  Zeroes of a linear filter
191  p : array_like
192  Poles of a linear filter
193  k : scalar
194  Gain of a linear filter
195  worN : {None, int, array_like}, optional
196  If None, then compute at 200 frequencies around the interesting parts
197  of the response curve (determined by pole-zero locations). If a single
198  integer, then compute at that many frequencies. Otherwise, compute the
199  response at the angular frequencies (e.g. rad/s) given in `worN`.
200 
201  Returns
202  -------
203  w : ndarray
204  The angular frequencies at which `h` was computed.
205  h : ndarray
206  The frequency response.
207 
208  See Also
209  --------
210  freqs : Compute the frequency response of an analog filter in TF form
211  freqz : Compute the frequency response of a digital filter in TF form
212  freqz_zpk : Compute the frequency response of a digital filter in ZPK form
213 
214  Notes
215  -----
216  .. versionadded: 0.19.0
217 
218  Examples
219  --------
220  >>> from scipy.signal import freqs_zpk, iirfilter
221 
222  >>> z, p, k = iirfilter(4, [1, 10], 1, 60, analog=True, ftype='cheby1',
223  ... output='zpk')
224 
225  >>> w, h = freqs_zpk(z, p, k, worN=np.logspace(-1, 2, 1000))
226 
227  >>> import matplotlib.pyplot as plt
228  >>> plt.semilogx(w, 20 * np.log10(abs(h)))
229  >>> plt.xlabel('Frequency')
230  >>> plt.ylabel('Amplitude response [dB]')
231  >>> plt.grid()
232  >>> plt.show()
233 
234  """
235  k = np.asarray(k)
236  if k.size > 1:
237  raise ValueError('k must be a single scalar gain')
238 
239  if worN is None:
240  w = findfreqs(z, p, 200, kind='zp')
241  elif isinstance(worN, int):
242  N = worN
243  w = findfreqs(z, p, N, kind='zp')
244  else:
245  w = worN
246 
247  w = atleast_1d(w)
248  s = 1j * w
249  num = polyvalfromroots(s, z)
250  den = polyvalfromroots(s, p)
251  h = k * num/den
252  return w, h

255 def freqz(b, a=1, worN=None, whole=False, plot=None):
256  """
257  Compute the frequency response of a digital filter.
258 
259  Given the M-order numerator `b` and N-order denominator `a` of a digital
260  filter, compute its frequency response::
261 
262  jw -jw -jwM
263  jw B(e ) b[0] + b[1]e + .... + b[M]e
264  H(e ) = ---- = -----------------------------------
265  jw -jw -jwN
266  A(e ) a[0] + a[1]e + .... + a[N]e
267 
268  Parameters
269  ----------
270  b : array_like
271  numerator of a linear filter
272  a : array_like
273  denominator of a linear filter
274  worN : {None, int, array_like}, optional
275  If None (default), then compute at 512 frequencies equally spaced
276  around the unit circle.
277  If a single integer, then compute at that many frequencies.
278  If an array_like, compute the response at the frequencies given (in
279  radians/sample).
280  whole : bool, optional
281  Normally, frequencies are computed from 0 to the Nyquist frequency,
282  pi radians/sample (upper-half of unit-circle). If `whole` is True,
283  compute frequencies from 0 to 2*pi radians/sample.
284  plot : callable
285  A callable that takes two arguments. If given, the return parameters
286  `w` and `h` are passed to plot. Useful for plotting the frequency
287  response inside `freqz`.
288 
289  Returns
290  -------
291  w : ndarray
292  The normalized frequencies at which `h` was computed, in
293  radians/sample.
294  h : ndarray
295  The frequency response, as complex numbers.
296 
297  See Also
298  --------
299  sosfreqz
300 
301  Notes
302  -----
303  Using Matplotlib's "plot" function as the callable for `plot` produces
304  unexpected results, this plots the real part of the complex transfer
305  function, not the magnitude. Try ``lambda w, h: plot(w, abs(h))``.
306 
307  Examples
308  --------
309  >>> from scipy import signal
310  >>> b = signal.firwin(80, 0.5, window=('kaiser', 8))
311  >>> w, h = signal.freqz(b)
312 
313  >>> import matplotlib.pyplot as plt
314  >>> fig = plt.figure()
315  >>> plt.title('Digital filter frequency response')
316  >>> ax1 = fig.add_subplot(111)
317 
318  >>> plt.plot(w, 20 * np.log10(abs(h)), 'b')
319  >>> plt.ylabel('Amplitude [dB]', color='b')
320  >>> plt.xlabel('Frequency [rad/sample]')
321 
322  >>> ax2 = ax1.twinx()
323  >>> angles = np.unwrap(np.angle(h))
324  >>> plt.plot(w, angles, 'g')
325  >>> plt.ylabel('Angle (radians)', color='g')
326  >>> plt.grid()
327  >>> plt.axis('tight')
328  >>> plt.show()
329 
330  """
331  b, a = map(atleast_1d, (b, a))
332  if whole:
333  lastpoint = 2 * pi
334  else:
335  lastpoint = pi
336  if worN is None:
337  N = 512
338  w = numpy.linspace(0, lastpoint, N, endpoint=False)
339  elif isinstance(worN, int):
340  N = worN
341  w = numpy.linspace(0, lastpoint, N, endpoint=False)
342  else:
343  w = worN
344  w = atleast_1d(w)
345  zm1 = exp(-1j * w)
346  h = polyval(b[::-1], zm1) / polyval(a[::-1], zm1)
347  if plot is not None:
348  plot(w, h)
349 
350  return w, h

375  around the unit circle.

377  If an array_like, compute the response at the frequencies given (in
378  radians/sample).
379  whole : bool, optional

381  pi radians/sample (upper-half of unit-circle). If `whole` is True,
382  compute frequencies from 0 to 2*pi radians/sample.
383 
384  Returns
385  -------
386  w : ndarray
387  The normalized frequencies at which `h` was computed, in
388  radians/sample.
389  h : ndarray

391

446 def group_delay(system, w=None, whole=False):
447  r"""Compute the group delay of a digital filter.
448 
449  The group delay measures by how many samples amplitude envelopes of
450  various spectral components of a signal are delayed by a filter.
451  It is formally defined as the derivative of continuous (unwrapped) phase::
452 
453  d jw
454  D(w) = - -- arg H(e)
455  dw
456 
457  Parameters
458  ----------
459  system : tuple of array_like (b, a)
460  Numerator and denominator coefficients of a filter transfer function.
461  w : {None, int, array-like}, optional
462  If None (default), then compute at 512 frequencies equally spaced
463  around the unit circle.
464  If a single integer, then compute at that many frequencies.
465  If array, compute the delay at the frequencies given
466  (in radians/sample).
467  whole : bool, optional
468  Normally, frequencies are computed from 0 to the Nyquist frequency,
469  pi radians/sample (upper-half of unit-circle). If `whole` is True,
470  compute frequencies from 0 to ``2*pi`` radians/sample.
471 
472  Returns
473  -------
474  w : ndarray
475  The normalized frequencies at which the group delay was computed,
476  in radians/sample.
477  gd : ndarray
478  The group delay.
479 
480  Notes
481  -----
482  The similar function in MATLAB is called `grpdelay`.
483 
484  If the transfer function :math:`H(z)` has zeros or poles on the unit
485  circle, the group delay at corresponding frequencies is undefined.
486  When such a case arises the warning is raised and the group delay
487  is set to 0 at those frequencies.
488 
489  For the details of numerical computation of the group delay refer to [1]_.
490 
491  .. versionadded: 0.16.0
492 
493  See Also
494  --------
495  freqz : Frequency response of a digital filter
496 
497  References
498  ----------
499  .. [1] Richard G. Lyons, "Understanding Digital Signal Processing,
500  3rd edition", p. 830.
501 
502  Examples
503  --------
504  >>> from scipy import signal
505  >>> b, a = signal.iirdesign(0.1, 0.3, 5, 50, ftype='cheby1')
506  >>> w, gd = signal.group_delay((b, a))
507 
508  >>> import matplotlib.pyplot as plt
509  >>> plt.title('Digital filter group delay')
510  >>> plt.plot(w, gd)
511  >>> plt.ylabel('Group delay [samples]')
512  >>> plt.xlabel('Frequency [rad/sample]')
513  >>> plt.show()
514 
515  """
516  if w is None:
517  w = 512
518 
519  if isinstance(w, int):
520  if whole:
521  w = np.linspace(0, 2 * pi, w, endpoint=False)
522  else:
523  w = np.linspace(0, pi, w, endpoint=False)
524 
525  w = np.atleast_1d(w)
526  b, a = map(np.atleast_1d, system)
527  c = np.convolve(b, a[::-1])
528  cr = c * np.arange(c.size)
529  z = np.exp(-1j * w)
530  num = np.polyval(cr[::-1], z)
531  den = np.polyval(c[::-1], z)
532  singular = np.absolute(den) < 10 * EPSILON
533  if np.any(singular):
534  warnings.warn(
535  "The group delay is singular at frequencies [{0}], setting to 0".
536  format(", ".join("{0:.3f}".format(ws) for ws in w[singular]))
537  )
538 
539  gd = np.zeros_like(w)
540  gd[~singular] = np.real(num[~singular] / den[~singular]) - a.size + 1
541  return w, gd

557 def sosfreqz(sos, worN=None, whole=False):
558  """
559  Compute the frequency response of a digital filter in SOS format.
560 
561  Given `sos`, an array with shape (n, 6) of second order sections of
562  a digital filter, compute the frequency response of the system function::
563 
564  B0(z) B1(z) B{n-1}(z)
565  H(z) = ----- * ----- * ... * ---------
566  A0(z) A1(z) A{n-1}(z)
567 
568  for z = exp(omega*1j), where B{k}(z) and A{k}(z) are numerator and
569  denominator of the transfer function of the k-th second order section.
570 
571  Parameters
572  ----------
573  sos : array_like
574  Array of second-order filter coefficients, must have shape
575  ``(n_sections, 6)``. Each row corresponds to a second-order
576  section, with the first three columns providing the numerator
577  coefficients and the last three providing the denominator
578  coefficients.
579  worN : {None, int, array_like}, optional
580  If None (default), then compute at 512 frequencies equally spaced
581  around the unit circle.
582  If a single integer, then compute at that many frequencies.
583  If an array_like, compute the response at the frequencies given (in
584  radians/sample).
585  whole : bool, optional
586  Normally, frequencies are computed from 0 to the Nyquist frequency,
587  pi radians/sample (upper-half of unit-circle). If `whole` is True,
588  compute frequencies from 0 to 2*pi radians/sample.
589 
590  Returns
591  -------
592  w : ndarray
593  The normalized frequencies at which `h` was computed, in
594  radians/sample.
595  h : ndarray
596  The frequency response, as complex numbers.
597 
598  See Also
599  --------
600  freqz, sosfilt
601 
602  Notes
603  -----
604 
605  .. versionadded:: 0.19.0
606 
607  Examples
608  --------
609  Design a 15th-order bandpass filter in SOS format.
610 
611  >>> from scipy import signal
612  >>> sos = signal.ellip(15, 0.5, 60, (0.2, 0.4), btype='bandpass',
613  ... output='sos')
614 
615  Compute the frequency response at 1500 points from DC to Nyquist.
616 
617  >>> w, h = signal.sosfreqz(sos, worN=1500)
618 
619  Plot the response.
620 
621  >>> import matplotlib.pyplot as plt
622  >>> plt.subplot(2, 1, 1)
623  >>> db = 20*np.log10(np.abs(h))
624  >>> plt.plot(w/np.pi, db)
625  >>> plt.ylim(-75, 5)
626  >>> plt.grid(True)
627  >>> plt.yticks([0, -20, -40, -60])
628  >>> plt.ylabel('Gain [dB]')
629  >>> plt.title('Frequency Response')
630  >>> plt.subplot(2, 1, 2)
631  >>> plt.plot(w/np.pi, np.angle(h))
632  >>> plt.grid(True)
633  >>> plt.yticks([-np.pi, -0.5*np.pi, 0, 0.5*np.pi, np.pi],
634  ... [r'$-\\pi$', r'$-\\pi/2$', '0', r'$\\pi/2$', r'$\\pi$'])
635  >>> plt.ylabel('Phase [rad]')
636  >>> plt.xlabel('Normalized frequency (1.0 = Nyquist)')
637  >>> plt.show()
638 
639  If the same filter is implemented as a single transfer function,
640  numerical error corrupts the frequency response:
641 
642  >>> b, a = signal.ellip(15, 0.5, 60, (0.2, 0.4), btype='bandpass',
643  ... output='ba')
644  >>> w, h = signal.freqz(b, a, worN=1500)
645  >>> plt.subplot(2, 1, 1)
646  >>> db = 20*np.log10(np.abs(h))
647  >>> plt.plot(w/np.pi, db)
648  >>> plt.subplot(2, 1, 2)
649  >>> plt.plot(w/np.pi, np.angle(h))
650  >>> plt.show()
651 
652  """
653 
654  sos, n_sections = _validate_sos(sos)
655  if n_sections == 0:
656  raise ValueError('Cannot compute frequencies with no sections')
657  h = 1.
658  for row in sos:

664 def _cplxreal(z, tol=None):
665  """
666  Split into complex and real parts, combining conjugate pairs.
667 
668  The 1D input vector `z` is split up into its complex (`zc`) and real (`zr`)
669  elements. Every complex element must be part of a complex-conjugate pair,
670  which are combined into a single number (with positive imaginary part) in
671  the output. Two complex numbers are considered a conjugate pair if their
672  real and imaginary parts differ in magnitude by less than ``tol * abs(z)``.
673 
674  Parameters
675  ----------
676  z : array_like
677  Vector of complex numbers to be sorted and split
678  tol : float, optional
679  Relative tolerance for testing realness and conjugate equality.
680  Default is ``100 * spacing(1)`` of `z`'s data type (i.e. 2e-14 for
681  float64)
682 
683  Returns
684  -------
685  zc : ndarray
686  Complex elements of `z`, with each pair represented by a single value
687  having positive imaginary part, sorted first by real part, and then
688  by magnitude of imaginary part. The pairs are averaged when combined
689  to reduce error.
690  zr : ndarray
691  Real elements of `z` (those having imaginary part less than
692  `tol` times their magnitude), sorted by value.
693 
694  Raises
695  ------
696  ValueError
697  If there are any complex numbers in `z` for which a conjugate
698  cannot be found.
699 
700  See Also
701  --------
702  _cplxpair
703 
704  Examples
705  --------
706  >>> a = [4, 3, 1, 2-2j, 2+2j, 2-1j, 2+1j, 2-1j, 2+1j, 1+1j, 1-1j]
707  >>> zc, zr = _cplxreal(a)
708  >>> print zc
709  [ 1.+1.j 2.+1.j 2.+1.j 2.+2.j]
710  >>> print zr
711  [ 1. 3. 4.]
712  """
713 
714  z = atleast_1d(z)
715  if z.size == 0:
716  return z, z
717  elif z.ndim != 1:
718  raise ValueError('_cplxreal only accepts 1D input')
719 
720  if tol is None:
721  # Get tolerance from dtype of input
722  tol = 100 * np.finfo((1.0 * z).dtype).eps
723 
724  # Sort by real part, magnitude of imaginary part (speed up further sorting)
725  z = z[np.lexsort((abs(z.imag), z.real))]
726 
727  # Split reals from conjugate pairs
728  real_indices = abs(z.imag) <= tol * abs(z)
729  zr = z[real_indices].real
730 
731  if len(zr) == len(z):
732  # Input is entirely real
733  return array([]), zr
734 
735  # Split positive and negative halves of conjugates
736  z = z[~real_indices]
737  zp = z[z.imag > 0]
738  zn = z[z.imag < 0]
739 
740  if len(zp) != len(zn):
741  raise ValueError('Array contains complex value with no matching '
742  'conjugate.')
743 
744  # Find runs of (approximately) the same real part
745  same_real = np.diff(zp.real) <= tol * abs(zp[:-1])
746  diffs = numpy.diff(concatenate(([0], same_real, [0])))
747  run_starts = numpy.where(diffs > 0)[0]
748  run_stops = numpy.where(diffs < 0)[0]
749 
750  # Sort each run by their imaginary parts
751  for i in range(len(run_starts)):
752  start = run_starts[i]
753  stop = run_stops[i] + 1
754  for chunk in (zp[start:stop], zn[start:stop]):
755  chunk[...] = chunk[np.lexsort([abs(chunk.imag)])]
756 
757  # Check that negatives match positives
758  if any(abs(zp - zn.conj()) > tol * abs(zn)):
759  raise ValueError('Array contains complex value with no matching '
760  'conjugate.')
761 
762  # Average out numerical inaccuracy in real vs imag parts of pairs
763  zc = (zp + zn.conj()) / 2
764 
765  return zc, zr

768 def _cplxpair(z, tol=None):
769  """
770  Sort into pairs of complex conjugates.
771 
772  Complex conjugates in `z` are sorted by increasing real part. In each
773  pair, the number with negative imaginary part appears first.
774 
775  If pairs have identical real parts, they are sorted by increasing
776  imaginary magnitude.
777 
778  Two complex numbers are considered a conjugate pair if their real and
779  imaginary parts differ in magnitude by less than ``tol * abs(z)``. The
780  pairs are forced to be exact complex conjugates by averaging the positive
781  and negative values.
782 
783  Purely real numbers are also sorted, but placed after the complex
784  conjugate pairs. A number is considered real if its imaginary part is
785  smaller than `tol` times the magnitude of the number.
786 
787  Parameters
788  ----------
789  z : array_like
790  1-dimensional input array to be sorted.
791  tol : float, optional
792  Relative tolerance for testing realness and conjugate equality.
793  Default is ``100 * spacing(1)`` of `z`'s data type (i.e. 2e-14 for
794  float64)
795 
796  Returns
797  -------
798  y : ndarray
799  Complex conjugate pairs followed by real numbers.
800 
801  Raises
802  ------
803  ValueError
804  If there are any complex numbers in `z` for which a conjugate
805  cannot be found.
806 
807  See Also
808  --------
809  _cplxreal
810 
811  Examples
812  --------
813  >>> a = [4, 3, 1, 2-2j, 2+2j, 2-1j, 2+1j, 2-1j, 2+1j, 1+1j, 1-1j]
814  >>> z = _cplxpair(a)
815  >>> print(z)
816  [ 1.-1.j 1.+1.j 2.-1.j 2.+1.j 2.-1.j 2.+1.j 2.-2.j 2.+2.j 1.+0.j
817  3.+0.j 4.+0.j]
818  """
819 
820  z = atleast_1d(z)
821  if z.size == 0 or np.isrealobj(z):
822  return np.sort(z)
823 
824  if z.ndim != 1:
825  raise ValueError('z must be 1-dimensional')

827  zc, zr = _cplxreal(z, tol)

831  zc = np.dstack((zc.conj(), zc)).flatten()

833  return z

836 def tf2zpk(b, a):
837  r"""Return zero, pole, gain (z, p, k) representation from a numerator,
838  denominator representation of a linear filter.
839 
840  Parameters
841  ----------
842  b : array_like
843  Numerator polynomial coefficients.
844  a : array_like
845  Denominator polynomial coefficients.
846 
847  Returns
848  -------
849  z : ndarray
850  Zeros of the transfer function.
851  p : ndarray
852  Poles of the transfer function.
853  k : float
854  System gain.
855 
856  Notes
857  -----
858  If some values of `b` are too close to 0, they are removed. In that case,
859  a BadCoefficients warning is emitted.
860 
861  The `b` and `a` arrays are interpreted as coefficients for positive,
862  descending powers of the transfer function variable. So the inputs
863  :math:`b = [b_0, b_1, ..., b_M]` and :math:`a =[a_0, a_1, ..., a_N]`
864  can represent an analog filter of the form:
865 
866  .. math::
867 
868  H(s) = \frac
869  {b_0 s^M + b_1 s^{(M-1)} + \cdots + b_M}
870  {a_0 s^N + a_1 s^{(N-1)} + \cdots + a_N}
871 
872  or a discrete-time filter of the form:
873 
874  .. math::
875 
876  H(z) = \frac
877  {b_0 z^M + b_1 z^{(M-1)} + \cdots + b_M}
878  {a_0 z^N + a_1 z^{(N-1)} + \cdots + a_N}
879 
880  This "positive powers" form is found more commonly in controls
881  engineering. If `M` and `N` are equal (which is true for all filters
882  generated by the bilinear transform), then this happens to be equivalent
883  to the "negative powers" discrete-time form preferred in DSP:
884 
885  .. math::
886 
887  H(z) = \frac
888  {b_0 + b_1 z^{-1} + \cdots + b_M z^{-M}}
889  {a_0 + a_1 z^{-1} + \cdots + a_N z^{-N}}
890 
891  Although this is true for common filters, remember that this is not true
892  in the general case. If `M` and `N` are not equal, the discrete-time
893  transfer function coefficients must first be converted to the "positive
894  powers" form before finding the poles and zeros.
895 
896  """
897  b, a = normalize(b, a)
898  b = (b + 0.0) / a[0]
899  a = (a + 0.0) / a[0]
900  k = b[0]
901  b /= b[0]
902  z = roots(b)
903  p = roots(a)
904  return z, p, k

1083 def zpk2sos(z, p, k, pairing='nearest'):
1084  """
1085  Return second-order sections from zeros, poles, and gain of a system
1086 
1087  Parameters
1088  ----------
1089  z : array_like
1090  Zeros of the transfer function.
1091  p : array_like
1092  Poles of the transfer function.
1093  k : float
1094  System gain.
1095  pairing : {'nearest', 'keep_odd'}, optional
1096  The method to use to combine pairs of poles and zeros into sections.
1097  See Notes below.
1098 
1099  Returns
1100  -------
1101  sos : ndarray
1102  Array of second-order filter coefficients, with shape
1103  ``(n_sections, 6)``. See `sosfilt` for the SOS filter format
1104  specification.
1105 
1106  See Also
1107  --------
1108  sosfilt
1109 
1110  Notes
1111  -----
1112  The algorithm used to convert ZPK to SOS format is designed to
1113  minimize errors due to numerical precision issues. The pairing
1114  algorithm attempts to minimize the peak gain of each biquadratic
1115  section. This is done by pairing poles with the nearest zeros, starting
1116  with the poles closest to the unit circle.
1117 
1118  *Algorithms*
1119 
1120  The current algorithms are designed specifically for use with digital
1121  filters. (The output coefficents are not correct for analog filters.)
1122 
1123  The steps in the ``pairing='nearest'`` and ``pairing='keep_odd'``
1124  algorithms are mostly shared. The ``nearest`` algorithm attempts to
1125  minimize the peak gain, while ``'keep_odd'`` minimizes peak gain under
1126  the constraint that odd-order systems should retain one section
1127  as first order. The algorithm steps and are as follows:
1128 
1129  As a pre-processing step, add poles or zeros to the origin as
1130  necessary to obtain the same number of poles and zeros for pairing.
1131  If ``pairing == 'nearest'`` and there are an odd number of poles,
1132  add an additional pole and a zero at the origin.
1133 
1134  The following steps are then iterated over until no more poles or
1135  zeros remain:
1136 
1137  1. Take the (next remaining) pole (complex or real) closest to the
1138  unit circle to begin a new filter section.
1139 
1140  2. If the pole is real and there are no other remaining real poles [#]_,
1141  add the closest real zero to the section and leave it as a first
1142  order section. Note that after this step we are guaranteed to be
1143  left with an even number of real poles, complex poles, real zeros,
1144  and complex zeros for subsequent pairing iterations.
1145 
1146  3. Else:
1147 
1148  1. If the pole is complex and the zero is the only remaining real
1149  zero*, then pair the pole with the *next* closest zero
1150  (guaranteed to be complex). This is necessary to ensure that
1151  there will be a real zero remaining to eventually create a
1152  first-order section (thus keeping the odd order).
1153 
1154  2. Else pair the pole with the closest remaining zero (complex or
1155  real).
1156 
1157  3. Proceed to complete the second-order section by adding another
1158  pole and zero to the current pole and zero in the section:
1159 
1160  1. If the current pole and zero are both complex, add their
1161  conjugates.
1162 
1163  2. Else if the pole is complex and the zero is real, add the
1164  conjugate pole and the next closest real zero.
1165 
1166  3. Else if the pole is real and the zero is complex, add the
1167  conjugate zero and the real pole closest to those zeros.
1168 
1169  4. Else (we must have a real pole and real zero) add the next
1170  real pole closest to the unit circle, and then add the real
1171  zero closest to that pole.
1172 
1173  .. [#] This conditional can only be met for specific odd-order inputs
1174  with the ``pairing == 'keep_odd'`` method.
1175 
1176  .. versionadded:: 0.16.0
1177 
1178  Examples
1179  --------
1180 
1181  Design a 6th order low-pass elliptic digital filter for a system with a
1182  sampling rate of 8000 Hz that has a pass-band corner frequency of
1183  1000 Hz. The ripple in the pass-band should not exceed 0.087 dB, and
1184  the attenuation in the stop-band should be at least 90 dB.
1185 
1186  In the following call to `signal.ellip`, we could use ``output='sos'``,
1187  but for this example, we'll use ``output='zpk'``, and then convert to SOS
1188  format with `zpk2sos`:
1189 
1190  >>> from scipy import signal
1191  >>> z, p, k = signal.ellip(6, 0.087, 90, 1000/(0.5*8000), output='zpk')
1192 
1193  Now convert to SOS format.
1194 
1195  >>> sos = signal.zpk2sos(z, p, k)
1196 
1197  The coefficients of the numerators of the sections:
1198 
1199  >>> sos[:, :3]
1200  array([[ 0.0014154 , 0.00248707, 0.0014154 ],
1201  [ 1. , 0.72965193, 1. ],
1202  [ 1. , 0.17594966, 1. ]])
1203 
1204  The symmetry in the coefficients occurs because all the zeros are on the
1205  unit circle.
1206 
1207  The coefficients of the denominators of the sections:
1208 
1209  >>> sos[:, 3:]
1210  array([[ 1. , -1.32543251, 0.46989499],
1211  [ 1. , -1.26117915, 0.6262586 ],
1212  [ 1. , -1.25707217, 0.86199667]])
1213 
1214  The next example shows the effect of the `pairing` option. We have a
1215  system with three poles and three zeros, so the SOS array will have
1216  shape (2, 6). The means there is, in effect, an extra pole and an extra
1217  zero at the origin in the SOS representation.
1218 
1219  >>> z1 = np.array([-1, -0.5-0.5j, -0.5+0.5j])
1220  >>> p1 = np.array([0.75, 0.8+0.1j, 0.8-0.1j])
1221 
1222  With ``pairing='nearest'`` (the default), we obtain
1223 
1224  >>> signal.zpk2sos(z1, p1, 1)
1225  array([[ 1. , 1. , 0.5 , 1. , -0.75, 0. ],
1226  [ 1. , 1. , 0. , 1. , -1.6 , 0.65]])
1227 
1228  The first section has the zeros {-0.5-0.05j, -0.5+0.5j} and the poles
1229  {0, 0.75}, and the second section has the zeros {-1, 0} and poles
1230  {0.8+0.1j, 0.8-0.1j}. Note that the extra pole and zero at the origin
1231  have been assigned to different sections.
1232 
1233  With ``pairing='keep_odd'``, we obtain:
1234 
1235  >>> signal.zpk2sos(z1, p1, 1, pairing='keep_odd')
1236  array([[ 1. , 1. , 0. , 1. , -0.75, 0. ],
1237  [ 1. , 1. , 0.5 , 1. , -1.6 , 0.65]])
1238 
1239  The extra pole and zero at the origin are in the same section.
1240  The first section is, in effect, a first-order section.
1241 
1242  """
1243  # TODO in the near future:
1244  # 1. Add SOS capability to `filtfilt`, `freqz`, etc. somehow (#3259).
1245  # 2. Make `decimate` use `sosfilt` instead of `lfilter`.
1246  # 3. Make sosfilt automatically simplify sections to first order
1247  # when possible. Note this might make `sosfiltfilt` a bit harder (ICs).
1248  # 4. Further optimizations of the section ordering / pole-zero pairing.
1249  # See the wiki for other potential issues.
1250 
1251  valid_pairings = ['nearest', 'keep_odd']
1252  if pairing not in valid_pairings:
1253  raise ValueError('pairing must be one of %s, not %s'
1254  % (valid_pairings, pairing))
1255  if len(z) == len(p) == 0:
1256  return array([[k, 0., 0., 1., 0., 0.]])
1257 
1258  # ensure we have the same number of poles and zeros, and make copies
1259  p = np.concatenate((p, np.zeros(max(len(z) - len(p), 0))))
1260  z = np.concatenate((z, np.zeros(max(len(p) - len(z), 0))))
1261  n_sections = (max(len(p), len(z)) + 1) // 2
1262  sos = zeros((n_sections, 6))
1263 
1264  if len(p) % 2 == 1 and pairing == 'nearest':
1265  p = np.concatenate((p, [0.]))
1266  z = np.concatenate((z, [0.]))
1267  assert len(p) == len(z)
1268 
1269  # Ensure we have complex conjugate pairs
1270  # (note that _cplxreal only gives us one element of each complex pair):
1271  z = np.concatenate(_cplxreal(z))
1272  p = np.concatenate(_cplxreal(p))
1273 
1274  p_sos = np.zeros((n_sections, 2), np.complex128)
1275  z_sos = np.zeros_like(p_sos)
1276  for si in range(n_sections):
1277  # Select the next "worst" pole
1278  p1_idx = np.argmin(np.abs(1 - np.abs(p)))
1279  p1 = p[p1_idx]
1280  p = np.delete(p, p1_idx)
1281 
1282  # Pair that pole with a zero
1283 
1284  if np.isreal(p1) and np.isreal(p).sum() == 0:
1285  # Special case to set a first-order section
1286  z1_idx = _nearest_real_complex_idx(z, p1, 'real')
1287  z1 = z[z1_idx]
1288  z = np.delete(z, z1_idx)
1289  p2 = z2 = 0
1290  else:
1291  if not np.isreal(p1) and np.isreal(z).sum() == 1:
1292  # Special case to ensure we choose a complex zero to pair
1293  # with so later (setting up a first-order section)
1294  z1_idx = _nearest_real_complex_idx(z, p1, 'complex')
1295  assert not np.isreal(z[z1_idx])
1296  else:
1297  # Pair the pole with the closest zero (real or complex)
1298  z1_idx = np.argmin(np.abs(p1 - z))
1299  z1 = z[z1_idx]
1300  z = np.delete(z, z1_idx)
1301 
1302  # Now that we have p1 and z1, figure out what p2 and z2 need to be
1303  if not np.isreal(p1):
1304  if not np.isreal(z1): # complex pole, complex zero
1305  p2 = p1.conj()
1306  z2 = z1.conj()
1307  else: # complex pole, real zero
1308  p2 = p1.conj()
1309  z2_idx = _nearest_real_complex_idx(z, p1, 'real')
1310  z2 = z[z2_idx]
1311  assert np.isreal(z2)
1312  z = np.delete(z, z2_idx)
1313  else:
1314  if not np.isreal(z1): # real pole, complex zero
1315  z2 = z1.conj()
1316  p2_idx = _nearest_real_complex_idx(p, z1, 'real')
1317  p2 = p[p2_idx]
1318  assert np.isreal(p2)
1319  else: # real pole, real zero
1320  # pick the next "worst" pole to use
1321  idx = np.where(np.isreal(p))[0]
1322  assert len(idx) > 0
1323  p2_idx = idx[np.argmin(np.abs(np.abs(p[idx]) - 1))]
1324  p2 = p[p2_idx]
1325  # find a real zero to match the added pole
1326  assert np.isreal(p2)
1327  z2_idx = _nearest_real_complex_idx(z, p2, 'real')
1328  z2 = z[z2_idx]
1329  assert np.isreal(z2)
1330  z = np.delete(z, z2_idx)
1331  p = np.delete(p, p2_idx)
1332  p_sos[si] = [p1, p2]
1333  z_sos[si] = [z1, z2]
1334  assert len(p) == len(z) == 0 # we've consumed all poles and zeros
1335  del p, z
1336 
1337  # Construct the system, reversing order so the "worst" are last
1338  p_sos = np.reshape(p_sos[::-1], (n_sections, 2))
1339  z_sos = np.reshape(z_sos[::-1], (n_sections, 2))
1340  gains = np.ones(n_sections)
1341  gains[0] = k
1342  for si in range(n_sections):

1344  sos[si] = np.concatenate(x)

1355  alignment when dealing with SIMO transfer functions.

1358  ----------

1361 
1362  Returns
1363  -------

1366  array with padded zeros for shorter numerators is returned. Otherwise
1367  returns ``np.asarray(nums)``.

1370  # The statement can throw a ValueError if one
1371  # of the numerators is a single digit and another
1372  # is array-like e.g. if nums = [5, [1, 2, 3]]
1373  nums = asarray(nums)
1374

1386

1394 def normalize(b, a):
1395  """Normalize numerator/denominator of a continuous-time transfer function.
1396 
1397  If values of `b` are too close to 0, they are removed. In that case, a
1398  BadCoefficients warning is emitted.
1399 
1400  Parameters
1401  ----------
1402  b: array_like
1403  Numerator of the transfer function. Can be a 2d array to normalize
1404  multiple transfer functions.
1405  a: array_like
1406  Denominator of the transfer function. At most 1d.
1407 
1408  Returns
1409  -------
1410  num: array
1411  The numerator of the normalized transfer function. At least a 1d
1412  array. A 2d-array if the input `num` is a 2d array.
1413  den: 1d-array
1414  The denominator of the normalized transfer function.
1415 
1416  Notes
1417  -----
1418  Coefficients for both the numerator and denominator should be specified in
1419  descending exponent order (e.g., ``s^2 + 3s + 5`` would be represented as
1420  ``[1, 3, 5]``).
1421  """
1422  num, den = b, a
1423 
1424  den = np.atleast_1d(den)
1425  num = np.atleast_2d(_align_nums(num))
1426 
1427  if den.ndim != 1:
1428  raise ValueError("Denominator polynomial must be rank-1 array.")
1429  if num.ndim > 2:
1430  raise ValueError("Numerator polynomial must be rank-1 or"
1431  " rank-2 array.")
1432  if np.all(den == 0):
1433  raise ValueError("Denominator must have at least on nonzero element.")
1434 
1435  # Trim leading zeros in denominator, leave at least one.
1436  den = np.trim_zeros(den, 'f')
1437 
1438  # Normalize transfer function
1439  num, den = num / den[0], den / den[0]
1440 
1441  # Count numerator columns that are all zero
1442  leading_zeros = 0
1443  for col in num.T:
1444  if np.allclose(col, 0, atol=1e-14):
1445  leading_zeros += 1
1446  else:
1447  break
1448 
1449  # Trim leading zeros of numerator
1450  if leading_zeros > 0:
1451  warnings.warn("Badly conditioned filter coefficients (numerator): the "
1452  "results may be meaningless", BadCoefficients)
1453  # Make sure at least one column remains
1454  if leading_zeros == num.shape[1]:
1455  leading_zeros -= 1
1456  num = num[:, leading_zeros:]
1457 
1458  # Squeeze first dimension if singular
1459  if num.shape[0] == 1:
1460  num = num[0, :]
1461 
1462  return num, den

1919 def _zpkbilinear(z, p, k, fs):
1920  """
1921  Return a digital filter from an analog one using a bilinear transform.
1922 
1923  Transform a set of poles and zeros from the analog s-plane to the digital
1924  z-plane using Tustin's method, which substitutes ``(z-1) / (z+1)`` for
1925  ``s``, maintaining the shape of the frequency response.
1926 
1927  Parameters
1928  ----------
1929  z : array_like
1930  Zeros of the analog IIR filter transfer function.
1931  p : array_like
1932  Poles of the analog IIR filter transfer function.
1933  k : float
1934  System gain of the analog IIR filter transfer function.
1935  fs : float
1936  Sample rate, as ordinary frequency (e.g. hertz). No prewarping is
1937  done in this function.
1938 
1939  Returns
1940  -------
1941  z : ndarray
1942  Zeros of the transformed digital filter transfer function.
1943  p : ndarray
1944  Poles of the transformed digital filter transfer function.
1945  k : float
1946  System gain of the transformed digital filter.
1947 
1948  """
1949  z = atleast_1d(z)
1950  p = atleast_1d(p)
1951 
1952  degree = _relative_degree(z, p)
1953 
1954  fs2 = 2*fs
1955 
1956  # Bilinear transform the poles and zeros
1957  z_z = (fs2 + z) / (fs2 - z)
1958  p_z = (fs2 + p) / (fs2 - p)
1959 
1960  # Any zeros that were at infinity get moved to the Nyquist frequency
1961  z_z = append(z_z, -ones(degree))
1962

1964  k_z = k * real(prod(fs2 - z) / prod(fs2 - p))

2723 def band_stop_obj(wp, ind, passb, stopb, gpass, gstop, type):
2724  """
2725  Band Stop Objective Function for order minimization.
2726 
2727  Returns the non-integer order for an analog band stop filter.
2728 
2729  Parameters
2730  ----------
2731  wp : scalar
2732  Edge of passband `passb`.
2733  ind : int, {0, 1}
2734  Index specifying which `passb` edge to vary (0 or 1).
2735  passb : ndarray
2736  Two element sequence of fixed passband edges.
2737  stopb : ndarray
2738  Two element sequence of fixed stopband edges.
2739  gstop : float
2740  Amount of attenuation in stopband in dB.
2741  gpass : float
2742  Amount of ripple in the passband in dB.
2743  type : {'butter', 'cheby', 'ellip'}
2744  Type of filter.
2745 
2746  Returns
2747  -------
2748  n : scalar
2749  Filter order (possibly non-integer).
2750 
2751  """
2752  passbC = passb.copy()
2753  passbC[ind] = wp
2754  nat = (stopb * (passbC[0] - passbC[1]) /
2755  (stopb ** 2 - passbC[0] * passbC[1]))
2756  nat = min(abs(nat))
2757 
2758  if type == 'butter':
2759  GSTOP = 10 ** (0.1 * abs(gstop))
2760  GPASS = 10 ** (0.1 * abs(gpass))
2761  n = (log10((GSTOP - 1.0) / (GPASS - 1.0)) / (2 * log10(nat)))
2762  elif type == 'cheby':

2764  GPASS = 10 ** (0.1 * abs(gpass))
2765  n = arccosh(sqrt((GSTOP - 1.0) / (GPASS - 1.0))) / arccosh(nat)
2766  elif type == 'ellip':

2768  GPASS = 10 ** (0.1 * gpass)
2769  arg1 = sqrt((GPASS - 1.0) / (GSTOP - 1.0))
2770  arg0 = 1.0 / nat
2771  d0 = special.ellipk([arg0 ** 2, 1 - arg0 ** 2])
2772  d1 = special.ellipk([arg1 ** 2, 1 - arg1 ** 2])
2773  n = (d0[0] * d1[1] / (d0[1] * d1[0]))
2774  else:
