36  """Attempt to determine the blocksize of a sparse matrix
37 
38  Returns a blocksize=(r,c) such that
39  - A.nnz / A.tobsr( (r,c) ).nnz > efficiency
40  """
41  if not (isspmatrix_csr(A) or isspmatrix_csc(A)):

44  if A.nnz == 0:

66  if e66 > efficiency:

68  else:

70  else:
