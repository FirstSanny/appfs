168  def test_spherical_in_inf_complex(self):
169  # http://dlmf.nist.gov/10.52.E5
170  # Ideally, i1n(n, 1j*inf) = 0 and i1n(n, (1+1j)*inf) = (1+1j)*inf, but
171  # this appears impossible to achieve because C99 regards any complex
