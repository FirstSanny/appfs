275 def tabulate(vals):
276  # From pfmoore on GitHub:
277  # https://github.com/pypa/pip/issues/3651#issuecomment-216932564
278  assert len(vals) > 0
279 
280  sizes = [0] * max(len(x) for x in vals)
281  for row in vals:
282  sizes = [max(s, len(str(c))) for s, c in zip_longest(sizes, row)]
283 
284  result = []
285  for row in vals:
286  display = " ".join([str(c).ljust(s) if c is not None else ''
287  for s, c in zip_longest(sizes, row)])
288  result.append(display)
289 
290  return result, sizes

293 def format_for_columns(pkgs, options):
294  """
295  Convert the package data into something usable
296  by output_package_listing_columns.
297  """
298  running_outdated = options.outdated
299  # Adjust the header for the `pip list --outdated` case.
300  if running_outdated:
301  header = ["Package", "Version", "Latest", "Type"]
302  else:
303  header = ["Package", "Version"]
304 
305  data = []
306  if any(dist_is_editable(x) for x in pkgs):
307  header.append("Location")
308 
309  for proj in pkgs:
310  # if we're working on the 'outdated' list, separate out the
311  # latest_version and type
312  row = [proj.project_name, proj.version]
313 
314  if running_outdated:
315  row.append(proj.latest_version)
316  row.append(proj.latest_filetype)
317 
318  if dist_is_editable(proj):
319  row.append(proj.location)
320 
321  data.append(row)
322
