33 """Some more special functions which may be useful for multivariate statistical
34 analysis."""
35 
36 from __future__ import division, print_function, absolute_import
37 
38 import numpy as np
39 from scipy.special import gammaln as loggam
40 
41 
42 __all__ = ['multigammaln']

45 def multigammaln(a, d):
46  r"""Returns the log of multivariate gamma, also sometimes called the
47  generalized gamma.
48 
49  Parameters
50  ----------
51  a : ndarray
52  The multivariate gamma is computed for each item of `a`.
53  d : int
54  The dimension of the space of integration.
55 
56  Returns
57  -------
58  res : ndarray
59  The values of the log multivariate gamma at the given points `a`.
60 
61  Notes
62  -----
63  The formal definition of the multivariate gamma of dimension d for a real
64  `a` is
65 
66  .. math::
67 
68  \Gamma_d(a) = \int_{A>0} e^{-tr(A)} |A|^{a - (d+1)/2} dA
69 
70  with the condition :math:`a > (d-1)/2`, and :math:`A > 0` being the set of
71  all the positive definite matrices of dimension `d`. Note that `a` is a
72  scalar: the integrand only is multivariate, the argument is not (the
73  function is defined over a subset of the real set).
74 
75  This can be proven to be equal to the much friendlier equation
76 
77  .. math::
78 
79  \Gamma_d(a) = \pi^{d(d-1)/4} \prod_{i=1}^{d} \Gamma(a - (i-1)/2).
80 
81  References
82  ----------
83  R. J. Muirhead, Aspects of multivariate statistical theory (Wiley Series in
84  probability and mathematical statistics).
85 
86  """
87  a = np.asarray(a)
88  if not np.isscalar(d) or (np.floor(d) != d):
89  raise ValueError("d should be a positive integer (dimension)")
90  if np.any(a <= 0.5 * (d - 1)):
91  raise ValueError("condition a (%f) > 0.5 * (d-1) (%f) not met"
92  % (a, 0.5 * (d-1)))
93 
94  res = (d * (d-1) * 0.25) * np.log(np.pi)
95  res += np.sum(loggam([(a - (j - 1.)/2) for j in range(1, d+1)]), axis=0)
96  return res
