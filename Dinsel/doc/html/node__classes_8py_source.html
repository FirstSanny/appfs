37 @decorators.raise_if_nothing_inferred
38 def unpack_infer(stmt, context=None):
39  """recursively generate nodes inferred by the given statement.
40  If the inferred value is a list or a tuple, recurse on the elements
41  """
42  if isinstance(stmt, (List, Tuple)):
43  for elt in stmt.elts:
44  if elt is util.Uninferable:
45  yield elt
46  continue
47  for inferred_elt in unpack_infer(elt, context):
48  yield inferred_elt
49  # Explicit StopIteration to return error information, see comment
50  # in raise_if_nothing_inferred.
51  raise StopIteration(dict(node=stmt, context=context))
52  # if inferred is a final node, return it and stop
53  inferred = next(stmt.infer(context))
54  if inferred is stmt:
55  yield inferred
56  # Explicit StopIteration to return error information, see comment
57  # in raise_if_nothing_inferred.
58  raise StopIteration(dict(node=stmt, context=context))
59  # else, infer recursively, except Uninferable object that should be returned as is
60  for inferred in stmt.infer(context):
61  if inferred is util.Uninferable:
62  yield inferred
63  else:
64  for inf_inf in unpack_infer(inferred, context):
65  yield inf_inf
66  raise StopIteration(dict(node=stmt, context=context))

69 def are_exclusive(stmt1, stmt2, exceptions=None): # pylint: disable=redefined-outer-name
70  """return true if the two given statements are mutually exclusive
71 
72  `exceptions` may be a list of exception names. If specified, discard If
73  branches and check one of the statement is in an exception handler catching
74  one of the given exceptions.
75 
76  algorithm :
77  1) index stmt1's parents
78  2) climb among stmt2's parents until we find a common parent
79  3) if the common parent is a If or TryExcept statement, look if nodes are
80  in exclusive branches
81  """
82  # index stmt1's parents
83  stmt1_parents = {}
84  children = {}
85  node = stmt1.parent
86  previous = stmt1
87  while node:
88  stmt1_parents[node] = 1
89  children[node] = previous
90  previous = node
91  node = node.parent
92  # climb among stmt2's parents until we find a common parent
93  node = stmt2.parent
94  previous = stmt2
95  while node:
96  if node in stmt1_parents:
97  # if the common parent is a If or TryExcept statement, look if
98  # nodes are in exclusive branches
99  if isinstance(node, If) and exceptions is None:
100  if (node.locate_child(previous)[1]
101  is not node.locate_child(children[node])[1]):
102  return True
103  elif isinstance(node, TryExcept):
104  c2attr, c2node = node.locate_child(previous)
105  c1attr, c1node = node.locate_child(children[node])
106  if c1node is not c2node:
107  first_in_body_caught_by_handlers = (
108  c2attr == 'handlers'
109  and c1attr == 'body'
110  and previous.catch(exceptions))
111  second_in_body_caught_by_handlers = (
112  c2attr == 'body'
113  and c1attr == 'handlers'
114  and children[node].catch(exceptions))
115  first_in_else_other_in_handlers = (
116  c2attr == 'handlers' and c1attr == 'orelse')
117  second_in_else_other_in_handlers = (
118  c2attr == 'orelse' and c1attr == 'handlers')
119  if any((first_in_body_caught_by_handlers,
120  second_in_body_caught_by_handlers,
121  first_in_else_other_in_handlers,
122  second_in_else_other_in_handlers)):
123  return True
124  elif c2attr == 'handlers' and c1attr == 'handlers':
125  return previous is not children[node]
126  return False
127  previous = node
128  node = node.parent
129  return False

137 def _slice_value(index, context=None):
138  """Get the value of the given slice index."""

728  def _filter_stmts(self, stmts, frame, offset):
729  """filter statements to remove ignorable statements.
730 
731  If self is not a frame itself and the name is found in the inner
732  frame locals, statements will be filtered to remove ignorable
733  statements according to self's location
734  """
735  # if offset == -1, my actual frame is not the inner frame but its parent
736  #
737  # class A(B): pass
738  #
739  # we need this to resolve B correctly
740  if offset == -1:
741  myframe = self.frame().parent.frame()
742  else:
743  myframe = self.frame()
744  # If the frame of this node is the same as the statement
745  # of this node, then the node is part of a class or
746  # a function definition and the frame of this node should be the
747  # the upper frame, not the frame of the definition.
748  # For more information why this is important,
749  # see Pylint issue #295.
750  # For example, for 'b', the statement is the same
751  # as the frame / scope:
752  #
753  # def test(b=1):
754  # ...
755 
756  if self.statement() is myframe and myframe.parent:
757  myframe = myframe.parent.frame()
758  mystmt = self.statement()
759  # line filtering if we are in the same frame
760  #
761  # take care node may be missing lineno information (this is the case for
762  # nodes inserted for living objects)
763  if myframe is frame and mystmt.fromlineno is not None:
764  assert mystmt.fromlineno is not None, mystmt
765  mylineno = mystmt.fromlineno + offset
766  else:
767  # disabling lineno filtering
768  mylineno = 0
769  _stmts = []
770  _stmt_parents = []
771  for node in stmts:
772  stmt = node.statement()
773  # line filtering is on and we have reached our location, break
774  if mylineno > 0 and stmt.fromlineno > mylineno:
775  break
776  assert hasattr(node, 'assign_type'), (node, node.scope(),
777  node.scope().locals)
778  assign_type = node.assign_type()
779  if node.has_base(self):
780  break
781 
782  _stmts, done = assign_type._get_filtered_stmts(self, node, _stmts, mystmt)
783  if done:
784  break
785 
786  optional_assign = assign_type.optional_assign
787  if optional_assign and assign_type.parent_of(self):
788  # we are inside a loop, loop var assignment is hiding previous
789  # assignment
790  _stmts = [node]
791  _stmt_parents = [stmt.parent]
792  continue
793 
794  # XXX comment various branches below!!!
795  try:
796  pindex = _stmt_parents.index(stmt.parent)
797  except ValueError:
798  pass
799  else:
800  # we got a parent index, this means the currently visited node
801  # is at the same block level as a previously visited node
802  if _stmts[pindex].assign_type().parent_of(assign_type):
803  # both statements are not at the same block level
804  continue
805  # if currently visited node is following previously considered
806  # assignment and both are not exclusive, we can drop the
807  # previous one. For instance in the following code ::
808  #
809  # if a:
810  # x = 1
811  # else:
812  # x = 2
813  # print x
814  #
815  # we can't remove neither x = 1 nor x = 2 when looking for 'x'
816  # of 'print x'; while in the following ::
817  #
818  # x = 1
819  # x = 2
820  # print x
821  #
822  # we can remove x = 1 when we see x = 2
823  #
824  # moreover, on loop assignment types, assignment won't
825  # necessarily be done if the loop has no iteration, so we don't
826  # want to clear previous assignments if any (hence the test on
827  # optional_assign)
828  if not (optional_assign or are_exclusive(_stmts[pindex], node)):
829  del _stmt_parents[pindex]
830  del _stmts[pindex]
831  if isinstance(node, AssignName):
832  if not optional_assign and stmt.parent is mystmt.parent:
833  _stmts = []
834  _stmt_parents = []
835  elif isinstance(node, DelName):
836  _stmts = []
837  _stmt_parents = []
838  continue
839  if not are_exclusive(self, node):
840  _stmts.append(node)
841  _stmt_parents.append(stmt.parent)
842  return _stmts
