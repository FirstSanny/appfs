139 def filemode(mode):
140  """Convert a file's mode to a string of the form '-rwxrwxrwx'."""
141  perm = []
142  for table in _filemode_table:
143  for bit, char in table:
144  if mode & bit == bit:
145  perm.append(char)
146  break
147  else:
148  perm.append("-")
149  return "".join(perm)
