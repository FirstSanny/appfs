121  # len of one number in exp format: sign + 1|0 + "." +
122  # number of digit for fractional part + 'E' + sign of exponent +

125  # Number of digits for fractional part

211 # Grammar for fortran format:
212 # format : LPAR format_string RPAR
213 # format_string : repeated | simple
214 # repeated : repeat simple
215 # simple : int_fmt | exp_fmt
216 # int_fmt : INT_ID width
217 # exp_fmt : simple_exp_fmt
218 # simple_exp_fmt : EXP_ID width DOT significand
219 # extended_exp_fmt : EXP_ID width DOT significand EXP_ID ndigits
220 # repeat : INT
221 # width : INT
222 # significand : INT
223 # ndigits : INT
