99 def argstoarray(*args):
100  """
101  Constructs a 2D array from a group of sequences.
102 
103  Sequences are filled with missing values to match the length of the longest
104  sequence.
105 
106  Parameters
107  ----------
108  args : sequences
109  Group of sequences.
110 
111  Returns
112  -------
113  argstoarray : MaskedArray
114  A ( `m` x `n` ) masked array, where `m` is the number of arguments and
115  `n` the length of the longest argument.
116 
117  Notes
118  -----
119  `numpy.ma.row_stack` has identical behavior, but is called with a sequence
120  of sequences.
121 
122  """
123  if len(args) == 1 and not isinstance(args[0], ndarray):
124  output = ma.asarray(args[0])
125  if output.ndim != 2:
126  raise ValueError("The input should be 2D")
127  else:
128  n = len(args)
129  m = max([len(k) for k in args])
130  output = ma.array(np.empty((n,m), dtype=float), mask=True)
131  for (k,v) in enumerate(args):
132  output[k,:len(v)] = v
133 
134  output[np.logical_not(np.isfinite(output._data))] = masked
135  return output

170 def count_tied_groups(x, use_missing=False):
171  """
172  Counts the number of tied values.
173 
174  Parameters
175  ----------
176  x : sequence
177  Sequence of data on which to counts the ties
178  use_missing : bool, optional
179  Whether to consider missing values as tied.
180 
181  Returns
182  -------
183  count_tied_groups : dict
184  Returns a dictionary (nb of ties: nb of groups).
185 
186  Examples
187  --------
188  >>> from scipy.stats import mstats
189  >>> z = [0, 0, 0, 2, 2, 2, 3, 3, 4, 5, 6]
190  >>> mstats.count_tied_groups(z)
191  {2: 1, 3: 2}
192 
193  In the above example, the ties were 0 (3x), 2 (3x) and 3 (2x).
194 
195  >>> z = np.ma.array([0, 0, 1, 2, 2, 2, 3, 3, 4, 5, 6])
196  >>> mstats.count_tied_groups(z)
197  {2: 2, 3: 1}
198  >>> z[[1,-1]] = np.ma.masked
199  >>> mstats.count_tied_groups(z, use_missing=True)
200  {2: 2, 3: 1}
201

204  # We need the copy as find_repeats will overwrite the initial data
205  data = ma.compressed(x).copy()
206  (ties, counts) = find_repeats(data)
207  nties = {}
208  if len(ties):
209  nties = dict(zip(np.unique(counts), itertools.repeat(1)))
210  nties.update(dict(zip(*find_repeats(counts))))
211 
212  if nmasked and use_missing:
213  try:
214  nties[nmasked] += 1
215  except KeyError:
216  nties[nmasked] = 1

221 def rankdata(data, axis=None, use_missing=False):
222  """Returns the rank (also known as order statistics) of each data point
223  along the given axis.
224 
225  If some values are tied, their rank is averaged.
226  If some values are masked, their rank is set to 0 if use_missing is False,
227  or set to the average rank of the unmasked values if use_missing is True.
228 
229  Parameters
230  ----------
231  data : sequence
232  Input data. The data is transformed to a masked array
233  axis : {None,int}, optional
234  Axis along which to perform the ranking.
235  If None, the array is first flattened. An exception is raised if
236  the axis is specified for arrays with a dimension larger than 2
237  use_missing : bool, optional
238  Whether the masked values have a rank of 0 (False) or equal to the
239  average rank of the unmasked values (True).
240 
241  """
242  def _rank1d(data, use_missing=False):
243  n = data.count()
244  rk = np.empty(data.size, dtype=float)
245  idx = data.argsort()
246  rk[idx[:n]] = np.arange(1,n+1)
247 
248  if use_missing:
249  rk[idx[n:]] = (n+1)/2.
250  else:
251  rk[idx[n:]] = 0
252 
253  repeats = find_repeats(data.copy())
254  for r in repeats[0]:
255  condition = (data == r).filled(False)
256  rk[condition] = rk[condition].mean()
257  return rk
258 
259  data = ma.array(data, copy=False)
260  if axis is None:
261  if data.ndim > 1:
262  return _rank1d(data.ravel(), use_missing).reshape(data.shape)
263  else:
264  return _rank1d(data, use_missing)
265  else:
266  return ma.apply_along_axis(_rank1d,axis,data,use_missing).view(ndarray)

348 def pearsonr(x,y):
349  """
350  Calculates a Pearson correlation coefficient and the p-value for testing
351  non-correlation.
352 
353  The Pearson correlation coefficient measures the linear relationship
354  between two datasets. Strictly speaking, Pearson's correlation requires
355  that each dataset be normally distributed. Like other correlation
356  coefficients, this one varies between -1 and +1 with 0 implying no
357  correlation. Correlations of -1 or +1 imply an exact linear
358  relationship. Positive correlations imply that as `x` increases, so does
359  `y`. Negative correlations imply that as `x` increases, `y` decreases.
360 
361  The p-value roughly indicates the probability of an uncorrelated system
362  producing datasets that have a Pearson correlation at least as extreme
363  as the one computed from these datasets. The p-values are not entirely
364  reliable but are probably reasonable for datasets larger than 500 or so.
365 
366  Parameters
367  ----------
368  x : 1-D array_like
369  Input
370  y : 1-D array_like
371  Input
372 
373  Returns
374  -------
375  pearsonr : float
376  Pearson's correlation coefficient, 2-tailed p-value.
377 
378  References
379  ----------
380  http://www.statsoft.com/textbook/glosp.html#Pearson%20Correlation
381 
382  """
383  (x, y, n) = _chk_size(x, y)
384  (x, y) = (x.ravel(), y.ravel())
385  # Get the common mask and the total nb of unmasked elements
386  m = ma.mask_or(ma.getmask(x), ma.getmask(y))
387  n -= m.sum()
388  df = n-2
389  if df < 0:
390  return (masked, masked)
391 
392  (mx, my) = (x.mean(), y.mean())
393  (xm, ym) = (x-mx, y-my)
394 
395  r_num = ma.add.reduce(xm*ym)
396  r_den = ma.sqrt(ma.dot(xm,xm) * ma.dot(ym,ym))
397  r = r_num / r_den
398  # Presumably, if r > 1, then it is only some small artifact of floating
399  # point arithmetic.
400  r = min(r, 1.0)
401  r = max(r, -1.0)
402  df = n - 2
403 
404  if r is masked or abs(r) == 1.0:
405  prob = 0.
406  else:
407  t_squared = (df / ((1.0 - r) * (1.0 + r))) * r * r
408  prob = _betai(0.5*df, 0.5, df/(df + t_squared))
409 
410  return r, prob

416 def spearmanr(x, y, use_ties=True):
417  """
418  Calculates a Spearman rank-order correlation coefficient and the p-value
419  to test for non-correlation.
420 
421  The Spearman correlation is a nonparametric measure of the linear
422  relationship between two datasets. Unlike the Pearson correlation, the
423  Spearman correlation does not assume that both datasets are normally
424  distributed. Like other correlation coefficients, this one varies
425  between -1 and +1 with 0 implying no correlation. Correlations of -1 or
426  +1 imply an exact linear relationship. Positive correlations imply that
427  as `x` increases, so does `y`. Negative correlations imply that as `x`
428  increases, `y` decreases.
429 
430  Missing values are discarded pair-wise: if a value is missing in `x`, the
431  corresponding value in `y` is masked.
432 
433  The p-value roughly indicates the probability of an uncorrelated system
434  producing datasets that have a Spearman correlation at least as extreme
435  as the one computed from these datasets. The p-values are not entirely
436  reliable but are probably reasonable for datasets larger than 500 or so.
437 
438  Parameters
439  ----------
440  x : array_like
441  The length of `x` must be > 2.
442  y : array_like
443  The length of `y` must be > 2.
444  use_ties : bool, optional
445  Whether the correction for ties should be computed.
446 
447  Returns
448  -------
449  correlation : float
450  Spearman correlation coefficient
451  pvalue : float
452  2-tailed p-value.
453 
454  References
455  ----------
456  [CRCProbStat2000] section 14.7
457 
458  """
459  (x, y, n) = _chk_size(x, y)
460  (x, y) = (x.ravel(), y.ravel())
461 
462  m = ma.mask_or(ma.getmask(x), ma.getmask(y))
463  n -= m.sum()
464  if m is not nomask:
465  x = ma.array(x, mask=m, copy=True)
466  y = ma.array(y, mask=m, copy=True)
467  df = n-2
468  if df < 0:
469  raise ValueError("The input must have at least 3 entries!")
470 
471  # Gets the ranks and rank differences
472  rankx = rankdata(x)
473  ranky = rankdata(y)
474  dsq = np.add.reduce((rankx-ranky)**2)
475  # Tie correction
476  if use_ties:
477  xties = count_tied_groups(x)
478  yties = count_tied_groups(y)
479  corr_x = np.sum(v*k*(k**2-1) for (k,v) in iteritems(xties))/12.
480  corr_y = np.sum(v*k*(k**2-1) for (k,v) in iteritems(yties))/12.
481  else:
482  corr_x = corr_y = 0
483 
484  denom = n*(n**2 - 1)/6.
485  if corr_x != 0 or corr_y != 0:
486  rho = denom - dsq - corr_x - corr_y
487  rho /= ma.sqrt((denom-2*corr_x)*(denom-2*corr_y))
488  else:
489  rho = 1. - dsq/denom
490 
491  t = ma.sqrt(ma.divide(df,(rho+1.0)*(1.0-rho))) * rho
492  if t is masked:
493  prob = 0.
494  else:
495  prob = _betai(0.5*df, 0.5, df/(df + t * t))
496 
497  return SpearmanrResult(rho, prob)

503 def kendalltau(x, y, use_ties=True, use_missing=False):
504  """
505  Computes Kendall's rank correlation tau on two variables *x* and *y*.
506 
507  Parameters
508  ----------
509  x : sequence
510  First data list (for example, time).
511  y : sequence
512  Second data list.
513  use_ties : {True, False}, optional
514  Whether ties correction should be performed.
515  use_missing : {False, True}, optional
516  Whether missing data should be allocated a rank of 0 (False) or the
517  average rank (True)
518 
519  Returns
520  -------
521  correlation : float
522  Kendall tau
523  pvalue : float
524  Approximate 2-side p-value.
525 
526  """
527  (x, y, n) = _chk_size(x, y)
528  (x, y) = (x.flatten(), y.flatten())
529  m = ma.mask_or(ma.getmask(x), ma.getmask(y))
530  if m is not nomask:
531  x = ma.array(x, mask=m, copy=True)
532  y = ma.array(y, mask=m, copy=True)
533  n -= m.sum()
534 
535  if n < 2:
536  return KendalltauResult(np.nan, np.nan)
537 
538  rx = ma.masked_equal(rankdata(x, use_missing=use_missing), 0)
539  ry = ma.masked_equal(rankdata(y, use_missing=use_missing), 0)
540  idx = rx.argsort()
541  (rx, ry) = (rx[idx], ry[idx])
542  C = np.sum([((ry[i+1:] > ry[i]) * (rx[i+1:] > rx[i])).filled(0).sum()
543  for i in range(len(ry)-1)], dtype=float)
544  D = np.sum([((ry[i+1:] < ry[i])*(rx[i+1:] > rx[i])).filled(0).sum()
545  for i in range(len(ry)-1)], dtype=float)
546  if use_ties:
547  xties = count_tied_groups(x)
548  yties = count_tied_groups(y)
549  corr_x = np.sum([v*k*(k-1) for (k,v) in iteritems(xties)], dtype=float)
550  corr_y = np.sum([v*k*(k-1) for (k,v) in iteritems(yties)], dtype=float)
551  denom = ma.sqrt((n*(n-1)-corr_x)/2. * (n*(n-1)-corr_y)/2.)
552  else:
553  denom = n*(n-1)/2.
554  tau = (C-D) / denom
555 
556  var_s = n*(n-1)*(2*n+5)
557  if use_ties:
558  var_s -= np.sum(v*k*(k-1)*(2*k+5)*1. for (k,v) in iteritems(xties))
559  var_s -= np.sum(v*k*(k-1)*(2*k+5)*1. for (k,v) in iteritems(yties))
560  v1 = np.sum([v*k*(k-1) for (k, v) in iteritems(xties)], dtype=float) *\
561  np.sum([v*k*(k-1) for (k, v) in iteritems(yties)], dtype=float)
562  v1 /= 2.*n*(n-1)
563  if n > 2:
564  v2 = np.sum([v*k*(k-1)*(k-2) for (k,v) in iteritems(xties)],
565  dtype=float) * \
566  np.sum([v*k*(k-1)*(k-2) for (k,v) in iteritems(yties)],
567  dtype=float)
568  v2 /= 9.*n*(n-1)*(n-2)
569  else:
570  v2 = 0
571  else:
572  v1 = v2 = 0
573 
574  var_s /= 18.
575  var_s += (v1 + v2)
576  z = (C-D)/np.sqrt(var_s)
577  prob = special.erfc(abs(z)/np.sqrt(2))
578  return KendalltauResult(tau, prob)

581 def kendalltau_seasonal(x):
582  """
583  Computes a multivariate Kendall's rank correlation tau, for seasonal data.
584 
585  Parameters
586  ----------
587  x : 2-D ndarray
588  Array of seasonal data, with seasons in columns.
589 
590  """
591  x = ma.array(x, subok=True, copy=False, ndmin=2)
592  (n,m) = x.shape
593  n_p = x.count(0)
594 
595  S_szn = np.sum(msign(x[i:]-x[i]).sum(0) for i in range(n))
596  S_tot = S_szn.sum()
597 
598  n_tot = x.count()
599  ties = count_tied_groups(x.compressed())
600  corr_ties = np.sum(v*k*(k-1) for (k,v) in iteritems(ties))
601  denom_tot = ma.sqrt(1.*n_tot*(n_tot-1)*(n_tot*(n_tot-1)-corr_ties))/2.
602 
603  R = rankdata(x, axis=0, use_missing=True)
604  K = ma.empty((m,m), dtype=int)
605  covmat = ma.empty((m,m), dtype=float)
606  denom_szn = ma.empty(m, dtype=float)
607  for j in range(m):
608  ties_j = count_tied_groups(x[:,j].compressed())
609  corr_j = np.sum(v*k*(k-1) for (k,v) in iteritems(ties_j))
610  cmb = n_p[j]*(n_p[j]-1)
611  for k in range(j,m,1):
612  K[j,k] = np.sum(msign((x[i:,j]-x[i,j])*(x[i:,k]-x[i,k])).sum()
613  for i in range(n))
614  covmat[j,k] = (K[j,k] + 4*(R[:,j]*R[:,k]).sum() -
615  n*(n_p[j]+1)*(n_p[k]+1))/3.
616  K[k,j] = K[j,k]
617  covmat[k,j] = covmat[j,k]
618 
619  denom_szn[j] = ma.sqrt(cmb*(cmb-corr_j)) / 2.
620 
621  var_szn = covmat.diagonal()
622 
623  z_szn = msign(S_szn) * (abs(S_szn)-1) / ma.sqrt(var_szn)
624  z_tot_ind = msign(S_tot) * (abs(S_tot)-1) / ma.sqrt(var_szn.sum())
625  z_tot_dep = msign(S_tot) * (abs(S_tot)-1) / ma.sqrt(covmat.sum())
626 
627  prob_szn = special.erfc(abs(z_szn)/np.sqrt(2))
628  prob_tot_ind = special.erfc(abs(z_tot_ind)/np.sqrt(2))
629  prob_tot_dep = special.erfc(abs(z_tot_dep)/np.sqrt(2))
630 
631  chi2_tot = (z_szn*z_szn).sum()
632  chi2_trd = m * z_szn.mean()**2
633  output = {'seasonal tau': S_szn/denom_szn,
634  'global tau': S_tot/denom_tot,

2014

2296 def mquantiles(a, prob=list([.25,.5,.75]), alphap=.4, betap=.4, axis=None,

2302  where ``x[j]`` is the j-th order statistic, and gamma is a function of
2303  ``j = floor(n*p + m)``, ``m = alphap + p*(1 - alphap - betap)`` and
2304  ``g = n*p + m - j``.
2305 
2306  Reinterpreting the above equations to compare to **R** lead to the
2307  equation: ``p(k) = (k - alphap)/(n + 1 - alphap - betap)``
2308

2310  - (0,1) : ``p(k) = k/n`` : linear interpolation of cdf
2311  (**R** type 4)

2313  (**R** type 5)
2314  - (0,0) : ``p(k) = k/(n+1)`` :
2315  (**R** type 6)
2316  - (1,1) : ``p(k) = (k-1)/(n-1)``: p(k) = mode[F(x[k])].
2317  (**R** type 7, **R** default)

2319  The resulting quantile estimates are approximately median-unbiased
2320  regardless of the distribution of x.
2321  (**R** type 8)
2322  - (3/8,3/8): ``p(k) = (k-3/8)/(n+1/4)``: Blom.
2323  The resulting quantile estimates are approximately unbiased
2324  if x is normally distributed
