70 def _is_owner_ignored(owner, name, ignored_classes, ignored_modules):
71  """Check if the given owner should be ignored
72 
73  This will verify if the owner's module is in *ignored_modules*
74  or the owner's module fully qualified name is in *ignored_modules*
75  or if the *ignored_modules* contains a pattern which catches
76  the fully qualified name of the module.
77 
78  Also, similar checks are done for the owner itself, if its name
79  matches any name from the *ignored_classes* or if its qualified
80  name can be found in *ignored_classes*.
81  """
82  ignored_modules = set(ignored_modules)
83  module_name = owner.root().name
84  module_qname = owner.root().qname()
85  if any(module_name in ignored_modules or
86  module_qname in ignored_modules or
87  fnmatch.fnmatch(module_qname, ignore) for ignore in ignored_modules):
88  return True
89 
90  ignored_classes = set(ignored_classes)
91  if hasattr(owner, 'qname'):
92  qname = owner.qname()
93  else:
94  qname = ''
95  return any(name == ignore or qname == ignore for ignore in ignored_classes)

121 def _string_distance(seq1, seq2):
122  seq2_length = len(seq2)
123 
124  row = list(range(1, seq2_length + 1)) + [0]
125  for seq1_index, seq1_char in enumerate(seq1):
126  last_row = row
127  row = [0] * seq2_length + [seq1_index + 1]
128 
129  for seq2_index, seq2_char in enumerate(seq2):
130  row[seq2_index] = min(
131  last_row[seq2_index] + 1,
132  row[seq2_index - 1] + 1,
133  last_row[seq2_index - 1] + (seq1_char != seq2_char)
134 
135  )
136 
137  return row[seq2_length - 1]

140 def _similar_names(owner, attrname, distance_threshold, max_choices):
141  """Given an owner and a name, try to find similar names
142 
143  The similar names are searched given a distance metric and only
144  a given number of choices will be returned.
145  """
146  possible_names = []
147  names = _node_names(owner)
148 
149  for name in names:
150  if name == attrname:
151  continue
152 
153  distance = _string_distance(attrname, name)
154  if distance <= distance_threshold:
155  possible_names.append((name, distance))
156 
157  # Now get back the values with a minimum, up to the given
158  # limit or choices.
159  picked = [name for (name, _) in
160  heapq.nsmallest(max_choices, possible_names,
161  key=operator.itemgetter(1))]
162  return sorted(picked)

271 def _emit_no_member(node, owner, owner_name, ignored_mixins):
272  """Try to see if no-member should be emitted for the given owner.
273 
274  The following cases are ignored:
275 
276  * the owner is a function and it has decorators.
277  * the owner is an instance and it has __getattr__, __getattribute__ implemented
278  * the module is explicitly ignored from no-member checks
279  * the owner is a class and the name can be found in its metaclass.
280  * The access node is protected by an except handler, which handles
281  AttributeError, Exception or bare except.
282  """
283  if node_ignores_exception(node, AttributeError):
284  return False
285  # skip None anyway
286  if isinstance(owner, astroid.Const) and owner.value is None:
287  return False
288  if is_super(owner) or getattr(owner, 'type', None) == 'metaclass':
289  return False
290  if ignored_mixins and owner_name[-5:].lower() == 'mixin':
291  return False
292  if isinstance(owner, astroid.FunctionDef) and owner.decorators:
293  return False
294  if isinstance(owner, (astroid.Instance, astroid.ClassDef)):
295  if owner.has_dynamic_getattr() or not has_known_bases(owner):
296  return False
297  if isinstance(owner, objects.Super):
298  # Verify if we are dealing with an invalid Super object.
299  # If it is invalid, then there's no point in checking that
300  # it has the required attribute. Also, don't fail if the
301  # MRO is invalid.
302  try:
303  owner.super_mro()
304  except (exceptions.MroError, exceptions.SuperError):
305  return False
306  if not all(map(has_known_bases, owner.type.mro())):
307  return False
308  return True

311 def _determine_callable(callable_obj):
312  # Ordering is important, since BoundMethod is a subclass of UnboundMethod,
313  # and Function inherits Lambda.
314  if isinstance(callable_obj, astroid.BoundMethod):
315  # Bound methods have an extra implicit 'self' argument.
316  return callable_obj, 1, callable_obj.type
317  elif isinstance(callable_obj, astroid.UnboundMethod):
318  return callable_obj, 0, 'unbound method'
319  elif isinstance(callable_obj, astroid.FunctionDef):
320  return callable_obj, 0, callable_obj.type
321  elif isinstance(callable_obj, astroid.Lambda):
322  return callable_obj, 0, 'lambda'
323  elif isinstance(callable_obj, astroid.ClassDef):
324  # Class instantiation, lookup __new__ instead.
325  # If we only find object.__new__, we can safely check __init__
326  # instead. If __new__ belongs to builtins, then we look
327  # again for __init__ in the locals, since we won't have
328  # argument information for the builtin __new__ function.
329  try:
330  # Use the last definition of __new__.
331  new = callable_obj.local_attr('__new__')[-1]
332  except exceptions.NotFoundError:
333  new = None
334 
335  from_object = new and new.parent.scope().name == 'object'
336  from_builtins = new and new.root().name in sys.builtin_module_names
337 
338  if not new or from_object or from_builtins:
339  try:
340  # Use the last definition of __init__.
341  callable_obj = callable_obj.local_attr('__init__')[-1]
342  except exceptions.NotFoundError:
343  # do nothing, covered by no-init.
344  raise ValueError
345  else:
346  callable_obj = new
347 
348  if not isinstance(callable_obj, astroid.FunctionDef):
349  raise ValueError
350  # both have an extra implicit 'cls'/'self' argument.
351  return callable_obj, 1, 'constructor'
352  else:
353  raise ValueError

400 def _no_context_variadic(node, variadic_name, variadic_type, variadics):
401  """Verify if the given call node has variadic nodes without context
402 
403  This is a workaround for handling cases of nested call functions
404  which don't have the specific call context at hand.
405  Variadic arguments (variable positional arguments and variable
406  keyword arguments) are inferred, inherently wrong, by astroid
407  as a Tuple, respectively a Dict with empty elements.
408  This can lead pylint to believe that a function call receives
409  too few arguments.
410  """
411  statement = node.statement()
412  for name in statement.nodes_of_class(astroid.Name):
413  if name.name != variadic_name:
414  continue
415 
416  inferred = safe_infer(name)
417  if isinstance(inferred, (astroid.List, astroid.Tuple)):
418  length = len(inferred.elts)
419  elif isinstance(inferred, astroid.Dict):
420  length = len(inferred.items)
421  else:
422  continue
423 
424  inferred_statement = inferred.statement()
425  if not length and isinstance(inferred_statement, astroid.FunctionDef):
426  is_in_starred_context = _has_parent_of_type(node, variadic_type, statement)
427  used_as_starred_argument = _is_name_used_as_variadic(name, variadics)
428  if is_in_starred_context or used_as_starred_argument:
429  return True
430  return False

446 def _infer_from_metaclass_constructor(cls, func):
447  """Try to infer what the given *func* constructor is building
448 
449  :param astroid.FunctionDef func:
450  A metaclass constructor. Metaclass definitions can be
451  functions, which should accept three arguments, the name of
452  the class, the bases of the class and the attributes.
453  The function could return anything, but usually it should
454  be a proper metaclass.
455  :param astroid.ClassDef cls:
456  The class for which the *func* parameter should generate
457  a metaclass.
458  :returns:
459  The class generated by the function or None,
460  if we couldn't infer it.
461  :rtype: astroid.ClassDef
462  """
463  context = astroid.context.InferenceContext()
464 
465  class_bases = astroid.List()
466  class_bases.postinit(elts=cls.bases)
467 
468  attrs = astroid.Dict()
469  local_names = [(name, values[-1]) for name, values in cls.locals.items()]
470  attrs.postinit(local_names)
471 
472  builder_args = astroid.Tuple()
473  builder_args.postinit([cls.name, class_bases, attrs])
474 
475  context.callcontext = astroid.context.CallContext(builder_args)
476  try:
477  inferred = next(func.infer_call_result(func, context), None)
478  except astroid.InferenceError:
479  return None
480  return inferred or None

786  @check_messages(*(list(MSGS.keys())))
787  def visit_call(self, node):
788  """check that called functions/methods are inferred to callable objects,
789  and that the arguments passed to the function match the parameters in
790  the inferred function's definition
791  """
792  # Build the set of keyword arguments, checking for duplicate keywords,
793  # and count the positional arguments.
794  call_site = astroid.arguments.CallSite.from_call(node)
795  num_positional_args = len(call_site.positional_arguments)
796  keyword_args = list(call_site.keyword_arguments.keys())
797 
798  # Determine if we don't have a context for our call and we use variadics.
799  if isinstance(node.scope(), astroid.FunctionDef):
800  has_no_context_positional_variadic = _no_context_variadic_positional(node)
801  has_no_context_keywords_variadic = _no_context_variadic_keywords(node)
802  else:
803  has_no_context_positional_variadic = has_no_context_keywords_variadic = False
804

806  # only function, generator and object defining __call__ are allowed
807  if called and not called.callable():

810  pass

818  called, implicit_args, callable_name = _determine_callable(called)
819  except ValueError:

821  # those errors are handled by different warnings.
822  return

827  return

831  # make sense of the function call in this case, so just return.

835  for keyword in call_site.duplicated_keywords:

844  num_mandatory_parameters = len(called.args.args) - len(called.args.defaults)

908  # Duplicate definition of function parameter.
909  self.add_message('redundant-keyword-arg', node=node,

916  else:
917  # Unexpected keyword argument.

944
