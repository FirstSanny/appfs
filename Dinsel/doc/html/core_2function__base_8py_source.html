25 def linspace(start, stop, num=50, endpoint=True, retstep=False, dtype=None):
26  """
27  Return evenly spaced numbers over a specified interval.
28 
29  Returns `num` evenly spaced samples, calculated over the
30  interval [`start`, `stop`].
31 
32  The endpoint of the interval can optionally be excluded.
33 
34  Parameters
35  ----------
36  start : scalar
37  The starting value of the sequence.
38  stop : scalar
39  The end value of the sequence, unless `endpoint` is set to False.
40  In that case, the sequence consists of all but the last of ``num + 1``
41  evenly spaced samples, so that `stop` is excluded. Note that the step
42  size changes when `endpoint` is False.
43  num : int, optional
44  Number of samples to generate. Default is 50. Must be non-negative.
45  endpoint : bool, optional
46  If True, `stop` is the last sample. Otherwise, it is not included.
47  Default is True.
48  retstep : bool, optional
49  If True, return (`samples`, `step`), where `step` is the spacing
50  between samples.
51  dtype : dtype, optional
52  The type of the output array. If `dtype` is not given, infer the data
53  type from the other input arguments.
54 
55  .. versionadded:: 1.9.0
56 
57  Returns
58  -------
59  samples : ndarray
60  There are `num` equally spaced samples in the closed interval
61  ``[start, stop]`` or the half-open interval ``[start, stop)``
62  (depending on whether `endpoint` is True or False).
63  step : float, optional
64  Only returned if `retstep` is True
65 
66  Size of spacing between samples.
67 
68 
69  See Also
70  --------
71  arange : Similar to `linspace`, but uses a step size (instead of the
72  number of samples).
73  logspace : Samples uniformly distributed in log space.
74 
75  Examples
76  --------
77  >>> np.linspace(2.0, 3.0, num=5)
78  array([ 2. , 2.25, 2.5 , 2.75, 3. ])
79  >>> np.linspace(2.0, 3.0, num=5, endpoint=False)
80  array([ 2. , 2.2, 2.4, 2.6, 2.8])
81  >>> np.linspace(2.0, 3.0, num=5, retstep=True)
82  (array([ 2. , 2.25, 2.5 , 2.75, 3. ]), 0.25)
83 
84  Graphical illustration:
85 
86  >>> import matplotlib.pyplot as plt
87  >>> N = 8
88  >>> y = np.zeros(N)
89  >>> x1 = np.linspace(0, 10, N, endpoint=True)
90  >>> x2 = np.linspace(0, 10, N, endpoint=False)
91  >>> plt.plot(x1, y, 'o')
92  [<matplotlib.lines.Line2D object at 0x...>]
93  >>> plt.plot(x2, y + 0.5, 'o')
94  [<matplotlib.lines.Line2D object at 0x...>]
95  >>> plt.ylim([-0.5, 1])
96  (-0.5, 1)
97  >>> plt.show()
98 
99  """
100  # 2016-02-25, 1.12
101  num = _index_deprecate(num)
102  if num < 0:
103  raise ValueError("Number of samples, %s, must be non-negative." % num)
104  div = (num - 1) if endpoint else num
105 
106  # Convert float/complex array scalars to float, gh-3504
107  # and make sure one can use variables that have an __array_interface__, gh-6634
108  start = asanyarray(start) * 1.0
109  stop = asanyarray(stop) * 1.0
110 
111  dt = result_type(start, stop, float(num))
112  if dtype is None:
113  dtype = dt
114 
115  y = _nx.arange(0, num, dtype=dt)
116 
117  delta = stop - start
118  if num > 1:
119  step = delta / div
120  if step == 0:
121  # Special handling for denormal numbers, gh-5437
122  y /= div
123  y = y * delta
124  else:
125  # One might be tempted to use faster, in-place multiplication here,
126  # but this prevents step from overriding what class is produced,
127  # and thus prevents, e.g., use of Quantities; see gh-7142.
128  y = y * step
129  else:
130  # 0 and 1 item long sequences have an undefined step
131  step = NaN
132  # Multiply with delta to allow possible override of output class.
133  y = y * delta
134 
135  y += start
136 
137  if endpoint and num > 1:
138  y[-1] = stop
139 
140  if retstep:
141  return y.astype(dtype, copy=False), step
142  else:
143  return y.astype(dtype, copy=False)

146 def logspace(start, stop, num=50, endpoint=True, base=10.0, dtype=None):
147  """
148  Return numbers spaced evenly on a log scale.
149 
150  In linear space, the sequence starts at ``base ** start``
151  (`base` to the power of `start`) and ends with ``base ** stop``
152  (see `endpoint` below).
153 
154  Parameters
155  ----------
156  start : float
157  ``base ** start`` is the starting value of the sequence.
158  stop : float
159  ``base ** stop`` is the final value of the sequence, unless `endpoint`
160  is False. In that case, ``num + 1`` values are spaced over the
161  interval in log-space, of which all but the last (a sequence of
162  length `num`) are returned.
163  num : integer, optional
164  Number of samples to generate. Default is 50.
165  endpoint : boolean, optional
166  If true, `stop` is the last sample. Otherwise, it is not included.
167  Default is True.
168  base : float, optional
169  The base of the log space. The step size between the elements in
170  ``ln(samples) / ln(base)`` (or ``log_base(samples)``) is uniform.
171  Default is 10.0.
172  dtype : dtype
173  The type of the output array. If `dtype` is not given, infer the data
174  type from the other input arguments.
175 
176  Returns
177  -------
178  samples : ndarray
179  `num` samples, equally spaced on a log scale.
180 
181  See Also
182  --------
183  arange : Similar to linspace, with the step size specified instead of the
184  number of samples. Note that, when used with a float endpoint, the
185  endpoint may or may not be included.
186  linspace : Similar to logspace, but with the samples uniformly distributed
187  in linear space, instead of log space.
188  geomspace : Similar to logspace, but with endpoints specified directly.
189 
190  Notes
191  -----
192  Logspace is equivalent to the code
193 
194  >>> y = np.linspace(start, stop, num=num, endpoint=endpoint)
195  ... # doctest: +SKIP
196  >>> power(base, y).astype(dtype)
197  ... # doctest: +SKIP
198 
199  Examples
200  --------
201  >>> np.logspace(2.0, 3.0, num=4)
202  array([ 100. , 215.443469 , 464.15888336, 1000. ])
203  >>> np.logspace(2.0, 3.0, num=4, endpoint=False)
204  array([ 100. , 177.827941 , 316.22776602, 562.34132519])
205  >>> np.logspace(2.0, 3.0, num=4, base=2.0)
206  array([ 4. , 5.0396842 , 6.34960421, 8. ])
207 
208  Graphical illustration:
209 
210  >>> import matplotlib.pyplot as plt
211  >>> N = 10
212  >>> x1 = np.logspace(0.1, 1, N, endpoint=True)
213  >>> x2 = np.logspace(0.1, 1, N, endpoint=False)
214  >>> y = np.zeros(N)
215  >>> plt.plot(x1, y, 'o')
216  [<matplotlib.lines.Line2D object at 0x...>]
217  >>> plt.plot(x2, y + 0.5, 'o')
218  [<matplotlib.lines.Line2D object at 0x...>]
219  >>> plt.ylim([-0.5, 1])
220  (-0.5, 1)
221  >>> plt.show()
222 
223  """
224  y = linspace(start, stop, num=num, endpoint=endpoint)
225  if dtype is None:
226  return _nx.power(base, y)
227  return _nx.power(base, y).astype(dtype)

230 def geomspace(start, stop, num=50, endpoint=True, dtype=None):
231  """
232  Return numbers spaced evenly on a log scale (a geometric progression).
233 
234  This is similar to `logspace`, but with endpoints specified directly.
235  Each output sample is a constant multiple of the previous.
236 
237  Parameters
238  ----------
239  start : scalar
240  The starting value of the sequence.
241  stop : scalar
242  The final value of the sequence, unless `endpoint` is False.
243  In that case, ``num + 1`` values are spaced over the
244  interval in log-space, of which all but the last (a sequence of
245  length `num`) are returned.
246  num : integer, optional
247  Number of samples to generate. Default is 50.
248  endpoint : boolean, optional
249  If true, `stop` is the last sample. Otherwise, it is not included.
250  Default is True.
251  dtype : dtype
252  The type of the output array. If `dtype` is not given, infer the data
253  type from the other input arguments.
254 
255  Returns
256  -------
257  samples : ndarray
258  `num` samples, equally spaced on a log scale.
259 
260  See Also
261  --------
262  logspace : Similar to geomspace, but with endpoints specified using log
263  and base.
264  linspace : Similar to geomspace, but with arithmetic instead of geometric
265  progression.
266  arange : Similar to linspace, with the step size specified instead of the
267  number of samples.
268 
269  Notes
270  -----
271  If the inputs or dtype are complex, the output will follow a logarithmic
272  spiral in the complex plane. (There are an infinite number of spirals
273  passing through two points; the output will follow the shortest such path.)
274 
275  Examples
276  --------
277  >>> np.geomspace(1, 1000, num=4)
278  array([ 1., 10., 100., 1000.])
279  >>> np.geomspace(1, 1000, num=3, endpoint=False)
280  array([ 1., 10., 100.])
281  >>> np.geomspace(1, 1000, num=4, endpoint=False)
282  array([ 1. , 5.62341325, 31.6227766 , 177.827941 ])
283  >>> np.geomspace(1, 256, num=9)
284  array([ 1., 2., 4., 8., 16., 32., 64., 128., 256.])
285 
286  Note that the above may not produce exact integers:
287 
288  >>> np.geomspace(1, 256, num=9, dtype=int)
289  array([ 1, 2, 4, 7, 16, 32, 63, 127, 256])
290  >>> np.around(np.geomspace(1, 256, num=9)).astype(int)
291  array([ 1, 2, 4, 8, 16, 32, 64, 128, 256])
292 
293  Negative, decreasing, and complex inputs are allowed:
294 
295  >>> geomspace(1000, 1, num=4)
296  array([ 1000., 100., 10., 1.])
297  >>> geomspace(-1000, -1, num=4)
298  array([-1000., -100., -10., -1.])
299  >>> geomspace(1j, 1000j, num=4) # Straight line
300  array([ 0. +1.j, 0. +10.j, 0. +100.j, 0.+1000.j])
301  >>> geomspace(-1+0j, 1+0j, num=5) # Circle
302  array([-1.00000000+0.j , -0.70710678+0.70710678j,
303  0.00000000+1.j , 0.70710678+0.70710678j,
304  1.00000000+0.j ])
305 
306  Graphical illustration of ``endpoint`` parameter:
307 
308  >>> import matplotlib.pyplot as plt
309  >>> N = 10
310  >>> y = np.zeros(N)
311  >>> plt.semilogx(np.geomspace(1, 1000, N, endpoint=True), y + 1, 'o')
312  >>> plt.semilogx(np.geomspace(1, 1000, N, endpoint=False), y + 2, 'o')
313  >>> plt.axis([0.5, 2000, 0, 3])
314  >>> plt.grid(True, color='0.7', linestyle='-', which='both', axis='both')
315  >>> plt.show()
316 
317  """
318  if start == 0 or stop == 0:
319  raise ValueError('Geometric sequence cannot include zero')
320 
321  dt = result_type(start, stop, float(num))
322  if dtype is None:
323  dtype = dt
324  else:
325  # complex to dtype('complex128'), for instance
326  dtype = _nx.dtype(dtype)
327 
328  # Avoid negligible real or imaginary parts in output by rotating to
329  # positive real, calculating, then undoing rotation
330  out_sign = 1
331  if start.real == stop.real == 0:
332  start, stop = start.imag, stop.imag
333  out_sign = 1j * out_sign
334  if _nx.sign(start) == _nx.sign(stop) == -1:
335  start, stop = -start, -stop
336  out_sign = -out_sign
337 
338  # Promote both arguments to the same dtype in case, for instance, one is
339  # complex and another is negative and log would produce NaN otherwise
340  start = start + (stop - stop)
341  stop = stop + (start - start)
342  if _nx.issubdtype(dtype, complex):
343  start = start + 0j
344  stop = stop + 0j
345 
346  log_start = _nx.log10(start)
347  log_stop = _nx.log10(stop)
348  result = out_sign * logspace(log_start, log_stop, num=num,
349  endpoint=endpoint, base=10.0, dtype=dtype)
350 
351  return result.astype(dtype)
