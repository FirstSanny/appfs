18 class PyTracer(object):
19  """Python implementation of the raw data tracer."""
20 
21  # Because of poor implementations of trace-function-manipulating tools,
22  # the Python trace function must be kept very simple. In particular, there
23  # must be only one function ever set as the trace function, both through
24  # sys.settrace, and as the return value from the trace function. Put
25  # another way, the trace function must always return itself. It cannot
26  # swap in other functions, or return None to avoid tracing a particular
27  # frame.
28  #
29  # The trace manipulator that introduced this restriction is DecoratorTools,
30  # which sets a trace function, and then later restores the pre-existing one
31  # by calling sys.settrace with a function it found in the current frame.
32  #
33  # Systems that use DecoratorTools (or similar trace manipulations) must use
34  # PyTracer to get accurate results. The command-line --timid argument is
35  # used to force the use of this tracer.

68  def _trace(self, frame, event, arg_unused):
69  """The trace function passed to sys.settrace."""
70 
71  if self.stopped:
72  return
73 
74  if self.last_exc_back:
75  if frame == self.last_exc_back:
76  # Someone forgot a return event.
77  if self.trace_arcs and self.cur_file_dict:
78  pair = (self.last_line, -self.last_exc_firstlineno)
79  self.cur_file_dict[pair] = None
80  self.cur_file_dict, self.last_line = self.data_stack.pop()
81  self.last_exc_back = None
82 
83  if event == 'call':
84  # Entering a new function context. Decide if we should trace
85  # in this file.
86  self._activity = True
87  self.data_stack.append((self.cur_file_dict, self.last_line))
88  filename = frame.f_code.co_filename
89  disp = self.should_trace_cache.get(filename)
90  if disp is None:
91  disp = self.should_trace(filename, frame)
92  self.should_trace_cache[filename] = disp
93 
94  self.cur_file_dict = None
95  if disp.trace:
96  tracename = disp.source_filename
97  if tracename not in self.data:
98  self.data[tracename] = {}
99  self.cur_file_dict = self.data[tracename]
100  # The call event is really a "start frame" event, and happens for
101  # function calls and re-entering generators. The f_lasti field is
102  # -1 for calls, and a real offset for generators. Use <0 as the
103  # line number for calls, and the real line number for generators.
104  if getattr(frame, 'f_lasti', -1) < 0:
105  self.last_line = -frame.f_code.co_firstlineno
106  else:
107  self.last_line = frame.f_lineno
108  elif event == 'line':
109  # Record an executed line.
110  if self.cur_file_dict is not None:
111  lineno = frame.f_lineno
112  if self.trace_arcs:
113  self.cur_file_dict[(self.last_line, lineno)] = None
114  else:
115  self.cur_file_dict[lineno] = None
116  self.last_line = lineno
117  elif event == 'return':
118  if self.trace_arcs and self.cur_file_dict:
119  # Record an arc leaving the function, but beware that a
120  # "return" event might just mean yielding from a generator.
121  # Jython seems to have an empty co_code, so just assume return.
122  code = frame.f_code.co_code
123  if (not code) or code[frame.f_lasti] != YIELD_VALUE:
124  first = frame.f_code.co_firstlineno
125  self.cur_file_dict[(self.last_line, -first)] = None
126  # Leaving this function, pop the filename stack.
127  self.cur_file_dict, self.last_line = self.data_stack.pop()
128  elif event == 'exception':
129  self.last_exc_back = frame.f_back
130  self.last_exc_firstlineno = frame.f_code.co_firstlineno
131  return self._trace

133  def start(self):
134  """Start this Tracer.
135 
136  Return a Python function suitable for use with sys.settrace().
137 
138  """
139  self.stopped = False
140  if self.threading:
141  if self.thread is None:
142  self.thread = self.threading.currentThread()
143  else:
144  if self.thread.ident != self.threading.currentThread().ident:
145  # Re-starting from a different thread!? Don't set the trace
146  # function, but we are marked as running again, so maybe it
147  # will be ok?
148  return self._trace
149 
150  sys.settrace(self._trace)
151  return self._trace

153  def stop(self):
154  """Stop this Tracer."""
155  self.stopped = True
156  if self.threading and self.thread.ident != self.threading.currentThread().ident:
157  # Called on a different thread than started us: we can't unhook
158  # ourselves, but we've set the flag that we should stop, so we
159  # won't do any more tracing.
160  return
161 
162  if self.warn:
163  # PyPy clears the trace function before running atexit functions,
164  # so don't warn if we are in atexit on PyPy and the trace function
165  # has changed to None.
166  tf = sys.gettrace()
167  dont_warn = (env.PYPY and env.PYPYVERSION >= (5, 4) and self.in_atexit and tf is None)
168  if (not dont_warn) and tf != self._trace:
169  self.warn(
170  "Trace function changed, measurement is likely wrong: %r" % (tf,),
171  slug="trace-changed",
172  )
