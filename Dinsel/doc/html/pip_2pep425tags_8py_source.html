165 def get_darwin_arches(major, minor, machine):
166  """Return a list of supported arches (including group arches) for
167  the given major, minor and machine architecture of an macOS machine.
168  """
169  arches = []

171  def _supports_arch(major, minor, arch):
172  # Looking at the application support for macOS versions in the chart
173  # provided by https://en.wikipedia.org/wiki/OS_X#Versions it appears
174  # our timeline looks roughly like:
175  #
176  # 10.0 - Introduces ppc support.
177  # 10.4 - Introduces ppc64, i386, and x86_64 support, however the ppc64
178  # and x86_64 support is CLI only, and cannot be used for GUI
179  # applications.
180  # 10.5 - Extends ppc64 and x86_64 support to cover GUI applications.
181  # 10.6 - Drops support for ppc64
182  # 10.7 - Drops support for ppc
183  #
184  # Given that we do not know if we're installing a CLI or a GUI
185  # application, we must be conservative and assume it might be a GUI
186  # application and behave as if ppc64 and x86_64 support did not occur
187  # until 10.5.
188  #
189  # Note: The above information is taken from the "Application support"
190  # column in the chart not the "Processor support" since I believe
191  # that we care about what instruction sets an application can use
192  # not which processors the OS supports.
193  if arch == 'ppc':
194  return (major, minor) <= (10, 5)
195  if arch == 'ppc64':
196  return (major, minor) == (10, 5)
197  if arch == 'i386':
198  return (major, minor) >= (10, 4)
199  if arch == 'x86_64':
200  return (major, minor) >= (10, 5)
201  if arch in groups:
202  for garch in groups[arch]:
203  if _supports_arch(major, minor, garch):
204  return True
205  return False
206 
207  groups = OrderedDict([
208  ("fat", ("i386", "ppc")),
209  ("intel", ("x86_64", "i386")),
210  ("fat64", ("x86_64", "ppc64")),
211  ("fat32", ("x86_64", "i386", "ppc")),
212  ])
213 
214  if _supports_arch(major, minor, machine):
215  arches.append(machine)
216 
217  for garch in groups:
218  if machine in groups[garch] and _supports_arch(major, minor, garch):
219  arches.append(garch)
220 
221  arches.append('universal')
222 
223  return arches

233  :param platform: specify the exact platform you want valid
234  tags for, or None. If None, use the local system platform.
235  :param impl: specify the exact implementation you want valid
236  tags for, or None. If None, use the local interpreter impl.
237  :param abi: specify the exact abi you want valid
238  tags for, or None. If None, use the local interpreter abi.
239  """

242  # Versions must be given with respect to the preference
243  if versions is None:
244  versions = []
245  version_info = get_impl_version_info()
246  major = version_info[:-1]
247  # Support all previous minor Python versions.
248  for minor in range(version_info[-1], -1, -1):
249  versions.append(''.join(map(str, major + (minor,))))
250 
251  impl = impl or get_abbr_impl()
252 
253  abis = []
254 
255  abi = abi or get_abi_tag()

259  abi3s = set()

261  for suffix in imp.get_suffixes():
262  if suffix[0].startswith('.abi'):
263  abi3s.add(suffix[0].split('.', 2)[1])
264 
265  abis.extend(sorted(list(abi3s)))
266 
267  abis.append('none')
268 
269  if not noarch:
270  arch = platform or get_platform()

272  # support macosx-10.6-intel on macosx-10.9-x86_64
273  match = _osx_arch_pat.match(arch)

276  tpl = '{0}_{1}_%i_%s'.format(name, major)
277  arches = []
278  for m in reversed(range(int(minor) + 1)):
279  for a in get_darwin_arches(int(major), m, actual_arch):
280  arches.append(tpl % (m, a))
281  else:

283  arches = [arch]

285  arches = [arch.replace('linux', 'manylinux1'), arch]
286  else:
287  arches = [arch]

289  # Current version, current API (built specifically for our Python):
290  for abi in abis:

292  supported.append(('%s%s' % (impl, versions[0]), abi, arch))
293 
294  # abi3 modules compatible with older version of Python
295  for version in versions[1:]:
296  # abi3 was introduced in Python 3.2
297  if version in ('31', '30'):

299  for abi in abi3s: # empty set if not Python 3
300  for arch in arches:
301  supported.append(("%s%s" % (impl, version), abi, arch))

304  for arch in arches:

308  supported.append(('%s%s' % (impl, versions[0]), 'none', 'any'))
309  # Tagged specifically as being cross-version compatible
310  # (with just the major version specified)
311  supported.append(('%s%s' % (impl, versions[0][0]), 'none', 'any'))
312 
313  # No abi / arch, generic Python
314  for i, version in enumerate(versions):
315  supported.append(('py%s' % (version,), 'none', 'any'))
316  if i == 0:
317  supported.append(('py%s' % (version[0]), 'none', 'any'))
318
