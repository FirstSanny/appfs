31 def _annotated_unpack_infer(stmt, context=None):
32  """
33  Recursively generate nodes inferred by the given statement.
34  If the inferred value is a list or a tuple, recurse on the elements.
35  Returns an iterator which yields tuples in the format
36  ('original node', 'infered node').
37  """
38  if isinstance(stmt, (astroid.List, astroid.Tuple)):
39  for elt in stmt.elts:
40  inferred = utils.safe_infer(elt)
41  if inferred and inferred is not astroid.YES:
42  yield elt, inferred
43  return
44  for infered in stmt.infer(context):
45  if infered is astroid.YES:
46  continue
47  yield stmt, infered

249  def _check_misplaced_bare_raise(self, node):
250  # Filter out if it's present in __exit__.
251  scope = node.scope()
252  if (isinstance(scope, astroid.FunctionDef)
253  and scope.is_method()
254  and scope.name == '__exit__'):

258  # Stop when a new scope is generated or when the raise

286  def _check_catching_non_exception(self, handler, exc, part):
287  if isinstance(exc, astroid.Tuple):
288  # Check if it is a tuple of exceptions.
289  inferred = [utils.safe_infer(elt) for elt in exc.elts]
290  if any(node is astroid.YES for node in inferred):
291  # Don't emit if we don't know every component.
292  return
293  if all(node and utils.inherit_from_std_ex(node)
294  for node in inferred):
295  return
296 
297  if not isinstance(exc, astroid.ClassDef):
298  # Don't emit the warning if the infered stmt
299  # is None, but the exception handler is something else,
300  # maybe it was redefined.
301  if (isinstance(exc, astroid.Const) and
302  exc.value is None):
303  if ((isinstance(handler.type, astroid.Const) and
304  handler.type.value is None) or
305  handler.type.parent_of(exc)):
306  # If the exception handler catches None or
307  # the exception component, which is None, is
308  # defined by the entire exception handler, then
309  # emit a warning.
310  self.add_message('catching-non-exception',
311  node=handler.type,
312  args=(part.as_string(), ))
313  else:
314  self.add_message('catching-non-exception',
315  node=handler.type,
316  args=(part.as_string(), ))
317  return

319  if (not utils.inherit_from_std_ex(exc) and
320  exc.name not in self._builtin_exceptions):
321  if utils.has_known_bases(exc):
322  self.add_message('catching-non-exception',

326  @utils.check_messages('bare-except', 'broad-except',
327  'binary-op-exception', 'bad-except-order',
328  'catching-non-exception', 'duplicate-except')
329  def visit_tryexcept(self, node):
330  """check for empty except"""
331  exceptions_classes = []
332  nb_handlers = len(node.handlers)
333  for index, handler in enumerate(node.handlers):
334  if handler.type is None:
335  if not utils.is_raising(handler.body):
336  self.add_message('bare-except', node=handler)
337  # check if a "except:" is followed by some other
338  # except
339  if index < (nb_handlers - 1):
340  msg = 'empty except clause should always appear last'
341  self.add_message('bad-except-order', node=node, args=msg)
342 
343  elif isinstance(handler.type, astroid.BoolOp):
344  self.add_message('binary-op-exception',
345  node=handler, args=handler.type.op)
346  else:
347  try:
348  excs = list(_annotated_unpack_infer(handler.type))
349  except astroid.InferenceError:
350  continue
351 
352  for part, exc in excs:
353  if exc is astroid.YES:
354  continue
355  if (isinstance(exc, astroid.Instance)
356  and utils.inherit_from_std_ex(exc)):
357  # pylint: disable=protected-access
358  exc = exc._proxied
359 
360  self._check_catching_non_exception(handler, exc, part)
361 
362  if not isinstance(exc, astroid.ClassDef):
363  continue
364 
365  exc_ancestors = [anc for anc in exc.ancestors()
366  if isinstance(anc, astroid.ClassDef)]
367 
368  for previous_exc in exceptions_classes:
369  if previous_exc in exc_ancestors:
370  msg = '%s is an ancestor class of %s' % (
371  previous_exc.name, exc.name)
372  self.add_message('bad-except-order',
373  node=handler.type, args=msg)
374  if (exc.name in self.config.overgeneral_exceptions
375  and exc.root().name == utils.EXCEPTIONS_MODULE
376  and not utils.is_raising(handler.body)):
377  self.add_message('broad-except',
378  args=exc.name, node=handler.type)
379 
380  if exc in exceptions_classes:
381  self.add_message('duplicate-except',
382  args=exc.name, node=handler.type)
383 
384  exceptions_classes += [exc for _, exc in excs]
