19 @wrapt.decorator
20 def cached(func, instance, args, kwargs):
21  """Simple decorator to cache result of method calls without args."""
22  cache = getattr(instance, '__cache', None)
23  if cache is None:
24  instance.__cache = cache = {}

26  return cache[func]

32 class cachedproperty(object):
33  """ Provides a cached property equivalent to the stacking of
34  @cached and @property, but more efficient.
35 
36  After first usage, the <property_name> becomes part of the object's
37  __dict__. Doing:
38 
39  del obj.<property_name> empties the cache.
40 
41  Idea taken from the pyramid_ framework and the mercurial_ project.
42 
43  .. _pyramid: http://pypi.python.org/pypi/pyramid
44  .. _mercurial: http://pypi.python.org/pypi/Mercurial
45  """
46  __slots__ = ('wrapped',)
47 
48  def __init__(self, wrapped):
49  try:
50  wrapped.__name__
51  except AttributeError:
52  util.reraise(TypeError('%s must have a __name__ attribute'
53  % wrapped))
54  self.wrapped = wrapped
55 
56  @property
57  def __doc__(self):
58  doc = getattr(self.wrapped, '__doc__', None)
59  return ('<wrapped by the cachedproperty decorator>%s'
60  % ('\n%s' % doc if doc else ''))
61 
62  def __get__(self, inst, objtype=None):
63  if inst is None:
64  return self
65  val = self.wrapped(inst)
66  setattr(inst, self.wrapped.__name__, val)
67  return val

73  @functools.wraps(func)
74  def wrapped(node, context=None, _func=func, **kwargs):
75  """wrapper function handling context"""
76  if context is None:
77  context = contextmod.InferenceContext()
78  if context.push(node):
79  return
80 
81  yielded = set()
82  generator = _func(node, context, **kwargs)
83  try:
84  while True:
85  res = next(generator)
86  # unproxy only true instance, not const, tuple, dict...
87  if res.__class__.__name__ == 'Instance':
88  ares = res._proxied
89  else:
90  ares = res
91  if ares not in yielded:
92  yield res
93  yielded.add(ares)
94  except StopIteration as error:
95  # Explicit StopIteration to return error information, see
96  # comment in raise_if_nothing_inferred.
97  if error.args:
98  raise StopIteration(error.args[0])
99  else:
100  raise StopIteration
101 
102  return wrapped

115 @wrapt.decorator
116 def raise_if_nothing_inferred(func, instance, args, kwargs):
117  '''All generators wrapped with raise_if_nothing_inferred *must*
118  explicitly raise StopIteration with information to create an
119  appropriate structured InferenceError.
120 
121  '''
122  # TODO: Explicitly raising StopIteration in a generator will cause
123  # a RuntimeError in Python >=3.7, as per
124  # http://legacy.python.org/dev/peps/pep-0479/ . Before 3.7 is
125  # released, this code will need to use one of four possible
126  # solutions: a decorator that restores the current behavior as
127  # described in
128  # http://legacy.python.org/dev/peps/pep-0479/#sub-proposal-decorator-to-explicitly-request-current-behaviour
129  # , dynamic imports or exec to generate different code for
130  # different versions, drop support for all Python versions <3.3,
131  # or refactoring to change how these decorators work. In any
132  # event, after dropping support for Python <3.3 this code should
133  # be refactored to use `yield from`.
134  inferred = False
135  try:
136  generator = func(*args, **kwargs)
137  while True:
138  yield next(generator)
139  inferred = True
140  except StopIteration as error:
141  if not inferred:
142  if error.args:
143  # pylint: disable=not-a-mapping
144  raise exceptions.InferenceError(**error.args[0])
145  else:
146  raise exceptions.InferenceError(
147  'StopIteration raised without any error information.')
