36 def translate_pattern(glob):
37  """
38  Translate a file path glob like '*.txt' in to a regular expression.
39  This differs from fnmatch.translate which allows wildcards to match
40  directory separators. It also knows about '**/' which matches any number of
41  directories.
42  """
43  pat = ''
44 
45  # This will split on '/' within [character classes]. This is deliberate.
46  chunks = glob.split(os.path.sep)
47 
48  sep = re.escape(os.sep)
49  valid_char = '[^%s]' % (sep,)
50 
51  for c, chunk in enumerate(chunks):
52  last_chunk = c == len(chunks) - 1
53 
54  # Chunks that are a literal ** are globstars. They match anything.
55  if chunk == '**':
56  if last_chunk:
57  # Match anything if this is the last component
58  pat += '.*'
59  else:
60  # Match '(name/)*'
61  pat += '(?:%s+%s)*' % (valid_char, sep)
62  continue # Break here as the whole path component has been handled
63 
64  # Find any special characters in the remainder
65  i = 0
66  chunk_len = len(chunk)
67  while i < chunk_len:
68  char = chunk[i]
69  if char == '*':
70  # Match any number of name characters
71  pat += valid_char + '*'
72  elif char == '?':
73  # Match a name character
74  pat += valid_char
75  elif char == '[':
76  # Character class
77  inner_i = i + 1
78  # Skip initial !/] chars
79  if inner_i < chunk_len and chunk[inner_i] == '!':
80  inner_i = inner_i + 1
81  if inner_i < chunk_len and chunk[inner_i] == ']':
82  inner_i = inner_i + 1
83 
84  # Loop till the closing ] is found
85  while inner_i < chunk_len and chunk[inner_i] != ']':
86  inner_i = inner_i + 1
87 
88  if inner_i >= chunk_len:
89  # Got to the end of the string without finding a closing ]
90  # Do not treat this as a matching group, but as a literal [
91  pat += re.escape(char)
92  else:
93  # Grab the insides of the [brackets]
94  inner = chunk[i + 1:inner_i]
95  char_class = ''
96 
97  # Class negation
98  if inner[0] == '!':
99  char_class = '^'
100  inner = inner[1:]
101 
102  char_class += re.escape(inner)
103  pat += '[%s]' % (char_class,)
104 
105  # Skip to the end ]
106  i = inner_i
107  else:
108  pat += re.escape(char)
109  i += 1
110 
111  # Join each chunk with the dir separator
112  if not last_chunk:
113  pat += sep
114 
115  pat += r'\Z'
116  return re.compile(pat, flags=re.MULTILINE|re.DOTALL)

317  def process_template_line(self, line):
318  # Parse the line: split it up, make sure the right number of words
319  # is there, and return the relevant words. 'action' is always
320  # defined: it's the first word of the line. Which of the other
321  # three are defined depends on the action; it'll be either
322  # patterns, (dir and patterns), or (dir_pattern).
323  (action, patterns, dir, dir_pattern) = self._parse_template_line(line)
324 
325  # OK, now we know that the action is valid and we have the
326  # right number of words on the line for that action -- so we
327  # can proceed with minimal error-checking.
328  if action == 'include':
329  self.debug_print("include " + ' '.join(patterns))
330  for pattern in patterns:
331  if not self.include(pattern):
332  log.warn("warning: no files found matching '%s'", pattern)
333 
334  elif action == 'exclude':
335  self.debug_print("exclude " + ' '.join(patterns))
336  for pattern in patterns:
337  if not self.exclude(pattern):
338  log.warn(("warning: no previously-included files "
339  "found matching '%s'"), pattern)
340 
341  elif action == 'global-include':
342  self.debug_print("global-include " + ' '.join(patterns))
343  for pattern in patterns:
344  if not self.global_include(pattern):
345  log.warn(("warning: no files found matching '%s' "
346  "anywhere in distribution"), pattern)
347 
348  elif action == 'global-exclude':
349  self.debug_print("global-exclude " + ' '.join(patterns))
350  for pattern in patterns:
351  if not self.global_exclude(pattern):
352  log.warn(("warning: no previously-included files matching "
353  "'%s' found anywhere in distribution"),
354  pattern)
355 
356  elif action == 'recursive-include':
357  self.debug_print("recursive-include %s %s" %
358  (dir, ' '.join(patterns)))
359  for pattern in patterns:
360  if not self.recursive_include(dir, pattern):
361  log.warn(("warning: no files found matching '%s' "
362  "under directory '%s'"),
363  pattern, dir)
364 
365  elif action == 'recursive-exclude':
366  self.debug_print("recursive-exclude %s %s" %
367  (dir, ' '.join(patterns)))
368  for pattern in patterns:
369  if not self.recursive_exclude(dir, pattern):
370  log.warn(("warning: no previously-included files matching "
371  "'%s' found under directory '%s'"),
372  pattern, dir)
373 
374  elif action == 'graft':
375  self.debug_print("graft " + dir_pattern)
376  if not self.graft(dir_pattern):
377  log.warn("warning: no directories found matching '%s'",
378  dir_pattern)
379 
380  elif action == 'prune':
381  self.debug_print("prune " + dir_pattern)
382  if not self.prune(dir_pattern):
383  log.warn(("no previously-included directories found "
384  "matching '%s'"), dir_pattern)
385 
386  else:
387  raise DistutilsInternalError(
388  "this cannot happen: invalid action '%s'" % action)

486  def _safe_path(self, path):
487  enc_warn = "'%s' not %s encodable -- skipping"
488 
489  # To avoid accidental trans-codings errors, first to unicode
490  u_path = unicode_utils.filesys_decode(path)
491  if u_path is None:
492  log.warn("'%s' in unexpected encoding -- skipping" % path)
493  return False
494 
495  # Must ensure utf-8 encodability
496  utf8_path = unicode_utils.try_encode(u_path, "utf-8")
497  if utf8_path is None:
498  log.warn(enc_warn, path, 'utf-8')
499  return False
500 
501  try:
502  # accept is either way checks out
503  if os.path.exists(u_path) or os.path.exists(utf8_path):
504  return True
505  # this will catch any encode errors decoding u_path
506  except UnicodeEncodeError:
507  log.warn(enc_warn, path, sys.getfilesystemencoding())
