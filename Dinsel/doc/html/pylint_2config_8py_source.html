226 def _expand_default(self, option):
227  """Patch OptionParser.expand_default with custom behaviour
228 
229  This will handle defaults to avoid overriding values in the
230  configuration file.
231  """
232  if self.parser is None or not self.default_tag:

296  # First, convert the value(s) to the right type. Howl if any

468  def register_options_provider(self, provider, own_group=True):

516  def optik_option(self, provider, opt, optdict):
517  """get our personal option definition and return a suitable form for
518  use with optik/optparse
519  """
520  optdict = copy.copy(optdict)
521  if 'action' in optdict:
522  self._nocallback_options[provider] = opt
523  else:
524  optdict['action'] = 'callback'
525  optdict['callback'] = self.cb_set_provider_option
526  # default is handled here and *must not* be given to optik if you
527  # want the whole machinery to work
528  if 'default' in optdict:
529  if ('help' in optdict
530  and optdict.get('default') is not None
531  and optdict['action'] not in ('store_true', 'store_false')):
532  optdict['help'] += ' [current: %default]'
533  del optdict['default']
534  args = ['--' + str(opt)]
535  if 'short' in optdict:
536  self._short_options[optdict['short']] = opt
537  args.append('-' + optdict['short'])
538  del optdict['short']
539  # cleanup option definition dict before giving it to optik
540  for key in list(optdict.keys()):
541  if key not in self._optik_option_attrs:
542  optdict.pop(key)
543  return args, optdict

562  def generate_config(self, stream=None, skipsections=(), encoding=None):
563  """write a configuration file according to the current configuration
564  into the given stream or stdout
565  """
566  options_by_section = {}
567  sections = []
568  for provider in self.options_providers:
569  for section, options in provider.options_by_section():
570  if section is None:
571  section = provider.name
572  if section in skipsections:
573  continue
574  options = [(n, d, v) for (n, d, v) in options
575  if d.get('type') is not None
576  and not d.get('deprecated')]
577  if not options:
578  continue
579  if section not in sections:
580  sections.append(section)
581  alloptions = options_by_section.setdefault(section, [])
582  alloptions += options
583  stream = stream or sys.stdout
584  encoding = utils._get_encoding(encoding, stream)
585  printed = False
586  for section in sections:
587  if printed:
588  print('\n', file=stream)
589  utils.format_section(stream, section.upper(),
590  sorted(options_by_section[section]),
591  encoding)
