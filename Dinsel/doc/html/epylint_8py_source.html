12 """Emacs and Flymake compatible Pylint.
13 
14 This script is for integration with emacs and is compatible with flymake mode.
15 
16 epylint walks out of python packages before invoking pylint. This avoids
17 reporting import errors that occur when a module within a package uses the
18 absolute import path to get another module within this package.
19 
20 For example:
21  - Suppose a package is structured as
22 
23  a/__init__.py
24  a/b/x.py
25  a/c/y.py
26 
27  - Then if y.py imports x as "from a.b import x" the following produces pylint
28  errors
29 
30  cd a/c; pylint y.py
31 
32  - The following obviously doesn't
33 
34  pylint a/c/y.py
35 
36  - As this script will be invoked by emacs within the directory of the file
37  we are checking we need to go out of it to avoid these false positives.
38 
39 
40 You may also use py_run to run pylint with desired options and get back (or not)
41 its output.
42 """
43 from __future__ import print_function
44

61 def lint(filename, options=None):
62  """Pylint the given file.
63 
64  When run from emacs we will be in the directory of a file, and passed its
65  filename. If this file is part of a package and is trying to import other
66  modules from within its own package or another package rooted in a directory
67  below it, pylint will classify it as a failed import.
68 
69  To get around this, we traverse down the directory tree to find the root of
70  the package this module is in. We then invoke pylint from this directory.
71 
72  Finally, we must correct the filenames in the output generated by pylint so
73  Emacs doesn't become confused (it will expect just the original filename,
74  while pylint may extend it with extra directories if we've traversed down
75  the tree)
76  """
77  # traverse downwards until we are out of a python package
78  full_path = osp.abspath(filename)
79  parent_path = osp.dirname(full_path)
80  child_path = osp.basename(full_path)
81 
82  while parent_path != "/" and osp.exists(osp.join(parent_path, '__init__.py')):
83  child_path = osp.join(osp.basename(parent_path), child_path)
84  parent_path = osp.dirname(parent_path)
85 
86  # Start pylint
87  # Ensure we use the python and pylint associated with the running epylint
88  run_cmd = "import sys; from pylint.lint import Run; Run(sys.argv[1:])"
89  options = options or ['--disable=C,R,I']
90  cmd = [sys.executable, "-c", run_cmd] + [
91  '--msg-template', '{path}:{line}: {category} ({msg_id}, {symbol}, {obj}) {msg}',
92  '-r', 'n', child_path] + options
93  process = Popen(cmd, stdout=PIPE, cwd=parent_path, env=_get_env(),
94  universal_newlines=True)
95 
96  for line in process.stdout:
97  # remove pylintrc warning
98  if line.startswith("No config file found"):
99  continue
100 
101  # modify the file name thats output to reverse the path traversal we made
102  parts = line.split(":")
103  if parts and parts[0] == child_path:
104  line = ":".join([filename] + parts[1:])
105  print(line, end=' ')
106 
107  process.wait()
108  return process.returncode

111 def py_run(command_options='', return_std=False, stdout=None, stderr=None):
112  """Run pylint from python
113 
114  ``command_options`` is a string containing ``pylint`` command line options;
115  ``return_std`` (boolean) indicates return of created standard output
116  and error (see below);
117  ``stdout`` and ``stderr`` are 'file-like' objects in which standard output
118  could be written.
119 
120  Calling agent is responsible for stdout/err management (creation, close).
121  Default standard output and error are those from sys,
122  or standalone ones (``subprocess.PIPE``) are used
123  if they are not set and ``return_std``.
124 
125  If ``return_std`` is set to ``True``, this function returns a 2-uple
126  containing standard output and error related to created process,
127  as follows: ``(stdout, stderr)``.
128 
129  A trivial usage could be as follows:
130  >>> py_run( '--version')
131  No config file found, using default configuration
132  pylint 0.18.1,
133  ...
134 
135  To silently run Pylint on a module, and get its standard output and error:
136  >>> (pylint_stdout, pylint_stderr) = py_run( 'module_name.py', True)
137  """
138  # Create command line to call pylint
139  epylint_part = [sys.executable, "-c", "from pylint import epylint;epylint.Run()"]
140  options = shlex.split(command_options)
141  cli = epylint_part + options
142 
143  # Providing standard output and/or error if not set
144  if stdout is None:
145  if return_std:
146  stdout = PIPE
147  else:
148  stdout = sys.stdout
149  if stderr is None:
150  if return_std:
151  stderr = PIPE
152  else:
153  stderr = sys.stderr
154  # Call pylint in a subprocess
155  process = Popen(cli, shell=False, stdout=stdout, stderr=stderr,
156  env=_get_env(), universal_newlines=True)
157  proc_stdout, proc_stderr = process.communicate()
158  # Return standard output and error
159  if return_std:
160  return six.moves.StringIO(proc_stdout), six.moves.StringIO(proc_stderr)
