18 def mintypecode(typechars,typeset='GDFgdf',default='d'):
19  """
20  Return the character for the minimum-size type to which given types can
21  be safely cast.
22 
23  The returned type character must represent the smallest size dtype such
24  that an array of the returned type can handle the data from an array of
25  all types in `typechars` (or if `typechars` is an array, then its
26  dtype.char).
27 
28  Parameters
29  ----------
30  typechars : list of str or array_like
31  If a list of strings, each string should represent a dtype.
32  If array_like, the character representation of the array dtype is used.
33  typeset : str or list of str, optional
34  The set of characters that the returned character is chosen from.
35  The default set is 'GDFgdf'.
36  default : str, optional
37  The default character, this is returned if none of the characters in
38  `typechars` matches a character in `typeset`.
39 
40  Returns
41  -------
42  typechar : str
43  The character representing the minimum-size type that was found.
44 
45  See Also
46  --------
47  dtype, sctype2char, maximum_sctype
48 
49  Examples
50  --------
51  >>> np.mintypecode(['d', 'f', 'S'])
52  'd'
53  >>> x = np.array([1.1, 2-3.j])
54  >>> np.mintypecode(x)
55  'D'
56 
57  >>> np.mintypecode('abceh', default='G')
58  'G'
59 
60  """
61  typecodes = [(isinstance(t, str) and t) or asarray(t).dtype.char
62  for t in typechars]
63  intersection = [t for t in typecodes if t in typeset]
64  if not intersection:
65  return default
66  if 'F' in intersection and 'd' in intersection:
67  return 'D'
68  l = []
69  for t in intersection:
70  i = _typecodes_by_elsize.index(t)
71  l.append((i, t))
72  l.sort()
73  return l[0][1]

318 def nan_to_num(x):
319  """
320  Replace nan with zero and inf with finite numbers.
321 
322  Returns an array or scalar replacing Not a Number (NaN) with zero,
323  (positive) infinity with a very large number and negative infinity
324  with a very small (or negative) number.
325 
326  Parameters
327  ----------
328  x : array_like
329  Input data.
330 
331  Returns
332  -------
333  out : ndarray
334  New Array with the same shape as `x` and dtype of the element in
335  `x` with the greatest precision. If `x` is inexact, then NaN is
336  replaced by zero, and infinity (-infinity) is replaced by the
337  largest (smallest or most negative) floating point value that fits
338  in the output dtype. If `x` is not inexact, then a copy of `x` is
339  returned.
340 
341  See Also
342  --------
343  isinf : Shows which elements are positive or negative infinity.
344  isneginf : Shows which elements are negative infinity.
345  isposinf : Shows which elements are positive infinity.
346  isnan : Shows which elements are Not a Number (NaN).
347  isfinite : Shows which elements are finite (not NaN, not infinity)
348 
349  Notes
350  -----
351  NumPy uses the IEEE Standard for Binary Floating-Point for Arithmetic
352  (IEEE 754). This means that Not a Number is not equivalent to infinity.
353 
354 
355  Examples
356  --------
357  >>> np.set_printoptions(precision=8)
358  >>> x = np.array([np.inf, -np.inf, np.nan, -128, 128])
359  >>> np.nan_to_num(x)
360  array([ 1.79769313e+308, -1.79769313e+308, 0.00000000e+000,
361  -1.28000000e+002, 1.28000000e+002])
362 
363  """
364  x = _nx.array(x, subok=True)
365  xtype = x.dtype.type
366  if not issubclass(xtype, _nx.inexact):
367  return x
368 
369  iscomplex = issubclass(xtype, _nx.complexfloating)
370  isscalar = (x.ndim == 0)
371 
372  x = x[None] if isscalar else x
373  dest = (x.real, x.imag) if iscomplex else (x,)
374  maxf, minf = _getmaxmin(x.real.dtype)
375  for d in dest:
376  _nx.copyto(d, 0.0, where=isnan(d))
377  _nx.copyto(d, maxf, where=isposinf(d))
378  _nx.copyto(d, minf, where=isneginf(d))
379  return x[0] if isscalar else x

383 def real_if_close(a,tol=100):
384  """
385  If complex input returns a real array if complex parts are close to zero.
386 
387  "Close to zero" is defined as `tol` * (machine epsilon of the type for
388  `a`).
389 
390  Parameters
391  ----------
392  a : array_like
393  Input array.
394  tol : float
395  Tolerance in machine epsilons for the complex part of the elements
396  in the array.
397 
398  Returns
399  -------
400  out : ndarray
401  If `a` is real, the type of `a` is used for the output. If `a`
402  has complex elements, the returned type is float.
403 
404  See Also
405  --------
406  real, imag, angle
407 
408  Notes
409  -----
410  Machine epsilon varies from machine to machine and between data types
411  but Python floats on most platforms have a machine epsilon equal to
412  2.2204460492503131e-16. You can use 'np.finfo(np.float).eps' to print
413  out the machine epsilon for floats.
414 
415  Examples
416  --------
417  >>> np.finfo(np.float).eps
418  2.2204460492503131e-16
419 
420  >>> np.real_if_close([2.1 + 4e-14j], tol=1000)
421  array([ 2.1])
422  >>> np.real_if_close([2.1 + 4e-13j], tol=1000)
423  array([ 2.1 +4.00000000e-13j])
424 
425  """
426  a = asanyarray(a)
427  if not issubclass(a.dtype.type, _nx.complexfloating):
428  return a
429  if tol > 1:
430  from numpy.core import getlimits
431  f = getlimits.finfo(a.dtype.type)
432  tol = f.eps * tol
433  if _nx.all(_nx.absolute(a.imag) < tol):
434  a = a.real
435  return a

550 def common_type(*arrays):
551  """
552  Return a scalar type which is common to the input arrays.
553 
554  The return type will always be an inexact (i.e. floating point) scalar
555  type, even if all the arrays are integer arrays. If one of the inputs is
556  an integer array, the minimum precision type that is returned is a
557  64-bit floating point dtype.
558 
559  All input arrays can be safely cast to the returned dtype without loss
560  of information.
561 
562  Parameters
563  ----------
564  array1, array2, ... : ndarrays
565  Input arrays.
566 
567  Returns
568  -------
569  out : data type code
570  Data type code.
571 
572  See Also
573  --------
574  dtype, mintypecode
575 
576  Examples
577  --------
578  >>> np.common_type(np.arange(2, dtype=np.float32))
579  <type 'numpy.float32'>
580  >>> np.common_type(np.arange(2, dtype=np.float32), np.arange(2))
581  <type 'numpy.float64'>
582  >>> np.common_type(np.arange(4), np.array([45, 6.j]), np.array([45.0]))
583  <type 'numpy.complex128'>
584 
585  """
586  is_complex = False
587  precision = 0
588  for a in arrays:
589  t = a.dtype.type
590  if iscomplexobj(a):
591  is_complex = True
592  if issubclass(t, _nx.integer):
593  p = 2 # array_precision[_nx.double]
594  else:
595  p = array_precision.get(t, None)
596  if p is None:
597  raise TypeError("can't get common type for non-numeric array")
598  precision = max(precision, p)
599  if is_complex:
600  return array_type[1][precision]
601  else:
602  return array_type[0][precision]
