15 def splprep(x, w=None, u=None, ub=None, ue=None, k=3, task=0, s=None, t=None,
16  full_output=0, nest=None, per=0, quiet=1):
17  """
18  Find the B-spline representation of an N-dimensional curve.
19 
20  Given a list of N rank-1 arrays, `x`, which represent a curve in
21  N-dimensional space parametrized by `u`, find a smooth approximating
22  spline curve g(`u`). Uses the FORTRAN routine parcur from FITPACK.
23 
24  Parameters
25  ----------
26  x : array_like
27  A list of sample vector arrays representing the curve.
28  w : array_like, optional
29  Strictly positive rank-1 array of weights the same length as `x[0]`.
30  The weights are used in computing the weighted least-squares spline
31  fit. If the errors in the `x` values have standard-deviation given by
32  the vector d, then `w` should be 1/d. Default is ``ones(len(x[0]))``.
33  u : array_like, optional
34  An array of parameter values. If not given, these values are
35  calculated automatically as ``M = len(x[0])``, where
36 
37  v[0] = 0
38 
39  v[i] = v[i-1] + distance(`x[i]`, `x[i-1]`)
40 
41  u[i] = v[i] / v[M-1]
42 
43  ub, ue : int, optional
44  The end-points of the parameters interval. Defaults to
45  u[0] and u[-1].
46  k : int, optional
47  Degree of the spline. Cubic splines are recommended.
48  Even values of `k` should be avoided especially with a small s-value.
49  ``1 <= k <= 5``, default is 3.
50  task : int, optional
51  If task==0 (default), find t and c for a given smoothing factor, s.
52  If task==1, find t and c for another value of the smoothing factor, s.
53  There must have been a previous call with task=0 or task=1
54  for the same set of data.
55  If task=-1 find the weighted least square spline for a given set of
56  knots, t.
57  s : float, optional
58  A smoothing condition. The amount of smoothness is determined by
59  satisfying the conditions: ``sum((w * (y - g))**2,axis=0) <= s``,
60  where g(x) is the smoothed interpolation of (x,y). The user can
61  use `s` to control the trade-off between closeness and smoothness
62  of fit. Larger `s` means more smoothing while smaller values of `s`
63  indicate less smoothing. Recommended values of `s` depend on the
64  weights, w. If the weights represent the inverse of the
65  standard-deviation of y, then a good `s` value should be found in
66  the range ``(m-sqrt(2*m),m+sqrt(2*m))``, where m is the number of
67  data points in x, y, and w.
68  t : int, optional
69  The knots needed for task=-1.
70  full_output : int, optional
71  If non-zero, then return optional outputs.
72  nest : int, optional
73  An over-estimate of the total number of knots of the spline to
74  help in determining the storage space. By default nest=m/2.
75  Always large enough is nest=m+k+1.
76  per : int, optional
77  If non-zero, data points are considered periodic with period
78  ``x[m-1] - x[0]`` and a smooth periodic spline approximation is
79  returned. Values of ``y[m-1]`` and ``w[m-1]`` are not used.
80  quiet : int, optional
81  Non-zero to suppress messages.
82  This parameter is deprecated; use standard Python warning filters
83  instead.
84 
85  Returns
86  -------
87  tck : tuple
88  (t,c,k) a tuple containing the vector of knots, the B-spline
89  coefficients, and the degree of the spline.
90  u : array
91  An array of the values of the parameter.
92  fp : float
93  The weighted sum of squared residuals of the spline approximation.
94  ier : int
95  An integer flag about splrep success. Success is indicated
96  if ier<=0. If ier in [1,2,3] an error occurred but was not raised.
97  Otherwise an error is raised.
98  msg : str
99  A message corresponding to the integer flag, ier.
100 
101  See Also
102  --------
103  splrep, splev, sproot, spalde, splint,
104  bisplrep, bisplev
105  UnivariateSpline, BivariateSpline
106  BSpline
107  make_interp_spline
108 
109  Notes
110  -----
111  See `splev` for evaluation of the spline and its derivatives.
112  The number of dimensions N must be smaller than 11.
113 
114  References
115  ----------
116  .. [1] P. Dierckx, "Algorithms for smoothing data with periodic and
117  parametric splines, Computer Graphics and Image Processing",
118  20 (1982) 171-184.
119  .. [2] P. Dierckx, "Algorithms for smoothing data with periodic and
120  parametric splines", report tw55, Dept. Computer Science,
121  K.U.Leuven, 1981.
122  .. [3] P. Dierckx, "Curve and surface fitting with splines", Monographs on
123  Numerical Analysis, Oxford University Press, 1993.
124 
125  Examples
126  --------
127  Generate a discretization of a limacon curve in the polar coordinates:
128 
129  >>> phi = np.linspace(0, 2.*np.pi, 40)
130  >>> r = 0.5 + np.cos(phi) # polar coords
131  >>> x, y = r * np.cos(phi), r * np.sin(phi) # convert to cartesian
132 
133  And interpolate:
134 
135  >>> from scipy.interpolate import splprep, splev
136  >>> tck, u = splprep([x, y], s=0)
137  >>> new_points = splev(u, tck)
138 
139  Notice that (i) we force interpolation by using `s=0`,
140  (ii) the parameterization, ``u``, is generated automatically.
141  Now plot the result:
142 
143  >>> import matplotlib.pyplot as plt
144  >>> fig, ax = plt.subplots()
145  >>> ax.plot(x, y, 'ro')
146  >>> ax.plot(new_points[0], new_points[1], 'r-')
147  >>> plt.show()
148 
149  """
150  res = _impl.splprep(x, w, u, ub, ue, k, task, s, t, full_output, nest, per,
151  quiet)
152  return res

155 def splrep(x, y, w=None, xb=None, xe=None, k=3, task=0, s=None, t=None,
156  full_output=0, per=0, quiet=1):
157  """
158  Find the B-spline representation of 1-D curve.
159 
160  Given the set of data points ``(x[i], y[i])`` determine a smooth spline
161  approximation of degree k on the interval ``xb <= x <= xe``.
162 
163  Parameters
164  ----------
165  x, y : array_like
166  The data points defining a curve y = f(x).
167  w : array_like, optional
168  Strictly positive rank-1 array of weights the same length as x and y.
169  The weights are used in computing the weighted least-squares spline
170  fit. If the errors in the y values have standard-deviation given by the
171  vector d, then w should be 1/d. Default is ones(len(x)).
172  xb, xe : float, optional
173  The interval to fit. If None, these default to x[0] and x[-1]
174  respectively.
175  k : int, optional
176  The degree of the spline fit. It is recommended to use cubic splines.
177  Even values of k should be avoided especially with small s values.
178  1 <= k <= 5
179  task : {1, 0, -1}, optional
180  If task==0 find t and c for a given smoothing factor, s.
181 
182  If task==1 find t and c for another value of the smoothing factor, s.
183  There must have been a previous call with task=0 or task=1 for the same
184  set of data (t will be stored an used internally)
185 
186  If task=-1 find the weighted least square spline for a given set of
187  knots, t. These should be interior knots as knots on the ends will be
188  added automatically.
189  s : float, optional
190  A smoothing condition. The amount of smoothness is determined by
191  satisfying the conditions: sum((w * (y - g))**2,axis=0) <= s where g(x)
192  is the smoothed interpolation of (x,y). The user can use s to control
193  the tradeoff between closeness and smoothness of fit. Larger s means
194  more smoothing while smaller values of s indicate less smoothing.
195  Recommended values of s depend on the weights, w. If the weights
196  represent the inverse of the standard-deviation of y, then a good s
197  value should be found in the range (m-sqrt(2*m),m+sqrt(2*m)) where m is
198  the number of datapoints in x, y, and w. default : s=m-sqrt(2*m) if
199  weights are supplied. s = 0.0 (interpolating) if no weights are
200  supplied.
201  t : array_like, optional
202  The knots needed for task=-1. If given then task is automatically set
203  to -1.
204  full_output : bool, optional
205  If non-zero, then return optional outputs.
206  per : bool, optional
207  If non-zero, data points are considered periodic with period x[m-1] -
208  x[0] and a smooth periodic spline approximation is returned. Values of
209  y[m-1] and w[m-1] are not used.
210  quiet : bool, optional
211  Non-zero to suppress messages.
212  This parameter is deprecated; use standard Python warning filters
213  instead.
214 
215  Returns
216  -------
217  tck : tuple
218  A tuple (t,c,k) containing the vector of knots, the B-spline
219  coefficients, and the degree of the spline.
220  fp : array, optional
221  The weighted sum of squared residuals of the spline approximation.
222  ier : int, optional
223  An integer flag about splrep success. Success is indicated if ier<=0.
224  If ier in [1,2,3] an error occurred but was not raised. Otherwise an
225  error is raised.
226  msg : str, optional
227  A message corresponding to the integer flag, ier.
228 
229  See Also
230  --------
231  UnivariateSpline, BivariateSpline
232  splprep, splev, sproot, spalde, splint
233  bisplrep, bisplev
234  BSpline
235  make_interp_spline
236 
237  Notes
238  -----
239  See `splev` for evaluation of the spline and its derivatives. Uses the
240  FORTRAN routine ``curfit`` from FITPACK.
241 
242  The user is responsible for assuring that the values of `x` are unique.
243  Otherwise, `splrep` will not return sensible results.
244 
245  If provided, knots `t` must satisfy the Schoenberg-Whitney conditions,
246  i.e., there must be a subset of data points ``x[j]`` such that
247  ``t[j] < x[j] < t[j+k+1]``, for ``j=0, 1,...,n-k-2``.
248 
249  References
250  ----------
251  Based on algorithms described in [1]_, [2]_, [3]_, and [4]_:
252 
253  .. [1] P. Dierckx, "An algorithm for smoothing, differentiation and
254  integration of experimental data using spline functions",
255  J.Comp.Appl.Maths 1 (1975) 165-184.
256  .. [2] P. Dierckx, "A fast algorithm for smoothing data on a rectangular
257  grid while using spline functions", SIAM J.Numer.Anal. 19 (1982)
258  1286-1304.
259  .. [3] P. Dierckx, "An improved algorithm for curve fitting with spline
260  functions", report tw54, Dept. Computer Science,K.U. Leuven, 1981.
261  .. [4] P. Dierckx, "Curve and surface fitting with splines", Monographs on
262  Numerical Analysis, Oxford University Press, 1993.
263 
264  Examples
265  --------
266 
267  >>> import matplotlib.pyplot as plt
268  >>> from scipy.interpolate import splev, splrep
269  >>> x = np.linspace(0, 10, 10)
270  >>> y = np.sin(x)
271  >>> spl = splrep(x, y)
272  >>> x2 = np.linspace(0, 10, 200)
273  >>> y2 = splev(x2, spl)
274  >>> plt.plot(x, y, 'o', x2, y2)
275  >>> plt.show()
276 
277  """
278  res = _impl.splrep(x, y, w, xb, xe, k, task, s, t, full_output, per, quiet)
279  return res

282 def splev(x, tck, der=0, ext=0):
283  """
284  Evaluate a B-spline or its derivatives.
285 
286  Given the knots and coefficients of a B-spline representation, evaluate
287  the value of the smoothing polynomial and its derivatives. This is a
288  wrapper around the FORTRAN routines splev and splder of FITPACK.
289 
290  Parameters
291  ----------
292  x : array_like
293  An array of points at which to return the value of the smoothed
294  spline or its derivatives. If `tck` was returned from `splprep`,
295  then the parameter values, u should be given.
296  tck : 3-tuple or a BSpline object
297  If a tuple, then it should be a sequence of length 3 returned by
298  `splrep` or `splprep` containing the knots, coefficients, and degree
299  of the spline. (Also see Notes.)
300  der : int, optional
301  The order of derivative of the spline to compute (must be less than
302  or equal to k).
303  ext : int, optional
304  Controls the value returned for elements of ``x`` not in the
305  interval defined by the knot sequence.
306 
307  * if ext=0, return the extrapolated value.
308  * if ext=1, return 0
309  * if ext=2, raise a ValueError
310  * if ext=3, return the boundary value.
311 
312  The default value is 0.
313 
314  Returns
315  -------
316  y : ndarray or list of ndarrays
317  An array of values representing the spline function evaluated at
318  the points in `x`. If `tck` was returned from `splprep`, then this
319  is a list of arrays representing the curve in N-dimensional space.
320 
321  Notes
322  -----
323  Manipulating the tck-tuples directly is not recommended. In new code,
324  prefer using `BSpline` objects.
325 
326  See Also
327  --------
328  splprep, splrep, sproot, spalde, splint
329  bisplrep, bisplev
330  BSpline
331 
332  References
333  ----------
334  .. [1] C. de Boor, "On calculating with b-splines", J. Approximation
335  Theory, 6, p.50-62, 1972.
336  .. [2] M. G. Cox, "The numerical evaluation of b-splines", J. Inst. Maths
337  Applics, 10, p.134-149, 1972.
338  .. [3] P. Dierckx, "Curve and surface fitting with splines", Monographs
339  on Numerical Analysis, Oxford University Press, 1993.
340 
341  """
342  if isinstance(tck, BSpline):
343  if tck.c.ndim > 1:
344  mesg = ("Calling splev() with BSpline objects with c.ndim > 1 is "
345  "not recommended. Use BSpline.__call__(x) instead.")
346  warnings.warn(mesg, DeprecationWarning)
347 
348  # remap the out-of-bounds behavior
349  try:
350  extrapolate = {0: True, }[ext]
351  except KeyError:
352  raise ValueError("Extrapolation mode %s is not supported "
353  "by BSpline." % ext)
354 
355  return tck(x, der, extrapolate=extrapolate)
356  else:
357  return _impl.splev(x, tck, der, ext)

526 def insert(x, tck, m=1, per=0):
527  """
528  Insert knots into a B-spline.
529 
530  Given the knots and coefficients of a B-spline representation, create a
531  new B-spline with a knot inserted `m` times at point `x`.
532  This is a wrapper around the FORTRAN routine insert of FITPACK.
533 
534  Parameters
535  ----------
536  x (u) : array_like
537  A 1-D point at which to insert a new knot(s). If `tck` was returned
538  from ``splprep``, then the parameter values, u should be given.
539  tck : a `BSpline` instance or a tuple
540  If tuple, then it is expected to be a tuple (t,c,k) containing
541  the vector of knots, the B-spline coefficients, and the degree of
542  the spline.
543  m : int, optional
544  The number of times to insert the given knot (its multiplicity).
545  Default is 1.
546  per : int, optional
547  If non-zero, the input spline is considered periodic.
548 
549  Returns
550  -------
551  BSpline instance or a tuple
552  A new B-spline with knots t, coefficients c, and degree k.
553  ``t(k+1) <= x <= t(n-k)``, where k is the degree of the spline.
554  In case of a periodic spline (``per != 0``) there must be
555  either at least k interior knots t(j) satisfying ``t(k+1)<t(j)<=x``
556  or at least k interior knots t(j) satisfying ``x<=t(j)<t(n-k)``.
557  A tuple is returned iff the input argument `tck` is a tuple, otherwise
558  a BSpline object is constructed and returned.
559 
560  Notes
561  -----
562  Based on algorithms from [1]_ and [2]_.
563 
564  Manipulating the tck-tuples directly is not recommended. In new code,
565  prefer using the `BSpline` objects.
566 
567  References
568  ----------
569  .. [1] W. Boehm, "Inserting new knots into b-spline curves.",
570  Computer Aided Design, 12, p.199-201, 1980.
571  .. [2] P. Dierckx, "Curve and surface fitting with splines, Monographs on
572  Numerical Analysis", Oxford University Press, 1993.
573 
574  """
575  if isinstance(tck, BSpline):
576 
577  t, c, k = tck.tck
578 
579  # FITPACK expects the interpolation axis to be last, so roll it over
580  # NB: if c array is 1D, transposes are no-ops
581  sh = tuple(range(c.ndim))
582  c = c.transpose(sh[1:] + (0,))
583  t_, c_, k_ = _impl.insert(x, (t, c, k), m, per)
584 
585  # and roll the last axis back
586  c_ = np.asarray(c_)
587  c_ = c_.transpose((sh[-1],) + sh[:-1])
588  return BSpline(t_, c_, k_)
589  else:
590  return _impl.insert(x, tck, m, per)
