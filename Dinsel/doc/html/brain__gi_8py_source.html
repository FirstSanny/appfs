25 def _gi_build_stub(parent):
26  """
27  Inspect the passed module recursively and build stubs for functions,
28  classes, etc.
29  """
30  classes = {}
31  functions = {}
32  constants = {}
33  methods = {}
34  for name in dir(parent):
35  if name.startswith("__"):
36  continue
37 
38  # Check if this is a valid name in python
39  if not re.match(_identifier_re, name):
40  continue
41 
42  try:
43  obj = getattr(parent, name)
44  except:
45  continue
46 
47  if inspect.isclass(obj):
48  classes[name] = obj
49  elif (inspect.isfunction(obj) or
50  inspect.isbuiltin(obj)):
51  functions[name] = obj
52  elif (inspect.ismethod(obj) or
53  inspect.ismethoddescriptor(obj)):
54  methods[name] = obj
55  elif (str(obj).startswith("<flags") or
56  str(obj).startswith("<enum ") or
57  str(obj).startswith("<GType ") or
58  inspect.isdatadescriptor(obj)):
59  constants[name] = 0
60  elif isinstance(obj, (int, str)):
61  constants[name] = obj
62  elif callable(obj):
63  # Fall back to a function for anything callable
64  functions[name] = obj
65  else:
66  # Assume everything else is some manner of constant
67  constants[name] = 0
68 
69  ret = ""
70 
71  if constants:
72  ret += "# %s constants\n\n" % parent.__name__
73  for name in sorted(constants):
74  if name[0].isdigit():
75  # GDK has some busted constant names like
76  # Gdk.EventType.2BUTTON_PRESS
77  continue
78 
79  val = constants[name]
80 
81  strval = str(val)
82  if isinstance(val, str):
83  strval = '"%s"' % str(val).replace("\\", "\\\\")
84  ret += "%s = %s\n" % (name, strval)
85 
86  if ret:
87  ret += "\n\n"
88  if functions:
89  ret += "# %s functions\n\n" % parent.__name__
90  for name in sorted(functions):
91  ret += "def %s(*args, **kwargs):\n" % name
92  ret += " pass\n"
93 
94  if ret:
95  ret += "\n\n"
96  if methods:
97  ret += "# %s methods\n\n" % parent.__name__
98  for name in sorted(methods):
99  ret += "def %s(self, *args, **kwargs):\n" % name
100  ret += " pass\n"
101 
102  if ret:
103  ret += "\n\n"
104  if classes:
105  ret += "# %s classes\n\n" % parent.__name__
106  for name in sorted(classes):
107  ret += "class %s(object):\n" % name
108 
109  classret = _gi_build_stub(classes[name])
110  if not classret:
111  classret = "pass\n"
112 
113  for line in classret.splitlines():
114  ret += " " + line + "\n"
115  ret += "\n"
116 
117  return ret

119 def _import_gi_module(modname):
120  # we only consider gi.repository submodules
121  if not modname.startswith('gi.repository.'):
122  raise AstroidBuildingError(modname=modname)
123  # build astroid representation unless we already tried so
124  if modname not in _inspected_modules:
125  modnames = [modname]
126  optional_modnames = []
127 
128  # GLib and GObject may have some special case handling
129  # in pygobject that we need to cope with. However at
130  # least as of pygobject3-3.13.91 the _glib module doesn't
131  # exist anymore, so if treat these modules as optional.
132  if modname == 'gi.repository.GLib':
133  optional_modnames.append('gi._glib')
134  elif modname == 'gi.repository.GObject':
135  optional_modnames.append('gi._gobject')
136 
137  try:
138  modcode = ''
139  for m in itertools.chain(modnames, optional_modnames):
140  try:
141  with warnings.catch_warnings():
142  # Just inspecting the code can raise gi deprecation
143  # warnings, so ignore them.
144  try:
145  from gi import PyGIDeprecationWarning, PyGIWarning
146  warnings.simplefilter("ignore", PyGIDeprecationWarning)
147  warnings.simplefilter("ignore", PyGIWarning)
148  except Exception:
149  pass
150 
151  __import__(m)
152  modcode += _gi_build_stub(sys.modules[m])
153  except ImportError:
154  if m not in optional_modnames:
155  raise
156  except ImportError:
157  astng = _inspected_modules[modname] = None
158  else:
159  astng = AstroidBuilder(MANAGER).string_build(modcode, modname)
160  _inspected_modules[modname] = astng
161  else:
162  astng = _inspected_modules[modname]
163  if astng is None:
164  raise AstroidBuildingError(modname=modname)
165  return astng

167 def _looks_like_require_version(node):
168  # Return whether this looks like a call to gi.require_version(<name>, <version>)
169  # Only accept function calls with two constant arguments
170  if len(node.args) != 2:
171  return False
172 
173  if not all(isinstance(arg, nodes.Const) for arg in node.args):
174  return False
175 
176  func = node.func
177  if isinstance(func, nodes.Attribute):
178  if func.attrname != 'require_version':
179  return False
180  if isinstance(func.expr, nodes.Name) and func.expr.name == 'gi':
181  return True
182 
183  return False
184 
185  if isinstance(func, nodes.Name):
186  return func.name == 'require_version'
187 
188  return False
