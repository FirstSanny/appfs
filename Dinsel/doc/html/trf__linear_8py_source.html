20 def regularized_lsq_with_qr(m, n, R, QTb, perm, diag, copy_R=True):
21  """Solve regularized least squares using information from QR-decomposition.
22 
23  The initial problem is to solve the following system in a least-squares
24  sense:
25  ::
26 
27  A x = b
28  D x = 0
29 
30  Where D is diagonal matrix. The method is based on QR decomposition
31  of the form A P = Q R, where P is a column permutation matrix, Q is an
32  orthogonal matrix and R is an upper triangular matrix.
33 
34  Parameters
35  ----------
36  m, n : int
37  Initial shape of A.
38  R : ndarray, shape (n, n)
39  Upper triangular matrix from QR decomposition of A.
40  QTb : ndarray, shape (n,)
41  First n components of Q^T b.
42  perm : ndarray, shape (n,)
43  Array defining column permutation of A, such that i-th column of
44  P is perm[i]-th column of identity matrix.
45  diag : ndarray, shape (n,)
46  Array containing diagonal elements of D.
47 
48  Returns
49  -------
50  x : ndarray, shape (n,)
51  Found least-squares solution.
52  """
53  if copy_R:
54  R = R.copy()
55  v = QTb.copy()
56 
57  givens_elimination(R, v, diag[perm])
58 
59  abs_diag_R = np.abs(np.diag(R))
60  threshold = EPS * max(m, n) * np.max(abs_diag_R)
61  nns, = np.nonzero(abs_diag_R > threshold)
62 
63  R = R[np.ix_(nns, nns)]
64  v = v[nns]
65 
66  x = np.zeros(n)
67  x[perm[nns]] = solve_triangular(R, v)
68 
69  return x

72 def backtracking(A, g, x, p, theta, p_dot_g, lb, ub):
73  """Find an appropriate step size using backtracking line search."""
74  alpha = 1
75  while True:
76  x_new, _ = reflective_transformation(x + alpha * p, lb, ub)
77  step = x_new - x
78  cost_change = -evaluate_quadratic(A, g, step)
79  if cost_change > -0.1 * alpha * p_dot_g:
80  break
81 
82  active = find_active_constraints(x_new, lb, ub)
83  if np.any(active != 0):
84  x_new, _ = reflective_transformation(x + theta * alpha * p, lb, ub)
85  x_new = make_strictly_feasible(x_new, lb, ub, rstep=0)
86  step = x_new - x
87  cost_change = -evaluate_quadratic(A, g, step)
88 
89  return x, step, cost_change

92 def select_step(x, A_h, g_h, c_h, p, p_h, d, lb, ub, theta):
93  """Select the best step according to Trust Region Reflective algorithm."""
94  if in_bounds(x + p, lb, ub):
95  return p
96 
97  p_stride, hits = step_size_to_bound(x, p, lb, ub)
98  r_h = np.copy(p_h)
99  r_h[hits.astype(bool)] *= -1
100  r = d * r_h
101 
102  # Restrict step, such that it hits the bound.
103  p *= p_stride
104  p_h *= p_stride
105  x_on_bound = x + p
106 
107  # Find the step size along reflected direction.
108  r_stride_u, _ = step_size_to_bound(x_on_bound, r, lb, ub)
109 
110  # Stay interior.
111  r_stride_l = (1 - theta) * r_stride_u
112  r_stride_u *= theta
113 
114  if r_stride_u > 0:
115  a, b, c = build_quadratic_1d(A_h, g_h, r_h, s0=p_h, diag=c_h)
116  r_stride, r_value = minimize_quadratic_1d(
117  a, b, r_stride_l, r_stride_u, c=c)
118  r_h = p_h + r_h * r_stride
119  r = d * r_h
120  else:
121  r_value = np.inf
122 
123  # Now correct p_h to make it strictly interior.
124  p_h *= theta
125  p *= theta
126  p_value = evaluate_quadratic(A_h, g_h, p_h, diag=c_h)
127 
128  ag_h = -g_h
129  ag = d * ag_h
130  ag_stride_u, _ = step_size_to_bound(x, ag, lb, ub)
131  ag_stride_u *= theta
132  a, b = build_quadratic_1d(A_h, g_h, ag_h, diag=c_h)
133  ag_stride, ag_value = minimize_quadratic_1d(a, b, 0, ag_stride_u)
134  ag *= ag_stride
135 
136  if p_value < r_value and p_value < ag_value:
137  return p
138  elif r_value < p_value and r_value < ag_value:
139  return r
140  else:
141  return ag

144 def trf_linear(A, b, x_lsq, lb, ub, tol, lsq_solver, lsmr_tol, max_iter,
145  verbose):

148  x = make_strictly_feasible(x, lb, ub, rstep=0.1)

164  elif lsmr_tol == 'auto':
165  auto_lsmr_tol = True

168  g = compute_grad(A, r)
169  cost = 0.5 * np.dot(r, r)
170  initial_cost = cost
171 
172  termination_status = None
173  step_norm = None
174  cost_change = None
175 
176  if max_iter is None:
177  max_iter = 100
178 
179  if verbose == 2:
180  print_header_linear()

182  for iteration in range(max_iter):
183  v, dv = CL_scaling_vector(x, g, lb, ub)
184  g_scaled = g * v
185  g_norm = norm(g_scaled, ord=np.inf)
186  if g_norm < tol:
187  termination_status = 1
188 
189  if verbose == 2:
190  print_iteration_linear(iteration, cost, cost_change,
191  step_norm, g_norm)

194  break

197  diag_root_h = diag_h ** 0.5
198  d = v ** 0.5

200

202  if lsq_solver == 'exact':
203  QTr[:k] = QT.dot(r)
204  p_h = -regularized_lsq_with_qr(m, n, R * d[perm], QTr, perm,
205  diag_root_h, copy_R=False)
206  elif lsq_solver == 'lsmr':
207  lsmr_op = regularized_lsq_operator(A_h, diag_root_h)
208  r_aug[:m] = r
209  if auto_lsmr_tol:
210  eta = 1e-2 * min(0.5, g_norm)
211  lsmr_tol = max(EPS, min(0.1, eta * g_norm))
212  p_h = -lsmr(lsmr_op, r_aug, atol=lsmr_tol, btol=lsmr_tol)[0]
213 
214  p = d * p_h
215 
216  p_dot_g = np.dot(p, g)
217  if p_dot_g > 0:
218  termination_status = -1
219 
220  theta = 1 - min(0.005, g_norm)
221  step = select_step(x, A_h, g_h, diag_h, p, p_h, d, lb, ub, theta)
222  cost_change = -evaluate_quadratic(A, g, step)
223 
224  # Perhaps almost never executed, the idea is that `p` is descent
225  # direction thus we must find acceptable cost decrease using simple
226  # "backtracking", otherwise algorithm's logic would break.
227  if cost_change < 0:
228  x, step, cost_change = backtracking(
229  A, g, x, p, theta, p_dot_g, lb, ub)
230  else:

232 
233  step_norm = norm(step)
234  r = A.dot(x) - b
235  g = compute_grad(A, r)
236 
237  if cost_change < tol * cost:
238  termination_status = 2
239 
240  cost = 0.5 * np.dot(r, r)
241 
242  if termination_status is None:
243  termination_status = 0
244 
245  active_mask = find_active_constraints(x, lb, ub, rtol=tol)
246 
247  return OptimizeResult(

249  nit=iteration + 1, status=termination_status,
