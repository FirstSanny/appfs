31 def hdquantiles(data, prob=list([.25,.5,.75]), axis=None, var=False,):
32  """
33  Computes quantile estimates with the Harrell-Davis method.
34 
35  The quantile estimates are calculated as a weighted linear combination
36  of order statistics.
37 
38  Parameters
39  ----------
40  data : array_like
41  Data array.
42  prob : sequence, optional
43  Sequence of quantiles to compute.
44  axis : int or None, optional
45  Axis along which to compute the quantiles. If None, use a flattened
46  array.
47  var : bool, optional
48  Whether to return the variance of the estimate.
49 
50  Returns
51  -------
52  hdquantiles : MaskedArray
53  A (p,) array of quantiles (if `var` is False), or a (2,p) array of
54  quantiles and variances (if `var` is True), where ``p`` is the
55  number of quantiles.
56 
57  """
58  def _hd_1D(data,prob,var):
59  "Computes the HD quantiles for a 1D array. Returns nan for invalid data."
60  xsorted = np.squeeze(np.sort(data.compressed().view(ndarray)))
61  # Don't use length here, in case we have a numpy scalar
62  n = xsorted.size
63 
64  hd = np.empty((2,len(prob)), float_)
65  if n < 2:
66  hd.flat = np.nan
67  if var:
68  return hd
69  return hd[0]
70 
71  v = np.arange(n+1) / float(n)
72  betacdf = beta.cdf
73  for (i,p) in enumerate(prob):
74  _w = betacdf(v, (n+1)*p, (n+1)*(1-p))
75  w = _w[1:] - _w[:-1]
76  hd_mean = np.dot(w, xsorted)
77  hd[0,i] = hd_mean
78  #
79  hd[1,i] = np.dot(w, (xsorted-hd_mean)**2)
80  #
81  hd[0, prob == 0] = xsorted[0]
82  hd[0, prob == 1] = xsorted[-1]
83  if var:
84  hd[1, prob == 0] = hd[1, prob == 1] = np.nan
85  return hd
86  return hd[0]
87  # Initialization & checks
88  data = ma.array(data, copy=False, dtype=float_)
89  p = np.array(prob, copy=False, ndmin=1)
90  # Computes quantiles along axis (or globally)
91  if (axis is None) or (data.ndim == 1):
92  result = _hd_1D(data, p, var)
93  else:
94  if data.ndim > 2:
95  raise ValueError("Array 'data' must be at most two dimensional, "
96  "but got data.ndim = %d" % data.ndim)
97  result = ma.apply_along_axis(_hd_1D, axis, data, p, var)
98 
99  return ma.fix_invalid(result, copy=False)

177 def trimmed_mean_ci(data, limits=(0.2,0.2), inclusive=(True,True),
178  alpha=0.05, axis=None):
179  """
180  Selected confidence interval of the trimmed mean along the given axis.
181 
182  Parameters
183  ----------
184  data : array_like
185  Input data.
186  limits : {None, tuple}, optional
187  None or a two item tuple.
188  Tuple of the percentages to cut on each side of the array, with respect
189  to the number of unmasked data, as floats between 0. and 1. If ``n``
190  is the number of unmasked data before trimming, then
191  (``n * limits[0]``)th smallest data and (``n * limits[1]``)th
192  largest data are masked. The total number of unmasked data after
193  trimming is ``n * (1. - sum(limits))``.
194  The value of one limit can be set to None to indicate an open interval.
195 
196  Defaults to (0.2, 0.2).
197  inclusive : (2,) tuple of boolean, optional
198  If relative==False, tuple indicating whether values exactly equal to
199  the absolute limits are allowed.
200  If relative==True, tuple indicating whether the number of data being
201  masked on each side should be rounded (True) or truncated (False).
202 
203  Defaults to (True, True).
204  alpha : float, optional
205  Confidence level of the intervals.
206 
207  Defaults to 0.05.
208  axis : int, optional
209  Axis along which to cut. If None, uses a flattened version of `data`.
210 
211  Defaults to None.
212 
213  Returns
214  -------
215  trimmed_mean_ci : (2,) ndarray
216  The lower and upper confidence intervals of the trimmed data.
217 
218  """
219  data = ma.array(data, copy=False)
220  trimmed = mstats.trimr(data, limits=limits, inclusive=inclusive, axis=axis)
221  tmean = trimmed.mean(axis)
222  tstde = mstats.trimmed_stde(data,limits=limits,inclusive=inclusive,axis=axis)
223  df = trimmed.count(axis) - 1
224  tppf = t.ppf(1-alpha/2.,df)
225  return np.array((tmean - tppf*tstde, tmean+tppf*tstde))

228 def mjci(data, prob=[0.25,0.5,0.75], axis=None):
229  """
230  Returns the Maritz-Jarrett estimators of the standard error of selected
231  experimental quantiles of the data.
232 
233  Parameters
234  ----------
235  data : ndarray
236  Data array.
237  prob : sequence, optional
238  Sequence of quantiles to compute.
239  axis : int or None, optional
240  Axis along which to compute the quantiles. If None, use a flattened
241  array.
242 
243  """
244  def _mjci_1D(data, p):
245  data = np.sort(data.compressed())
246  n = data.size
247  prob = (np.array(p) * n + 0.5).astype(int_)
248  betacdf = beta.cdf
249 
250  mj = np.empty(len(prob), float_)
251  x = np.arange(1,n+1, dtype=float_) / n
252  y = x - 1./n
253  for (i,m) in enumerate(prob):
254  W = betacdf(x,m-1,n-m) - betacdf(y,m-1,n-m)
255  C1 = np.dot(W,data)
256  C2 = np.dot(W,data**2)
257  mj[i] = np.sqrt(C2 - C1**2)
258  return mj
259 
260  data = ma.array(data, copy=False)
261  if data.ndim > 2:
262  raise ValueError("Array 'data' must be at most two dimensional, "
263  "but got data.ndim = %d" % data.ndim)
264 
265  p = np.array(prob, copy=False, ndmin=1)
266  # Computes quantiles along axis (or globally)
267  if (axis is None):
268  return _mjci_1D(data, p)
269  else:
270  return ma.apply_along_axis(_mjci_1D, axis, data, p)

273 def mquantiles_cimj(data, prob=[0.25,0.50,0.75], alpha=0.05, axis=None):
274  """
275  Computes the alpha confidence interval for the selected quantiles of the
276  data, with Maritz-Jarrett estimators.
277 
278  Parameters
279  ----------
280  data : ndarray
281  Data array.
282  prob : sequence, optional
283  Sequence of quantiles to compute.
284  alpha : float, optional
285  Confidence level of the intervals.
286  axis : int or None, optional
287  Axis along which to compute the quantiles.
288  If None, use a flattened array.
289 
290  """
291  alpha = min(alpha, 1-alpha)
292  z = norm.ppf(1-alpha/2.)
293  xq = mstats.mquantiles(data, prob, alphap=0, betap=0, axis=axis)

298 def median_cihs(data, alpha=0.05, axis=None):
299  """
300  Computes the alpha-level confidence interval for the median of the data.
301 
302  Uses the Hettmasperger-Sheather method.
303 
304  Parameters
305  ----------
306  data : array_like
307  Input data. Masked values are discarded. The input should be 1D only,
308  or `axis` should be set to None.
309  alpha : float, optional
310  Confidence level of the intervals.
311  axis : int or None, optional
312  Axis along which to compute the quantiles. If None, use a flattened
313  array.
314 
315  Returns
316  -------
317  median_cihs
318  Alpha level confidence interval.
319 
320  """
321  def _cihs_1D(data, alpha):
322  data = np.sort(data.compressed())
323  n = len(data)
324  alpha = min(alpha, 1-alpha)
325  k = int(binom._ppf(alpha/2., n, 0.5))
326  gk = binom.cdf(n-k,n,0.5) - binom.cdf(k-1,n,0.5)
327  if gk < 1-alpha:
328  k -= 1
329  gk = binom.cdf(n-k,n,0.5) - binom.cdf(k-1,n,0.5)
330  gkk = binom.cdf(n-k-1,n,0.5) - binom.cdf(k,n,0.5)
331  I = (gk - 1 + alpha)/(gk - gkk)
332  lambd = (n-k) * I / float(k + (n-2*k)*I)
333  lims = (lambd*data[k] + (1-lambd)*data[k-1],
334  lambd*data[n-k-1] + (1-lambd)*data[n-k])
335  return lims
336  data = ma.rray(data, copy=False)
337  # Computes quantiles along axis (or globally)
338  if (axis is None):
339  result = _cihs_1D(data.compressed(), alpha)
340  else:
341  if data.ndim > 2:
342  raise ValueError("Array 'data' must be at most two dimensional, "
343  "but got data.ndim = %d" % data.ndim)
344  result = ma.apply_along_axis(_cihs_1D, axis, data, alpha)
345 
346  return result

349 def compare_medians_ms(group_1, group_2, axis=None):
350  """
351  Compares the medians from two independent groups along the given axis.
352 
353  The comparison is performed using the McKean-Schrader estimate of the
354  standard error of the medians.
355 
356  Parameters
357  ----------
358  group_1 : array_like
359  First dataset.
360  group_2 : array_like
361  Second dataset.
362  axis : int, optional
363  Axis along which the medians are estimated. If None, the arrays are
364  flattened. If `axis` is not None, then `group_1` and `group_2`
365  should have the same shape.
366 
367  Returns
368  -------
369  compare_medians_ms : {float, ndarray}
370  If `axis` is None, then returns a float, otherwise returns a 1-D
371  ndarray of floats with a length equal to the length of `group_1`
372  along `axis`.
373 
374  """
375  (med_1, med_2) = (ma.median(group_1,axis=axis), ma.median(group_2,axis=axis))
376  (std_1, std_2) = (mstats.stde_median(group_1, axis=axis),
377  mstats.stde_median(group_2, axis=axis))
378  W = np.abs(med_1 - med_2) / ma.sqrt(std_1**2 + std_2**2)
379  return 1 - norm.cdf(W)

382 def idealfourths(data, axis=None):
383  """
384  Returns an estimate of the lower and upper quartiles.
385 
386  Uses the ideal fourths algorithm.
387 
388  Parameters
389  ----------
390  data : array_like
391  Input array.
392  axis : int, optional
393  Axis along which the quartiles are estimated. If None, the arrays are
394  flattened.
395 
396  Returns
397  -------
398  idealfourths : {list of floats, masked array}
399  Returns the two internal values that divide `data` into four parts
400  using the ideal fourths algorithm either along the flattened array
401  (if `axis` is None) or along `axis` of `data`.
402 
403  """
404  def _idf(data):
405  x = data.compressed()
406  n = len(x)
407  if n < 3:
408  return [np.nan,np.nan]
409  (j,h) = divmod(n/4. + 5/12.,1)
410  j = int(j)
411  qlo = (1-h)*x[j-1] + h*x[j]
412  k = n - j
413  qup = (1-h)*x[k] + h*x[k-1]
414  return [qlo, qup]
415  data = ma.sort(data, axis=axis).view(MaskedArray)
416  if (axis is None):
417  return _idf(data)
418  else:
419  return ma.apply_along_axis(_idf, axis, data)

422 def rsh(data, points=None):
423  """
424  Evaluates Rosenblatt's shifted histogram estimators for each point
425  on the dataset 'data'.
426 
427  Parameters
428  ----------
429  data : sequence
430  Input data. Masked values are ignored.
431  points : sequence or None, optional
432  Sequence of points where to evaluate Rosenblatt shifted histogram.
433  If None, use the data.
434 
435  """
436  data = ma.array(data, copy=False)
437  if points is None:
438  points = data
439  else:
440  points = np.array(points, copy=False, ndmin=1)
441 
442  if data.ndim != 1:
443  raise AttributeError("The input array should be 1D only !")
444 
445  n = data.count()
446  r = idealfourths(data, axis=None)
447  h = 1.2 * (r[-1]-r[0]) / n**(1./5)
448  nhi = (data[:,None] <= points[None,:] + h).sum(0)
449  nlo = (data[:,None] < points[None,:] - h).sum(0)
450  return (nhi-nlo) / (2.*n*h)
