188 def get_thunk_type_set():
189  """
190  Get a list containing cartesian product of data types, plus a getter routine.
191 
192  Returns
193  -------
194  i_types : list [(j, I_typenum, None, I_type, None), ...]
195  Pairing of index type numbers and the corresponding C++ types,
196  and an unique index `j`. This is for routines that are parameterized
197  only by I but not by T.
198  it_types : list [(j, I_typenum, T_typenum, I_type, T_type), ...]
199  Same as `i_types`, but for routines parameterized both by T and I.
200  getter_code : str
201  C++ code for a function that takes I_typenum, T_typenum and returns
202  the unique index corresponding to the lists, or -1 if no match was
203  found.
204 
205  """
206  it_types = []
207  i_types = []
208 
209  j = 0
210 
211  getter_code = " if (0) {}"
212 
213  for I_typenum, I_type in I_TYPES:
214  piece = """
215  else if (I_typenum == %(I_typenum)s) {
216  if (T_typenum == -1) { return %(j)s; }"""
217  getter_code += piece % dict(I_typenum=I_typenum, j=j)
218 
219  i_types.append((j, I_typenum, None, I_type, None))
220  j += 1
221 
222  for T_typenum, T_type in T_TYPES:
223  piece = """
224  else if (T_typenum == %(T_typenum)s) { return %(j)s; }"""
225  getter_code += piece % dict(T_typenum=T_typenum, j=j)
226 
227  it_types.append((j, I_typenum, T_typenum, I_type, T_type))
228  j += 1
229 
230  getter_code += """
231  }"""
232 
233  return i_types, it_types, GET_THUNK_CASE_TEMPLATE % dict(content=getter_code)

236 def parse_routine(name, args, types):
237  """
238  Generate thunk and method code for a given routine.
239 
240  Parameters
241  ----------
242  name : str
243  Name of the C++ routine
244  args : str
245  Argument list specification (in format explained above)
246  types : list
247  List of types to instantiate, as returned `get_thunk_type_set`
248 
249  """
250 
251  ret_spec = args[0]
252  arg_spec = args[1:]
253 
254  def get_arglist(I_type, T_type):
255  """
256  Generate argument list for calling the C++ function
257  """
258  args = []
259  next_is_writeable = False
260  j = 0
261  for t in arg_spec:
262  const = '' if next_is_writeable else 'const '
263  next_is_writeable = False
264  if t == '*':
265  next_is_writeable = True
266  continue
267  elif t == 'i':
268  args.append("*(%s*)a[%d]" % (const + I_type, j))
269  elif t == 'I':
270  args.append("(%s*)a[%d]" % (const + I_type, j))
271  elif t == 'T':
272  args.append("(%s*)a[%d]" % (const + T_type, j))
273  elif t == 'B':
274  args.append("(npy_bool_wrapper*)a[%d]" % (j,))
275  elif t == 'V':
276  if const:
277  raise ValueError("'V' argument must be an output arg")
278  args.append("(std::vector<%s>*)a[%d]" % (I_type, j,))
279  elif t == 'W':
280  if const:
281  raise ValueError("'W' argument must be an output arg")
282  args.append("(std::vector<%s>*)a[%d]" % (T_type, j,))
283  else:
284  raise ValueError("Invalid spec character %r" % (t,))
285  j += 1
286  return ", ".join(args)
287 
288  # Generate thunk code: a giant switch statement with different
289  # type combinations inside.
290  thunk_content = """int j = get_thunk_case(I_typenum, T_typenum);
291  switch (j) {"""
292  for j, I_typenum, T_typenum, I_type, T_type in types:
293  arglist = get_arglist(I_type, T_type)
294  if T_type is None:
295  dispatch = "%s" % (I_type,)
296  else:
297  dispatch = "%s,%s" % (I_type, T_type)
298  if 'B' in arg_spec:
299  dispatch += ",npy_bool_wrapper"
300 
301  piece = """
302  case %(j)s:"""
303  if ret_spec == 'v':
304  piece += """
305  (void)%(name)s<%(dispatch)s>(%(arglist)s);
306  return 0;"""
307  else:
308  piece += """
309  return %(name)s<%(dispatch)s>(%(arglist)s);"""
310  thunk_content += piece % dict(j=j, I_type=I_type, T_type=T_type,
311  I_typenum=I_typenum, T_typenum=T_typenum,
312  arglist=arglist, name=name,
313  dispatch=dispatch)
314 
315  thunk_content += """
316  default:
317  throw std::runtime_error("internal error: invalid argument typenums");
318  }"""
319 
320  thunk_code = THUNK_TEMPLATE % dict(name=name,
321  thunk_content=thunk_content)
322 
323  # Generate method code
324  method_code = METHOD_TEMPLATE % dict(name=name,
325  ret_spec=ret_spec,
326  arg_spec=arg_spec)
327 
328  return thunk_code, method_code
