268  def _inequality(self, other, op, op_name, bad_scalar_msg):
269  # Scalar other.
270  if isscalarlike(other):
271  if 0 == other and op_name in ('_le_', '_ge_'):
272  raise NotImplementedError(" >= and <= don't work with 0.")

385  def multiply(self, other):
386  """Point-wise multiplication by another matrix, vector, or
387  scalar.
388  """
389  # Scalar multiplication.
390  if isscalarlike(other):
391  return self._mul_scalar(other)
392  # Sparse matrix or vector.
393  if isspmatrix(other):
394  if self.shape == other.shape:
395  other = self.__class__(other)
396  return self._binopt(other, '_elmul_')
397  # Single element.
398  elif other.shape == (1,1):
399  return self._mul_scalar(other.toarray()[0, 0])
400  elif self.shape == (1,1):
401  return other._mul_scalar(self.toarray()[0, 0])
402  # A row times a column.
403  elif self.shape[1] == 1 and other.shape[0] == 1:
404  return self._mul_sparse_matrix(other.tocsc())
405  elif self.shape[0] == 1 and other.shape[1] == 1:
406  return other._mul_sparse_matrix(self.tocsc())
407  # Row vector times matrix. other is a row.
408  elif other.shape[0] == 1 and self.shape[1] == other.shape[1]:
409  other = dia_matrix((other.toarray().ravel(), [0]),
410  shape=(other.shape[1], other.shape[1]))
411  return self._mul_sparse_matrix(other)
412  # self is a row.
413  elif self.shape[0] == 1 and self.shape[1] == other.shape[1]:
414  copy = dia_matrix((self.toarray().ravel(), [0]),
415  shape=(self.shape[1], self.shape[1]))
416  return other._mul_sparse_matrix(copy)
417  # Column vector times matrix. other is a column.
418  elif other.shape[1] == 1 and self.shape[0] == other.shape[0]:
419  other = dia_matrix((other.toarray().ravel(), [0]),
420  shape=(other.shape[0], other.shape[0]))
421  return other._mul_sparse_matrix(self)
422  # self is a column.
423  elif self.shape[1] == 1 and self.shape[0] == other.shape[0]:
424  copy = dia_matrix((self.toarray().ravel(), [0]),
425  shape=(self.shape[0], self.shape[0]))
426  return copy._mul_sparse_matrix(other)
427  else:
428  raise ValueError("inconsistent shapes")
429 
430  # Assume other is a dense matrix/array, which produces a single-item
431  # object array if other isn't convertible to ndarray.
432  other = np.atleast_2d(other)
433 
434  if other.ndim != 2:
435  return np.multiply(self.toarray(), other)
436  # Single element / wrapped object.
437  if other.size == 1:
438  return self._mul_scalar(other.flat[0])
439  # Fast case for trivial sparse matrix.
440  elif self.shape == (1, 1):
441  return np.multiply(self.toarray()[0,0], other)
442 
443  from .coo import coo_matrix
444  ret = self.tocoo()
445  # Matching shapes.
446  if self.shape == other.shape:
447  data = np.multiply(ret.data, other[ret.row, ret.col])
448  # Sparse row vector times...
449  elif self.shape[0] == 1:
450  if other.shape[1] == 1: # Dense column vector.
451  data = np.multiply(ret.data, other)
452  elif other.shape[1] == self.shape[1]: # Dense matrix.
453  data = np.multiply(ret.data, other[:, ret.col])
454  else:
455  raise ValueError("inconsistent shapes")
456  row = np.repeat(np.arange(other.shape[0]), len(ret.row))
457  col = np.tile(ret.col, other.shape[0])
458  return coo_matrix((data.view(np.ndarray).ravel(), (row, col)),
459  shape=(other.shape[0], self.shape[1]),
460  copy=False)
461  # Sparse column vector times...
462  elif self.shape[1] == 1:
463  if other.shape[0] == 1: # Dense row vector.
464  data = np.multiply(ret.data[:, None], other)
465  elif other.shape[0] == self.shape[0]: # Dense matrix.
466  data = np.multiply(ret.data[:, None], other[ret.row])
467  else:
468  raise ValueError("inconsistent shapes")
469  row = np.repeat(ret.row, other.shape[1])
470  col = np.tile(np.arange(other.shape[1]), len(ret.col))
471  return coo_matrix((data.view(np.ndarray).ravel(), (row, col)),
472  shape=(self.shape[0], other.shape[1]),
473  copy=False)
474  # Sparse matrix times dense row vector.
475  elif other.shape[0] == 1 and self.shape[1] == other.shape[1]:
476  data = np.multiply(ret.data, other[:, ret.col].ravel())
477  # Sparse matrix times dense column vector.
478  elif other.shape[1] == 1 and self.shape[0] == other.shape[0]:
479  data = np.multiply(ret.data, other[ret.row].ravel())
480  else:
481  raise ValueError("inconsistent shapes")
482  ret.data = data.view(np.ndarray).ravel()
483  return ret
