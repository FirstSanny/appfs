204 def _precache_zipimporters(path=None):
205  pic = sys.path_importer_cache
206 
207  # When measured, despite having the same complexity (O(n)),
208  # converting to tuples and then caching the conversion to sets
209  # and the set difference is faster than converting to sets
210  # and then only caching the set difference.
211 
212  req_paths = tuple(path or sys.path)
213  cached_paths = tuple(pic)
214  new_paths = _cached_set_diff(req_paths, cached_paths)
215  for entry_path in new_paths:

249 def find_spec(modpath, path=None):
250  """Find a spec for the given module.
251 
252  :type modpath: list or tuple
253  :param modpath:
254  split module's name (i.e name of a module or package split
255  on '.'), with leading empty strings for explicit relative import
256 
257  :type path: list or None
258  :param path:
259  optional list of path where the module or package should be
260  searched (use sys.path if nothing or None is given)
261 
262  :rtype: ModuleSpec
263  :return: A module spec, which describes how the module was
264  found and where.
265  """
266  _path = path or sys.path
267 
268  # Need a copy for not mutating the argument.
269  modpath = modpath[:]
270 
271  submodule_path = None
272  module_parts = modpath[:]
273  processed = []
274 
275  while modpath:
276  modname = modpath.pop(0)
277  finder, spec = _find_spec_with_path(_path, modname,
278  module_parts, processed,
279  submodule_path or path)
280  processed.append(modname)
281  if modpath:
282  submodule_path = finder.contribute_to_path(spec, processed)
283 
284  if spec.type == ModuleType.PKG_DIRECTORY:
285  spec = spec._replace(submodule_search_locations=submodule_path)
286 
287  return spec
