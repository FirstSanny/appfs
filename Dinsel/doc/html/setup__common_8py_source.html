15 # How to change C_API_VERSION ?
16 # - increase C_API_VERSION value
17 # - record the hash for the new C API with the script cversions.py
18 # and add the hash to cversions.txt
19 # The hash values are used to remind developers when the C API number was not
20 # updated - generates a MismatchCAPIWarning warning which is turned into an
21 # exception for released version.
22 
23 # Binary compatibility version number. This number is increased whenever the
24 # C-API is changed such that binary compatibility is broken, i.e. whenever a
25 # recompile of extension modules is needed.
26 C_ABI_VERSION = 0x01000009
27 
28 # Minor API version. This number is increased whenever a change is made to the
29 # C-API -- whether it breaks binary compatibility or not. Some changes, such
30 # as adding a function pointer to the end of the function table, can be made
31 # without breaking binary compatibility. In this case, only the C_API_VERSION
32 # (*not* C_ABI_VERSION) would be increased. Whenever binary compatibility is
33 # broken, both C_API_VERSION and C_ABI_VERSION should be increased.
34 #
35 # 0x00000008 - 1.7.x
36 # 0x00000009 - 1.8.x
37 # 0x00000009 - 1.9.x
38 # 0x0000000a - 1.10.x
39 # 0x0000000a - 1.11.x
40 # 0x0000000a - 1.12.x
41 C_API_VERSION = 0x0000000a

248 def pyod(filename):
249  """Python implementation of the od UNIX utility (od -b, more exactly).
250 
251  Parameters
252  ----------
253  filename : str
254  name of the file to get the dump from.
255 
256  Returns
257  -------
258  out : seq
259  list of lines of od output
260 
261  Note
262  ----
263  We only implement enough to get the necessary information for long double
264  representation, this is not intended as a compatible replacement for od.
265  """
266  def _pyod2():
267  out = []
268 
269  fid = open(filename, 'rb')
270  try:
271  yo = [int(oct(int(binascii.b2a_hex(o), 16))) for o in fid.read()]
272  for i in range(0, len(yo), 16):
273  line = ['%07d' % int(oct(i))]
274  line.extend(['%03d' % c for c in yo[i:i+16]])
275  out.append(" ".join(line))
276  return out
277  finally:
278  fid.close()
279 
280  def _pyod3():
281  out = []
282 
283  fid = open(filename, 'rb')
284  try:
285  yo2 = [oct(o)[2:] for o in fid.read()]
286  for i in range(0, len(yo2), 16):
287  line = ['%07d' % int(oct(i)[2:])]
288  line.extend(['%03d' % int(c) for c in yo2[i:i+16]])
289  out.append(" ".join(line))
290  return out
291  finally:
292  fid.close()
293 
294  if sys.version_info[0] < 3:
295  return _pyod2()
296  else:
297  return _pyod3()

319 def long_double_representation(lines):
320  """Given a binary dump as given by GNU od -b, look for long double
321  representation."""
322 
323  # Read contains a list of 32 items, each item is a byte (in octal
324  # representation, as a string). We 'slide' over the output until read is of
325  # the form before_seq + content + after_sequence, where content is the long double
326  # representation:
327  # - content is 12 bytes: 80 bits Intel representation
328  # - content is 16 bytes: 80 bits Intel representation (64 bits) or quad precision
329  # - content is 8 bytes: same as double (not implemented yet)
330  read = [''] * 32
331  saw = None
332  for line in lines:
333  # we skip the first word, as od -b output an index at the beginning of
334  # each line
335  for w in line.split()[1:]:
336  read.pop(0)
337  read.append(w)
338 
339  # If the end of read is equal to the after_sequence, read contains
340  # the long double
341  if read[-8:] == _AFTER_SEQ:
342  saw = copy.copy(read)
343  if read[:12] == _BEFORE_SEQ[4:]:
344  if read[12:-8] == _INTEL_EXTENDED_12B:
345  return 'INTEL_EXTENDED_12_BYTES_LE'
346  if read[12:-8] == _MOTOROLA_EXTENDED_12B:
347  return 'MOTOROLA_EXTENDED_12_BYTES_BE'
348  elif read[:8] == _BEFORE_SEQ[8:]:
349  if read[8:-8] == _INTEL_EXTENDED_16B:
350  return 'INTEL_EXTENDED_16_BYTES_LE'
351  elif read[8:-8] == _IEEE_QUAD_PREC_BE:
352  return 'IEEE_QUAD_BE'
353  elif read[8:-8] == _IEEE_QUAD_PREC_LE:
354  return 'IEEE_QUAD_LE'
355  elif read[8:-8] == _DOUBLE_DOUBLE_BE:
356  return 'DOUBLE_DOUBLE_BE'
357  elif read[8:-8] == _DOUBLE_DOUBLE_LE:
358  return 'DOUBLE_DOUBLE_LE'
359  elif read[:16] == _BEFORE_SEQ:
360  if read[16:-8] == _IEEE_DOUBLE_LE:
361  return 'IEEE_DOUBLE_LE'
362  elif read[16:-8] == _IEEE_DOUBLE_BE:
363  return 'IEEE_DOUBLE_BE'
364 
365  if saw is not None:
366  raise ValueError("Unrecognized format (%s)" % saw)
367  else:
368  # We never detected the after_sequence
369  raise ValueError("Could not lock sequences (%s)" % saw)
