76  def evaluate(self, node, filename=None):
77  """
78  Evaluate a source string or node, using ``filename`` when
79  displaying errors.
80  """
81  if isinstance(node, string_types):
82  self.source = node
83  kwargs = {'mode': 'eval'}
84  if filename:
85  kwargs['filename'] = filename
86  try:
87  node = ast.parse(node, **kwargs)
88  except SyntaxError as e:
89  s = self.get_fragment(e.offset)
90  raise SyntaxError('syntax error %s' % s)
91  node_type = node.__class__.__name__.lower()
92  handler = self.get_handler(node_type)
93  if handler is None:
94  if self.source is None:
95  s = '(source not available)'
96  else:
97  s = self.get_fragment(node.col_offset)
98  raise SyntaxError("don't know how to evaluate %r %s" % (
99  node_type, s))
100  return handler(node)
