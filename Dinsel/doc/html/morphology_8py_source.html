47 def _center_is_true(structure, origin):
48  structure = numpy.array(structure)
49  coor = tuple([oo + ss // 2 for ss, oo in zip(structure.shape,
50  origin)])
51  return bool(structure[coor])

54 def iterate_structure(structure, iterations, origin=None):
55  """
56  Iterate a structure by dilating it with itself.
57 
58  Parameters
59  ----------
60  structure : array_like
61  Structuring element (an array of bools, for example), to be dilated with
62  itself.
63  iterations : int
64  number of dilations performed on the structure with itself
65  origin : optional
66  If origin is None, only the iterated structure is returned. If
67  not, a tuple of the iterated structure and the modified origin is
68  returned.
69 
70  Returns
71  -------
72  iterate_structure : ndarray of bools
73  A new structuring element obtained by dilating `structure`
74  (`iterations` - 1) times with itself.
75 
76  See also
77  --------
78  generate_binary_structure
79 
80  Examples
81  --------
82  >>> from scipy import ndimage
83  >>> struct = ndimage.generate_binary_structure(2, 1)
84  >>> struct.astype(int)
85  array([[0, 1, 0],
86  [1, 1, 1],
87  [0, 1, 0]])
88  >>> ndimage.iterate_structure(struct, 2).astype(int)
89  array([[0, 0, 1, 0, 0],
90  [0, 1, 1, 1, 0],
91  [1, 1, 1, 1, 1],
92  [0, 1, 1, 1, 0],
93  [0, 0, 1, 0, 0]])
94  >>> ndimage.iterate_structure(struct, 3).astype(int)
95  array([[0, 0, 0, 1, 0, 0, 0],
96  [0, 0, 1, 1, 1, 0, 0],
97  [0, 1, 1, 1, 1, 1, 0],
98  [1, 1, 1, 1, 1, 1, 1],
99  [0, 1, 1, 1, 1, 1, 0],
100  [0, 0, 1, 1, 1, 0, 0],
101  [0, 0, 0, 1, 0, 0, 0]])
102 
103  """
104  structure = numpy.asarray(structure)
105  if iterations < 2:
106  return structure.copy()
107  ni = iterations - 1
108  shape = [ii + ni * (ii - 1) for ii in structure.shape]
109  pos = [ni * (structure.shape[ii] // 2) for ii in range(len(shape))]
110  slc = [slice(pos[ii], pos[ii] + structure.shape[ii], None)
111  for ii in range(len(shape))]
112  out = numpy.zeros(shape, bool)
113  out[slc] = structure != 0
114  out = binary_dilation(out, structure, iterations=ni)
115  if origin is None:
116  return out
117  else:
118  origin = _ni_support._normalize_sequence(origin, structure.ndim)
119  origin = [iterations * o for o in origin]
120  return out, origin

123 def generate_binary_structure(rank, connectivity):
124  """
125  Generate a binary structure for binary morphological operations.
126 
127  Parameters
128  ----------
129  rank : int
130  Number of dimensions of the array to which the structuring element
131  will be applied, as returned by `np.ndim`.
132  connectivity : int
133  `connectivity` determines which elements of the output array belong
134  to the structure, i.e. are considered as neighbors of the central
135  element. Elements up to a squared distance of `connectivity` from
136  the center are considered neighbors. `connectivity` may range from 1
137  (no diagonal elements are neighbors) to `rank` (all elements are
138  neighbors).
139 
140  Returns
141  -------
142  output : ndarray of bools
143  Structuring element which may be used for binary morphological
144  operations, with `rank` dimensions and all dimensions equal to 3.
145 
146  See also
147  --------
148  iterate_structure, binary_dilation, binary_erosion
149 
150  Notes
151  -----
152  `generate_binary_structure` can only create structuring elements with
153  dimensions equal to 3, i.e. minimal dimensions. For larger structuring
154  elements, that are useful e.g. for eroding large objects, one may either
155  use `iterate_structure`, or create directly custom arrays with
156  numpy functions such as `numpy.ones`.
157 
158  Examples
159  --------
160  >>> from scipy import ndimage
161  >>> struct = ndimage.generate_binary_structure(2, 1)
162  >>> struct
163  array([[False, True, False],
164  [ True, True, True],
165  [False, True, False]], dtype=bool)
166  >>> a = np.zeros((5,5))
167  >>> a[2, 2] = 1
168  >>> a
169  array([[ 0., 0., 0., 0., 0.],
170  [ 0., 0., 0., 0., 0.],
171  [ 0., 0., 1., 0., 0.],
172  [ 0., 0., 0., 0., 0.],
173  [ 0., 0., 0., 0., 0.]])
174  >>> b = ndimage.binary_dilation(a, structure=struct).astype(a.dtype)
175  >>> b
176  array([[ 0., 0., 0., 0., 0.],
177  [ 0., 0., 1., 0., 0.],
178  [ 0., 1., 1., 1., 0.],
179  [ 0., 0., 1., 0., 0.],
180  [ 0., 0., 0., 0., 0.]])
181  >>> ndimage.binary_dilation(b, structure=struct).astype(a.dtype)
182  array([[ 0., 0., 1., 0., 0.],
183  [ 0., 1., 1., 1., 0.],
184  [ 1., 1., 1., 1., 1.],
185  [ 0., 1., 1., 1., 0.],
186  [ 0., 0., 1., 0., 0.]])
187  >>> struct = ndimage.generate_binary_structure(2, 2)
188  >>> struct
189  array([[ True, True, True],
190  [ True, True, True],
191  [ True, True, True]], dtype=bool)
192  >>> struct = ndimage.generate_binary_structure(3, 1)
193  >>> struct # no diagonal elements
194  array([[[False, False, False],
195  [False, True, False],
196  [False, False, False]],
197  [[False, True, False],
198  [ True, True, True],
199  [False, True, False]],
200  [[False, False, False],
201  [False, True, False],
202  [False, False, False]]], dtype=bool)
203 
204  """
205  if connectivity < 1:
206  connectivity = 1
207  if rank < 1:
208  if connectivity < 1:
209  return numpy.array(0, dtype=bool)
210  else:
211  return numpy.array(1, dtype=bool)
212  output = numpy.fabs(numpy.indices([3] * rank) - 1)
213  output = numpy.add.reduce(output, 0)
214  return numpy.asarray(output <= connectivity, dtype=bool)

217 def _binary_erosion(input, structure, iterations, mask, output,
218  border_value, origin, invert, brute_force):
219  input = numpy.asarray(input)
220  if numpy.iscomplexobj(input):
221  raise TypeError('Complex type not supported')
222  if structure is None:
223  structure = generate_binary_structure(input.ndim, 1)
224  else:
225  structure = numpy.asarray(structure)
226  structure = structure.astype(bool)
227  if structure.ndim != input.ndim:
228  raise RuntimeError('structure and input must have same dimensionality')
229  if not structure.flags.contiguous:
230  structure = structure.copy()
231  if numpy.product(structure.shape,axis=0) < 1:
232  raise RuntimeError('structure must not be empty')
233  if mask is not None:
234  mask = numpy.asarray(mask)
235  if mask.shape != input.shape:
236  raise RuntimeError('mask and input must have equal sizes')
237  origin = _ni_support._normalize_sequence(origin, input.ndim)
238  cit = _center_is_true(structure, origin)
239  if isinstance(output, numpy.ndarray):
240  if numpy.iscomplexobj(output):
241  raise TypeError('Complex output type not supported')
242  else:
243  output = bool
244  output, return_value = _ni_support._get_output(output, input)
245 
246  if iterations == 1:
247  _nd_image.binary_erosion(input, structure, mask, output,
248  border_value, origin, invert, cit, 0)
249  return return_value
250  elif cit and not brute_force:
251  changed, coordinate_list = _nd_image.binary_erosion(input,
252  structure, mask, output, border_value, origin, invert, cit, 1)
253  structure = structure[tuple([slice(None, None, -1)] *
254  structure.ndim)]
255  for ii in range(len(origin)):
256  origin[ii] = -origin[ii]
257  if not structure.shape[ii] & 1:
258  origin[ii] -= 1
259  if mask is not None:
260  msk = numpy.asarray(mask)
261  msk = mask.astype(numpy.int8)
262  if msk is mask:
263  msk = mask.copy()
264  mask = msk
265  if not structure.flags.contiguous:
266  structure = structure.copy()
267  _nd_image.binary_erosion2(output, structure, mask, iterations - 1,
268  origin, invert, coordinate_list)
269  return return_value
270  else:
271  tmp_in = numpy.zeros(input.shape, bool)
272  if return_value is None:
273  tmp_out = output
274  else:
275  tmp_out = numpy.zeros(input.shape, bool)
276  if not iterations & 1:
277  tmp_in, tmp_out = tmp_out, tmp_in
278  changed = _nd_image.binary_erosion(input, structure, mask,
279  tmp_out, border_value, origin, invert, cit, 0)
280  ii = 1
281  while (ii < iterations) or (iterations < 1) and changed:
282  tmp_in, tmp_out = tmp_out, tmp_in
283  changed = _nd_image.binary_erosion(tmp_in, structure, mask,
284  tmp_out, border_value, origin, invert, cit, 0)
285  ii += 1
286  if return_value is not None:
287  return tmp_out

290 def binary_erosion(input, structure=None, iterations=1, mask=None,
291  output=None, border_value=0, origin=0, brute_force=False):
292  """
293  Multi-dimensional binary erosion with a given structuring element.
294 
295  Binary erosion is a mathematical morphology operation used for image
296  processing.
297 
298  Parameters
299  ----------
300  input : array_like
301  Binary image to be eroded. Non-zero (True) elements form
302  the subset to be eroded.
303  structure : array_like, optional
304  Structuring element used for the erosion. Non-zero elements are
305  considered True. If no structuring element is provided, an element
306  is generated with a square connectivity equal to one.
307  iterations : {int, float}, optional
308  The erosion is repeated `iterations` times (one, by default).
309  If iterations is less than 1, the erosion is repeated until the
310  result does not change anymore.
311  mask : array_like, optional
312  If a mask is given, only those elements with a True value at
313  the corresponding mask element are modified at each iteration.
314  output : ndarray, optional
315  Array of the same shape as input, into which the output is placed.
316  By default, a new array is created.
317  origin : int or tuple of ints, optional
318  Placement of the filter, by default 0.
319  border_value : int (cast to 0 or 1), optional
320  Value at the border in the output array.
321 
322  Returns
323  -------
324  binary_erosion : ndarray of bools
325  Erosion of the input by the structuring element.
326 
327  See also
328  --------
329  grey_erosion, binary_dilation, binary_closing, binary_opening,
330  generate_binary_structure
331 
332  Notes
333  -----
334  Erosion [1]_ is a mathematical morphology operation [2]_ that uses a
335  structuring element for shrinking the shapes in an image. The binary
336  erosion of an image by a structuring element is the locus of the points
337  where a superimposition of the structuring element centered on the point
338  is entirely contained in the set of non-zero elements of the image.
339 
340  References
341  ----------
342  .. [1] http://en.wikipedia.org/wiki/Erosion_%28morphology%29
343  .. [2] http://en.wikipedia.org/wiki/Mathematical_morphology
344 
345  Examples
346  --------
347  >>> from scipy import ndimage
348  >>> a = np.zeros((7,7), dtype=int)
349  >>> a[1:6, 2:5] = 1
350  >>> a
351  array([[0, 0, 0, 0, 0, 0, 0],
352  [0, 0, 1, 1, 1, 0, 0],
353  [0, 0, 1, 1, 1, 0, 0],
354  [0, 0, 1, 1, 1, 0, 0],
355  [0, 0, 1, 1, 1, 0, 0],
356  [0, 0, 1, 1, 1, 0, 0],
357  [0, 0, 0, 0, 0, 0, 0]])
358  >>> ndimage.binary_erosion(a).astype(a.dtype)
359  array([[0, 0, 0, 0, 0, 0, 0],
360  [0, 0, 0, 0, 0, 0, 0],
361  [0, 0, 0, 1, 0, 0, 0],
362  [0, 0, 0, 1, 0, 0, 0],
363  [0, 0, 0, 1, 0, 0, 0],
364  [0, 0, 0, 0, 0, 0, 0],
365  [0, 0, 0, 0, 0, 0, 0]])
366  >>> #Erosion removes objects smaller than the structure
367  >>> ndimage.binary_erosion(a, structure=np.ones((5,5))).astype(a.dtype)
368  array([[0, 0, 0, 0, 0, 0, 0],
369  [0, 0, 0, 0, 0, 0, 0],
370  [0, 0, 0, 0, 0, 0, 0],
371  [0, 0, 0, 0, 0, 0, 0],
372  [0, 0, 0, 0, 0, 0, 0],
373  [0, 0, 0, 0, 0, 0, 0],
374  [0, 0, 0, 0, 0, 0, 0]])
375 
376  """
377  return _binary_erosion(input, structure, iterations, mask,
378  output, border_value, origin, 0, brute_force)

381 def binary_dilation(input, structure=None, iterations=1, mask=None,
382  output=None, border_value=0, origin=0, brute_force=False):
383  """
384  Multi-dimensional binary dilation with the given structuring element.
385 
386  Parameters
387  ----------
388  input : array_like
389  Binary array_like to be dilated. Non-zero (True) elements form
390  the subset to be dilated.
391  structure : array_like, optional
392  Structuring element used for the dilation. Non-zero elements are
393  considered True. If no structuring element is provided an element
394  is generated with a square connectivity equal to one.
395  iterations : {int, float}, optional
396  The dilation is repeated `iterations` times (one, by default).
397  If iterations is less than 1, the dilation is repeated until the
398  result does not change anymore.
399  mask : array_like, optional
400  If a mask is given, only those elements with a True value at
401  the corresponding mask element are modified at each iteration.
402  output : ndarray, optional
403  Array of the same shape as input, into which the output is placed.
404  By default, a new array is created.
405  origin : int or tuple of ints, optional
406  Placement of the filter, by default 0.
407  border_value : int (cast to 0 or 1), optional
408  Value at the border in the output array.
409 
410  Returns
411  -------
412  binary_dilation : ndarray of bools
413  Dilation of the input by the structuring element.
414 
415  See also
416  --------
417  grey_dilation, binary_erosion, binary_closing, binary_opening,
418  generate_binary_structure
419 
420  Notes
421  -----
422  Dilation [1]_ is a mathematical morphology operation [2]_ that uses a
423  structuring element for expanding the shapes in an image. The binary
424  dilation of an image by a structuring element is the locus of the points
425  covered by the structuring element, when its center lies within the
426  non-zero points of the image.
427 
428  References
429  ----------
430  .. [1] http://en.wikipedia.org/wiki/Dilation_%28morphology%29
431  .. [2] http://en.wikipedia.org/wiki/Mathematical_morphology
432 
433  Examples
434  --------
435  >>> from scipy import ndimage
436  >>> a = np.zeros((5, 5))
437  >>> a[2, 2] = 1
438  >>> a
439  array([[ 0., 0., 0., 0., 0.],
440  [ 0., 0., 0., 0., 0.],
441  [ 0., 0., 1., 0., 0.],
442  [ 0., 0., 0., 0., 0.],
443  [ 0., 0., 0., 0., 0.]])
444  >>> ndimage.binary_dilation(a)
445  array([[False, False, False, False, False],
446  [False, False, True, False, False],
447  [False, True, True, True, False],
448  [False, False, True, False, False],
449  [False, False, False, False, False]], dtype=bool)
450  >>> ndimage.binary_dilation(a).astype(a.dtype)
451  array([[ 0., 0., 0., 0., 0.],
452  [ 0., 0., 1., 0., 0.],
453  [ 0., 1., 1., 1., 0.],
454  [ 0., 0., 1., 0., 0.],
455  [ 0., 0., 0., 0., 0.]])
456  >>> # 3x3 structuring element with connectivity 1, used by default
457  >>> struct1 = ndimage.generate_binary_structure(2, 1)
458  >>> struct1
459  array([[False, True, False],
460  [ True, True, True],
461  [False, True, False]], dtype=bool)
462  >>> # 3x3 structuring element with connectivity 2
463  >>> struct2 = ndimage.generate_binary_structure(2, 2)
464  >>> struct2
465  array([[ True, True, True],
466  [ True, True, True],
467  [ True, True, True]], dtype=bool)
468  >>> ndimage.binary_dilation(a, structure=struct1).astype(a.dtype)
469  array([[ 0., 0., 0., 0., 0.],
470  [ 0., 0., 1., 0., 0.],
471  [ 0., 1., 1., 1., 0.],
472  [ 0., 0., 1., 0., 0.],
473  [ 0., 0., 0., 0., 0.]])
474  >>> ndimage.binary_dilation(a, structure=struct2).astype(a.dtype)
475  array([[ 0., 0., 0., 0., 0.],
476  [ 0., 1., 1., 1., 0.],
477  [ 0., 1., 1., 1., 0.],
478  [ 0., 1., 1., 1., 0.],
479  [ 0., 0., 0., 0., 0.]])
480  >>> ndimage.binary_dilation(a, structure=struct1,\\
481  ... iterations=2).astype(a.dtype)
482  array([[ 0., 0., 1., 0., 0.],
483  [ 0., 1., 1., 1., 0.],
484  [ 1., 1., 1., 1., 1.],
485  [ 0., 1., 1., 1., 0.],
486  [ 0., 0., 1., 0., 0.]])
487 
488  """
489  input = numpy.asarray(input)
490  if structure is None:
491  structure = generate_binary_structure(input.ndim, 1)
492  origin = _ni_support._normalize_sequence(origin, input.ndim)
493  structure = numpy.asarray(structure)
494  structure = structure[tuple([slice(None, None, -1)] *
495  structure.ndim)]
496  for ii in range(len(origin)):
497  origin[ii] = -origin[ii]
498  if not structure.shape[ii] & 1:
499  origin[ii] -= 1
500 
501  return _binary_erosion(input, structure, iterations, mask,
502  output, border_value, origin, 1, brute_force)

505 def binary_opening(input, structure=None, iterations=1, output=None,
506  origin=0):
507  """
508  Multi-dimensional binary opening with the given structuring element.
509 
510  The *opening* of an input image by a structuring element is the
511  *dilation* of the *erosion* of the image by the structuring element.
512 
513  Parameters
514  ----------
515  input : array_like
516  Binary array_like to be opened. Non-zero (True) elements form
517  the subset to be opened.
518  structure : array_like, optional
519  Structuring element used for the opening. Non-zero elements are
520  considered True. If no structuring element is provided an element
521  is generated with a square connectivity equal to one (i.e., only
522  nearest neighbors are connected to the center, diagonally-connected
523  elements are not considered neighbors).
524  iterations : {int, float}, optional
525  The erosion step of the opening, then the dilation step are each
526  repeated `iterations` times (one, by default). If `iterations` is
527  less than 1, each operation is repeated until the result does
528  not change anymore.
529  output : ndarray, optional
530  Array of the same shape as input, into which the output is placed.
531  By default, a new array is created.
532  origin : int or tuple of ints, optional
533  Placement of the filter, by default 0.
534 
535  Returns
536  -------
537  binary_opening : ndarray of bools
538  Opening of the input by the structuring element.
539 
540  See also
541  --------
542  grey_opening, binary_closing, binary_erosion, binary_dilation,
543  generate_binary_structure
544 
545  Notes
546  -----
547  *Opening* [1]_ is a mathematical morphology operation [2]_ that
548  consists in the succession of an erosion and a dilation of the
549  input with the same structuring element. Opening therefore removes
550  objects smaller than the structuring element.
551 
552  Together with *closing* (`binary_closing`), opening can be used for
553  noise removal.
554 
555  References
556  ----------
557  .. [1] http://en.wikipedia.org/wiki/Opening_%28morphology%29
558  .. [2] http://en.wikipedia.org/wiki/Mathematical_morphology
559 
560  Examples
561  --------
562  >>> from scipy import ndimage
563  >>> a = np.zeros((5,5), dtype=int)
564  >>> a[1:4, 1:4] = 1; a[4, 4] = 1
565  >>> a
566  array([[0, 0, 0, 0, 0],
567  [0, 1, 1, 1, 0],
568  [0, 1, 1, 1, 0],
569  [0, 1, 1, 1, 0],
570  [0, 0, 0, 0, 1]])
571  >>> # Opening removes small objects
572  >>> ndimage.binary_opening(a, structure=np.ones((3,3))).astype(int)
573  array([[0, 0, 0, 0, 0],
574  [0, 1, 1, 1, 0],
575  [0, 1, 1, 1, 0],
576  [0, 1, 1, 1, 0],
577  [0, 0, 0, 0, 0]])
578  >>> # Opening can also smooth corners
579  >>> ndimage.binary_opening(a).astype(int)
580  array([[0, 0, 0, 0, 0],
581  [0, 0, 1, 0, 0],
582  [0, 1, 1, 1, 0],
583  [0, 0, 1, 0, 0],
584  [0, 0, 0, 0, 0]])
585  >>> # Opening is the dilation of the erosion of the input
586  >>> ndimage.binary_erosion(a).astype(int)
587  array([[0, 0, 0, 0, 0],
588  [0, 0, 0, 0, 0],
589  [0, 0, 1, 0, 0],
590  [0, 0, 0, 0, 0],
591  [0, 0, 0, 0, 0]])
592  >>> ndimage.binary_dilation(ndimage.binary_erosion(a)).astype(int)
593  array([[0, 0, 0, 0, 0],
594  [0, 0, 1, 0, 0],
595  [0, 1, 1, 1, 0],
596  [0, 0, 1, 0, 0],
597  [0, 0, 0, 0, 0]])
598 
599  """
600  input = numpy.asarray(input)
601  if structure is None:
602  rank = input.ndim
603  structure = generate_binary_structure(rank, 1)
604 
605  tmp = binary_erosion(input, structure, iterations, None, None, 0,
606  origin)
607  return binary_dilation(tmp, structure, iterations, None, output, 0,
608  origin)

978 def binary_fill_holes(input, structure=None, output=None, origin=0):

980  Fill the holes in binary objects.
981 
982 
983  Parameters
984  ----------
985  input : array_like
986  n-dimensional binary array with holes to be filled
987  structure : array_like, optional
988  Structuring element used in the computation; large-size elements
989  make computations faster but may miss holes separated from the
990  background by thin regions. The default element (with a square
991  connectivity equal to one) yields the intuitive result where all
992  holes in the input have been filled.
993  output : ndarray, optional
994  Array of the same shape as input, into which the output is placed.
995  By default, a new array is created.
996  origin : int, tuple of ints, optional
997  Position of the structuring element.
998
