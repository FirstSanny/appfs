150 def _format_marker(marker, first=True):
151  assert isinstance(marker, (list, tuple, string_types))
152 
153  # Sometimes we have a structure like [[...]] which is a single item list
154  # where the single item is itself it's own list. In that case we want skip
155  # the rest of this function so that we don't get extraneous () on the
156  # outside.
157  if (isinstance(marker, list) and len(marker) == 1 and
158  isinstance(marker[0], (list, tuple))):
159  return _format_marker(marker[0])
160 
161  if isinstance(marker, list):
162  inner = (_format_marker(m, first=False) for m in marker)
163  if first:
164  return " ".join(inner)
165  else:
166  return "(" + " ".join(inner) + ")"
167  elif isinstance(marker, tuple):
168  return " ".join([m.serialize() for m in marker])
169  else:
170  return marker

185 def _eval_op(lhs, op, rhs):
186  try:
187  spec = Specifier("".join([op.serialize(), rhs]))
188  except InvalidSpecifier:
189  pass
190  else:
191  return spec.contains(lhs)

216 def _evaluate_markers(markers, environment):
217  groups = [[]]
218 
219  for marker in markers:
220  assert isinstance(marker, (list, tuple, string_types))
221 
222  if isinstance(marker, list):
223  groups[-1].append(_evaluate_markers(marker, environment))
224  elif isinstance(marker, tuple):
225  lhs, op, rhs = marker
226 
227  if isinstance(lhs, Variable):
228  lhs_value = _get_env(environment, lhs.value)
229  rhs_value = rhs.value
230  else:
231  lhs_value = lhs.value
232  rhs_value = _get_env(environment, rhs.value)
233 
234  groups[-1].append(_eval_op(lhs_value, op, rhs_value))
235  else:
236  assert marker in ["and", "or"]
237  if marker == "or":
238  groups.append([])
239 
240  return any(all(item) for item in groups)
