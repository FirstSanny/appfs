329  def _wrap(self, line):
330  """
331  Returns an import wrapped to the specified line-length, if possible.
332  """
333  wrap_mode = self.config['multi_line_output']
334  if len(line) > self.config['line_length'] and wrap_mode != settings.WrapModes.NOQA:
335  for splitter in ("import", ".", "as"):
336  exp = r"\b" + re.escape(splitter) + r"\b"
337  if re.search(exp, line) and not line.strip().startswith(splitter):
338  line_parts = re.split(exp, line)
339  next_line = []
340  while (len(line) + 2) > (self.config['wrap_length'] or self.config['line_length']) and line_parts:
341  next_line.append(line_parts.pop())
342  line = splitter.join(line_parts)
343  if not line:
344  line = next_line.pop()
345 
346  cont_line = self._wrap(self.config['indent'] + splitter.join(next_line).lstrip())
347  if self.config['use_parentheses']:
348  output = "{0}{1} (\n{2}{3}{4})".format(
349  line, splitter, cont_line,
350  "," if self.config['include_trailing_comma'] else "",
351  "\n" if wrap_mode in (
352  settings.WrapModes.VERTICAL_HANGING_INDENT,
353  settings.WrapModes.VERTICAL_GRID_GROUPED,
354  ) else "")
355  lines = output.split('\n')
356  if ' #' in lines[-1] and lines[-1].endswith(')'):
357  line, comment = lines[-1].split(' #', 1)
358  lines[-1] = line + ') #' + comment[:-1]
359  return '\n'.join(lines)
360  return "{0}{1} \\\n{2}".format(line, splitter, cont_line)
361  elif len(line) > self.config['line_length'] and wrap_mode == settings.WrapModes.NOQA:
362  if "# NOQA" not in line:
363  return "{0} # NOQA".format(line)
364 
365  return line
