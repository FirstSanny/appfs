18 class CallSite(object):
19  """Class for understanding arguments passed into a call site
20 
21  It needs a call context, which contains the arguments and the
22  keyword arguments that were passed into a given call site.
23  In order to infer what an argument represents, call
24  :meth:`infer_argument` with the corresponding function node
25  and the argument name.
26  """
27 
28  def __init__(self, callcontext):
29  args = callcontext.args
30  keywords = callcontext.keywords
31  self.duplicated_keywords = set()
32  self._unpacked_args = self._unpack_args(args)
33  self._unpacked_kwargs = self._unpack_keywords(keywords)
34 
35  self.positional_arguments = [
36  arg for arg in self._unpacked_args
37  if arg is not util.Uninferable
38  ]
39  self.keyword_arguments = {
40  key: value for key, value in self._unpacked_kwargs.items()
41  if value is not util.Uninferable
42  }
43 
44  @classmethod
45  def from_call(cls, call_node):
46  """Get a CallSite object from the given Call node."""
47  callcontext = contextmod.CallContext(call_node.args,
48  call_node.keywords)
49  return cls(callcontext)

51  def has_invalid_arguments(self):
52  """Check if in the current CallSite were passed *invalid* arguments
53 
54  This can mean multiple things. For instance, if an unpacking
55  of an invalid object was passed, then this method will return True.
56  Other cases can be when the arguments can't be inferred by astroid,
57  for example, by passing objects which aren't known statically.
58  """
59  return len(self.positional_arguments) != len(self._unpacked_args)
60 
61  def has_invalid_keywords(self):
62  """Check if in the current CallSite were passed *invalid* keyword arguments
63 
64  For instance, unpacking a dictionary with integer keys is invalid
65  (**{1:2}), because the keys must be strings, which will make this
66  method to return True. Other cases where this might return True if
67  objects which can't be inferred were passed.
68  """
69  return len(self.keyword_arguments) != len(self._unpacked_kwargs)

71  def _unpack_keywords(self, keywords):
72  values = {}
73  context = contextmod.InferenceContext()
74  for name, value in keywords:
75  if name is None:
76  # Then it's an unpacking operation (**)
77  try:
78  inferred = next(value.infer(context=context))
79  except exceptions.InferenceError:
80  values[name] = util.Uninferable
81  continue
82 
83  if not isinstance(inferred, nodes.Dict):
84  # Not something we can work with.
85  values[name] = util.Uninferable
86  continue
87 
88  for dict_key, dict_value in inferred.items:
89  try:
90  dict_key = next(dict_key.infer(context=context))
91  except exceptions.InferenceError:
92  values[name] = util.Uninferable
93  continue
94  if not isinstance(dict_key, nodes.Const):
95  values[name] = util.Uninferable
96  continue
97  if not isinstance(dict_key.value, six.string_types):
98  values[name] = util.Uninferable
99  continue
100  if dict_key.value in values:
101  # The name is already in the dictionary
102  values[dict_key.value] = util.Uninferable
103  self.duplicated_keywords.add(dict_key.value)
104  continue
105  values[dict_key.value] = dict_value
106  else:
107  values[name] = value
108  return values

110  @staticmethod
111  def _unpack_args(args):
112  values = []
113  context = contextmod.InferenceContext()
114  for arg in args:
115  if isinstance(arg, nodes.Starred):
116  try:
117  inferred = next(arg.value.infer(context=context))
118  except exceptions.InferenceError:
119  values.append(util.Uninferable)
120  continue
121 
122  if inferred is util.Uninferable:
123  values.append(util.Uninferable)
124  continue
125  if not hasattr(inferred, 'elts'):
126  values.append(util.Uninferable)
127  continue
128  values.extend(inferred.elts)
129  else:
130  values.append(arg)
131  return values

133  def infer_argument(self, funcnode, name, context):
134  """infer a function argument value according to the call context
135 
136  Arguments:
137  funcnode: The function being called.
138  name: The name of the argument whose value is being inferred.
139  context: TODO
140  """
141  if name in self.duplicated_keywords:
142  raise exceptions.InferenceError('The arguments passed to {func!r} '
143  ' have duplicate keywords.',
144  call_site=self, func=funcnode,
145  arg=name, context=context)
146 
147  # Look into the keywords first, maybe it's already there.
148  try:
149  return self.keyword_arguments[name].infer(context)
150  except KeyError:
151  pass
152 
153  # Too many arguments given and no variable arguments.
154  if len(self.positional_arguments) > len(funcnode.args.args):
155  if not funcnode.args.vararg:
156  raise exceptions.InferenceError('Too many positional arguments '
157  'passed to {func!r} that does '
158  'not have *args.',
159  call_site=self, func=funcnode,
160  arg=name, context=context)
161 
162  positional = self.positional_arguments[:len(funcnode.args.args)]
163  vararg = self.positional_arguments[len(funcnode.args.args):]
164  argindex = funcnode.args.find_argname(name)[0]
165  kwonlyargs = set(arg.name for arg in funcnode.args.kwonlyargs)
166  kwargs = {
167  key: value for key, value in self.keyword_arguments.items()
168  if key not in kwonlyargs
169  }
170  # If there are too few positionals compared to
171  # what the function expects to receive, check to see
172  # if the missing positional arguments were passed
173  # as keyword arguments and if so, place them into the
174  # positional args list.
175  if len(positional) < len(funcnode.args.args):
176  for func_arg in funcnode.args.args:
177  if func_arg.name in kwargs:
178  arg = kwargs.pop(func_arg.name)
179  positional.append(arg)
180 
181  if argindex is not None:
182  # 2. first argument of instance/class method
183  if argindex == 0 and funcnode.type in ('method', 'classmethod'):
184  if context.boundnode is not None:
185  boundnode = context.boundnode
186  else:
187  # XXX can do better ?
188  boundnode = funcnode.parent.frame()
189 
190  if isinstance(boundnode, nodes.ClassDef):
191  # Verify that we're accessing a method
192  # of the metaclass through a class, as in
193  # `cls.metaclass_method`. In this case, the
194  # first argument is always the class.
195  method_scope = funcnode.parent.scope()
196  if method_scope is boundnode.metaclass():
197  return iter((boundnode, ))
198 
199  if funcnode.type == 'method':
200  if not isinstance(boundnode, bases.Instance):
201  boundnode = bases.Instance(boundnode)
202  return iter((boundnode,))
203  if funcnode.type == 'classmethod':
204  return iter((boundnode,))
205  # if we have a method, extract one position
206  # from the index, so we'll take in account
207  # the extra parameter represented by `self` or `cls`
208  if funcnode.type in ('method', 'classmethod'):
209  argindex -= 1
210  # 2. search arg index
211  try:
212  return self.positional_arguments[argindex].infer(context)
213  except IndexError:
214  pass
215 
216  if funcnode.args.kwarg == name:
217  # It wants all the keywords that were passed into
218  # the call site.
219  if self.has_invalid_keywords():
220  raise exceptions.InferenceError(
221  "Inference failed to find values for all keyword arguments "
222  "to {func!r}: {unpacked_kwargs!r} doesn't correspond to "
223  "{keyword_arguments!r}.",
224  keyword_arguments=self.keyword_arguments,
225  unpacked_kwargs=self._unpacked_kwargs,
226  call_site=self, func=funcnode, arg=name, context=context)
227  kwarg = nodes.Dict(lineno=funcnode.args.lineno,
228  col_offset=funcnode.args.col_offset,
229  parent=funcnode.args)
230  kwarg.postinit([(nodes.const_factory(key), value)
231  for key, value in kwargs.items()])
232  return iter((kwarg, ))
233  elif funcnode.args.vararg == name:
234  # It wants all the args that were passed into
235  # the call site.
236  if self.has_invalid_arguments():
237  raise exceptions.InferenceError(
238  "Inference failed to find values for all positional "
239  "arguments to {func!r}: {unpacked_args!r} doesn't "
240  "correspond to {positional_arguments!r}.",
241  positional_arguments=self.positional_arguments,
242  unpacked_args=self._unpacked_args,
243  call_site=self, func=funcnode, arg=name, context=context)
244  args = nodes.Tuple(lineno=funcnode.args.lineno,
245  col_offset=funcnode.args.col_offset,
246  parent=funcnode.args)
247  args.postinit(vararg)
248  return iter((args, ))
249 
250  # Check if it's a default parameter.
251  try:
252  return funcnode.args.default_value(name).infer(context)
253  except exceptions.NoDefault:
254  pass
255  raise exceptions.InferenceError('No value found for argument {name} to '
256  '{func!r}', call_site=self,
257  func=funcnode, arg=name, context=context)
