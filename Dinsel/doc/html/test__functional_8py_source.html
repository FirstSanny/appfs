170 def get_expected_messages(stream):
171  """Parses a file and get expected messages.
172 
173  :param stream: File-like input stream.
174  :returns: A dict mapping line,msg-symbol tuples to the count on this line.
175  """
176  messages = collections.Counter()
177  for i, line in enumerate(stream):
178  match = _EXPECTED_RE.search(line)
179  if match is None:
180  continue
181  line = match.group('line')
182  if line is None:
183  line = i + 1
184  elif line.startswith('+') or line.startswith('-'):
185  line = i + 1 + int(line)
186  else:
187  line = int(line)
188 
189  version = match.group('version')
190  op = match.group('op')
191  if version:
192  required = parse_python_version(version)
193  if not _OPERATORS[op](sys.version_info, required):
194  continue
195 
196  for msg_id in match.group('msgs').split(','):
197  messages[line, msg_id.strip()] += 1
198  return messages

201 def multiset_difference(left_op, right_op):
202  """Takes two multisets and compares them.
203 
204  A multiset is a dict with the cardinality of the key as the value.
205 
206  :param left_op: The expected entries.
207  :param right_op: Actual entries.
208 
209  :returns: The two multisets of missing and unexpected messages.
210  """
211  missing = left_op.copy()
212  missing.subtract(right_op)
213  unexpected = {}
214  for key, value in list(six.iteritems(missing)):
215  if value <= 0:
216  missing.pop(key)
217  if value < 0:
218  unexpected[key] = -value
219  return missing, unexpected
