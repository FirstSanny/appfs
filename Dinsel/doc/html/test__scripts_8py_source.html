20 def run_command(cmd, check_code=True):
21  """ Run command sequence `cmd` returning exit code, stdout, stderr
22 
23  Parameters
24  ----------
25  cmd : str or sequence
26  string with command name or sequence of strings defining command
27  check_code : {True, False}, optional
28  If True, raise error for non-zero return code
29 
30  Returns
31  -------
32  returncode : int
33  return code from execution of `cmd`
34  stdout : bytes (python 3) or str (python 2)
35  stdout from `cmd`
36  stderr : bytes (python 3) or str (python 2)
37  stderr from `cmd`
38 
39  Raises
40  ------
41  RuntimeError
42  If `check_code` is True, and return code !=0
43  """
44  cmd = [cmd] if isinstance(cmd, basestring) else list(cmd)
45  if os.name == 'nt':
46  # Quote any arguments with spaces. The quotes delimit the arguments
47  # on Windows, and the arguments might be file paths with spaces.
48  # On Unix the list elements are each separate arguments.
49  cmd = ['"{0}"'.format(c) if ' ' in c else c for c in cmd]
50  proc = Popen(cmd, stdout=PIPE, stderr=PIPE)
51  stdout, stderr = proc.communicate()
52  if proc.poll() is None:
53  proc.terminate()
54  if check_code and proc.returncode != 0:
55  raise RuntimeError('\n'.join(
56  ['Command "{0}" failed with',
57  'stdout', '------', '{1}', '',
58  'stderr', '------', '{2}']).format(cmd, stdout, stderr))
59  return proc.returncode, stdout, stderr
