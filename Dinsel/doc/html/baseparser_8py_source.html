199  def _update_defaults(self, defaults):
200  """Updates the given defaults with values from the config files and
201  the environ. Does a little special handling for certain types of
202  options (lists)."""
203  # Then go and look for the other sources of configuration:
204  config = {}
205  # 1. config files
206  for section in ('global', self.name):
207  config.update(
208  self.normalize_keys(self.get_config_section(section))
209  )
210  # 2. environmental variables
211  if not self.isolated:
212  config.update(self.normalize_keys(self.get_environ_vars()))
213  # Accumulate complex default state.
214  self.values = optparse.Values(self.defaults)
215  late_eval = set()
216  # Then set the options with those values
217  for key, val in config.items():
218  # ignore empty values
219  if not val:
220  continue
221 
222  option = self.get_option(key)
223  # Ignore options not present in this parser. E.g. non-globals put
224  # in [global] by users that want them to apply to all applicable
225  # commands.
226  if option is None:
227  continue
228 
229  if option.action in ('store_true', 'store_false', 'count'):
230  val = strtobool(val)
231  elif option.action == 'append':
232  val = val.split()
233  val = [self.check_default(option, key, v) for v in val]
234  elif option.action == 'callback':
235  late_eval.add(option.dest)
236  opt_str = option.get_opt_string()
237  val = option.convert_value(opt_str, val)
238  # From take_action
239  args = option.callback_args or ()
240  kwargs = option.callback_kwargs or {}
241  option.callback(option, opt_str, val, self, *args, **kwargs)
242  else:
243  val = self.check_default(option, key, val)
244 
245  defaults[option.dest] = val
246 
247  for key in late_eval:
248  defaults[key] = getattr(self.values, key)
249  self.values = None
250  return defaults
