302 class _fromnxfunction_args(_fromnxfunction):
303  """
304  A version of `_fromnxfunction` that is called with multiple array
305  arguments. The first non-array-like input marks the beginning of the
306  arguments that are passed verbatim for both the data and mask calls.
307  Array arguments are processed independently and the results are
308  returned in a list. If only one array is found, the return value is
309  just the processed array instead of a list.
310  """
311  def __call__(self, *args, **params):
312  func = getattr(np, self.__name__)
313  arrays = []
314  args = list(args)
315  while len(args) > 0 and issequence(args[0]):
316  arrays.append(args.pop(0))
317  res = []
318  for x in arrays:
319  _d = func(np.asarray(x), *args, **params)
320  _m = func(getmaskarray(x), *args, **params)
321  res.append(masked_array(_d, mask=_m))
322  if len(arrays) == 1:
323  return res[0]
324  return res

521 def average(a, axis=None, weights=None, returned=False):
522  """
523  Return the weighted average of array over the given axis.
524 
525  Parameters
526  ----------
527  a : array_like
528  Data to be averaged.
529  Masked entries are not taken into account in the computation.
530  axis : int, optional
531  Axis along which to average `a`. If `None`, averaging is done over
532  the flattened array.
533  weights : array_like, optional
534  The importance that each element has in the computation of the average.
535  The weights array can either be 1-D (in which case its length must be
536  the size of `a` along the given axis) or of the same shape as `a`.
537  If ``weights=None``, then all data in `a` are assumed to have a
538  weight equal to one. If `weights` is complex, the imaginary parts
539  are ignored.
540  returned : bool, optional
541  Flag indicating whether a tuple ``(result, sum of weights)``
542  should be returned as output (True), or just the result (False).
543  Default is False.
544 
545  Returns
546  -------
547  average, [sum_of_weights] : (tuple of) scalar or MaskedArray
548  The average along the specified axis. When returned is `True`,
549  return a tuple with the average as the first element and the sum
550  of the weights as the second element. The return type is `np.float64`
551  if `a` is of integer type and floats smaller than `float64`, or the
552  input data-type, otherwise. If returned, `sum_of_weights` is always
553  `float64`.
554 
555  Examples
556  --------
557  >>> a = np.ma.array([1., 2., 3., 4.], mask=[False, False, True, True])
558  >>> np.ma.average(a, weights=[3, 1, 0, 0])
559  1.25
560 
561  >>> x = np.ma.arange(6.).reshape(3, 2)
562  >>> print(x)
563  [[ 0. 1.]
564  [ 2. 3.]
565  [ 4. 5.]]
566  >>> avg, sumweights = np.ma.average(x, axis=0, weights=[1, 2, 3],
567  ... returned=True)
568  >>> print(avg)
569  [2.66666666667 3.66666666667]
570 
571  """
572  a = asarray(a)
573  m = getmask(a)
574 
575  # inspired by 'average' in numpy/lib/function_base.py
576 
577  if weights is None:
578  avg = a.mean(axis)
579  scl = avg.dtype.type(a.count(axis))
580  else:
581  wgt = np.asanyarray(weights)
582 
583  if issubclass(a.dtype.type, (np.integer, np.bool_)):
584  result_dtype = np.result_type(a.dtype, wgt.dtype, 'f8')
585  else:
586  result_dtype = np.result_type(a.dtype, wgt.dtype)
587 
588  # Sanity checks
589  if a.shape != wgt.shape:
590  if axis is None:
591  raise TypeError(
592  "Axis must be specified when shapes of a and weights "
593  "differ.")
594  if wgt.ndim != 1:
595  raise TypeError(
596  "1D weights expected when shapes of a and weights differ.")
597  if wgt.shape[0] != a.shape[axis]:
598  raise ValueError(
599  "Length of weights not compatible with specified axis.")
600 
601  # setup wgt to broadcast along axis
602  wgt = np.broadcast_to(wgt, (a.ndim-1)*(1,) + wgt.shape)
603  wgt = wgt.swapaxes(-1, axis)
604 
605  if m is not nomask:
606  wgt = wgt*(~a.mask)
607 
608  scl = wgt.sum(axis=axis, dtype=result_dtype)
609  avg = np.multiply(a, wgt, dtype=result_dtype).sum(axis)/scl
610 
611  if returned:
612  if scl.shape != avg.shape:
613  scl = np.broadcast_to(scl, avg.shape).copy()
614  return avg, scl
615  else:
616  return avg

619 def median(a, axis=None, out=None, overwrite_input=False, keepdims=False):
620  """
621  Compute the median along the specified axis.
622 
623  Returns the median of the array elements.
624 
625  Parameters
626  ----------
627  a : array_like
628  Input array or object that can be converted to an array.
629  axis : int, optional
630  Axis along which the medians are computed. The default (None) is
631  to compute the median along a flattened version of the array.
632  out : ndarray, optional
633  Alternative output array in which to place the result. It must
634  have the same shape and buffer length as the expected output
635  but the type will be cast if necessary.
636  overwrite_input : bool, optional
637  If True, then allow use of memory of input array (a) for
638  calculations. The input array will be modified by the call to
639  median. This will save memory when you do not need to preserve
640  the contents of the input array. Treat the input as undefined,
641  but it will probably be fully or partially sorted. Default is
642  False. Note that, if `overwrite_input` is True, and the input
643  is not already an `ndarray`, an error will be raised.
644  keepdims : bool, optional
645  If this is set to True, the axes which are reduced are left
646  in the result as dimensions with size one. With this option,
647  the result will broadcast correctly against the input array.
648 
649  .. versionadded:: 1.10.0
650 
651  Returns
652  -------
653  median : ndarray
654  A new array holding the result is returned unless out is
655  specified, in which case a reference to out is returned.
656  Return data-type is `float64` for integers and floats smaller than
657  `float64`, or the input data-type, otherwise.
658 
659  See Also
660  --------
661  mean
662 
663  Notes
664  -----
665  Given a vector ``V`` with ``N`` non masked values, the median of ``V``
666  is the middle value of a sorted copy of ``V`` (``Vs``) - i.e.
667  ``Vs[(N-1)/2]``, when ``N`` is odd, or ``{Vs[N/2 - 1] + Vs[N/2]}/2``
668  when ``N`` is even.
669 
670  Examples
671  --------
672  >>> x = np.ma.array(np.arange(8), mask=[0]*4 + [1]*4)
673  >>> np.ma.median(x)
674  1.5
675 
676  >>> x = np.ma.array(np.arange(10).reshape(2, 5), mask=[0]*6 + [1]*4)
677  >>> np.ma.median(x)
678  2.5
679  >>> np.ma.median(x, axis=-1, overwrite_input=True)
680  masked_array(data = [ 2. 5.],
681  mask = False,
682  fill_value = 1e+20)
683 
684  """
685  if not hasattr(a, 'mask'):
686  m = np.median(getdata(a, subok=True), axis=axis,

701 def _median(a, axis=None, out=None, overwrite_input=False):
702  # when an unmasked NaN is present return it, so we need to sort the NaN
703  # values behind the mask
704  if np.issubdtype(a.dtype, np.inexact):
705  fill_value = np.inf

716  asorted = sort(a, axis=axis, fill_value=fill_value)

719  axis = 0

721  axis += asorted.ndim
722

724  # for empty axis integer indices fail so use slicing to get same result
725  # as median (which is mean of empty slice = nan)
726  indexer = [slice(None)] * asorted.ndim
727  indexer[axis] = slice(0, 0)
728  return np.ma.mean(asorted[indexer], axis=axis, out=out)

731  counts = count(asorted)
732  idx, odd = divmod(count(asorted), 2)

735  # avoid inf / x = masked
736  s = mid.sum(out=out)

738  s = np.true_divide(s, 2., casting='safe', out=out)

740  # fix this to return the same in the nd path
741  if out is not None:
742  s = out

744  else:
745  s = mid.mean(out=out)
746

748  # minimum_fill_value so that it would be the median or all values
749  # masked

751  return np.ma.minimum_fill_value(asorted)
752  return s

754  counts = count(asorted, axis=axis)

779  replace_masked(high)

793  np.true_divide(s.data, 2., casting='unsafe', out=s.data)

795  s = np.lib.utils._median_nancheck(asorted, s, axis, out)

1276 def cov(x, y=None, rowvar=True, bias=False, allow_masked=True, ddof=None):
1277  """
1278  Estimate the covariance matrix.
1279 
1280  Except for the handling of missing data this function does the same as
1281  `numpy.cov`. For more details and examples, see `numpy.cov`.
1282 
1283  By default, masked values are recognized as such. If `x` and `y` have the
1284  same shape, a common mask is allocated: if ``x[i,j]`` is masked, then
1285  ``y[i,j]`` will also be masked.
1286  Setting `allow_masked` to False will raise an exception if values are
1287  missing in either of the input arrays.
1288 
1289  Parameters
1290  ----------
1291  x : array_like
1292  A 1-D or 2-D array containing multiple variables and observations.
1293  Each row of `x` represents a variable, and each column a single
1294  observation of all those variables. Also see `rowvar` below.
1295  y : array_like, optional
1296  An additional set of variables and observations. `y` has the same
1297  form as `x`.
1298  rowvar : bool, optional
1299  If `rowvar` is True (default), then each row represents a
1300  variable, with observations in the columns. Otherwise, the relationship
1301  is transposed: each column represents a variable, while the rows
1302  contain observations.
1303  bias : bool, optional
1304  Default normalization (False) is by ``(N-1)``, where ``N`` is the
1305  number of observations given (unbiased estimate). If `bias` is True,
1306  then normalization is by ``N``. This keyword can be overridden by
1307  the keyword ``ddof`` in numpy versions >= 1.5.
1308  allow_masked : bool, optional
1309  If True, masked values are propagated pair-wise: if a value is masked
1310  in `x`, the corresponding value is masked in `y`.
1311  If False, raises a `ValueError` exception when some values are missing.
1312  ddof : {None, int}, optional
1313  If not ``None`` normalization is by ``(N - ddof)``, where ``N`` is
1314  the number of observations; this overrides the value implied by
1315  ``bias``. The default value is ``None``.
1316 
1317  .. versionadded:: 1.5
1318 
1319  Raises
1320  ------
1321  ValueError
1322  Raised if some values are missing and `allow_masked` is False.
1323 
1324  See Also
1325  --------
1326  numpy.cov
1327 
1328  """
1329  # Check inputs
1330  if ddof is not None and ddof != int(ddof):
1331  raise ValueError("ddof must be an integer")
1332  # Set up ddof
1333  if ddof is None:
1334  if bias:
1335  ddof = 0
1336  else:
1337  ddof = 1
1338 
1339  (x, xnotmask, rowvar) = _covhelper(x, y, rowvar, allow_masked)
1340  if not rowvar:
1341  fact = np.dot(xnotmask.T, xnotmask) * 1. - ddof
1342  result = (dot(x.T, x.conj(), strict=False) / fact).squeeze()
1343  else:
1344  fact = np.dot(xnotmask, xnotmask.T) * 1. - ddof
1345  result = (dot(x, x.T.conj(), strict=False) / fact).squeeze()
1346  return result
