98 class Run(ConfigurationMixIn):
99  """base class providing common behaviour for pyreverse commands"""

103  def __init__(self, args):
104  ConfigurationMixIn.__init__(self, usage=__doc__)

106  args = self.load_command_line_configuration()

112  def run(self, args):
113  """checking arguments and run project"""
114  if not args:
115  print(self.help())
116  return 1
117  # insert current working directory to the python path to recognize
118  # dependencies to local modules even if cwd is not in the PYTHONPATH
119  sys.path.insert(0, os.getcwd())
120  try:
121  project = project_from_files(args, project_name=self.config.project,
122  black_list=self.config.black_list)
123  linker = Linker(project, tag=True)
124  handler = DiadefsHandler(self.config)
125  diadefs = handler.get_diadefs(project, linker)
126  finally:
127  sys.path.pop(0)
128 
129  if self.config.output_format == "vcg":
130  writer.VCGWriter(self.config).write(diadefs)
131  else:
132  writer.DotWriter(self.config).write(diadefs)
133  return 0
