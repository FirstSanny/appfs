14 def central_diff_weights(Np, ndiv=1):
15  """
16  Return weights for an Np-point central derivative.
17 
18  Assumes equally-spaced function points.
19 
20  If weights are in the vector w, then
21  derivative is w[0] * f(x-ho*dx) + ... + w[-1] * f(x+h0*dx)
22 
23  Parameters
24  ----------
25  Np : int
26  Number of points for the central derivative.
27  ndiv : int, optional
28  Number of divisions. Default is 1.
29 
30  Notes
31  -----
32  Can be inaccurate for large number of points.
33 
34  """
35  if Np < ndiv + 1:
36  raise ValueError("Number of points must be at least the derivative order + 1.")
37  if Np % 2 == 0:
38  raise ValueError("The number of points must be odd.")
39  from scipy import linalg
40  ho = Np >> 1
41  x = arange(-ho,ho+1.0)
42  x = x[:,newaxis]
43  X = x**0.0
44  for k in range(1,Np):
45  X = hstack([X,x**k])
46  w = product(arange(1,ndiv+1),axis=0)*linalg.inv(X)[ndiv]
47  return w

50 def derivative(func, x0, dx=1.0, n=1, args=(), order=3):
51  """
52  Find the n-th derivative of a function at a point.
53 
54  Given a function, use a central difference formula with spacing `dx` to
55  compute the `n`-th derivative at `x0`.
56 
57  Parameters
58  ----------
59  func : function
60  Input function.
61  x0 : float
62  The point at which `n`-th derivative is found.
63  dx : float, optional
64  Spacing.
65  n : int, optional
66  Order of the derivative. Default is 1.
67  args : tuple, optional
68  Arguments
69  order : int, optional
70  Number of points to use, must be odd.
71 
72  Notes
73  -----
74  Decreasing the step size too small can result in round-off error.
75 
76  Examples
77  --------
78  >>> from scipy.misc import derivative
79  >>> def f(x):
80  ... return x**3 + x**2
81  >>> derivative(f, 1.0, dx=1e-6)
82  4.9999999999217337
83 
84  """
85  if order < n + 1:
86  raise ValueError("'order' (the number of points used to compute the derivative), "
87  "must be at least the derivative order 'n' + 1.")
88  if order % 2 == 0:
89  raise ValueError("'order' (the number of points used to compute the derivative) "
90  "must be odd.")
91  # pre-computed for n=1 and 2 and low-order for speed.
92  if n == 1:
93  if order == 3:
94  weights = array([-1,0,1])/2.0
95  elif order == 5:
96  weights = array([1,-8,0,8,-1])/12.0
97  elif order == 7:
98  weights = array([-1,9,-45,0,45,-9,1])/60.0
99  elif order == 9:
100  weights = array([3,-32,168,-672,0,672,-168,32,-3])/840.0
101  else:
102  weights = central_diff_weights(order,1)
103  elif n == 2:
104  if order == 3:
105  weights = array([1,-2.0,1])
106  elif order == 5:
107  weights = array([-1,16,-30,16,-1])/12.0
108  elif order == 7:
109  weights = array([2,-27,270,-490,270,-27,2])/180.0
110  elif order == 9:
111  weights = array([-9,128,-1008,8064,-14350,8064,-1008,128,-9])/5040.0
112  else:
113  weights = central_diff_weights(order,2)
114  else:
115  weights = central_diff_weights(order, n)
116  val = 0.0
117  ho = order >> 1
118  for k in range(order):
119  val += weights[k]*func(x0+(k-ho)*dx,*args)
120  return val / product((dx,)*n,axis=0)

194 def face(gray=False):
195  """
196  Get a 1024 x 768, color image of a raccoon face.
197 
198  raccoon-procyon-lotor.jpg at http://www.public-domain-image.com
199 
200  Parameters
201  ----------
202  gray : bool, optional
203  If True return 8-bit grey-scale image, otherwise return a color image
204 
205  Returns
206  -------
207  face : ndarray
208  image of a racoon face
209 
210  Examples
211  --------
212  >>> import scipy.misc
213  >>> face = scipy.misc.face()
214  >>> face.shape
215  (768, 1024, 3)
216  >>> face.max()
217  255
218  >>> face.dtype
219  dtype('uint8')
220 
221  >>> import matplotlib.pyplot as plt
222  >>> plt.gray()
223  >>> plt.imshow(face)
224  >>> plt.show()
225 
226  """
227  import bz2
228  import os
229  with open(os.path.join(os.path.dirname(__file__), 'face.dat'), 'rb') as f:
230  rawdata = f.read()
231  data = bz2.decompress(rawdata)
232  face = fromstring(data, dtype='uint8')
233  face.shape = (768, 1024, 3)
234  if gray is True:
235  face = (0.21 * face[:,:,0] + 0.71 * face[:,:,1] + 0.07 * face[:,:,2]).astype('uint8')
236  return face
