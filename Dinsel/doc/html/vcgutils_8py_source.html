156  def _write_attributes(self, attributes_dict, **args):
157  """write graph, node or edge attributes
158  """
159  for key, value in args.items():
160  try:
161  _type = attributes_dict[key]
162  except KeyError:
163  raise Exception('''no such attribute %s
164 possible attributes are %s''' % (key, attributes_dict.keys()))
165 
166  if not _type:
167  self._stream.write('%s%s:"%s"\n' % (self._indent, key, value))
168  elif _type == 1:
169  self._stream.write('%s%s:%s\n' % (self._indent, key,
170  int(value)))
171  elif value in _type:
172  self._stream.write('%s%s:%s\n' % (self._indent, key, value))
173  else:
174  raise Exception('''value %s isn\'t correct for attribute %s
175 correct values are %s''' % (value, key, _type))
