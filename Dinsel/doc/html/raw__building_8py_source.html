109 def build_function(name, args=None, defaults=None, doc=None):
110  """create and initialize a astroid FunctionDef node"""
111  args, defaults = args or [], defaults or []
112  # first argument is now a list of decorators
113  func = nodes.FunctionDef(name, doc)
114  func.args = argsnode = nodes.Arguments()
115  argsnode.args = []
116  for arg in args:
117  argsnode.args.append(nodes.Name())
118  argsnode.args[-1].name = arg
119  argsnode.args[-1].parent = argsnode
120  argsnode.defaults = []
121  for default in defaults:
122  argsnode.defaults.append(nodes.const_factory(default))
123  argsnode.defaults[-1].parent = argsnode
124  argsnode.kwarg = None
125  argsnode.vararg = None
126  argsnode.parent = func

128  register_arguments(func)

192 def _base_class_object_build(node, member, basenames, name=None, localname=None):
193  """create astroid for a living class object, with a given set of base names
194  (e.g. ancestors)
195  """
196  klass = build_class(name or getattr(member, '__name__', None) or localname,
197  basenames, member.__doc__)
198  klass._newstyle = isinstance(member, type)
199  node.add_local_node(klass, localname)
200  try:
201  # limit the instantiation trick since it's too dangerous
202  # (such as infinite test execution...)
203  # this at least resolves common case such as Exception.args,
204  # OSError.errno
205  if issubclass(member, Exception):
206  instdict = member().__dict__
207  else:
208  raise TypeError
209  except: # pylint: disable=bare-except
210  pass
211  else:
212  for item_name, obj in instdict.items():
213  valnode = nodes.EmptyNode()
214  valnode.object = obj
215  valnode.parent = klass
216  valnode.lineno = 1
217  klass.instance_attrs[item_name] = [valnode]
218  return klass

221 def _build_from_function(node, name, member, module):
222  # verify this is not an imported function
223  try:
224  code = six.get_function_code(member)
225  except AttributeError:
226  # Some implementations don't provide the code object,
227  # such as Jython.
228  code = None
229  filename = getattr(code, 'co_filename', None)
230  if filename is None:
231  assert isinstance(member, object)
232  object_build_methoddescriptor(node, member, name)
233  elif filename != getattr(module, '__file__', None):
234  attach_dummy_node(node, name, member)
235  else:

252  def inspect_build(self, module, modname=None, path=None):
253  """build astroid from a living module (i.e. using inspect)
254  this is used when there is no python source code available (either
255  because it's a built-in module or because the .py is not available)
256  """
257  self._module = module
258  if modname is None:
259  modname = module.__name__
260  try:
261  node = build_module(modname, module.__doc__)
262  except AttributeError:
263  # in jython, java modules have no __doc__ (see #109562)
264  node = build_module(modname)
265  node.file = node.path = os.path.abspath(path) if path else path
266  node.name = modname
267  MANAGER.cache_module(node)
268  node.package = hasattr(module, '__path__')
269  self._done = {}
270  self.object_build(node, module)
271  return node

273  def object_build(self, node, obj):
274  """recursive method which create a partial ast from real objects
275  (only function, class, and method are handled)
276  """
277  if obj in self._done:
278  return self._done[obj]
279  self._done[obj] = node
280  for name in dir(obj):
281  try:
282  member = getattr(obj, name)
283  except AttributeError:
284  # damned ExtensionClass.Base, I know you're there !
285  attach_dummy_node(node, name)
286  continue
287  if inspect.ismethod(member):
288  member = six.get_method_function(member)
289  if inspect.isfunction(member):
290  _build_from_function(node, name, member, self._module)
291  elif inspect.isbuiltin(member):
292  if (not _io_discrepancy(member) and
293  self.imported_member(node, member, name)):
294  continue
295  object_build_methoddescriptor(node, member, name)
296  elif inspect.isclass(member):
297  if self.imported_member(node, member, name):
298  continue
299  if member in self._done:
300  class_node = self._done[member]
301  if class_node not in node.locals.get(name, ()):
302  node.add_local_node(class_node, name)
303  else:
304  class_node = object_build_class(node, member, name)
305  # recursion
306  self.object_build(class_node, member)
307  if name == '__class__' and class_node.parent is None:
308  class_node.parent = self._done[self._module]
309  elif inspect.ismethoddescriptor(member):
310  assert isinstance(member, object)
311  object_build_methoddescriptor(node, member, name)
312  elif inspect.isdatadescriptor(member):
313  assert isinstance(member, object)
314  object_build_datadescriptor(node, member, name)
315  elif isinstance(member, _CONSTANTS):
316  attach_const_node(node, name, member)
317  elif inspect.isroutine(member):
318  # This should be called for Jython, where some builtin
319  # methods aren't caught by isbuiltin branch.
320  _build_from_function(node, name, member, self._module)
321  else:
322  # create an empty node so that the name is actually defined
323  attach_dummy_node(node, name, member)

325  def imported_member(self, node, member, name):
326  """verify this is not an imported class or handle it"""
327  # /!\ some classes like ExtensionClass doesn't have a __module__
328  # attribute ! Also, this may trigger an exception on badly built module
329  # (see http://www.logilab.org/ticket/57299 for instance)
330  try:
331  modname = getattr(member, '__module__', None)
332  except: # pylint: disable=bare-except
333  _LOG.exception('unexpected error while building '
334  'astroid from living object')
335  modname = None
336  if modname is None:
337  if (name in ('__new__', '__subclasshook__')
338  or (name in _BUILTINS and _JYTHON)):
339  # Python 2.5.1 (r251:54863, Sep 1 2010, 22:03:14)
340  # >>> print object.__new__.__module__
341  # None
342  modname = six.moves.builtins.__name__
343  else:
344  attach_dummy_node(node, name, member)
345  return True

347  real_name = {
348  'gtk': 'gtk_gtk',
349  '_io': 'io',
350  }.get(modname, modname)

352  if real_name != self._module.__name__:
