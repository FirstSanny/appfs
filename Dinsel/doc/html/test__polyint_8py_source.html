464 class TestCubicSpline(object):
465  @staticmethod
466  def check_correctness(S, bc_start='not-a-knot', bc_end='not-a-knot',
467  tol=1e-14):
468  """Check that spline coefficients satisfy the continuity and boundary
469  conditions."""
470  x = S.x
471  c = S.c
472  dx = np.diff(x)
473  dx = dx.reshape([dx.shape[0]] + [1] * (c.ndim - 2))
474  dxi = dx[:-1]
475 
476  # Check C2 continuity.
477  assert_allclose(c[3, 1:], c[0, :-1] * dxi**3 + c[1, :-1] * dxi**2 +
478  c[2, :-1] * dxi + c[3, :-1], rtol=tol, atol=tol)
479  assert_allclose(c[2, 1:], 3 * c[0, :-1] * dxi**2 +
480  2 * c[1, :-1] * dxi + c[2, :-1], rtol=tol, atol=tol)
481  assert_allclose(c[1, 1:], 3 * c[0, :-1] * dxi + c[1, :-1],
482  rtol=tol, atol=tol)
483 
484  # Check that we found a parabola, the third derivative is 0.
485  if x.size == 3 and bc_start == 'not-a-knot' and bc_end == 'not-a-knot':
486  assert_allclose(c[0], 0, rtol=tol, atol=tol)
487  return
488 
489  # Check periodic boundary conditions.
490  if bc_start == 'periodic':
491  assert_allclose(S(x[0], 0), S(x[-1], 0), rtol=tol, atol=tol)
492  assert_allclose(S(x[0], 1), S(x[-1], 1), rtol=tol, atol=tol)
493  assert_allclose(S(x[0], 2), S(x[-1], 2), rtol=tol, atol=tol)
494  return
495 
496  # Check other boundary conditions.
497  if bc_start == 'not-a-knot':
498  if x.size == 2:
499  slope = (S(x[1]) - S(x[0])) / dx[0]
500  assert_allclose(S(x[0], 1), slope, rtol=tol, atol=tol)
501  else:
