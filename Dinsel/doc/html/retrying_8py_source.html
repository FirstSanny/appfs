194  def call(self, fn, *args, **kwargs):
195  start_time = int(round(time.time() * 1000))
196  attempt_number = 1
197  while True:
198  try:
199  attempt = Attempt(fn(*args, **kwargs), attempt_number, False)
200  except:

208  if self.stop(attempt_number, delay_since_first_attempt_ms):

210  # get() on an attempt with an exception should cause it to be raised, but raise just in case
211  raise attempt.get()

237  """
238  Return the return value of this Attempt instance or raise an Exception.
239  If wrap_exception is true, this Attempt is wrapped inside of a
240  RetryError before being raised.
241  """
242  if self.has_exception:
243  if wrap_exception:
244  raise RetryError(self)
