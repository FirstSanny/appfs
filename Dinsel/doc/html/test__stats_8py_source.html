781  def wkq(x, y, rank, weigher, add):
782  tot = conc = disc = u = v = 0
783  for i in range(len(x)):
784  for j in range(len(x)):
785  w = weigher(rank[i]) + weigher(rank[j]) if add else weigher(rank[i]) * weigher(rank[j])
786  tot += w
787  if x[i] == x[j]:
788  u += w
789  if y[i] == y[j]:
790  v += w
791  if x[i] < x[j] and y[i] < y[j] or x[i] > x[j] and y[i] > y[j]:
792  conc += w
793  elif x[i] < x[j] and y[i] > y[j] or x[i] > x[j] and y[i] < y[j]:
794  disc += w
795  return (conc - disc) / np.sqrt(tot - u) / np.sqrt(tot - v)

1642 class _numpy_version_warn_context_mgr(object):
1643  """
1644  A simple context maneger class to avoid retyping the same code for
1645  different versions of numpy when the only difference is that older
1646  versions raise warnings.
1647 
1648  This manager does not apply for cases where the old code returns
1649  different values.
1650  """
1651  def __init__(self, min_numpy_version, warning_type, num_warnings):
1652  if NumpyVersion(np.__version__) < min_numpy_version:
1653  self.numpy_is_old = True
1654  self.warning_type = warning_type
1655  self.num_warnings = num_warnings
1656  self.delegate = warnings.catch_warnings(record = True)
1657  else:
1658  self.numpy_is_old = False
1659 
1660  def __enter__(self):
1661  if self.numpy_is_old:
1662  self.warn_list = self.delegate.__enter__()
1663  warnings.simplefilter("always")
1664  return None
1665 
1666  def __exit__(self, exc_type, exc_value, traceback):
1667  if self.numpy_is_old:
1668  self.delegate.__exit__(exc_type, exc_value, traceback)
1669  _check_warnings(self.warn_list, self.warning_type, self.num_warnings)
