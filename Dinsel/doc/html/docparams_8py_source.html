21 class DocstringParameterChecker(BaseChecker):
22  """Checker for Sphinx, Google, or Numpy style docstrings
23 
24  * Check that all function, method and constructor parameters are mentioned
25  in the params and types part of the docstring. Constructor parameters
26  can be documented in either the class docstring or ``__init__`` docstring,
27  but not both.
28  * Check that there are no naming inconsistencies between the signature and
29  the documentation, i.e. also report documented parameters that are missing
30  in the signature. This is important to find cases where parameters are
31  renamed only in the code, not in the documentation.
32  * Check that all explicitly raised exceptions in a function are documented
33  in the function docstring. Caught exceptions are ignored.
34 
35  Activate this checker by adding the line::
36 
37  load-plugins=pylint.extensions.docparams
38 
39  to the ``MASTER`` section of your ``.pylintrc``.
40 
41  :param linter: linter object
42  :type linter: :class:`pylint.lint.PyLinter`
43  """
44  __implements__ = IAstroidChecker

261  def check_arguments_in_docstring(self, doc, arguments_node, warning_node,
262  accept_no_param_doc=None):
263  """Check that all parameters in a function, method or class constructor
264  on the one hand and the parameters mentioned in the parameter
265  documentation (e.g. the Sphinx tags 'param' and 'type') on the other
266  hand are consistent with each other.
267 
268  * Undocumented parameters except 'self' are noticed.
269  * Undocumented parameter types except for 'self' and the ``*<args>``
270  and ``**<kwargs>`` parameters are noticed.
271  * Parameters mentioned in the parameter documentation that don't or no
272  longer exist in the function parameter list are noticed.
273  * If the text "For the parameters, see" or "For the other parameters,
274  see" (ignoring additional whitespace) is mentioned in the docstring,
275  missing parameter documentation is tolerated.
276  * If there's no Sphinx style, Google style or NumPy style parameter
277  documentation at all, i.e. ``:param`` is never mentioned etc., the
278  checker assumes that the parameters are documented in another format
279  and the absence is tolerated.
280 
281  :param doc: Docstring for the function, method or class.
282  :type doc: str
283 
284  :param arguments_node: Arguments node for the function, method or
285  class constructor.
286  :type arguments_node: :class:`astroid.scoped_nodes.Arguments`
287 
288  :param warning_node: The node to assign the warnings to
289  :type warning_node: :class:`astroid.scoped_nodes.Node`
290 
291  :param accept_no_param_doc: Whether or not to allow no parameters
292  to be documented.
293  If None then this value is read from the configuration.
294  :type accept_no_param_doc: bool or None
295  """
296  # Tolerate missing param or type declarations if there is a link to
297  # another method carrying the same name.
298  if not doc.doc:
299  return
300 
301  if accept_no_param_doc is None:
302  accept_no_param_doc = self.config.accept_no_param_doc
303  tolerate_missing_params = doc.params_documented_elsewhere()
304 
305  # Collect the function arguments.
306  expected_argument_names = set(arg.name for arg in arguments_node.args)
307  expected_argument_names.update(arg.name for arg in arguments_node.kwonlyargs)
308  not_needed_type_in_docstring = (
309  self.not_needed_param_in_docstring.copy())
310 
311  if arguments_node.vararg is not None:
312  expected_argument_names.add(arguments_node.vararg)
313  not_needed_type_in_docstring.add(arguments_node.vararg)
314  if arguments_node.kwarg is not None:
315  expected_argument_names.add(arguments_node.kwarg)
316  not_needed_type_in_docstring.add(arguments_node.kwarg)
317  params_with_doc, params_with_type = doc.match_param_docs()
318 
319  # Tolerate no parameter documentation at all.
320  if (not params_with_doc and not params_with_type
321  and accept_no_param_doc):
322  tolerate_missing_params = True

324  def _compare_missing_args(found_argument_names, message_id,
325  not_needed_names):
326  """Compare the found argument names with the expected ones and
327  generate a message if there are arguments missing.
328 
329  :param set found_argument_names: argument names found in the
330  docstring
331 
332  :param str message_id: pylint message id
333 
334  :param not_needed_names: names that may be omitted
335  :type not_needed_names: set of str
336  """
337  if not tolerate_missing_params:
338  missing_argument_names = (
339  (expected_argument_names - found_argument_names)
340  - not_needed_names)
341  if missing_argument_names:
342  self.add_message(
343  message_id,
344  args=(', '.join(
345  sorted(missing_argument_names)),),
346  node=warning_node)
