174 def get_date_format(atrv):
175  r_date = re.compile(r"[Dd][Aa][Tt][Ee]\s+[\"']?(.+?)[\"']?$")
176  m = r_date.match(atrv)
177  if m:
178  pattern = m.group(1).strip()
179  # convert time pattern from Java's SimpleDateFormat to C's format
180  datetime_unit = None
181  if "yyyy" in pattern:
182  pattern = pattern.replace("yyyy", "%Y")
183  datetime_unit = "Y"
184  elif "yy":
185  pattern = pattern.replace("yy", "%y")
186  datetime_unit = "Y"
187  if "MM" in pattern:
188  pattern = pattern.replace("MM", "%m")
189  datetime_unit = "M"
190  if "dd" in pattern:
191  pattern = pattern.replace("dd", "%d")
192  datetime_unit = "D"
193  if "HH" in pattern:
194  pattern = pattern.replace("HH", "%H")
195  datetime_unit = "h"
196  if "mm" in pattern:
197  pattern = pattern.replace("mm", "%M")
198  datetime_unit = "m"
199  if "ss" in pattern:
200  pattern = pattern.replace("ss", "%S")
201  datetime_unit = "s"
202  if "z" in pattern or "Z" in pattern:
203  raise ValueError("Date type attributes with time zone not "
204  "supported, yet")
205 
206  if datetime_unit is None:
207  raise ValueError("Invalid or unsupported date format")
208 
209  return pattern, datetime_unit
210  else:
211  raise ValueError("Invalid or no date format")

224 def tokenize_attribute(iterable, attribute):
225  """Parse a raw string in header (eg starts by @attribute).
226 
227  Given a raw string attribute, try to get the name and type of the
228  attribute. Constraints:
229 
230  * The first line must start with @attribute (case insensitive, and
231  space like characters before @attribute are allowed)
232  * Works also if the attribute is spread on multilines.
233  * Works if empty lines or comments are in between
234 
235  Parameters
236  ----------
237  attribute : str
238  the attribute string.
239 
240  Returns
241  -------
242  name : str
243  name of the attribute
244  value : str
245  value of the attribute
246  next : str
247  next line to be parsed
248 
249  Examples
250  --------
251  If attribute is a string defined in python as r"floupi real", will
252  return floupi as name, and real as value.
253 
254  >>> iterable = iter([0] * 10) # dummy iterator
255  >>> tokenize_attribute(iterable, r"@attribute floupi real")
256  ('floupi', 'real', 0)
257 
258  If attribute is r"'floupi 2' real", will return 'floupi 2' as name,
259  and real as value.
260 
261  >>> tokenize_attribute(iterable, r" @attribute 'floupi 2' real ")
262  ('floupi 2', 'real', 0)
263 
264  """
265  sattr = attribute.strip()
266  mattr = r_attribute.match(sattr)
267  if mattr:
268  # atrv is everything after @attribute
269  atrv = mattr.group(1)
270  if r_comattrval.match(atrv):
271  name, type = tokenize_single_comma(atrv)
272  next_item = next(iterable)
273  elif r_wcomattrval.match(atrv):
274  name, type = tokenize_single_wcomma(atrv)
275  next_item = next(iterable)
276  else:
277  # Not sure we should support this, as it does not seem supported by
278  # weka.
279  raise ValueError("multi line not supported yet")
280  #name, type, next_item = tokenize_multilines(iterable, atrv)
281  else:
282  raise ValueError("First line unparsable: %s" % sattr)
283 
284  if type == 'relational':
285  raise ValueError("relational attributes not supported yet")
286  return name, type, next_item

319 def read_header(ofile):
320  """Read the header of the iterable ofile."""
321  i = next(ofile)
322 
323  # Pass first comments
324  while r_comment.match(i):
325  i = next(ofile)
326 
327  # Header is everything up to DATA attribute ?
328  relation = None
329  attributes = []
330  while not r_datameta.match(i):
331  m = r_headerline.match(i)
332  if m:
333  isattr = r_attribute.match(i)
334  if isattr:
335  name, type, i = tokenize_attribute(ofile, i)
336  attributes.append((name, type))
337  else:
338  isrel = r_relation.match(i)
339  if isrel:
340  relation = isrel.group(1)
341  else:
342  raise ValueError("Error parsing line %s" % i)
343  i = next(ofile)
344  else:
345  i = next(ofile)
346 
347  return relation, attributes

467 def loadarff(f):
468  """
469  Read an arff file.
470 
471  The data is returned as a record array, which can be accessed much like
472  a dictionary of numpy arrays. For example, if one of the attributes is
473  called 'pressure', then its first 10 data points can be accessed from the
474  ``data`` record array like so: ``data['pressure'][0:10]``
475 
476 
477  Parameters
478  ----------
479  f : file-like or str
480  File-like object to read from, or filename to open.
481 
482  Returns
483  -------
484  data : record array
485  The data of the arff file, accessible by attribute names.
486  meta : `MetaData`
487  Contains information about the arff file such as name and
488  type of attributes, the relation (name of the dataset), etc...
489 
490  Raises
491  ------
492  ParseArffError
493  This is raised if the given file is not ARFF-formatted.
494  NotImplementedError
495  The ARFF file has an attribute which is not supported yet.
496 
497  Notes
498  -----
499 
500  This function should be able to read most arff files. Not
501  implemented functionality include:
502 
503  * date type attributes
504  * string type attributes
505 
506  It can read files with numeric and nominal attributes. It cannot read
507  files with sparse data ({} in the file). However, this function can
508  read files with missing data (? in the file), representing the data
509  points as NaNs.
510 
511  Examples
512  --------
513  >>> from scipy.io import arff
514  >>> from cStringIO import StringIO
515  >>> content = \"\"\"
516  ... @relation foo
517  ... @attribute width numeric
518  ... @attribute height numeric
519  ... @attribute color {red,green,blue,yellow,black}
520  ... @data
521  ... 5.0,3.25,blue
522  ... 4.5,3.75,green
523  ... 3.0,4.00,red
524  ... \"\"\"
525  >>> f = StringIO(content)
526  >>> data, meta = arff.loadarff(f)
527  >>> data
528  array([(5.0, 3.25, 'blue'), (4.5, 3.75, 'green'), (3.0, 4.0, 'red')],
529  dtype=[('width', '<f8'), ('height', '<f8'), ('color', '|S6')])
530  >>> meta
531  Dataset: foo
532  \twidth's type is numeric
533  \theight's type is numeric
534  \tcolor's type is nominal, range is ('red', 'green', 'blue', 'yellow', 'black')
535 
536  """
537  if hasattr(f, 'read'):
538  ofile = f
539  else:
540  ofile = open(f, 'rt')
541  try:
542  return _loadarff(ofile)
543  finally:
544  if ofile is not f: # only close what we opened
545  ofile.close()

548 def _loadarff(ofile):
549  # Parse the header file
550  try:
551  rel, attr = read_header(ofile)
552  except ValueError as e:
553  msg = "Error while parsing header, error was: " + str(e)
554  raise ParseArffError(msg)
555 
556  # Check whether we have a string attribute (not supported yet)
557  hasstr = False
558  for name, value in attr:
559  type = parse_type(value)
560  if type == 'string':
561  hasstr = True
562 
563  meta = MetaData(rel, attr)
564 
565  # XXX The following code is not great
566  # Build the type descriptor descr and the list of convertors to convert
567  # each attribute to the suitable type (which should match the one in
568  # descr).
569 
570  # This can be used once we want to support integer as integer values and
571  # not as numeric anymore (using masked arrays ?).
572  acls2dtype = {'real': float, 'integer': float, 'numeric': float}
573  acls2conv = {'real': safe_float,
574  'integer': safe_float,
575  'numeric': safe_float}
576  descr = []
577  convertors = []
578  if not hasstr:
579  for name, value in attr:
580  type = parse_type(value)
581  if type == 'date':
582  date_format, datetime_unit = get_date_format(value)
583  descr.append((name, "datetime64[%s]" % datetime_unit))
584  convertors.append(partial(safe_date, date_format=date_format,
585  datetime_unit=datetime_unit))
586  elif type == 'nominal':
587  n = maxnomlen(value)
588  descr.append((name, 'S%d' % n))
589  pvalue = get_nom_val(value)
590  convertors.append(partial(safe_nominal, pvalue=pvalue))
591  else:
592  descr.append((name, acls2dtype[type]))
593  convertors.append(safe_float)
594  #dc.append(acls2conv[type])
595  #sdescr.append((name, acls2sdtype[type]))
596  else:
597  # How to support string efficiently ? Ideally, we should know the max
598  # size of the string before allocating the numpy array.
599  raise NotImplementedError("String attributes not supported yet, sorry")

601  ni = len(convertors)

605  # could be made more efficiently:
606  # - We could for example "compile" the function, because some values
607  # do not change here.
608  # - The function to convert a line to dtyped values could also be
609  # generated on the fly from a string and be executed instead of
610  # looping.
611  # - The regex are overkill: for comments, checking that a line starts
612  # by % should be enough and faster, and for empty lines, same thing
613  # --> this does not seem to change anything.
614 
615  # 'compiling' the range since it does not change
616  # Note, I have already tried zipping the converters and
617  # row elements and got slightly worse performance.
618  elems = list(range(ni))
619 
620  for raw in row_iter:
621  # We do not abstract skipping comments and empty lines for

623  if r_comment.match(raw) or r_empty.match(raw):

625  row = raw.split(delim)
