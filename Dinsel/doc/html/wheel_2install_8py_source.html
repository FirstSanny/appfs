105  def tags(self):
106  """A wheel file is compatible with the Cartesian product of the
107  period-delimited tags in its filename.
108  To choose a wheel file among several candidates having the same
109  distribution version 'ver', an installer ranks each triple of
110  (pyver, abi, plat) that its Python installation can run, sorting
111  the wheels by the best-ranked tag it supports and then by their
112  arity which is just len(list(compatibility_tags)).
113  """
114  tags = self.parsed_filename.groupdict()
115  for pyver in tags['pyver'].split('.'):
116  for abi in tags['abi'].split('.'):
117  for plat in tags['plat'].split('.'):
118  yield (pyver, abi, plat)

264  def get_path(key):

273  # Parse all the names in the archive
274  name_trans = {}
275  for info in self.zipfile.infolist():
276  name = info.filename
277  # Zip files can contain entries representing directories.
278  # These end in a '/'.
279  # We ignore these, as we create directories on demand.
280  if name.endswith('/'):

284  # In theory, paths could start with ./ or have other oddities
285  # but this won't happen in practical cases of well-formed wheels.
286  # We'll cover the simple case of an initial './' as it's both easy
287  # to do and more common than most other oddities.
288  if name.startswith('./'):
289  name = name[2:]
290

292  # installed in non-root locations
293  basedir, sep, filename = name.partition('/')
294  if sep and basedir == self.datadir_name:
295  # Data file. Target destination is elsewhere
296  key, sep, filename = filename.partition('/')
297  if not sep:
298  raise ValueError("Invalid filename in wheel: {0}".format(name))
299  target = get_path(key)
300  else:
301  # Normal file. Target destination is root
302  key = ''
303  target = root
304  filename = name
305 
306  # Map the actual filename from the zipfile to its intended target
307  # directory and the pathname relative to that directory.
308  dest = os.path.normpath(os.path.join(target, filename))

312  # is as follows:

315  # 2. Overwrite check - do any of the files to be installed already

318  # 4. Update RECORD - write a suitably modified RECORD file to

369  def verify(self, zipfile=None):
370  """Configure the VerifyingZipFile `zipfile` by verifying its signature
371  and setting expected hashes for every hash in RECORD.
372  Caller must complete the verification process by completely reading
373  every file in the archive (e.g. with extractall)."""
374  sig = None
375  if zipfile is None:
376  zipfile = self.zipfile
377  zipfile.strict = True
378 
379  record_name = '/'.join((self.distinfo_name, 'RECORD'))
380  sig_name = '/'.join((self.distinfo_name, 'RECORD.jws'))
381  # tolerate s/mime signatures:
382  smime_sig_name = '/'.join((self.distinfo_name, 'RECORD.p7s'))
383  zipfile.set_expected_hash(record_name, None)
384  zipfile.set_expected_hash(sig_name, None)
385  zipfile.set_expected_hash(smime_sig_name, None)
386  record = zipfile.read(record_name)
387 
388  record_digest = urlsafe_b64encode(hashlib.sha256(record).digest())
389  try:
390  sig = from_json(native(zipfile.read(sig_name)))
391  except KeyError: # no signature
392  pass
393  if sig:
394  headers, payload = signatures.verify(sig)
395  if payload['hash'] != "sha256=" + native(record_digest):
396  msg = "RECORD.sig claimed RECORD hash {0} != computed hash {1}."
397  raise BadWheelFile(msg.format(payload['hash'],
398  native(record_digest)))
399 
400  reader = csv.reader((native(r) for r in record.splitlines()))
401 
402  for row in reader:
403  filename = row[0]
404  hash = row[1]
405  if not hash:
406  if filename not in (record_name, sig_name):
407  sys.stderr.write("%s has no hash!\n" % filename)
408  continue
409  algo, data = row[1].split('=', 1)
410  assert algo == "sha256", "Unsupported hash algorithm"
411  zipfile.set_expected_hash(filename, urlsafe_b64decode(binary(data)))

436  """Return file-like object for 'name'."""

438  ef = zipfile.ZipFile.open(self, name_or_info, mode, pwd)

441  else:
442  name = name_or_info

446  try:
447  _update_crc_orig = ef._update_crc
448  except AttributeError:
449  warnings.warn('Need ZipExtFile._update_crc to implement '
450  'file hash verification (in Python >= 2.7)')
451  return ef
452  running_hash = self._hash_algorithm()
453  if hasattr(ef, '_eof'): # py33
454  def _update_crc(data):
455  _update_crc_orig(data)
456  running_hash.update(data)
457  if ef._eof and running_hash.digest() != expected_hash:
458  raise BadWheelFile("Bad hash for file %r" % ef.name)
459  else:
460  def _update_crc(data, eof=None):
461  _update_crc_orig(data, eof=eof)
462  running_hash.update(data)
463  if eof and running_hash.digest() != expected_hash:
464  raise BadWheelFile("Bad hash for file %r" % ef.name)
465  ef._update_crc = _update_crc
466  elif self.strict and name not in self._expected_hashes:
467  raise BadWheelFile("No expected hash for file %r" % ef.name)
