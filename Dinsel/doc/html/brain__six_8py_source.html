21 def _indent(text, prefix, predicate=None):
22  """Adds 'prefix' to the beginning of selected lines in 'text'.
23 
24  If 'predicate' is provided, 'prefix' will only be added to the lines
25  where 'predicate(line)' is True. If 'predicate' is not provided,
26  it will default to adding 'prefix' to all non-empty lines that do not
27  consist solely of whitespace characters.
28  """
29  if predicate is None:
30  predicate = lambda line: line.strip()
31 
32  def prefixed_lines():
33  for line in text.splitlines(True):
34  yield prefix + line if predicate(line) else line
35  return ''.join(prefixed_lines())

250 def transform_six_add_metaclass(node):
251  """Check if the given class node is decorated with *six.add_metaclass*
252 
253  If so, inject its argument as the metaclass of the underlying class.
254  """
255  if not node.decorators:
256  return
257 
258  for decorator in node.decorators.nodes:
259  if not isinstance(decorator, nodes.Call):
260  continue
261 
262  try:
263  func = next(decorator.func.infer())
264  except InferenceError:
265  continue
266  if func.qname() == SIX_ADD_METACLASS and decorator.args:
267  metaclass = decorator.args[0]
268  node._metaclass = metaclass
269  return node
