106  NpzFile(fid)
107 
108  A dictionary-like object with lazy-loading of files in the zipped
109  archive provided on construction.
110

112  format. It assumes that files in the archive have a ``.npy`` extension,
113  other files are ignored.
114

116  getitem access using ``obj['key']`` or attribute lookup using
117  ``obj.f.key``. A list of all files (without ``.npy`` extensions) can
118  be obtained with ``obj.files`` and the ZipFile object itself using
119  ``obj.zip``.
120

124  List of all files in the archive with a ``.npy`` extension.
125  zip : ZipFile instance

127  f : BagObj instance

129  to getitem access on the `NpzFile` instance itself.

131  Allow loading pickled data. Default: True
132  pickle_kwargs : dict, optional
133  Additional keyword arguments to pass on to pickle.load.
134  These are only useful when loading object arrays saved on
135  Python 2 when using Python 3.
136 
137  Parameters
138  ----------
139  fid : file or str
140  The zipped archive to open. This is either a file-like object
141  or a string containing the path to the archive.

266 def load(file, mmap_mode=None, allow_pickle=True, fix_imports=True,
267  encoding='ASCII'):
268  """
269  Load arrays or pickled objects from ``.npy``, ``.npz`` or pickled files.
270 
271  Parameters
272  ----------
273  file : file-like object, string, or pathlib.Path
274  The file to read. File-like objects must support the
275  ``seek()`` and ``read()`` methods. Pickled files require that the
276  file-like object support the ``readline()`` method as well.
277  mmap_mode : {None, 'r+', 'r', 'w+', 'c'}, optional
278  If not None, then memory-map the file, using the given mode (see
279  `numpy.memmap` for a detailed description of the modes). A
280  memory-mapped array is kept on disk. However, it can be accessed
281  and sliced like any ndarray. Memory mapping is especially useful
282  for accessing small fragments of large files without reading the
283  entire file into memory.
284  allow_pickle : bool, optional
285  Allow loading pickled object arrays stored in npy files. Reasons for
286  disallowing pickles include security, as loading pickled data can
287  execute arbitrary code. If pickles are disallowed, loading object
288  arrays will fail.
289  Default: True
290  fix_imports : bool, optional
291  Only useful when loading Python 2 generated pickled files on Python 3,
292  which includes npy/npz files containing object arrays. If `fix_imports`
293  is True, pickle will try to map the old Python 2 names to the new names
294  used in Python 3.
295  encoding : str, optional
296  What encoding to use when reading Python 2 strings. Only useful when
297  loading Python 2 generated pickled files on Python 3, which includes
298  npy/npz files containing object arrays. Values other than 'latin1',
299  'ASCII', and 'bytes' are not allowed, as they can corrupt numerical
300  data. Default: 'ASCII'
301 
302  Returns
303  -------
304  result : array, tuple, dict, etc.
305  Data stored in the file. For ``.npz`` files, the returned instance
306  of NpzFile class must be closed to avoid leaking file descriptors.
307 
308  Raises
309  ------
310  IOError
311  If the input file does not exist or cannot be read.
312  ValueError
313  The file contains an object array, but allow_pickle=False given.
314 
315  See Also
316  --------
317  save, savez, savez_compressed, loadtxt
318  memmap : Create a memory-map to an array stored in a file on disk.
319  lib.format.open_memmap : Create or load a memory-mapped ``.npy`` file.
320 
321  Notes
322  -----
323  - If the file contains pickle data, then whatever object is stored
324  in the pickle is returned.
325  - If the file is a ``.npy`` file, then a single array is returned.
326  - If the file is a ``.npz`` file, then a dictionary-like object is
327  returned, containing ``{filename: array}`` key-value pairs, one for
328  each file in the archive.
329  - If the file is a ``.npz`` file, the returned value supports the
330  context manager protocol in a similar fashion to the open function::
331 
332  with load('foo.npz') as data:
333  a = data['a']
334 
335  The underlying file descriptor is closed when exiting the 'with'
336  block.
337 
338  Examples
339  --------
340  Store data to disk, and load it again:
341 
342  >>> np.save('/tmp/123', np.array([[1, 2, 3], [4, 5, 6]]))
343  >>> np.load('/tmp/123.npy')
344  array([[1, 2, 3],
345  [4, 5, 6]])
346 
347  Store compressed data to disk, and load it again:
348 
349  >>> a=np.array([[1, 2, 3], [4, 5, 6]])
350  >>> b=np.array([1, 2])
351  >>> np.savez('/tmp/123.npz', a=a, b=b)
352  >>> data = np.load('/tmp/123.npz')
353  >>> data['a']
354  array([[1, 2, 3],
355  [4, 5, 6]])
356  >>> data['b']
357  array([1, 2])
358  >>> data.close()
359 
360  Mem-map the stored array, and then access the second row
361  directly from disk:
362 
363  >>> X = np.load('/tmp/123.npy', mmap_mode='r')
364  >>> X[1, :]
365  memmap([4, 5, 6])
366 
367  """
368  own_fid = False
369  if isinstance(file, basestring):
370  fid = open(file, "rb")
371  own_fid = True
372  elif is_pathlib_path(file):
373  fid = file.open("rb")
374  own_fid = True
375  else:
376  fid = file
377 
378  if encoding not in ('ASCII', 'latin1', 'bytes'):
379  # The 'encoding' value for pickle also affects what encoding
380  # the serialized binary data of NumPy arrays is loaded
381  # in. Pickle does not pass on the encoding information to
382  # NumPy. The unpickling code in numpy.core.multiarray is
383  # written to assume that unicode data appearing where binary
384  # should be is in 'latin1'. 'bytes' is also safe, as is 'ASCII'.
385  #
386  # Other encoding values can corrupt binary data, and we
387  # purposefully disallow them. For the same reason, the errors=
388  # argument is not exposed, as values other than 'strict'
389  # result can similarly silently corrupt numerical data.
390  raise ValueError("encoding must be 'ASCII', 'latin1', or 'bytes'")
391 
392  if sys.version_info[0] >= 3:
393  pickle_kwargs = dict(encoding=encoding, fix_imports=fix_imports)
394  else:
395  # Nothing to do on Python 2
396  pickle_kwargs = {}
397 
398  try:
399  # Code to distinguish from NumPy binary files and pickles.
400  _ZIP_PREFIX = asbytes('PK\x03\x04')
401  N = len(format.MAGIC_PREFIX)
402  magic = fid.read(N)
403  # If the file size is less than N, we need to make sure not
404  # to seek past the beginning of the file
405  fid.seek(-min(N, len(magic)), 1) # back-up
406  if magic.startswith(_ZIP_PREFIX):
407  # zip-file (assume .npz)
408  # Transfer file ownership to NpzFile
409  tmp = own_fid
410  own_fid = False
411  return NpzFile(fid, own_fid=tmp, allow_pickle=allow_pickle,
412  pickle_kwargs=pickle_kwargs)
413  elif magic == format.MAGIC_PREFIX:
414  # .npy file
415  if mmap_mode:
416  return format.open_memmap(file, mode=mmap_mode)
417  else:
418  return format.read_array(fid, allow_pickle=allow_pickle,
419  pickle_kwargs=pickle_kwargs)
420  else:
421  # Try a pickle
422  if not allow_pickle:
423  raise ValueError("allow_pickle=False, but file does not contain "
424  "non-pickled data")
425  try:
426  return pickle.load(fid, **pickle_kwargs)
427  except:
428  raise IOError(
429  "Failed to interpret file %s as a pickle" % repr(file))
430  finally:
431  if own_fid:
432  fid.close()

622 def _savez(file, args, kwds, compress, allow_pickle=True, pickle_kwargs=None):
623  # Import is postponed to here since zipfile depends on gzip, an optional
624  # component of the so-called standard library.
625  import zipfile
626  # Import deferred for startup time improvement
627  import tempfile
628 
629  if isinstance(file, basestring):
630  if not file.endswith('.npz'):
631  file = file + '.npz'
632  elif is_pathlib_path(file):
633  if not file.name.endswith('.npz'):
634  file = file.parent / (file.name + '.npz')
635 
636  namedict = kwds
637  for i, val in enumerate(args):
638  key = 'arr_%d' % i
639  if key in namedict.keys():
640  raise ValueError(
641  "Cannot use un-named variables and keyword %s" % key)
642  namedict[key] = val
643 
644  if compress:
645  compression = zipfile.ZIP_DEFLATED
646  else:
647  compression = zipfile.ZIP_STORED
648 
649  zipf = zipfile_factory(file, mode="w", compression=compression)
650 
651  # Stage arrays in a temporary file on disk, before writing to zip.
652 
653  # Since target file might be big enough to exceed capacity of a global
654  # temporary directory, create temp file side-by-side with the target file.
655  file_dir, file_prefix = os.path.split(file) if _is_string_like(file) else (None, 'tmp')
656  fd, tmpfile = tempfile.mkstemp(prefix=file_prefix, dir=file_dir, suffix='-numpy.npy')
657  os.close(fd)
658  try:
659  for key, val in namedict.items():
660  fname = key + '.npy'

662  try:
663  format.write_array(fid, np.asanyarray(val),
664  allow_pickle=allow_pickle,
665  pickle_kwargs=pickle_kwargs)
666  fid.close()
667  fid = None
668  zipf.write(tmpfile, arcname=fname)
669  except IOError as exc:
670  raise IOError("Failed to write to %s: %s" % (tmpfile, exc))
671  finally:
672  if fid:
673  fid.close()
674  finally:

714  Load data from a text file.
715 
716  Each row in the text file must have the same number of values.
717 
718  Parameters
719  ----------
720  fname : file, str, or pathlib.Path
721  File, filename, or generator to read. If the filename extension is
722  ``.gz`` or ``.bz2``, the file is first decompressed. Note that
723  generators should return byte strings for Python 3k.
724  dtype : data-type, optional
725  Data-type of the resulting array; default: float. If this is a
726  structured data-type, the resulting array will be 1-dimensional, and
727  each row will be interpreted as an element of the array. In this
728  case, the number of columns used must match the number of fields in
729  the data-type.
730  comments : str or sequence, optional

732  comment;
733  default: '#'.

740  ``converters = {0: datestr2num}``. Converters can also be used to
741  provide a default value for missing data (but see also `genfromtxt`):
742  ``converters = {3: lambda s: float(s.strip() or 0)}``. Default: None.
743  skiprows : int, optional

912  def split_line(line):
913  """Chop off comments, strip, and split at delimiter.
914 
915  Note that although the file is opened as text, this function
916  returns bytes.
917 
918  """
919  line = asbytes(line)
920  if comments is not None:
921  line = regex_comments.split(asbytes(line), maxsplit=1)[0]
922  line = line.strip(asbytes('\r\n'))
923  if line:
924  return line.split(delimiter)
925  else:
926  return []
927 
928  try:
929  # Make sure we're dealing with a proper dtype
930  dtype = np.dtype(dtype)
931  defconv = _getconv(dtype)
932 
933  # Skip the first `skiprows` lines
934  for i in range(skiprows):
935  next(fh)
936 
937  # Read until we find a line with some values, and use
938  # it to estimate the number of columns, N.
939  first_vals = None
940  try:
941  while not first_vals:
942  first_line = next(fh)
943  first_vals = split_line(first_line)
944  except StopIteration:
945  # End of lines reached
946  first_line = ''
947  first_vals = []

949  N = len(usecols or first_vals)
950 
951  dtype_types, packing = flatten_dtype(dtype)
952  if len(dtype_types) > 1:
953  # We're dealing with a structured array, each field of
954  # the dtype matches a column
955  converters = [_getconv(dt) for dt in dtype_types]
956  else:
957  # All fields have the same dtype
958  converters = [defconv for i in range(N)]
959  if N > 1:
960  packing = [(N, tuple)]
961 
962  # By preference, use the converters specified by the user
963  for i, conv in (user_converters or {}).items():
964  if usecols:
965  try:
966  i = usecols.index(i)
967  except ValueError:
968  # Unused converter specified
969  continue
970  converters[i] = conv
971 
972  # Parse each line, including the first
973  for i, line in enumerate(itertools.chain([first_line], fh)):
974  vals = split_line(line)
975  if len(vals) == 0:
976  continue
977  if usecols:
978  vals = [vals[i] for i in usecols]
979  if len(vals) != N:
980  line_num = i + skiprows + 1
981  raise ValueError("Wrong number of columns at line %d"
982  % line_num)
983 
984  # Convert each value according to its column and store
985  items = [conv(val) for (conv, val) in zip(converters, vals)]
986  # Then pack it according to the dtype's nesting
987  items = pack_items(items, packing)
988  X.append(items)
989  finally:
990  if fown:
991  fh.close()

994  # Multicolumn data are returned with shape (1, N, M), i.e.
995  # (1, 1, M) for a single row - remove the singleton dimension there
996  if X.ndim == 3 and X.shape[:2] == (1, 1):
997  X.shape = (1, -1)
998 
999  # Verify that the array has at least dimensions `ndmin`.
1000  # Check correctness of the values of `ndmin`
1001  if ndmin not in [0, 1, 2]:
1002  raise ValueError('Illegal value of ndmin keyword: %s' % ndmin)
1003  # Tweak the size and shape of the arrays - remove extraneous dimensions
1004  if X.ndim > ndmin:
1005  X = np.squeeze(X)
1006  # and ensure we have the minimum number of dimensions asked for
1007  # - has to be in this order for the odd case ndmin=1, X.squeeze().ndim=0
1008  if X.ndim < ndmin:
1009  if ndmin == 1:
1010  X = np.atleast_1d(X)
1011  elif ndmin == 2:
1012  X = np.atleast_2d(X).T
1013 
1014  if unpack:
1015  if len(dtype_types) > 1:
1016  # For structured arrays, return an array for each field.
1017  return [X[field] for field in dtype.names]
1018  else:
1019  return X.T

1021  return X

1024 def savetxt(fname, X, fmt='%.18e', delimiter=' ', newline='\n', header='',
1025  footer='', comments='# '):
1026  """
1027  Save an array to a text file.
1028 
1029  Parameters
1030  ----------
1031  fname : filename or file handle
1032  If the filename ends in ``.gz``, the file is automatically saved in
1033  compressed gzip format. `loadtxt` understands gzipped files
1034  transparently.
1035  X : array_like
1036  Data to be saved to a text file.
1037  fmt : str or sequence of strs, optional
1038  A single format (%10.5f), a sequence of formats, or a
1039  multi-format string, e.g. 'Iteration %d -- %10.5f', in which
1040  case `delimiter` is ignored. For complex `X`, the legal options
1041  for `fmt` are:
1042  a) a single specifier, `fmt='%.4e'`, resulting in numbers formatted
1043  like `' (%s+%sj)' % (fmt, fmt)`
1044  b) a full string specifying every real and imaginary part, e.g.
1045  `' %.4e %+.4ej %.4e %+.4ej %.4e %+.4ej'` for 3 columns
1046  c) a list of specifiers, one per column - in this case, the real
1047  and imaginary part must have separate specifiers,
1048  e.g. `['%.3e + %.3ej', '(%.15e%+.15ej)']` for 2 columns
1049  delimiter : str, optional
1050  String or character separating columns.
1051  newline : str, optional
1052  String or character separating lines.
1053 
1054  .. versionadded:: 1.5.0
1055  header : str, optional
1056  String that will be written at the beginning of the file.
1057 
1058  .. versionadded:: 1.7.0
1059  footer : str, optional
1060  String that will be written at the end of the file.
1061 
1062  .. versionadded:: 1.7.0
1063  comments : str, optional
1064  String that will be prepended to the ``header`` and ``footer`` strings,
1065  to mark them as comments. Default: '# ', as expected by e.g.
1066  ``numpy.loadtxt``.
1067 
1068  .. versionadded:: 1.7.0
1069 
1070 
1071  See Also
1072  --------
1073  save : Save an array to a binary file in NumPy ``.npy`` format
1074  savez : Save several arrays into an uncompressed ``.npz`` archive
1075  savez_compressed : Save several arrays into a compressed ``.npz`` archive
1076 
1077  Notes
1078  -----
1079  Further explanation of the `fmt` parameter
1080  (``%[flag]width[.precision]specifier``):
1081 
1082  flags:
1083  ``-`` : left justify
1084 
1085  ``+`` : Forces to precede result with + or -.
1086 
1087  ``0`` : Left pad the number with zeros instead of space (see width).
1088 
1089  width:
1090  Minimum number of characters to be printed. The value is not truncated
1091  if it has more characters.
1092 
1093  precision:
1094  - For integer specifiers (eg. ``d,i,o,x``), the minimum number of
1095  digits.
1096  - For ``e, E`` and ``f`` specifiers, the number of digits to print
1097  after the decimal point.
1098  - For ``g`` and ``G``, the maximum number of significant digits.
1099  - For ``s``, the maximum number of characters.
1100 
1101  specifiers:
1102  ``c`` : character
1103 
1104  ``d`` or ``i`` : signed decimal integer
1105 
1106  ``e`` or ``E`` : scientific notation with ``e`` or ``E``.
1107 
1108  ``f`` : decimal floating point
1109 
1110  ``g,G`` : use the shorter of ``e,E`` or ``f``
1111 
1112  ``o`` : signed octal
1113 
1114  ``s`` : string of characters
1115 
1116  ``u`` : unsigned decimal integer
1117 
1118  ``x,X`` : unsigned hexadecimal integer
1119 
1120  This explanation of ``fmt`` is not complete, for an exhaustive
1121  specification see [1]_.
1122 
1123  References
1124  ----------
1125  .. [1] `Format Specification Mini-Language
1126  <http://docs.python.org/library/string.html#
1127  format-specification-mini-language>`_, Python Documentation.
1128 
1129  Examples
1130  --------
1131  >>> x = y = z = np.arange(0.0,5.0,1.0)
1132  >>> np.savetxt('test.out', x, delimiter=',') # X is an array
1133  >>> np.savetxt('test.out', (x,y,z)) # x,y,z equal sized 1D arrays
1134  >>> np.savetxt('test.out', x, fmt='%1.4e') # use exponential notation
1135 
1136  """
1137 
1138  # Py3 conversions first
1139  if isinstance(fmt, bytes):
1140  fmt = asstr(fmt)
1141  delimiter = asstr(delimiter)
1142 
1143  own_fh = False
1144  if is_pathlib_path(fname):
1145  fname = str(fname)
1146  if _is_string_like(fname):
1147  own_fh = True
1148  if fname.endswith('.gz'):
1149  import gzip
1150  fh = gzip.open(fname, 'wb')
1151  else:
1152  if sys.version_info[0] >= 3:
1153  fh = open(fname, 'wb')
1154  else:
1155  fh = open(fname, 'w')

1157  fh = fname

1159  raise ValueError('fname must be a string or file handle')

1162  X = np.asarray(X)
1163 
1164  # Handle 1-dimensional arrays
1165  if X.ndim == 1:
1166  # Common case -- 1d array of numbers
1167  if X.dtype.names is None:
1168  X = np.atleast_2d(X).T
1169  ncol = 1
1170 
1171  # Complex dtype -- each field indicates a separate column
1172  else:
1173  ncol = len(X.dtype.descr)
1174  else:
1175  ncol = X.shape[1]
1176 
1177  iscomplex_X = np.iscomplexobj(X)
1178  # `fmt` can be a string with multiple insertion points or a

1180  if type(fmt) in (list, tuple):

1182  raise AttributeError('fmt has wrong shape. %s' % str(fmt))

1209  row2.append(number.real)

1215  fh.write(asbytes(format % tuple(row) + newline))
1216  except TypeError:
1217  raise TypeError("Mismatch between array dtype ('%s') and "

1221  footer = footer.replace('\n', '\n' + comments)
1222  fh.write(asbytes(comments + footer + newline))
1223  finally:
1224  if own_fh:
1225  fh.close()

1231

1233  all matches of the regular expression in the file. Groups in the regular
1234  expression are converted to fields of the structured array.
1235

1237  ----------
1238  file : str or file
1239  File name or file object to read.

1243  dtype : dtype or list of dtypes
1244  Dtype for the structured array.
1245 
1246  Returns
1247  -------
1248  output : ndarray

1250  was matched by `regexp`. `output` is always a structured array.

1256

1259  fromstring, loadtxt

1264  forms specify at least the data type and field name. For details see
1265  `doc.structured_arrays`.
