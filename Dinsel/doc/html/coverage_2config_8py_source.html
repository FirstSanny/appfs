50  def get(self, section, *args, **kwargs): # pylint: disable=arguments-differ
51  """Get a value, replacing environment variables also.
52 
53  The arguments are the same as `RawConfigParser.get`, but in the found
54  value, ``$WORD`` or ``${WORD}`` are replaced by the value of the
55  environment variable ``WORD``.
56 
57  Returns the finished value.
58 
59  """
60  section = self.section_prefix + section
61  v = configparser.RawConfigParser.get(self, section, *args, **kwargs)
62  def dollar_replace(m):
63  """Called for each $replacement."""
64  # Only one of the groups will have matched, just get its text.
65  word = next(w for w in m.groups() if w is not None) # pragma: part covered
66  if word == "$":
67  return "$"
68  else:
69  return os.environ.get(word, '')
70 
71  dollar_pattern = r"""(?x) # Use extended regex syntax
72  \$(?: # A dollar sign, then
73  (?P<v1>\w+) | # a plain word,
74  {(?P<v2>\w+)} | # or a {-wrapped word,
75  (?P<char>[$]) # or a dollar sign.
76  )
77  """
78  v = re.sub(dollar_pattern, dollar_replace, v)
79  return v

83

101

103  list of regexes. Each value is stripped of whitespace.

105  Returns the list of strings.

206  def from_file(self, filename, section_prefix=""):
207  """Read configuration from a .rc file.
208 
209  `filename` is a file name to read.
210 
211  Returns True or False, whether the file could be read, and it had some
212  coverage.py settings in it.
213 
214  """
215  self.attempted_config_files.append(filename)
216 
217  cp = HandyConfigParser(section_prefix)
218  try:
219  files_read = cp.read(filename)
220  except configparser.Error as err:
221  raise CoverageException("Couldn't read config file %s: %s" % (filename, err))
222  if not files_read:
223  return False
224 
225  self.config_files.extend(files_read)
226 
227  any_set = False
228  try:
229  for option_spec in self.CONFIG_FILE_OPTIONS:
230  was_set = self._set_attr_from_config_option(cp, *option_spec)
231  if was_set:
232  any_set = True
233  except ValueError as err:
234  raise CoverageException("Couldn't read config file %s: %s" % (filename, err))
235 
236  # Check that there are no unrecognized options.
237  all_options = collections.defaultdict(set)
238  for option_spec in self.CONFIG_FILE_OPTIONS:
239  section, option = option_spec[1].split(":")
240  all_options[section].add(option)
241 
242  for section, options in iitems(all_options):
243  if cp.has_section(section):
244  for unknown in set(cp.options(section)) - options:
245  if section_prefix:
246  section = section_prefix + section
247  raise CoverageException(
248  "Unrecognized option '[%s] %s=' in config file %s" % (
249  section, unknown, filename

254  if cp.has_section('paths'):
255  for option in cp.options('paths'):
256  self.paths[option] = cp.getlist('paths', option)
257  any_set = True
258

262  self.plugin_options[plugin] = cp.get_section(plugin)
263  any_set = True
264 
265  # Was this file used as a config file? If no prefix, then it was used.
266  # If a prefix, then it was only used if we found some settings in it.

269  else:

388 def read_coverage_config(config_file, **kwargs):
389  """Read the coverage.py configuration.
390 
391  Arguments:
392  config_file: a boolean or string, see the `Coverage` class for the
393  tricky details.
394  all others: keyword arguments from the `Coverage` class, used for
395  setting values in the configuration.
396 
397  Returns:
398  config_file, config:
399  config_file is the value to use for config_file in other
400  invocations of coverage.
401 
402  config is a CoverageConfig object read from the appropriate
403  configuration file.
404 
405  """
406  # Build the configuration from a number of sources:
407  # 1) defaults:
408  config = CoverageConfig()
409 
410  # 2) from a file:
411  if config_file:
412  # Some API users were specifying ".coveragerc" to mean the same as
413  # True, so make it so.
414  if config_file == ".coveragerc":
415  config_file = True
416  specified_file = (config_file is not True)
417  if not specified_file:
418  config_file = ".coveragerc"
419 
420  for fname, prefix in [(config_file, ""),
421  ("setup.cfg", "coverage:"),
422  ("tox.ini", "coverage:")]:
423  config_read = config.from_file(fname, section_prefix=prefix)
424  is_config_file = fname == config_file
425 
426  if not config_read and is_config_file and specified_file:
427  raise CoverageException("Couldn't read '%s' as a config file" % fname)
428 
429  if config_read:
430  break
431 
432  # 3) from environment variables:
433  env_data_file = os.environ.get('COVERAGE_FILE')
434  if env_data_file:
435  config.data_file = env_data_file
436  debugs = os.environ.get('COVERAGE_DEBUG')
437  if debugs:
438  config.debug.extend(d.strip() for d in debugs.split(","))
439 
440  # 4) from constructor arguments:
441  config.from_args(**kwargs)
442 
443  return config_file, config
