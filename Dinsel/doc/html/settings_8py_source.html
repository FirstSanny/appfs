142 def _update_with_config_file(file_path, sections, computed_settings):
143  settings = _get_config_data(file_path, sections).copy()
144  if not settings:
145  return
146 
147  if file_path.endswith('.editorconfig'):
148  indent_style = settings.pop('indent_style', '').strip()
149  indent_size = settings.pop('indent_size', '').strip()
150  if indent_style == 'space':
151  computed_settings['indent'] = ' ' * (indent_size and int(indent_size) or 4)
152  elif indent_style == 'tab':
153  computed_settings['indent'] = '\t' * (indent_size and int(indent_size) or 1)
154 
155  max_line_length = settings.pop('max_line_length', '').strip()
156  if max_line_length:
157  computed_settings['line_length'] = float('inf') if max_line_length == 'off' else int(max_line_length)
158 
159  for key, value in itemsview(settings):
160  access_key = key.replace('not_', '').lower()
161  existing_value_type = type(default.get(access_key, ''))
162  if existing_value_type in (list, tuple):
163  # sections has fixed order values; no adding or substraction from any set
164  if access_key == 'sections':
165  computed_settings[access_key] = tuple(_as_list(value))
166  else:
167  existing_data = set(computed_settings.get(access_key, default.get(access_key)))
168  if key.startswith('not_'):
169  computed_settings[access_key] = list(existing_data.difference(_as_list(value)))
170  else:
171  computed_settings[access_key] = list(existing_data.union(_as_list(value)))
172  elif existing_value_type == bool and value.lower().strip() == 'false':
173  computed_settings[access_key] = False
174  elif key.startswith('known_'):
175  computed_settings[access_key] = list(_as_list(value))
176  elif key == 'force_grid_wrap':
177  try:
178  result = existing_value_type(value)
179  except ValueError:
180  # backwards compat
181  result = default.get(access_key) if value.lower().strip() == 'false' else 2
182  computed_settings[access_key] = result
183  else:
184  computed_settings[access_key] = existing_value_type(value)

216 def should_skip(filename, config, path='/'):
217  """Returns True if the file should be skipped based on the passed in settings."""
218  for skip_path in config['skip']:
219  if os.path.join(path, filename).endswith('/' + skip_path.lstrip('/')):
220  return True
221 
222  position = os.path.split(filename)
223  while position[1]:
224  if position[1] in config['skip']:
225  return True
226  position = os.path.split(position[0])
227 
228  for glob in config['skip_glob']:
229  if fnmatch.fnmatch(filename, glob):
230  return True
231 
232  return False
