44 def upfirdn_naive(x, h, up=1, down=1):
45  """Naive upfirdn processing in Python
46 
47  Note: arg order (x, h) differs to facilitate apply_along_axis use.
48  """
49  h = np.asarray(h)
50  out = np.zeros(len(x) * up, x.dtype)
51  out[::up] = x
52  out = np.convolve(h, out)[::down][:_output_len(len(h), len(x), up, down)]
53  return out

81  x += 1j * self.rng.randn(*size)

88  def scrub(self, x, axis=-1):
89  yr = np.apply_along_axis(upfirdn_naive, axis, x,
90  self.h, self.up, self.down)
91  y = upfirdn(self.h, x, self.up, self.down, axis=axis)
92  dtypes = (self.h.dtype, x.dtype)
93  if all(d == np.complex64 for d in dtypes):
94  assert_equal(y.dtype, np.complex64)
95  elif np.complex64 in dtypes and np.float32 in dtypes:
96  assert_equal(y.dtype, np.complex64)
97  elif all(d == np.float32 for d in dtypes):
98  assert_equal(y.dtype, np.float32)
99  elif np.complex128 in dtypes or np.complex64 in dtypes:
100  assert_equal(y.dtype, np.complex128)
101  else:
102  assert_equal(y.dtype, np.float64)
103  assert_allclose(yr, y)
