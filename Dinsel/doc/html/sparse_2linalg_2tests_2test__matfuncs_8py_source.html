31 def _burkardt_13_power(n, p):
32  """
33  A helper function for testing matrix functions.
34 
35  Parameters
36  ----------
37  n : integer greater than 1
38  Order of the square matrix to be returned.
39  p : non-negative integer
40  Power of the matrix.
41 
42  Returns
43  -------
44  out : ndarray representing a square matrix
45  A Forsythe matrix of order n, raised to the power p.
46 
47  """
48  # Input validation.
49  if n != int(n) or n < 2:
50  raise ValueError('n must be an integer greater than 1')
51  n = int(n)
52  if p != int(p) or p < 0:
53  raise ValueError('p must be a non-negative integer')
54  p = int(p)
55 
56  # Construct the matrix explicitly.
57  a, b = divmod(p, n)
58  large = np.power(10.0, -n*a)
59  small = large * np.power(10.0, -n)
60  return np.diag([large]*(n-b), b) + np.diag([small]*b, b-n)

214  # Review of "Linear System Theory" by Joao Hespanha,
215  # SIAM Review,
216  # Volume 52, Number 4, December 2010, pages 779--781.
