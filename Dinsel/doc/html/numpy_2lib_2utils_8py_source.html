186 def byte_bounds(a):
187  """
188  Returns pointers to the end-points of an array.
189 
190  Parameters
191  ----------
192  a : ndarray
193  Input array. It must conform to the Python-side of the array
194  interface.
195 
196  Returns
197  -------
198  (low, high) : tuple of 2 integers
199  The first integer is the first byte of the array, the second
200  integer is just past the last byte of the array. If `a` is not
201  contiguous it will not use every byte between the (`low`, `high`)
202  values.
203 
204  Examples
205  --------
206  >>> I = np.eye(2, dtype='f'); I.dtype
207  dtype('float32')
208  >>> low, high = np.byte_bounds(I)
209  >>> high - low == I.size*I.itemsize
210  True
211  >>> I = np.eye(2, dtype='G'); I.dtype
212  dtype('complex192')
213  >>> low, high = np.byte_bounds(I)
214  >>> high - low == I.size*I.itemsize
215  True
216 
217  """
218  ai = a.__array_interface__
219  a_data = ai['data'][0]
220  astrides = ai['strides']
221  ashape = ai['shape']
222  bytes_a = asarray(a).dtype.itemsize
223 
224  a_low = a_high = a_data
225  if astrides is None:
226  # contiguous case
227  a_high += a.size * bytes_a
228  else:
229  for shape, stride in zip(ashape, astrides):
230  if stride < 0:
231  a_low += (shape-1)*stride
232  else:
233  a_high += (shape-1)*stride
234  a_high += bytes_a
235  return a_low, a_high

243 def who(vardict=None):
244  """
245  Print the NumPy arrays in the given dictionary.
246 
247  If there is no dictionary passed in or `vardict` is None then returns
248  NumPy arrays in the globals() dictionary (all NumPy arrays in the
249  namespace).
250 
251  Parameters
252  ----------
253  vardict : dict, optional
254  A dictionary possibly containing ndarrays. Default is globals().
255 
256  Returns
257  -------
258  out : None
259  Returns 'None'.
260 
261  Notes
262  -----
263  Prints out the name, shape, bytes and type of all of the ndarrays
264  present in `vardict`.
265 
266  Examples
267  --------
268  >>> a = np.arange(10)
269  >>> b = np.ones(20)
270  >>> np.who()

272  ===========================================================
273  a 10 40 int32
274  b 20 160 float64
275  Upper bound on total bytes = 200
276 
277  >>> d = {'x': np.arange(2.0), 'y': np.arange(3.0), 'txt': 'Some str',
278  ... 'idx':5}
279  >>> np.who(d)
280  Name Shape Bytes Type
281  ===========================================================
282  y 3 24 float64
283  x 2 16 float64
284  Upper bound on total bytes = 40
285 
286  """
287  if vardict is None:
288  frame = sys._getframe().f_back
289  vardict = frame.f_globals
290  sta = []
291  cache = {}
292  for name in vardict.keys():
293  if isinstance(vardict[name], ndarray):
294  var = vardict[name]
295  idv = id(var)
296  if idv in cache.keys():
297  namestr = name + " (%s)" % cache[idv]
298  original = 0
299  else:
300  cache[idv] = name
301  namestr = name
302  original = 1
303  shapestr = " x ".join(map(str, var.shape))
304  bytestr = str(var.nbytes)
305  sta.append([namestr, shapestr, bytestr, var.dtype.name,
306  original])
307 
308  maxname = 0
309  maxshape = 0
310  maxbyte = 0
311  totalbytes = 0
312  for k in range(len(sta)):
313  val = sta[k]
314  if maxname < len(val[0]):
315  maxname = len(val[0])
316  if maxshape < len(val[1]):
317  maxshape = len(val[1])
318  if maxbyte < len(val[2]):
319  maxbyte = len(val[2])
320  if val[4]:
321  totalbytes += int(val[2])
322 
323  if len(sta) > 0:
324  sp1 = max(10, maxname)
325  sp2 = max(10, maxshape)
326  sp3 = max(10, maxbyte)
327  prval = "Name %s Shape %s Bytes %s Type" % (sp1*' ', sp2*' ', sp3*' ')
328  print(prval + "\n" + "="*(len(prval)+5) + "\n")
329 
330  for k in range(len(sta)):
331  val = sta[k]
332  print("%s %s %s %s %s %s %s" % (val[0], ' '*(sp1-len(val[0])+4),
333  val[1], ' '*(sp2-len(val[1])+5),
334  val[2], ' '*(sp3-len(val[2])+5),
335  val[3]))
336  print("\nUpper bound on total bytes = %d" % totalbytes)
337  return

345 # combine name and arguments and split to multiple lines of width

349  firstwidth = len(name)
350  k = firstwidth
351  newstr = name
352  sepstr = ", "
353  arglist = arguments.split(sepstr)
354  for argument in arglist:
355  if k == firstwidth:
356  addstr = ""
357  else:
358  addstr = sepstr
359  k = k + len(argument) + len(addstr)
360  if k > width:
361  k = firstwidth + 1 + len(argument)
362  newstr = newstr + ",\n" + " "*(firstwidth+2) + argument
363  else:
364  newstr = newstr + addstr + argument
365  return newstr

370 # Traverse all module directories underneath globals
371 # to see if something is defined
372 def _makenamedict(module='numpy'):
373  module = __import__(module, globals(), locals(), [])
374  thedict = {module.__name__:module.__dict__}
375  dictlist = [module.__name__]
376  totraverse = [module.__dict__]
377  while True:
378  if len(totraverse) == 0:
379  break
380  thisdict = totraverse.pop(0)
381  for x in thisdict.keys():
382  if isinstance(thisdict[x], types.ModuleType):
383  modname = thisdict[x].__name__
384  if modname not in dictlist:
385  moddict = thisdict[x].__dict__
386  dictlist.append(modname)
387  totraverse.append(moddict)
388  thedict[modname] = moddict
389  return thedict, dictlist

705 def lookfor(what, module=None, import_modules=True, regenerate=False,
706  output=None):
707  """
708  Do a keyword search on docstrings.
709 
710  A list of of objects that matched the search is displayed,
711  sorted by relevance. All given keywords need to be found in the
712  docstring for it to be returned as a result, but the order does
713  not matter.
714 
715  Parameters
716  ----------
717  what : str
718  String containing words to look for.
719  module : str or list, optional
720  Name of module(s) whose docstrings to go through.
721  import_modules : bool, optional
722  Whether to import sub-modules in packages. Default is True.
723  regenerate : bool, optional
724  Whether to re-generate the docstring cache. Default is False.
725  output : file-like, optional
726  File-like object to write the output to. If omitted, use a pager.
727 
728  See Also
729  --------
730  source, info
731 
732  Notes
733  -----
734  Relevance is determined only roughly, by checking if the keywords occur
735  in the function name, at the start of a docstring, etc.
736 
737  Examples
