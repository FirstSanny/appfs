22 def calc_circumcenters(tetrahedrons):
23  """ Calculates the cirumcenters of the circumspheres of tetrahedrons.
24 
25  An implementation based on
26  http://mathworld.wolfram.com/Circumsphere.html
27 
28  Parameters
29  ----------
30  tetrahedrons : an array of shape (N, 4, 3)
31  consisting of N tetrahedrons defined by 4 points in 3D
32 
33  Returns
34  ----------
35  circumcenters : an array of shape (N, 3)
36  consisting of the N circumcenters of the tetrahedrons in 3D
37 
38  """
39 
40  num = tetrahedrons.shape[0]
41  a = np.concatenate((tetrahedrons, np.ones((num, 4, 1))), axis=2)
42 
43  sums = np.sum(tetrahedrons ** 2, axis=2)
44  d = np.concatenate((sums[:, :, np.newaxis], a), axis=2)
45 
46  dx = np.delete(d, 1, axis=2)
47  dy = np.delete(d, 2, axis=2)
48  dz = np.delete(d, 3, axis=2)
49 
50  dx = np.linalg.det(dx)
51  dy = -np.linalg.det(dy)
52  dz = np.linalg.det(dz)
53  a = np.linalg.det(a)
54 
55  nominator = np.vstack((dx, dy, dz))
56  denominator = 2*a
57  return (nominator / denominator).T

60 def project_to_sphere(points, center, radius):
61  """
62  Projects the elements of points onto the sphere defined
63  by center and radius.
64 
65  Parameters
66  ----------
67  points : array of floats of shape (npoints, ndim)
68  consisting of the points in a space of dimension ndim
69  center : array of floats of shape (ndim,)
70  the center of the sphere to project on
71  radius : float
72  the radius of the sphere to project on
73 
74  returns: array of floats of shape (npoints, ndim)
75  the points projected onto the sphere
76  """
77 
78  lengths = scipy.spatial.distance.cdist(points, np.array([center]))
79  return (points - center) / lengths * radius + center

82 class SphericalVoronoi:
83  """ Voronoi diagrams on the surface of a sphere.
84 
85  .. versionadded:: 0.18.0
86 
87  Parameters
88  ----------
89  points : ndarray of floats, shape (npoints, 3)
90  Coordinates of points to construct a spherical
91  Voronoi diagram from
92  radius : float, optional
93  Radius of the sphere (Default: 1)
94  center : ndarray of floats, shape (3,)
95  Center of sphere (Default: origin)
96 
97  Attributes
98  ----------
99  points : double array of shape (npoints, 3)
100  the points in 3D to generate the Voronoi diagram from
101  radius : double
102  radius of the sphere
103  Default: None (forces estimation, which is less precise)
104  center : double array of shape (3,)
105  center of the sphere
106  Default: None (assumes sphere is centered at origin)
107  vertices : double array of shape (nvertices, 3)
108  Voronoi vertices corresponding to points
109  regions : list of list of integers of shape (npoints, _ )
110  the n-th entry is a list consisting of the indices
111  of the vertices belonging to the n-th point in points
112 
113  Notes
114  ----------
115  The spherical Voronoi diagram algorithm proceeds as follows. The Convex
116  Hull of the input points (generators) is calculated, and is equivalent to
117  their Delaunay triangulation on the surface of the sphere [Caroli]_.
118  A 3D Delaunay tetrahedralization is obtained by including the origin of
119  the coordinate system as the fourth vertex of each simplex of the Convex
120  Hull. The circumcenters of all tetrahedra in the system are calculated and
121  projected to the surface of the sphere, producing the Voronoi vertices.
122  The Delaunay tetrahedralization neighbour information is then used to
123  order the Voronoi region vertices around each generator. The latter
124  approach is substantially less sensitive to floating point issues than
125  angle-based methods of Voronoi region vertex sorting.
126 
127  The surface area of spherical polygons is calculated by decomposing them
128  into triangles and using L'Huilier's Theorem to calculate the spherical
129  excess of each triangle [Weisstein]_. The sum of the spherical excesses is
130  multiplied by the square of the sphere radius to obtain the surface area
131  of the spherical polygon. For nearly-degenerate spherical polygons an area
132  of approximately 0 is returned by default, rather than attempting the
133  unstable calculation.
134 
135  Empirical assessment of spherical Voronoi algorithm performance suggests
136  quadratic time complexity (loglinear is optimal, but algorithms are more
137  challenging to implement). The reconstitution of the surface area of the
138  sphere, measured as the sum of the surface areas of all Voronoi regions,
139  is closest to 100 % for larger (>> 10) numbers of generators.
140 
141  References
142  ----------
143 
144  .. [Caroli] Caroli et al. Robust and Efficient Delaunay triangulations of
145  points on or close to a sphere. Research Report RR-7004, 2009.
146  .. [Weisstein] "L'Huilier's Theorem." From MathWorld -- A Wolfram Web
147  Resource. http://mathworld.wolfram.com/LHuiliersTheorem.html
148 
149  See Also
150  --------
151  Voronoi : Conventional Voronoi diagrams in N dimensions.

195  def __init__(self, points, radius=None, center=None):
196  """
197  Initializes the object and starts the computation of the Voronoi
198  diagram.
199 
200  points : The generator points of the Voronoi diagram assumed to be
201  all on the sphere with radius supplied by the radius parameter and
202  center supplied by the center parameter.
203  radius : The radius of the sphere. Will default to 1 if not supplied.
204  center : The center of the sphere. Will default to the origin if not
205  supplied.
206  """
207 
208  self.points = points
209  if np.any(center):
210  self.center = center
211  else:
212  self.center = np.zeros(3)
213  if radius:
214  self.radius = radius
215  else:
216  self.radius = 1
217  self.vertices = None
218  self.regions = None
219  self._tri = None
220  self._calc_vertices_regions()

222  def _calc_vertices_regions(self):
223  """
224  Calculates the Voronoi vertices and regions of the generators stored
225  in self.points. The vertices will be stored in self.vertices and the
226  regions in self.regions.
227 
228  This algorithm was discussed at PyData London 2015 by
229  Tyler Reddy, Ross Hemsley and Nikolai Nowaczyk
230  """
231 
232  # perform 3D Delaunay triangulation on data set
233  # (here ConvexHull can also be used, and is faster)
234  self._tri = scipy.spatial.ConvexHull(self.points)
235 
236  # add the center to each of the simplices in tri to get the same
237  # tetrahedrons we'd have gotten from Delaunay tetrahedralization
238  # tetrahedrons will have shape: (2N-4, 4, 3)
239  tetrahedrons = self._tri.points[self._tri.simplices]
240  tetrahedrons = np.insert(
241  tetrahedrons,
242  3,
243  np.array([self.center]),
244  axis=1
245  )
246 
247  # produce circumcenters of tetrahedrons from 3D Delaunay
248  # circumcenters will have shape: (2N-4, 3)
249  circumcenters = calc_circumcenters(tetrahedrons)
250 
251  # project tetrahedron circumcenters to the surface of the sphere
252  # self.vertices will have shape: (2N-4, 3)
253  self.vertices = project_to_sphere(
254  circumcenters,
255  self.center,
256  self.radius
257  )
258 
259  # calculate regions from triangulation
260  # simplex_indices will have shape: (2N-4,)
261  simplex_indices = np.arange(self._tri.simplices.shape[0])
262  # tri_indices will have shape: (6N-12,)
263  tri_indices = np.column_stack([simplex_indices, simplex_indices,
264  simplex_indices]).ravel()
265  # point_indices will have shape: (6N-12,)
266  point_indices = self._tri.simplices.ravel()
267 
268  # array_associations will have shape: (6N-12, 2)
269  array_associations = np.dstack((point_indices, tri_indices))[0]
270  array_associations = array_associations[np.lexsort((
271  array_associations[...,1],
272  array_associations[...,0]))]
273  array_associations = array_associations.astype(np.intp)
274 
275  # group by generator indices to produce
276  # unsorted regions in nested list
277  groups = []
278  for k, g in itertools.groupby(array_associations,
279  lambda t: t[0]):
280  groups.append(list(list(zip(*list(g)))[1]))
281 
282  self.regions = groups

284  def sort_vertices_of_regions(self):
285  """
286  For each region in regions, it sorts the indices of the Voronoi
287  vertices such that the resulting points are in a clockwise or
288  counterclockwise order around the generator point.
289 
290  This is done as follows: Recall that the n-th region in regions
291  surrounds the n-th generator in points and that the k-th
292  Voronoi vertex in vertices is the projected circumcenter of the
293  tetrahedron obtained by the k-th triangle in _tri.simplices (and the
294  origin). For each region n, we choose the first triangle (=Voronoi
295  vertex) in _tri.simplices and a vertex of that triangle not equal to
296  the center n. These determine a unique neighbor of that triangle,
297  which is then chosen as the second triangle. The second triangle
298  will have a unique vertex not equal to the current vertex or the
299  center. This determines a unique neighbor of the second triangle,
300  which is then chosen as the third triangle and so forth. We proceed
301  through all the triangles (=Voronoi vertices) belonging to the
302  generator in points and obtain a sorted version of the vertices
303  of its surrounding region.
304  """
305 
306  _voronoi.sort_vertices_of_regions(self._tri.simplices,
307  self.regions)
