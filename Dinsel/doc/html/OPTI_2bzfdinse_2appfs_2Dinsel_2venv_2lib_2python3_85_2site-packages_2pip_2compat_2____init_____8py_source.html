101 def get_path_uid(path):
102  """
103  Return path's uid.
104 
105  Does not follow symlinks:
106  https://github.com/pypa/pip/pull/935#discussion_r5307003
107 
108  Placed this function in compat due to differences on AIX and
109  Jython, that should eventually go away.
110 
111  :raises OSError: When path is a symlink or can't be read.
112  """
113  if hasattr(os, 'O_NOFOLLOW'):
114  fd = os.open(path, os.O_RDONLY | os.O_NOFOLLOW)
115  file_uid = os.fstat(fd).st_uid
116  os.close(fd)
117  else: # AIX and Jython
118  # WARNING: time of check vulnerability, but best we can do w/o NOFOLLOW
119  if not os.path.islink(path):
120  # older versions of Jython don't have `os.fstat`
121  file_uid = os.stat(path).st_uid
122  else:
123  # raise OSError for parity with os.O_NOFOLLOW above
124  raise OSError(
125  "%s is a symlink; Will not return uid for symlinks" % path
126  )
127  return file_uid
