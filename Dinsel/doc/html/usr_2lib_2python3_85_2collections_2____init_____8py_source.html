71 class OrderedDict(dict):
72  'Dictionary that remembers insertion order'
73  # An inherited dict maps keys to values.
74  # The inherited dict provides __getitem__, __len__, __contains__, and get.
75  # The remaining methods are order-aware.
76  # Big-O running times for all methods are the same as regular dictionaries.
77 
78  # The internal self.__map dict maps keys to links in a doubly linked list.
79  # The circular doubly linked list starts and ends with a sentinel element.
80  # The sentinel element never gets deleted (this simplifies the algorithm).
81  # The sentinel is in self.__hardroot with a weakref proxy in self.__root.
82  # The prev links are weakref proxies (to prevent circular references).
83  # Individual links are kept alive by the hard reference in self.__map.
84  # Those hard references disappear when a key is deleted from an OrderedDict.
85

107  def __setitem__(self, key, value,
108  dict_setitem=dict.__setitem__, proxy=_proxy, Link=_Link):
109  'od.__setitem__(i, y) <==> od[i]=y'
110  # Setting a new item creates a new link at the end of the linked list,
111  # and the inherited dictionary is updated with the new key/value pair.
112  if key not in self:
113  self.__map[key] = link = Link()
114  root = self.__root
115  last = root.prev
116  link.prev, link.next, link.key = last, root, key
117  last.next = link
118  root.prev = proxy(link)

121  def __delitem__(self, key, dict_delitem=dict.__delitem__):
122  'od.__delitem__(y) <==> del od[y]'

124  # removed by updating the links in the predecessor and successor nodes.
125  dict_delitem(self, key)

127  link_prev = link.prev
128  link_next = link.next

356 def namedtuple(typename, field_names, verbose=False, rename=False):
357  """Returns a new subclass of tuple with named fields.
358 
359  >>> Point = namedtuple('Point', ['x', 'y'])
360  >>> Point.__doc__ # docstring for the new class
361  'Point(x, y)'
362  >>> p = Point(11, y=22) # instantiate with positional args or keywords
363  >>> p[0] + p[1] # indexable like a plain tuple
364  33
365  >>> x, y = p # unpack like a regular tuple
366  >>> x, y
367  (11, 22)
368  >>> p.x + p.y # fields also accessible by name
369  33
370  >>> d = p._asdict() # convert to a dictionary
371  >>> d['x']
372  11
373  >>> Point(**d) # convert from a dictionary
374  Point(x=11, y=22)
375  >>> p._replace(x=100) # _replace() is like str.replace() but targets named fields
376  Point(x=100, y=22)
377 
378  """
379 
380  # Validate the field names. At the user's option, either generate an error
381  # message or automatically replace the field name with a valid name.
382  if isinstance(field_names, str):
383  field_names = field_names.replace(',', ' ').split()
384  field_names = list(map(str, field_names))
385  typename = str(typename)
386  if rename:
387  seen = set()
388  for index, name in enumerate(field_names):
389  if (not name.isidentifier()
390  or _iskeyword(name)
391  or name.startswith('_')
392  or name in seen):
393  field_names[index] = '_%d' % index
394  seen.add(name)
395  for name in [typename] + field_names:
396  if type(name) != str:
397  raise TypeError('Type names and field names must be strings')
398  if not name.isidentifier():
399  raise ValueError('Type names and field names must be valid '
400  'identifiers: %r' % name)
401  if _iskeyword(name):
402  raise ValueError('Type names and field names cannot be a '
403  'keyword: %r' % name)
404  seen = set()
405  for name in field_names:
406  if name.startswith('_') and not rename:
407  raise ValueError('Field names cannot start with an underscore: '
408  '%r' % name)
409  if name in seen:
410  raise ValueError('Encountered duplicate field name: %r' % name)
411  seen.add(name)
412 
413  # Fill-in the class template
414  class_definition = _class_template.format(
415  typename = typename,
416  field_names = tuple(field_names),
417  num_fields = len(field_names),
418  arg_list = repr(tuple(field_names)).replace("'", "")[1:-1],
419  repr_fmt = ', '.join(_repr_template.format(name=name)
420  for name in field_names),
421  field_defs = '\n'.join(_field_template.format(index=index, name=name)
422  for index, name in enumerate(field_names))
423  )
424 
425  # Execute the template string in a temporary namespace and support
426  # tracing utilities by setting a value for frame.f_globals['__name__']
427  namespace = dict(__name__='namedtuple_%s' % typename)
428  exec(class_definition, namespace)
429  result = namespace[typename]
430  result._source = class_definition
431  if verbose:
432  print(result._source)
433 
434  # For pickling to work, the __module__ variable needs to be set to the frame
435  # where the named tuple is created. Bypass this step in environments where
436  # sys._getframe is not defined (Jython for example) or sys._getframe is not
437  # defined for arguments greater than 0 (IronPython).
438  try:
439  result.__module__ = _sys._getframe(1).f_globals.get('__name__', '__main__')
440  except (AttributeError, ValueError):
441  pass
442 
443  return result

461 class Counter(dict):
462  '''Dict subclass for counting hashable items. Sometimes called a bag
463  or multiset. Elements are stored as dictionary keys and their counts
464  are stored as dictionary values.
465 
466  >>> c = Counter('abcdeabcdabcaba') # count elements from a string
467 
468  >>> c.most_common(3) # three most common elements
469  [('a', 5), ('b', 4), ('c', 3)]
470  >>> sorted(c) # list all unique elements
471  ['a', 'b', 'c', 'd', 'e']
472  >>> ''.join(sorted(c.elements())) # list elements with repetitions
473  'aaaaabbbbcccdde'
474  >>> sum(c.values()) # total of all counts
475  15
476 
477  >>> c['a'] # count of letter 'a'
478  5
479  >>> for elem in 'shazam': # update counts from an iterable
480  ... c[elem] += 1 # by adding 1 to each element's count
481  >>> c['a'] # now there are seven 'a'
482  7
483  >>> del c['b'] # remove all 'b'
484  >>> c['b'] # now there are zero 'b'
485  0
486 
487  >>> d = Counter('simsalabim') # make another counter
488  >>> c.update(d) # add in the second counter
489  >>> c['a'] # now there are nine 'a'
490  9
491 
492  >>> c.clear() # empty the counter
493  >>> c
494  Counter()
495 
496  Note: If a count is set to zero or reduced to zero, it will remain
497  in the counter until the entry is deleted or the counter is cleared:
498 
499  >>> c = Counter('aaabbc')
500  >>> c['b'] -= 2 # reduce the count of 'b' by two
501  >>> c.most_common() # 'b' is still in, but its count is zero
502  [('a', 3), ('c', 1), ('b', 0)]

581  def update(*args, **kwds):
582  '''Like dict.update() but add counts instead of replacing them.
583 
584  Source can be an iterable, a dictionary, or another Counter instance.
585 
586  >>> c = Counter('which')
587  >>> c.update('witch') # add elements from another iterable
588  >>> d = Counter('watch')
589  >>> c.update(d) # add elements from another counter
590  >>> c['h'] # four 'h' in which, witch, and watch
591  4
592 
593  '''
594  # The regular dict.update() operation makes no sense here because the
595  # replace behavior results in the some of original untouched counts
596  # being mixed-in with all of the other counts for a mismash that
597  # doesn't have a straight-forward interpretation in most counting
598  # contexts. Instead, we implement straight-addition. Both the inputs
599  # and outputs are allowed to contain zero and negative counts.
600 
601  if not args:
602  raise TypeError("descriptor 'update' of 'Counter' object "
603  "needs an argument")
604  self, *args = args
605  if len(args) > 1:
606  raise TypeError('expected at most 1 arguments, got %d' % len(args))
607  iterable = args[0] if args else None
608  if iterable is not None:
609  if isinstance(iterable, Mapping):
610  if self:
611  self_get = self.get
612  for elem, count in iterable.items():
613  self[elem] = count + self_get(elem, 0)
614  else:
615  super(Counter, self).update(iterable) # fast path when counter is empty
616  else:
617  _count_elements(self, iterable)
618  if kwds:
619  self.update(kwds)
