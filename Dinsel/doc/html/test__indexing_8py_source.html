740  def _get_multi_index(self, arr, indices):
741  """Mimic multi dimensional indexing.
742 
743  Parameters
744  ----------
745  arr : ndarray
746  Array to be indexed.
747  indices : tuple of index objects
748 
749  Returns
750  -------
751  out : ndarray
752  An array equivalent to the indexing operation (but always a copy).
753  `arr[indices]` should be identical.
754  no_copy : bool
755  Whether the indexing operation requires a copy. If this is `True`,
756  `np.may_share_memory(arr, arr[indicies])` should be `True` (with
757  some exceptions for scalars and possibly 0-d arrays).
758 
759  Notes
760  -----
761  While the function may mostly match the errors of normal indexing this
762  is generally not the case.
763  """
764  in_indices = list(indices)
765  indices = []
766  # if False, this is a fancy or boolean index
767  no_copy = True
768  # number of fancy/scalar indexes that are not consecutive
769  num_fancy = 0
770  # number of dimensions indexed by a "fancy" index
771  fancy_dim = 0
772  # NOTE: This is a funny twist (and probably OK to change).
773  # The boolean array has illegal indexes, but this is
774  # allowed if the broadcast fancy-indices are 0-sized.
775  # This variable is to catch that case.
776  error_unless_broadcast_to_empty = False
777 
778  # We need to handle Ellipsis and make arrays from indices, also
779  # check if this is fancy indexing (set no_copy).
780  ndim = 0
781  ellipsis_pos = None # define here mostly to replace all but first.
782  for i, indx in enumerate(in_indices):
783  if indx is None:
784  continue
785  if isinstance(indx, np.ndarray) and indx.dtype == bool:
786  no_copy = False
787  if indx.ndim == 0:
788  raise IndexError
789  # boolean indices can have higher dimensions
790  ndim += indx.ndim
791  fancy_dim += indx.ndim
792  continue
793  if indx is Ellipsis:
794  if ellipsis_pos is None:
795  ellipsis_pos = i
796  continue # do not increment ndim counter
797  raise IndexError
798  if isinstance(indx, slice):
799  ndim += 1
800  continue
801  if not isinstance(indx, np.ndarray):
802  # This could be open for changes in numpy.
803  # numpy should maybe raise an error if casting to intp
804  # is not safe. It rejects np.array([1., 2.]) but not
805  # [1., 2.] as index (same for ie. np.take).
806  # (Note the importance of empty lists if changing this here)
807  indx = np.array(indx, dtype=np.intp)
808  in_indices[i] = indx
809  elif indx.dtype.kind != 'b' and indx.dtype.kind != 'i':
810  raise IndexError('arrays used as indices must be of '
811  'integer (or boolean) type')
812  if indx.ndim != 0:
813  no_copy = False
814  ndim += 1
815  fancy_dim += 1
816 
817  if arr.ndim - ndim < 0:
818  # we can't take more dimensions then we have, not even for 0-d
819  # arrays. since a[()] makes sense, but not a[(),]. We will
820  # raise an error later on, unless a broadcasting error occurs
821  # first.
822  raise IndexError
823 
824  if ndim == 0 and None not in in_indices:
825  # Well we have no indexes or one Ellipsis. This is legal.
826  return arr.copy(), no_copy
827 
828  if ellipsis_pos is not None:
829  in_indices[ellipsis_pos:ellipsis_pos+1] = ([slice(None, None)] *
830  (arr.ndim - ndim))
831 
832  for ax, indx in enumerate(in_indices):
833  if isinstance(indx, slice):
834  # convert to an index array
835  indx = np.arange(*indx.indices(arr.shape[ax]))
836  indices.append(['s', indx])
837  continue
838  elif indx is None:
839  # this is like taking a slice with one element from a new axis:
840  indices.append(['n', np.array([0], dtype=np.intp)])
841  arr = arr.reshape((arr.shape[:ax] + (1,) + arr.shape[ax:]))
842  continue
843  if isinstance(indx, np.ndarray) and indx.dtype == bool:
844  if indx.shape != arr.shape[ax:ax+indx.ndim]:
845  raise IndexError
846 
847  try:
848  flat_indx = np.ravel_multi_index(np.nonzero(indx),
849  arr.shape[ax:ax+indx.ndim], mode='raise')
850  except:
851  error_unless_broadcast_to_empty = True
852  # fill with 0s instead, and raise error later
853  flat_indx = np.array([0]*indx.sum(), dtype=np.intp)
854  # concatenate axis into a single one:
855  if indx.ndim != 0:
856  arr = arr.reshape((arr.shape[:ax]
857  + (np.prod(arr.shape[ax:ax+indx.ndim]),)
858  + arr.shape[ax+indx.ndim:]))
859  indx = flat_indx
860  else:
861  # This could be changed, a 0-d boolean index can
862  # make sense (even outside the 0-d indexed array case)
863  # Note that originally this is could be interpreted as
864  # integer in the full integer special case.
865  raise IndexError
866  else:
867  # If the index is a singleton, the bounds check is done
868  # before the broadcasting. This used to be different in <1.9
869  if indx.ndim == 0:
870  if indx >= arr.shape[ax] or indx < -arr.shape[ax]:
871  raise IndexError
872  if indx.ndim == 0:
873  # The index is a scalar. This used to be two fold, but if
874  # fancy indexing was active, the check was done later,
875  # possibly after broadcasting it away (1.7. or earlier).
876  # Now it is always done.
877  if indx >= arr.shape[ax] or indx < - arr.shape[ax]:
878  raise IndexError
879  if (len(indices) > 0 and
880  indices[-1][0] == 'f' and
881  ax != ellipsis_pos):
882  # NOTE: There could still have been a 0-sized Ellipsis
883  # between them. Checked that with ellipsis_pos.
884  indices[-1].append(indx)
885  else:
886  # We have a fancy index that is not after an existing one.
887  # NOTE: A 0-d array triggers this as well, while one may
888  # expect it to not trigger it, since a scalar would not be
889  # considered fancy indexing.
890  num_fancy += 1
891  indices.append(['f', indx])
892 
893  if num_fancy > 1 and not no_copy:
894  # We have to flush the fancy indexes left
895  new_indices = indices[:]
896  axes = list(range(arr.ndim))
897  fancy_axes = []
898  new_indices.insert(0, ['f'])
899  ni = 0
900  ai = 0
901  for indx in indices:
902  ni += 1
903  if indx[0] == 'f':
904  new_indices[0].extend(indx[1:])
905  del new_indices[ni]
906  ni -= 1
907  for ax in range(ai, ai + len(indx[1:])):
908  fancy_axes.append(ax)
909  axes.remove(ax)
910  ai += len(indx) - 1 # axis we are at
911  indices = new_indices
912  # and now we need to transpose arr:
913  arr = arr.transpose(*(fancy_axes + axes))
914 
915  # We only have one 'f' index now and arr is transposed accordingly.
916  # Now handle newaxis by reshaping...
917  ax = 0
918  for indx in indices:
919  if indx[0] == 'f':
920  if len(indx) == 1:
921  continue
922  # First of all, reshape arr to combine fancy axes into one:
923  orig_shape = arr.shape
924  orig_slice = orig_shape[ax:ax + len(indx[1:])]
925  arr = arr.reshape((arr.shape[:ax]
926  + (np.prod(orig_slice).astype(int),)
927  + arr.shape[ax + len(indx[1:]):]))
928 
929  # Check if broadcasting works
930  res = np.broadcast(*indx[1:])
931  # unfortunately the indices might be out of bounds. So check
932  # that first, and use mode='wrap' then. However only if
933  # there are any indices...
934  if res.size != 0:
935  if error_unless_broadcast_to_empty:
936  raise IndexError
937  for _indx, _size in zip(indx[1:], orig_slice):
938  if _indx.size == 0:
939  continue
940  if np.any(_indx >= _size) or np.any(_indx < -_size):
941  raise IndexError
942  if len(indx[1:]) == len(orig_slice):
943  if np.product(orig_slice) == 0:
944  # Work around for a crash or IndexError with 'wrap'
945  # in some 0-sized cases.
946  try:
947  mi = np.ravel_multi_index(indx[1:], orig_slice,
948  mode='raise')
949  except:
950  # This happens with 0-sized orig_slice (sometimes?)
951  # here it is a ValueError, but indexing gives a:
952  raise IndexError('invalid index into 0-sized')
953  else:
954  mi = np.ravel_multi_index(indx[1:], orig_slice,
955  mode='wrap')
956  else:
957  # Maybe never happens...
958  raise ValueError
959  arr = arr.take(mi.ravel(), axis=ax)
960  arr = arr.reshape((arr.shape[:ax]
961  + mi.shape
962  + arr.shape[ax+1:]))
963  ax += mi.ndim
964  continue
965 
966  # If we are here, we have a 1D array for take:
967  arr = arr.take(indx[1], axis=ax)
968  ax += 1
969 
970  return arr, no_copy
