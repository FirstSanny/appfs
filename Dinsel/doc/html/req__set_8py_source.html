215  def add_requirement(self, install_req, parent_req_name=None,
216  extras_requested=None):
217  """Add install_req as a requirement to install.
218 
219  :param parent_req_name: The name of the requirement that needed this
220  added. The name is used because when multiple unnamed requirements
221  resolve to the same name, we could otherwise end up with dependency
222  links that point outside the Requirements set. parent_req must
223  already be added. Note that None implies that this is a user
224  supplied requirement, vs an inferred one.
225  :param extras_requested: an iterable of extras used to evaluate the
226  environement markers.
227  :return: Additional requirements to scan. That is either [] if
228  the requirement is not applicable, or [install_req] if the
229  requirement is applicable and has just been added.
230  """
231  name = install_req.name
232  if not install_req.match_markers(extras_requested):
233  logger.warning("Ignoring %s: markers '%s' don't match your "
234  "environment", install_req.name,
235  install_req.markers)
236  return []
237 
238  # This check has to come after we filter requirements with the
239  # environment markers.
240  if install_req.link and install_req.link.is_wheel:
241  wheel = Wheel(install_req.link.filename)
242  if not wheel.supported():
243  raise InstallationError(
244  "%s is not a supported wheel on this platform." %
245  wheel.filename
246  )
247 
248  install_req.as_egg = self.as_egg
249  install_req.use_user_site = self.use_user_site
250  install_req.target_dir = self.target_dir
251  install_req.pycompile = self.pycompile
252  install_req.is_direct = (parent_req_name is None)
253 
254  if not name:
255  # url or path requirement w/o an egg fragment
256  self.unnamed_requirements.append(install_req)
257  return [install_req]
258  else:
259  try:
260  existing_req = self.get_requirement(name)
261  except KeyError:
262  existing_req = None
263  if (parent_req_name is None and existing_req and not
264  existing_req.constraint and
265  existing_req.extras == install_req.extras and not
266  existing_req.req.specifier == install_req.req.specifier):
267  raise InstallationError(
268  'Double requirement given: %s (already in %s, name=%r)'
269  % (install_req, existing_req, name))
270  if not existing_req:
271  # Add requirement
272  self.requirements[name] = install_req
273  # FIXME: what about other normalizations? E.g., _ vs. -?
274  if name.lower() != name:
275  self.requirement_aliases[name.lower()] = name
276  result = [install_req]
277  else:
278  # Assume there's no need to scan, and that we've already
279  # encountered this for scanning.
280  result = []
281  if not install_req.constraint and existing_req.constraint:
282  if (install_req.link and not (existing_req.link and
283  install_req.link.path == existing_req.link.path)):
284  self.reqs_to_cleanup.append(install_req)
285  raise InstallationError(
286  "Could not satisfy constraints for '%s': "
287  "installation from path or url cannot be "
288  "constrained to a version" % name)
289  # If we're now installing a constraint, mark the existing
290  # object for real installation.
291  existing_req.constraint = False
292  existing_req.extras = tuple(
293  sorted(set(existing_req.extras).union(
294  set(install_req.extras))))
295  logger.debug("Setting %s extras to: %s",
296  existing_req, existing_req.extras)
297  # And now we need to scan this.
298  result = [existing_req]
299  # Canonicalise to the already-added object for the backref
300  # check below.
301  install_req = existing_req
302  if parent_req_name:
303  parent_req = self.get_requirement(parent_req_name)
304  self._dependencies[parent_req].append(install_req)
305  return result

395  def _check_skip_installed(self, req_to_install, finder):
396  """Check if req_to_install should be skipped.
397 
398  This will check if the req is installed, and whether we should upgrade
399  or reinstall it, taking into account all the relevant user options.
400 
401  After calling this req_to_install will only have satisfied_by set to
402  None if the req_to_install is to be upgraded/reinstalled etc. Any
403  other value will be a dist recording the current thing installed that
404  satisfies the requirement.
405 
406  Note that for vcs urls and the like we can't assess skipping in this
407  routine - we simply identify that we need to pull the thing down,
408  then later on it is pulled down and introspected to assess upgrade/
409  reinstalls etc.
410 
411  :return: A text reason for why it was skipped, or None.
412  """
413  # Check whether to upgrade/reinstall this req or not.
414  req_to_install.check_if_exists()
415  if req_to_install.satisfied_by:
416  upgrade_allowed = self._is_upgrade_allowed(req_to_install)
417 
418  # Is the best version is installed.
419  best_installed = False
420 
421  if upgrade_allowed:
422  # For link based requirements we have to pull the
423  # tree down and inspect to assess the version #, so
424  # its handled way down.
425  if not (self.force_reinstall or req_to_install.link):
426  try:
427  finder.find_requirement(
428  req_to_install, upgrade_allowed)
429  except BestVersionAlreadyInstalled:
430  best_installed = True
431  except DistributionNotFound:
432  # No distribution found, so we squash the
433  # error - it will be raised later when we
434  # re-try later to do the install.
435  # Why don't we just raise here?
436  pass
437 
438  if not best_installed:
439  # don't uninstall conflict if user install and
440  # conflict is not user install
441  if not (self.use_user_site and not
442  dist_in_usersite(req_to_install.satisfied_by)):
443  req_to_install.conflicts_with = \
444  req_to_install.satisfied_by
445  req_to_install.satisfied_by = None
446 
447  # Figure out a nice message to say why we're skipping this.
448  if best_installed:
449  skip_reason = 'already up-to-date'
450  elif self.upgrade_strategy == "only-if-needed":
451  skip_reason = 'not upgraded as not directly required'
452  else:
453  skip_reason = 'already satisfied'
454 
455  return skip_reason
456  else:
457  return None

459  def _prepare_file(self,
460  finder,
461  req_to_install,
462  require_hashes=False,
463  ignore_dependencies=False):
464  """Prepare a single requirements file.
465 
466  :return: A list of additional InstallRequirements to also install.
467  """
468  # Tell user what we are doing for this requirement:
469  # obtain (editable), skipping, processing (local url), collecting
470  # (remote url or package name)
471  if req_to_install.constraint or req_to_install.prepared:
472  return []
473 
474  req_to_install.prepared = True
475 
476  # ###################### #
477  # # print log messages # #
478  # ###################### #
479  if req_to_install.editable:
480  logger.info('Obtaining %s', req_to_install)
481  else:
482  # satisfied_by is only evaluated by calling _check_skip_installed,
483  # so it must be None here.
484  assert req_to_install.satisfied_by is None
485  if not self.ignore_installed:
486  skip_reason = self._check_skip_installed(
487  req_to_install, finder)
488 
489  if req_to_install.satisfied_by:
490  assert skip_reason is not None, (
491  '_check_skip_installed returned None but '
492  'req_to_install.satisfied_by is set to %r'
493  % (req_to_install.satisfied_by,))
494  logger.info(
495  'Requirement %s: %s', skip_reason,
496  req_to_install)
497  else:
498  if (req_to_install.link and
499  req_to_install.link.scheme == 'file'):
500  path = url_to_path(req_to_install.link.url)
501  logger.info('Processing %s', display_path(path))
502  else:
503  logger.info('Collecting %s', req_to_install)
504 
505  with indent_log():
506  # ################################ #
507  # # vcs update or unpack archive # #
508  # ################################ #
509  if req_to_install.editable:
510  if require_hashes:
511  raise InstallationError(
512  'The editable requirement %s cannot be installed when '
513  'requiring hashes, because there is no single file to '
514  'hash.' % req_to_install)
515  req_to_install.ensure_has_source_dir(self.src_dir)
516  req_to_install.update_editable(not self.is_download)
517  abstract_dist = make_abstract_dist(req_to_install)
518  abstract_dist.prep_for_dist()
519  if self.is_download:
520  req_to_install.archive(self.download_dir)
521  req_to_install.check_if_exists()
522  elif req_to_install.satisfied_by:
523  if require_hashes:
524  logger.debug(
525  'Since it is already installed, we are trusting this '
526  'package without checking its hash. To ensure a '
527  'completely repeatable environment, install into an '
528  'empty virtualenv.')
529  abstract_dist = Installed(req_to_install)
530  else:
531  # @@ if filesystem packages are not marked
532  # editable in a req, a non deterministic error
533  # occurs when the script attempts to unpack the
534  # build directory
535  req_to_install.ensure_has_source_dir(self.build_dir)
536  # If a checkout exists, it's unwise to keep going. version
537  # inconsistencies are logged later, but do not fail the
538  # installation.
539  # FIXME: this won't upgrade when there's an existing
540  # package unpacked in `req_to_install.source_dir`
541  if os.path.exists(
542  os.path.join(req_to_install.source_dir, 'setup.py')):
543  raise PreviousBuildDirError(
544  "pip can't proceed with requirements '%s' due to a"
545  " pre-existing build directory (%s). This is "
546  "likely due to a previous installation that failed"
547  ". pip is being responsible and not assuming it "
548  "can delete this. Please delete it and try again."
549  % (req_to_install, req_to_install.source_dir)
550  )
551  req_to_install.populate_link(
552  finder,
553  self._is_upgrade_allowed(req_to_install),
554  require_hashes
555  )
556  # We can't hit this spot and have populate_link return None.
557  # req_to_install.satisfied_by is None here (because we're
558  # guarded) and upgrade has no impact except when satisfied_by
559  # is not None.
560  # Then inside find_requirement existing_applicable -> False
561  # If no new versions are found, DistributionNotFound is raised,
562  # otherwise a result is guaranteed.
563  assert req_to_install.link
564  link = req_to_install.link
565 
566  # Now that we have the real link, we can tell what kind of
567  # requirements we have and raise some more informative errors
568  # than otherwise. (For example, we can raise VcsHashUnsupported
569  # for a VCS URL rather than HashMissing.)
570  if require_hashes:
571  # We could check these first 2 conditions inside
572  # unpack_url and save repetition of conditions, but then
573  # we would report less-useful error messages for
574  # unhashable requirements, complaining that there's no
575  # hash provided.
576  if is_vcs_url(link):
577  raise VcsHashUnsupported()
578  elif is_file_url(link) and is_dir_url(link):
579  raise DirectoryUrlHashUnsupported()
580  if (not req_to_install.original_link and
581  not req_to_install.is_pinned):
582  # Unpinned packages are asking for trouble when a new
583  # version is uploaded. This isn't a security check, but
584  # it saves users a surprising hash mismatch in the
585  # future.
586  #
587  # file:/// URLs aren't pinnable, so don't complain
588  # about them not being pinned.
589  raise HashUnpinned()
590  hashes = req_to_install.hashes(
591  trust_internet=not require_hashes)
592  if require_hashes and not hashes:
593  # Known-good hashes are missing for this requirement, so
594  # shim it with a facade object that will provoke hash
595  # computation and then raise a HashMissing exception
596  # showing the user what the hash should be.
597  hashes = MissingHashes()
598 
599  try:
600  download_dir = self.download_dir
601  # We always delete unpacked sdists after pip ran.
602  autodelete_unpacked = True
603  if req_to_install.link.is_wheel \
604  and self.wheel_download_dir:
605  # when doing 'pip wheel` we download wheels to a
606  # dedicated dir.
607  download_dir = self.wheel_download_dir
608  if req_to_install.link.is_wheel:
609  if download_dir:
610  # When downloading, we only unpack wheels to get
611  # metadata.
612  autodelete_unpacked = True
613  else:
614  # When installing a wheel, we use the unpacked
615  # wheel.
616  autodelete_unpacked = False
617  unpack_url(
618  req_to_install.link, req_to_install.source_dir,
619  download_dir, autodelete_unpacked,
620  session=self.session, hashes=hashes)
621  except requests.HTTPError as exc:
622  logger.critical(
623  'Could not install requirement %s because '
624  'of error %s',
625  req_to_install,
626  exc,
627  )
628  raise InstallationError(
629  'Could not install requirement %s because '
630  'of HTTP error %s for URL %s' %
631  (req_to_install, exc, req_to_install.link)
632  )
633  abstract_dist = make_abstract_dist(req_to_install)
634  abstract_dist.prep_for_dist()
635  if self.is_download:
636  # Make a .zip of the source_dir we already created.
637  if req_to_install.link.scheme in vcs.all_schemes:
638  req_to_install.archive(self.download_dir)
639  # req_to_install.req is only avail after unpack for URL
640  # pkgs repeat check_if_exists to uninstall-on-upgrade
641  # (#14)

731  def _to_install(self):
732  """Create the installation order.
733 
734  The installation order is topological - requirements are installed
735  before the requiring thing. We break cycles at an arbitrary point,
736  and make no other guarantees.
737  """
738  # The current implementation, which we may change at any point
739  # installs the user specified things in the order given, except when
740  # dependencies must come earlier to achieve topological order.
741  order = []
742  ordered_reqs = set()
743 
744  def schedule(req):
745  if req.satisfied_by or req in ordered_reqs:
746  return
747  if req.constraint:
748  return
749  ordered_reqs.add(req)
750  for dep in self._dependencies[req]:
751  schedule(dep)
752  order.append(req)
753  for install_req in self.requirements.values():
754  schedule(install_req)
755  return order
