145  def multi_fcn(self, B, x):
146  if (x < 0.0).any():
147  raise OdrStop
148  theta = pi*B[3]/2.
149  ctheta = np.cos(theta)
150  stheta = np.sin(theta)
151  omega = np.power(2.*pi*x*np.exp(-B[2]), B[3])
152  phi = np.arctan2((omega*stheta), (1.0 + omega*ctheta))
153  r = (B[0] - B[1]) * np.power(np.sqrt(np.power(1.0 + omega*ctheta, 2) +
154  np.power(omega*stheta, 2)), -B[4])
155  ret = np.vstack([B[1] + r*np.cos(B[4]*phi),
156  r*np.sin(B[4]*phi)])
157  return ret
