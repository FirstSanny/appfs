20 class memmap(ndarray):
21  """Create a memory-map to an array stored in a *binary* file on disk.
22 
23  Memory-mapped files are used for accessing small segments of large files
24  on disk, without reading the entire file into memory. NumPy's
25  memmap's are array-like objects. This differs from Python's ``mmap``
26  module, which uses file-like objects.
27 
28  This subclass of ndarray has some unpleasant interactions with
29  some operations, because it doesn't quite fit properly as a subclass.
30  An alternative to using this subclass is to create the ``mmap``
31  object yourself, then create an ndarray with ndarray.__new__ directly,
32  passing the object created in its 'buffer=' parameter.
33 
34  This class may at some point be turned into a factory function
35  which returns a view into an mmap buffer.
36 
37  Delete the memmap instance to close.
38 
39 
40  Parameters
41  ----------
42  filename : str, file-like object, or pathlib.Path instance
43  The file name or file object to be used as the array data buffer.
44  dtype : data-type, optional
45  The data-type used to interpret the file contents.
46  Default is `uint8`.
47  mode : {'r+', 'r', 'w+', 'c'}, optional
48  The file is opened in this mode:
49 
50  +------+-------------------------------------------------------------+
51  | 'r' | Open existing file for reading only. |
52  +------+-------------------------------------------------------------+
53  | 'r+' | Open existing file for reading and writing. |
54  +------+-------------------------------------------------------------+
55  | 'w+' | Create or overwrite existing file for reading and writing. |
56  +------+-------------------------------------------------------------+
57  | 'c' | Copy-on-write: assignments affect data in memory, but |
58  | | changes are not saved to disk. The file on disk is |
59  | | read-only. |
60  +------+-------------------------------------------------------------+
61 
62  Default is 'r+'.
63  offset : int, optional
64  In the file, array data starts at this offset. Since `offset` is
65  measured in bytes, it should normally be a multiple of the byte-size
66  of `dtype`. When ``mode != 'r'``, even positive offsets beyond end of
67  file are valid; The file will be extended to accommodate the
68  additional data. By default, ``memmap`` will start at the beginning of
69  the file, even if ``filename`` is a file pointer ``fp`` and
70  ``fp.tell() != 0``.
71  shape : tuple, optional
72  The desired shape of the array. If ``mode == 'r'`` and the number
73  of remaining bytes after `offset` is not a multiple of the byte-size
74  of `dtype`, you must specify `shape`. By default, the returned array
75  will be 1-D with the number of elements determined by file size
76  and data-type.
77  order : {'C', 'F'}, optional
78  Specify the order of the ndarray memory layout:
79  :term:`row-major`, C-style or :term:`column-major`,
80  Fortran-style. This only has an effect if the shape is
81  greater than 1-D. The default order is 'C'.
82 
83  Attributes
84  ----------
85  filename : str or pathlib.Path instance
86  Path to the mapped file.
87  offset : int
88  Offset position in the file.
89  mode : str
90  File mode.
91 
92  Methods
93  -------
94  flush
95  Flush any changes in memory to file on disk.
96  When you delete a memmap object, flush is called first to write
97  changes to disk before removing the object.
98 
99 
100  See also
101  --------
102  lib.format.open_memmap : Create or load a memory-mapped ``.npy`` file.
103 
104  Notes
105  -----
106  The memmap object can be used anywhere an ndarray is accepted.
107  Given a memmap ``fp``, ``isinstance(fp, numpy.ndarray)`` returns
108  ``True``.
109 
110  Memory-mapped files cannot be larger than 2GB on 32-bit systems.
111 
112  When a memmap causes a file to be created or extended beyond its
113  current size in the filesystem, the contents of the new part are
114  unspecified. On systems with POSIX filesystem semantics, the extended
115  part will be filled with zero bytes.
116 
117  Examples
118  --------
119  >>> data = np.arange(12, dtype='float32')
120  >>> data.resize((3,4))
121 
122  This example uses a temporary file so that doctest doesn't write
123  files to your directory. You would use a 'normal' filename.
124 
125  >>> from tempfile import mkdtemp
126  >>> import os.path as path
127  >>> filename = path.join(mkdtemp(), 'newfile.dat')
128 
129  Create a memmap with dtype and shape that matches our data:
130 
131  >>> fp = np.memmap(filename, dtype='float32', mode='w+', shape=(3,4))
132  >>> fp
133  memmap([[ 0., 0., 0., 0.],
134  [ 0., 0., 0., 0.],
135  [ 0., 0., 0., 0.]], dtype=float32)
136 
137  Write data to memmap array:
138 
139  >>> fp[:] = data[:]
140  >>> fp
141  memmap([[ 0., 1., 2., 3.],
142  [ 4., 5., 6., 7.],
143  [ 8., 9., 10., 11.]], dtype=float32)
144 
145  >>> fp.filename == path.abspath(filename)
146  True
147 
148  Deletion flushes memory changes to disk before removing the object:
149 
150  >>> del fp
151 
152  Load the memmap and verify data was stored:
153 
154  >>> newfp = np.memmap(filename, dtype='float32', mode='r', shape=(3,4))
155  >>> newfp
156  memmap([[ 0., 1., 2., 3.],
157  [ 4., 5., 6., 7.],
158  [ 8., 9., 10., 11.]], dtype=float32)
159 
160  Read-only memmap:
161 
162  >>> fpr = np.memmap(filename, dtype='float32', mode='r', shape=(3,4))
163  >>> fpr.flags.writeable
164  False
165 
166  Copy-on-write memmap:
167 
168  >>> fpc = np.memmap(filename, dtype='float32', mode='c', shape=(3,4))
169  >>> fpc.flags.writeable
170  True
171 
172  It's possible to assign to copy-on-write array, but values are only
173  written into the memory copy of the array, and not written to disk:
174 
175  >>> fpc
176  memmap([[ 0., 1., 2., 3.],
177  [ 4., 5., 6., 7.],
178  [ 8., 9., 10., 11.]], dtype=float32)
179  >>> fpc[0,:] = 0
180  >>> fpc
181  memmap([[ 0., 0., 0., 0.],
182  [ 4., 5., 6., 7.],
183  [ 8., 9., 10., 11.]], dtype=float32)
184 
185  File on disk is unchanged:
186 
187  >>> fpr
188  memmap([[ 0., 1., 2., 3.],
189  [ 4., 5., 6., 7.],
190  [ 8., 9., 10., 11.]], dtype=float32)
191 
192  Offset into a memmap:
193 
194  >>> fpo = np.memmap(filename, dtype='float32', mode='r', offset=16)
195  >>> fpo
196  memmap([ 4., 5., 6., 7., 8., 9., 10., 11.], dtype=float32)
197

202  def __new__(subtype, filename, dtype=uint8, mode='r+', offset=0,
203  shape=None, order='C'):
204  # Import here to minimize 'import numpy' overhead
205  import mmap
206  import os.path
207  try:
208  mode = mode_equivalents[mode]
209  except KeyError:
210  if mode not in valid_filemodes:
211  raise ValueError("mode must be one of %s" %
212  (valid_filemodes + list(mode_equivalents.keys())))
213 
214  if hasattr(filename, 'read'):
215  fid = filename
216  own_file = False
217  elif is_pathlib_path(filename):
218  fid = filename.open((mode == 'c' and 'r' or mode)+'b')
219  own_file = True
220  else:
221  fid = open(filename, (mode == 'c' and 'r' or mode)+'b')
222  own_file = True
223 
224  if (mode == 'w+') and shape is None:
225  raise ValueError("shape must be given")
226 
227  fid.seek(0, 2)
228  flen = fid.tell()
229  descr = dtypedescr(dtype)
230  _dbytes = descr.itemsize
231 
232  if shape is None:
233  bytes = flen - offset
234  if (bytes % _dbytes):
235  fid.close()
236  raise ValueError("Size of available data is not a "
237  "multiple of the data-type size.")
238  size = bytes // _dbytes
239  shape = (size,)
240  else:
241  if not isinstance(shape, tuple):
242  shape = (shape,)
243  size = 1
244  for k in shape:
245  size *= k
246 
247  bytes = long(offset + size*_dbytes)
248 
249  if mode == 'w+' or (mode == 'r+' and flen < bytes):
250  fid.seek(bytes - 1, 0)
251  fid.write(np.compat.asbytes('\0'))
252  fid.flush()
253 
254  if mode == 'c':
255  acc = mmap.ACCESS_COPY
256  elif mode == 'r':
257  acc = mmap.ACCESS_READ
258  else:
259  acc = mmap.ACCESS_WRITE
260 
261  start = offset - offset % mmap.ALLOCATIONGRANULARITY
262  bytes -= start
263  offset -= start
264  mm = mmap.mmap(fid.fileno(), bytes, access=acc, offset=start)
265 
266  self = ndarray.__new__(subtype, shape, dtype=descr, buffer=mm,
267  offset=offset, order=order)
268  self._mmap = mm
269  self.offset = offset
270  self.mode = mode
271 
272  if isinstance(filename, basestring):
273  self.filename = os.path.abspath(filename)
274  elif is_pathlib_path(filename):
275  self.filename = filename.resolve()
276  # py3 returns int for TemporaryFile().name
277  elif (hasattr(filename, "name") and
278  isinstance(filename.name, basestring)):
279  self.filename = os.path.abspath(filename.name)
280  # same as memmap copies (e.g. memmap + 1)
281  else:
282  self.filename = None
283 
284  if own_file:
285  fid.close()
286 
287  return self
