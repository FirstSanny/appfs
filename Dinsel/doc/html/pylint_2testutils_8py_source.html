45 def get_tests_info(input_dir, msg_dir, prefix, suffix):
46  """get python input examples and output messages
47 
48  We use following conventions for input files and messages:
49  for different inputs:
50  test for python >= x.y -> input = <name>_pyxy.py
51  test for python < x.y -> input = <name>_py_xy.py
52  for one input and different messages:
53  message for python >= x.y -> message = <name>_pyxy.txt
54  lower versions -> message with highest num
55  """
56  result = []
57  for fname in glob(join(input_dir, prefix + '*' + suffix)):
58  infile = basename(fname)
59  fbase = splitext(infile)[0]
60  # filter input files :
61  pyrestr = fbase.rsplit('_py', 1)[-1] # like _26 or 26
62  if pyrestr.isdigit(): # '24', '25'...
63  if SYS_VERS_STR < pyrestr:
64  continue
65  if pyrestr.startswith('_') and pyrestr[1:].isdigit():
66  # skip test for higher python versions
67  if SYS_VERS_STR >= pyrestr[1:]:
68  continue
69  messages = glob(join(msg_dir, fbase + '*.txt'))
70  # the last one will be without ext, i.e. for all or upper versions:
71  if messages:
72  for outfile in sorted(messages, reverse=True):
73  py_rest = outfile.rsplit('_py', 1)[-1][:-4]
74  if py_rest.isdigit() and SYS_VERS_STR >= py_rest:
75  break
76  else:
77  # This will provide an error message indicating the missing filename.
78  outfile = join(msg_dir, fbase + '.txt')
79  result.append((infile, outfile))
80  return result

352 @contextlib.contextmanager
353 def create_tempfile(content=None):
354  """Create a new temporary file.
355 
356  If *content* parameter is given, then it will be written
357  in the temporary file, before passing it back.
358  This is a context manager and should be used with a *with* statement.
359  """
360  # Can't use tempfile.NamedTemporaryFile here
361  # because on Windows the file must be closed before writing to it,
362  # see http://bugs.python.org/issue14243
363  file_handle, tmp = tempfile.mkstemp()
364  if content:
365  if sys.version_info >= (3, 0):
366  # erff
367  os.write(file_handle, bytes(content, 'ascii'))
368  else:
369  os.write(file_handle, content)
370  try:
371  yield tmp
372  finally:
373  os.close(file_handle)
374  os.remove(tmp)
