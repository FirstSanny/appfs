92 def _infer_map(node, context):
93  """Infer all values based on Dict.items"""
94  values = {}
95  for name, value in node.items:
96  if isinstance(name, nodes.DictUnpack):
97  double_starred = helpers.safe_infer(value, context)
98  if double_starred in (None, util.Uninferable):
99  raise exceptions.InferenceError
100  if not isinstance(double_starred, nodes.Dict):
101  raise exceptions.InferenceError(node=node,
102  context=context)
103  values.update(_infer_map(double_starred, context))
104  else:
105  key = helpers.safe_infer(name, context=context)
106  value = helpers.safe_infer(value, context=context)

109  context=context)
110  values[key] = value

117 def _higher_function_scope(node):
118  """ Search for the first function which encloses the given
119  scope. This can be used for looking up in that function's
120  scope, in case looking up in a lower scope for a particular
121  name fails.
122 
123  :param node: A scope node.
124  :returns:
125  ``None``, if no parent function scope was found,
126  otherwise an instance of :class:`astroid.scoped_nodes.Function`,
127  which encloses the given node.
128  """
129  current = node
130  while current.parent and not isinstance(current.parent, nodes.FunctionDef):
131  current = current.parent
132  if current and current.parent:
133  return current.parent

234 @decorators.raise_if_nothing_inferred
235 def infer_attribute(self, context=None):
236  """infer an Attribute node by using getattr on the associated object"""
237  for owner in self.expr.infer(context):
238  if owner is util.Uninferable:
239  yield owner
240  continue
241 
242  if context and context.boundnode:
243  # This handles the situation where the attribute is accessed through a subclass
244  # of a base class and the attribute is defined at the base class's level,
245  # by taking in consideration a redefinition in the subclass.
246  if (isinstance(owner, bases.Instance)
247  and isinstance(context.boundnode, bases.Instance)):
248  try:
249  if helpers.is_subtype(helpers.object_type(context.boundnode),
250  helpers.object_type(owner)):
251  owner = context.boundnode
252  except exceptions._NonDeducibleTypeHierarchy:
253  # Can't determine anything useful.
254  pass
255 
256  try:
257  context.boundnode = owner
258  for obj in owner.igetattr(self.attrname, context):
259  yield obj
260  context.boundnode = None
261  except (exceptions.AttributeInferenceError, exceptions.InferenceError):
262  context.boundnode = None
263  except AttributeError:
264  # XXX method / function
265  context.boundnode = None
266  # Explicit StopIteration to return error information, see comment
267  # in raise_if_nothing_inferred.
268  raise StopIteration(dict(node=self, context=context))

290 @decorators.raise_if_nothing_inferred
291 def infer_subscript(self, context=None):
292  """Inference for subscripts
293 
294  We're understanding if the index is a Const
295  or a slice, passing the result of inference
296  to the value's `getitem` method, which should
297  handle each supported index type accordingly.
298  """
299 
300  value = next(self.value.infer(context))
301  if value is util.Uninferable:
302  yield util.Uninferable
303  return
304 
305  index = next(self.slice.infer(context))
306  if index is util.Uninferable:
307  yield util.Uninferable
308  return
309 
310  # Try to deduce the index value.
311  index_value = _SUBSCRIPT_SENTINEL
312  if value.__class__ == bases.Instance:
313  index_value = index
314  else:
315  if index.__class__ == bases.Instance:
316  instance_as_index = helpers.class_instance_as_index(index)
317  if instance_as_index:
318  index_value = instance_as_index
319  else:
320  index_value = index
321  if index_value is _SUBSCRIPT_SENTINEL:
322  raise exceptions.InferenceError(node=self, context=context)
323 
324  try:
325  assigned = value.getitem(index_value, context)
326  except (exceptions.AstroidTypeError,
327  exceptions.AstroidIndexError,
328  exceptions.AttributeInferenceError,
329  AttributeError) as exc:
330  util.reraise(exceptions.InferenceError(node=self, error=exc,
331  context=context))
332 
333  # Prevent inferring if the inferred subscript
334  # is the same as the original subscripted object.
335  if self is assigned or assigned is util.Uninferable:
336  yield util.Uninferable
337  return
338  for inferred in assigned.infer(context):
339  yield inferred
340 
341  # Explicit StopIteration to return error information, see comment
342  # in raise_if_nothing_inferred.
343  raise StopIteration(dict(node=self, context=context))

349 @decorators.raise_if_nothing_inferred
350 @decorators.path_wrapper
351 def _infer_boolop(self, context=None):
352  """Infer a boolean operation (and / or / not).
353 
354  The function will calculate the boolean operation
355  for all pairs generated through inference for each component
356  node.
357  """
358  values = self.values
359  if self.op == 'or':
360  predicate = operator.truth
361  else:
362  predicate = operator.not_
363 
364  try:
365  values = [value.infer(context=context) for value in values]
366  except exceptions.InferenceError:
367  yield util.Uninferable
368  return
369 
370  for pair in itertools.product(*values):
371  if any(item is util.Uninferable for item in pair):
372  # Can't infer the final result, just yield Uninferable.
373  yield util.Uninferable
374  continue
375 
376  bool_values = [item.bool_value() for item in pair]
377  if any(item is util.Uninferable for item in bool_values):
378  # Can't infer the final result, just yield Uninferable.
379  yield util.Uninferable
380  continue
381 
382  # Since the boolean operations are short circuited operations,
383  # this code yields the first value for which the predicate is True
384  # and if no value respected the predicate, then the last value will
385  # be returned (or Uninferable if there was no last value).
386  # This is conforming to the semantics of `and` and `or`:
387  # 1 and 0 -> 1
388  # 0 and 1 -> 0
389  # 1 or 0 -> 1
390  # 0 or 1 -> 1
391  value = util.Uninferable
392  for value, bool_value in zip(pair, bool_values):
393  if predicate(bool_value):
394  yield value
395  break
396  else:
397  yield value
398 
399  # Explicit StopIteration to return error information, see comment
400  # in raise_if_nothing_inferred.
401  raise StopIteration(dict(node=self, context=context))

419 def _infer_unaryop(self, context=None):
420  """Infer what an UnaryOp should return when evaluated."""
421  for operand in self.operand.infer(context):
422  try:
423  yield operand.infer_unary_op(self.op)
424  except TypeError as exc:
425  # The operand doesn't support this operation.
426  yield util.BadUnaryOperationMessage(operand, self.op, exc)
427  except AttributeError as exc:
428  meth = protocols.UNARY_OP_METHOD[self.op]
429  if meth is None:
430  # `not node`. Determine node's boolean
431  # value and negate its result, unless it is
432  # Uninferable, which will be returned as is.
433  bool_value = operand.bool_value()
434  if bool_value is not util.Uninferable:
435  yield nodes.const_factory(not bool_value)
436  else:
437  yield util.Uninferable
438  else:
439  if not isinstance(operand, (bases.Instance, nodes.ClassDef)):
440  # The operation was used on something which
441  # doesn't support it.
442  yield util.BadUnaryOperationMessage(operand, self.op, exc)
443  continue
444 
445  try:
446  try:
447  methods = dunder_lookup.lookup(operand, meth)
448  except exceptions.AttributeInferenceError:
449  yield util.BadUnaryOperationMessage(operand, self.op, exc)
450  continue
451 
452  meth = methods[0]
453  inferred = next(meth.infer(context=context))
454  if inferred is util.Uninferable or not inferred.callable():
455  continue
456 
457  context = contextmod.copy_context(context)
458  context.callcontext = contextmod.CallContext(args=[operand])
459  call_results = inferred.infer_call_result(self, context=context)
460  result = next(call_results, None)
461  if result is None:
462  # Failed to infer, return the same type.
463  yield operand
464  else:
465  yield result
466  except exceptions.AttributeInferenceError as exc:
467  # The unary operation special method was not found.
468  yield util.BadUnaryOperationMessage(operand, self.op, exc)
469  except exceptions.InferenceError:
470  yield util.Uninferable

548 def _get_binop_flow(left, left_type, binary_opnode, right, right_type,
549  context, reverse_context):
550  """Get the flow for binary operations.
551 
552  The rules are a bit messy:
553 
554  * if left and right have the same type, then only one
555  method will be called, left.__op__(right)
556  * if left and right are unrelated typewise, then first
557  left.__op__(right) is tried and if this does not exist
558  or returns NotImplemented, then right.__rop__(left) is tried.
559  * if left is a subtype of right, then only left.__op__(right)
560  is tried.
561  * if left is a supertype of right, then right.__rop__(left)
562  is first tried and then left.__op__(right)
563  """
564  op = binary_opnode.op
565  if _same_type(left_type, right_type):
566  methods = [_bin_op(left, binary_opnode, op, right, context)]
567  elif helpers.is_subtype(left_type, right_type):
568  methods = [_bin_op(left, binary_opnode, op, right, context)]
569  elif helpers.is_supertype(left_type, right_type):
570  methods = [_bin_op(right, binary_opnode, op, left, reverse_context, reverse=True),
571  _bin_op(left, binary_opnode, op, right, context)]
572  else:
573  methods = [_bin_op(left, binary_opnode, op, right, context),
574  _bin_op(right, binary_opnode, op, left, reverse_context, reverse=True)]
575  return methods

578 def _get_aug_flow(left, left_type, aug_opnode, right, right_type,
579  context, reverse_context):
580  """Get the flow for augmented binary operations.
581 
582  The rules are a bit messy:
583 
584  * if left and right have the same type, then left.__augop__(right)
585  is first tried and then left.__op__(right).
586  * if left and right are unrelated typewise, then
587  left.__augop__(right) is tried, then left.__op__(right)
588  is tried and then right.__rop__(left) is tried.
589  * if left is a subtype of right, then left.__augop__(right)
590  is tried and then left.__op__(right).
591  * if left is a supertype of right, then left.__augop__(right)
592  is tried, then right.__rop__(left) and then
593  left.__op__(right)
594  """
595  bin_op = aug_opnode.op.strip("=")
596  aug_op = aug_opnode.op
597  if _same_type(left_type, right_type):
598  methods = [_aug_op(left, aug_opnode, aug_op, right, context),
599  _bin_op(left, aug_opnode, bin_op, right, context)]
600  elif helpers.is_subtype(left_type, right_type):
601  methods = [_aug_op(left, aug_opnode, aug_op, right, context),
602  _bin_op(left, aug_opnode, bin_op, right, context)]
603  elif helpers.is_supertype(left_type, right_type):
604  methods = [_aug_op(left, aug_opnode, aug_op, right, context),
605  _bin_op(right, aug_opnode, bin_op, left, reverse_context, reverse=True),
606  _bin_op(left, aug_opnode, bin_op, right, context)]
607  else:
608  methods = [_aug_op(left, aug_opnode, aug_op, right, context),
609  _bin_op(left, aug_opnode, bin_op, right, context),
610  _bin_op(right, aug_opnode, bin_op, left, reverse_context, reverse=True)]
611  return methods

614 def _infer_binary_operation(left, right, binary_opnode, context, flow_factory):
615  """Infer a binary operation between a left operand and a right operand
616 
617  This is used by both normal binary operations and augmented binary
618  operations, the only difference is the flow factory used.
619  """
620 
621  context, reverse_context = _get_binop_contexts(context, left, right)
622  left_type = helpers.object_type(left)
623  right_type = helpers.object_type(right)
624  methods = flow_factory(left, left_type, binary_opnode, right, right_type,
625  context, reverse_context)
626  for method in methods:
627  try:
628  results = list(method())
629  except AttributeError:
630  continue
631  except exceptions.AttributeInferenceError:
632  continue
633  except exceptions.InferenceError:
634  yield util.Uninferable
635  return
636  else:
637  if any(result is util.Uninferable for result in results):
638  yield util.Uninferable
639  return
640 
641  # TODO(cpopa): since the inference engine might return
642  # more values than are actually possible, we decide
643  # to return util.Uninferable if we have union types.
644  if all(map(_is_not_implemented, results)):
645  continue
646  not_implemented = sum(1 for result in results
647  if _is_not_implemented(result))
648  if not_implemented and not_implemented != len(results):
649  # Can't decide yet what this is, not yet though.
650  yield util.Uninferable
651  return
652 
653  for result in results:
654  yield result
655  return
