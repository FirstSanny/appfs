77 def _infer_stmts(stmts, context, frame=None):
78  """Return an iterator on statements inferred by each statement in *stmts*."""
79  stmt = None
80  inferred = False
81  if context is not None:
82  name = context.lookupname
83  context = context.clone()
84  else:
85  name = None
86  context = contextmod.InferenceContext()
87

92  continue
93  context.lookupname = stmt._infer_name(frame, name)
94  try:
95  for inferred in stmt.infer(context=context):

97  inferred = True
98  except exceptions.NameInferenceError:
99  continue
100  except exceptions.InferenceError:
101  yield util.Uninferable
102  inferred = True

104  raise exceptions.InferenceError(

106  stmts=stmts, frame=frame, context=context)

109 def _infer_method_result_truth(instance, method_name, context):
110  # Get the method from the instance and try to infer
111  # its return's truth value.
112  meth = next(instance.igetattr(method_name, context=context), None)
113  if meth and hasattr(meth, 'infer_call_result'):
114  if not meth.callable():
115  return util.Uninferable

117  if value is util.Uninferable:

159  def igetattr(self, name, context=None):
160  """inferred getattr"""
161  if not context:
162  context = contextmod.InferenceContext()
163  try:
164  # avoid recursively inferring the same attr on the same class
165  if context.push((self._proxied, name)):
166  return
167 
168  # XXX frame should be self._proxied, or not ?
169  get_attr = self.getattr(name, context, lookupclass=False)
170  for stmt in _infer_stmts(self._wrap_attr(get_attr, context),
171  context, frame=self):
172  yield stmt
173  except exceptions.AttributeInferenceError:
174  try:
175  # fallback to class.igetattr since it has some logic to handle
176  # descriptors
177  attrs = self._proxied.igetattr(name, context, class_context=False)
178  for stmt in self._wrap_attr(attrs, context):
179  yield stmt
180  except exceptions.AttributeInferenceError as error:
181  util.reraise(exceptions.InferenceError(**vars(error)))

183  def _wrap_attr(self, attrs, context=None):
184  """wrap bound methods of attrs in a InstanceMethod proxies"""
185  for attr in attrs:
186  if isinstance(attr, UnboundMethod):
187  if _is_property(attr):
188  for inferred in attr.infer_call_result(self, context):
189  yield inferred
190  else:
191  yield BoundMethod(attr, self)
192  elif hasattr(attr, 'name') and attr.name == '<lambda>':
193  # This is a lambda function defined at class level,
194  # since its scope is the underlying _proxied class.
195  # Unfortunately, we can't do an isinstance check here,
196  # because of the circular dependency between astroid.bases
197  # and astroid.scoped_nodes.
198  if attr.statement().scope() == self._proxied:
199  if attr.args.args and attr.args.args[0].name == 'self':
200  yield BoundMethod(attr, self)
201  continue
202  yield attr
203  else:
204  yield attr

206  def infer_call_result(self, caller, context=None):
207  """infer what a class instance is returning when called"""
208  inferred = False
209  for node in self._proxied.igetattr('__call__', context):
210  if node is util.Uninferable or not node.callable():
211  continue

213  inferred = True

247  def bool_value(self):
248  """Infer the truth value for an Instance
249 
250  The truth value of an instance is determined by these conditions:
251 
252  * if it implements __bool__ on Python 3 or __nonzero__
253  on Python 2, then its bool value will be determined by
254  calling this special method and checking its result.
255  * when this method is not defined, __len__() is called, if it
256  is defined, and the object is considered true if its result is
257  nonzero. If a class defines neither __len__() nor __bool__(),
258  all its instances are considered true.
259  """
260  context = contextmod.InferenceContext()
261  context.callcontext = contextmod.CallContext(args=[])
262  context.boundnode = self
263 
264  try:
265  result = _infer_method_result_truth(self, BOOL_SPECIAL_METHOD, context)
266  except (exceptions.InferenceError, exceptions.AttributeInferenceError):
267  # Fallback to __len__.
268  try:
269  result = _infer_method_result_truth(self, '__len__', context)
270  except (exceptions.AttributeInferenceError, exceptions.InferenceError):
271  return True
272  return result

331  def _infer_type_new_call(self, caller, context):
332  """Try to infer what type.__new__(mcs, name, bases, attrs) returns.
333 
334  In order for such call to be valid, the metaclass needs to be
335  a subtype of ``type``, the name needs to be a string, the bases
336  needs to be a tuple of classes and the attributes a dictionary
337  of strings to values.
338  """
339  from astroid import node_classes
340  # Verify the metaclass
341  mcs = next(caller.args[0].infer(context=context))
342  if mcs.__class__.__name__ != 'ClassDef':
343  # Not a valid first argument.
344  return
345  if not mcs.is_subtype_of("%s.type" % BUILTINS):
346  # Not a valid metaclass.
347  return
348 
349  # Verify the name
350  name = next(caller.args[1].infer(context=context))
351  if name.__class__.__name__ != 'Const':
352  # Not a valid name, needs to be a const.
353  return
354  if not isinstance(name.value, str):
355  # Needs to be a string.
356  return
357 
358  # Verify the bases
359  bases = next(caller.args[2].infer(context=context))
360  if bases.__class__.__name__ != 'Tuple':
361  # Needs to be a tuple.
362  return
363  inferred_bases = [next(elt.infer(context=context))
364  for elt in bases.elts]
365  if any(base.__class__.__name__ != 'ClassDef'
366  for base in inferred_bases):
367  # All the bases needs to be Classes
368  return
369 
370  # Verify the attributes.
371  attrs = next(caller.args[3].infer(context=context))
372  if attrs.__class__.__name__ != 'Dict':
373  # Needs to be a dictionary.
374  return
375  cls_locals = collections.defaultdict(list)
376  for key, value in attrs.items:
377  key = next(key.infer(context=context))
378  value = next(value.infer(context=context))
379  if key.__class__.__name__ != 'Const':
380  # Something invalid as an attribute.
381  return
382  if not isinstance(key.value, str):
383  # Not a proper attribute.
384  return
385  cls_locals[key.value].append(value)
386 
387  # Build the class from now.
388  cls = mcs.__class__(name=name.value, lineno=caller.lineno,
389  col_offset=caller.col_offset,
390  parent=caller)
391  empty = node_classes.Pass()
392  cls.postinit(bases=bases.elts, body=[empty], decorators=[],
393  newstyle=True, metaclass=mcs, keywords=[])
394  cls.locals = cls_locals
395  return cls
