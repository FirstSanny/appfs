282  def _check_barycentric_transforms(self, tri, err_msg="",
283  unit_cube=False,
284  unit_cube_tol=0):
285  """Check that a triangulation has reasonable barycentric transforms"""
286  vertices = tri.points[tri.vertices]
287  sc = 1/(tri.ndim + 1.0)
288  centroids = vertices.sum(axis=1) * sc
289 
290  # Either: (i) the simplex has a `nan` barycentric transform,
291  # or, (ii) the centroid is in the simplex
292 
293  def barycentric_transform(tr, x):
294  ndim = tr.shape[1]
295  r = tr[:,-1,:]
296  Tinv = tr[:,:-1,:]
297  return np.einsum('ijk,ik->ij', Tinv, x - r)
298 
299  eps = np.finfo(float).eps
300 
301  c = barycentric_transform(tri.transform, centroids)
302  olderr = np.seterr(invalid="ignore")
303  try:
304  ok = np.isnan(c).all(axis=1) | (abs(c - sc)/sc < 0.1).all(axis=1)
305  finally:
306  np.seterr(**olderr)
307 
308  assert_(ok.all(), "%s %s" % (err_msg, np.where(~ok)))
309 
310  # Invalid simplices must be (nearly) zero volume
311  q = vertices[:,:-1,:] - vertices[:,-1,None,:]
312  volume = np.array([np.linalg.det(q[k,:,:])
313  for k in range(tri.nsimplex)])
314  ok = np.isfinite(tri.transform[:,0,0]) | (volume < np.sqrt(eps))
315  assert_(ok.all(), "%s %s" % (err_msg, np.where(~ok)))
316 
317  # Also, find_simplex for the centroid should end up in some
318  # simplex for the non-degenerate cases
319  j = tri.find_simplex(centroids)
320  ok = (j != -1) | np.isnan(tri.transform[:,0,0])
321  assert_(ok.all(), "%s %s" % (err_msg, np.where(~ok)))
322 
323  if unit_cube:
324  # If in unit cube, no interior point should be marked out of hull
325  at_boundary = (centroids <= unit_cube_tol).any(axis=1)
326  at_boundary |= (centroids >= 1 - unit_cube_tol).any(axis=1)
327 
328  ok = (j != -1) | at_boundary
329  assert_(ok.all(), "%s %s" % (err_msg, np.where(~ok)))

575 def assert_hulls_equal(points, facets_1, facets_2):
576  # Check that two convex hulls constructed from the same point set
577  # are equal
578 
579  facets_1 = set(map(sorted_tuple, facets_1))
580  facets_2 = set(map(sorted_tuple, facets_2))
581 
582  if facets_1 != facets_2 and points.shape[1] == 2:
583  # The direct check fails for the pathological cases
584  # --- then the convex hull from Delaunay differs (due
585  # to rounding error etc.) from the hull computed
586  # otherwise, by the question whether (tricoplanar)
587  # points that lie almost exactly on the hull are
588  # included as vertices of the hull or not.
589  #
590  # So we check the result, and accept it if the Delaunay
591  # hull line segments are a subset of the usual hull.
592 
593  eps = 1000 * np.finfo(float).eps
594 
595  for a, b in facets_1:
596  for ap, bp in facets_2:
597  t = points[bp] - points[ap]
598  t /= np.linalg.norm(t) # tangent
599  n = np.array([-t[1], t[0]]) # normal
600 
601  # check that the two line segments are parallel
602  # to the same line
603  c1 = np.dot(n, points[b] - points[ap])
604  c2 = np.dot(n, points[a] - points[ap])
605  if not np.allclose(np.dot(c1, n), 0):
606  continue
607  if not np.allclose(np.dot(c2, n), 0):
608  continue
609 
610  # Check that the segment (a, b) is contained in (ap, bp)
611  c1 = np.dot(t, points[a] - points[ap])
612  c2 = np.dot(t, points[b] - points[ap])
613  c3 = np.dot(t, points[bp] - points[ap])
614  if c1 < -eps or c1 > c3 + eps:
615  continue
616  if c2 < -eps or c2 > c3 + eps:
617  continue
618 
619  # OK:
620  break
621  else:
622  raise AssertionError("comparison fails")
623 
624  # it was OK
625  return
626 
627  assert_equal(facets_1, facets_2)
