20 # Some notes on function parameters:
21 #
22 # `cutoff` and `width` are given as a numbers between 0 and 1. These
23 # are relative frequencies, expressed as a fraction of the Nyquist rate.
24 # For example, if the Nyquist rate is 2KHz, then width=0.15 is a width
25 # of 300 Hz.
26 #
27 # The `order` of a FIR filter is one less than the number of taps.
28 # This is a potential source of confusion, so in the following code,
29 # we will always use the number of taps as the parameterization of
30 # the 'size' of the filter. The "number of taps" means the number
31 # of coefficients, which is the same as the length of the impulse
32 # response of the filter.

35 def kaiser_beta(a):
36  """Compute the Kaiser parameter `beta`, given the attenuation `a`.
37 
38  Parameters
39  ----------
40  a : float
41  The desired attenuation in the stopband and maximum ripple in
42  the passband, in dB. This should be a *positive* number.
43 
44  Returns
45  -------
46  beta : float
47  The `beta` parameter to be used in the formula for a Kaiser window.
48 
49  References
50  ----------
51  Oppenheim, Schafer, "Discrete-Time Signal Processing", p.475-476.
52  """
53  if a > 50:
54  beta = 0.1102 * (a - 8.7)
55  elif a > 21:
56  beta = 0.5842 * (a - 21) ** 0.4 + 0.07886 * (a - 21)
57  else:
58  beta = 0.0
59  return beta

62 def kaiser_atten(numtaps, width):
63  """Compute the attenuation of a Kaiser FIR filter.
64 
65  Given the number of taps `N` and the transition width `width`, compute the
66  attenuation `a` in dB, given by Kaiser's formula:
67 
68  a = 2.285 * (N - 1) * pi * width + 7.95
69 
70  Parameters
71  ----------
72  numtaps : int
73  The number of taps in the FIR filter.
74  width : float
75  The desired width of the transition region between passband and
76  stopband (or, in general, at any discontinuity) for the filter.
77 
78  Returns
79  -------
80  a : float
81  The attenuation of the ripple, in dB.
82 
83  See Also
84  --------
85  kaiserord, kaiser_beta
86  """
87  a = 2.285 * (numtaps - 1) * np.pi * width + 7.95
88  return a

91 def kaiserord(ripple, width):
92  """
93  Design a Kaiser window to limit ripple and width of transition region.
94 
95  Parameters
96  ----------
97  ripple : float
98  Positive number specifying maximum ripple in passband (dB) and minimum
99  ripple in stopband.
100  width : float
101  Width of transition region (normalized so that 1 corresponds to pi
102  radians / sample).
103 
104  Returns
105  -------
106  numtaps : int
107  The length of the kaiser window.
108  beta : float
109  The beta parameter for the kaiser window.
110 
111  See Also
112  --------
113  kaiser_beta, kaiser_atten
114 
115  Notes
116  -----
117  There are several ways to obtain the Kaiser window:
118 
119  - ``signal.kaiser(numtaps, beta, sym=True)``
120  - ``signal.get_window(beta, numtaps)``
121  - ``signal.get_window(('kaiser', beta), numtaps)``
122 
123  The empirical equations discovered by Kaiser are used.
124 
125  References
126  ----------
127  Oppenheim, Schafer, "Discrete-Time Signal Processing", p.475-476.
128 
129  """
130  A = abs(ripple) # in case somebody is confused as to what's meant
131  if A < 8:
132  # Formula for N is not valid in this range.
133  raise ValueError("Requested maximum ripple attentuation %f is too "
134  "small for the Kaiser formula." % A)
135  beta = kaiser_beta(A)
136 
137  # Kaiser's formula (as given in Oppenheim and Schafer) is for the filter
138  # order, so we have to add 1 to get the number of taps.
139  numtaps = (A - 7.95) / 2.285 / (np.pi * width) + 1
140 
141  return int(ceil(numtaps)), beta

144 def firwin(numtaps, cutoff, width=None, window='hamming', pass_zero=True,
145  scale=True, nyq=1.0):
146  """
147  FIR filter design using the window method.
148 
149  This function computes the coefficients of a finite impulse response
150  filter. The filter will have linear phase; it will be Type I if
151  `numtaps` is odd and Type II if `numtaps` is even.
152 
153  Type II filters always have zero response at the Nyquist rate, so a
154  ValueError exception is raised if firwin is called with `numtaps` even and
155  having a passband whose right end is at the Nyquist rate.
156 
157  Parameters
158  ----------
159  numtaps : int
160  Length of the filter (number of coefficients, i.e. the filter
161  order + 1). `numtaps` must be even if a passband includes the
162  Nyquist frequency.
163  cutoff : float or 1D array_like
164  Cutoff frequency of filter (expressed in the same units as `nyq`)
165  OR an array of cutoff frequencies (that is, band edges). In the
166  latter case, the frequencies in `cutoff` should be positive and
167  monotonically increasing between 0 and `nyq`. The values 0 and
168  `nyq` must not be included in `cutoff`.
169  width : float or None, optional
170  If `width` is not None, then assume it is the approximate width
171  of the transition region (expressed in the same units as `nyq`)
172  for use in Kaiser FIR filter design. In this case, the `window`
173  argument is ignored.
174  window : string or tuple of string and parameter values, optional
175  Desired window to use. See `scipy.signal.get_window` for a list
176  of windows and required parameters.
177  pass_zero : bool, optional
178  If True, the gain at the frequency 0 (i.e. the "DC gain") is 1.
179  Otherwise the DC gain is 0.
180  scale : bool, optional
181  Set to True to scale the coefficients so that the frequency
182  response is exactly unity at a certain frequency.
183  That frequency is either:
184 
185  - 0 (DC) if the first passband starts at 0 (i.e. pass_zero
186  is True)
187  - `nyq` (the Nyquist rate) if the first passband ends at
188  `nyq` (i.e the filter is a single band highpass filter);
189  center of first passband otherwise
190 
191  nyq : float, optional
192  Nyquist frequency. Each frequency in `cutoff` must be between 0
193  and `nyq`.
194 
195  Returns
196  -------
197  h : (numtaps,) ndarray
198  Coefficients of length `numtaps` FIR filter.
199 
200  Raises
201  ------
202  ValueError
203  If any value in `cutoff` is less than or equal to 0 or greater
204  than or equal to `nyq`, if the values in `cutoff` are not strictly
205  monotonically increasing, or if `numtaps` is even but a passband
206  includes the Nyquist frequency.
207 
208  See also
209  --------
210  firwin2
211  firls
212  minimum_phase
213  remez
214 
215  Examples
216  --------
217  Low-pass from 0 to f:
218 
219  >>> from scipy import signal
220  >>> numtaps = 3
221  >>> f = 0.1
222  >>> signal.firwin(numtaps, f)
223  array([ 0.06799017, 0.86401967, 0.06799017])
224 
225  Use a specific window function:
226 
227  >>> signal.firwin(numtaps, f, window='nuttall')
228  array([ 3.56607041e-04, 9.99286786e-01, 3.56607041e-04])
229 
230  High-pass ('stop' from 0 to f):
231 
232  >>> signal.firwin(numtaps, f, pass_zero=False)
233  array([-0.00859313, 0.98281375, -0.00859313])
234 
235  Band-pass:
236 
237  >>> f1, f2 = 0.1, 0.2
238  >>> signal.firwin(numtaps, [f1, f2], pass_zero=False)
239  array([ 0.06301614, 0.88770441, 0.06301614])
240 
241  Band-stop:
242 
243  >>> signal.firwin(numtaps, [f1, f2])
244  array([-0.00801395, 1.0160279 , -0.00801395])
245 
246  Multi-band (passbands are [0, f1], [f2, f3] and [f4, 1]):
247 
248  >>> f3, f4 = 0.3, 0.4
249  >>> signal.firwin(numtaps, [f1, f2, f3, f4])
250  array([-0.01376344, 1.02752689, -0.01376344])
251 
252  Multi-band (passbands are [f1, f2] and [f3,f4]):
253 
254  >>> signal.firwin(numtaps, [f1, f2, f3, f4], pass_zero=False)
255  array([ 0.04890915, 0.91284326, 0.04890915])
256 
257  """
258 
259  # The major enhancements to this function added in November 2010 were
260  # developed by Tom Krauss (see ticket #902).
261 
262  cutoff = np.atleast_1d(cutoff) / float(nyq)
263 
264  # Check for invalid input.
265  if cutoff.ndim > 1:
266  raise ValueError("The cutoff argument must be at most "
267  "one-dimensional.")
268  if cutoff.size == 0:
269  raise ValueError("At least one cutoff frequency must be given.")
270  if cutoff.min() <= 0 or cutoff.max() >= 1:
271  raise ValueError("Invalid cutoff frequency: frequencies must be "
272  "greater than 0 and less than nyq.")
273  if np.any(np.diff(cutoff) <= 0):
274  raise ValueError("Invalid cutoff frequencies: the frequencies "
275  "must be strictly increasing.")
276 
277  if width is not None:
278  # A width was given. Find the beta parameter of the Kaiser window
279  # and set `window`. This overrides the value of `window` passed in.
280  atten = kaiser_atten(numtaps, float(width) / nyq)
281  beta = kaiser_beta(atten)
282  window = ('kaiser', beta)
283 
284  pass_nyquist = bool(cutoff.size & 1) ^ pass_zero
285  if pass_nyquist and numtaps % 2 == 0:
286  raise ValueError("A filter with an even number of coefficients must "
287  "have zero response at the Nyquist rate.")
288 
289  # Insert 0 and/or 1 at the ends of cutoff so that the length of cutoff
290  # is even, and each pair in cutoff corresponds to passband.
291  cutoff = np.hstack(([0.0] * pass_zero, cutoff, [1.0] * pass_nyquist))
292 
293  # `bands` is a 2D array; each row gives the left and right edges of
294  # a passband.
295  bands = cutoff.reshape(-1, 2)
296 
297  # Build up the coefficients.
298  alpha = 0.5 * (numtaps - 1)
299  m = np.arange(0, numtaps) - alpha
300  h = 0
301  for left, right in bands:
302  h += right * sinc(right * m)
303  h -= left * sinc(left * m)
304 
305  # Get and apply the window function.
306  from .signaltools import get_window
307  win = get_window(window, numtaps, fftbins=False)
308  h *= win
309 
310  # Now handle scaling if desired.
311  if scale:
312  # Get the first passband.
313  left, right = bands[0]
314  if left == 0:
315  scale_frequency = 0.0
316  elif right == 1:
317  scale_frequency = 1.0
318  else:
319  scale_frequency = 0.5 * (left + right)
320  c = np.cos(np.pi * m * scale_frequency)
321  s = np.sum(h * c)
322  h /= s
323 
324  return h

331 def firwin2(numtaps, freq, gain, nfreqs=None, window='hamming', nyq=1.0,
332  antisymmetric=False):
333  """
334  FIR filter design using the window method.
335 
336  From the given frequencies `freq` and corresponding gains `gain`,
337  this function constructs an FIR filter with linear phase and
338  (approximately) the given frequency response.
339 
340  Parameters
341  ----------
342  numtaps : int
343  The number of taps in the FIR filter. `numtaps` must be less than
344  `nfreqs`.
345  freq : array_like, 1D
346  The frequency sampling points. Typically 0.0 to 1.0 with 1.0 being
347  Nyquist. The Nyquist frequency can be redefined with the argument
348  `nyq`.
349  The values in `freq` must be nondecreasing. A value can be repeated
350  once to implement a discontinuity. The first value in `freq` must
351  be 0, and the last value must be `nyq`.
352  gain : array_like
353  The filter gains at the frequency sampling points. Certain
354  constraints to gain values, depending on the filter type, are applied,
355  see Notes for details.
356  nfreqs : int, optional
357  The size of the interpolation mesh used to construct the filter.
358  For most efficient behavior, this should be a power of 2 plus 1
359  (e.g, 129, 257, etc). The default is one more than the smallest
360  power of 2 that is not less than `numtaps`. `nfreqs` must be greater
361  than `numtaps`.
362  window : string or (string, float) or float, or None, optional
363  Window function to use. Default is "hamming". See
364  `scipy.signal.get_window` for the complete list of possible values.
365  If None, no window function is applied.
366  nyq : float, optional
367  Nyquist frequency. Each frequency in `freq` must be between 0 and
368  `nyq` (inclusive).
369  antisymmetric : bool, optional
370  Whether resulting impulse response is symmetric/antisymmetric.
371  See Notes for more details.
372 
373  Returns
374  -------
375  taps : ndarray
376  The filter coefficients of the FIR filter, as a 1-D array of length
377  `numtaps`.
378 
379  See also
380  --------
381  firls
382  firwin
383  minimum_phase
384  remez
385 
386  Notes
387  -----
388  From the given set of frequencies and gains, the desired response is
389  constructed in the frequency domain. The inverse FFT is applied to the
390  desired response to create the associated convolution kernel, and the
391  first `numtaps` coefficients of this kernel, scaled by `window`, are
392  returned.
393 
394  The FIR filter will have linear phase. The type of filter is determined by
395  the value of 'numtaps` and `antisymmetric` flag.
396  There are four possible combinations:
397 
398  - odd `numtaps`, `antisymmetric` is False, type I filter is produced
399  - even `numtaps`, `antisymmetric` is False, type II filter is produced
400  - odd `numtaps`, `antisymmetric` is True, type III filter is produced
401  - even `numtaps`, `antisymmetric` is True, type IV filter is produced
402 
403  Magnitude response of all but type I filters are subjects to following
404  constraints:
405 
406  - type II -- zero at the Nyquist frequency
407  - type III -- zero at zero and Nyquist frequencies
408  - type IV -- zero at zero frequency
409 
410  .. versionadded:: 0.9.0
411 
412  References
413  ----------
414  .. [1] Oppenheim, A. V. and Schafer, R. W., "Discrete-Time Signal
415  Processing", Prentice-Hall, Englewood Cliffs, New Jersey (1989).
416  (See, for example, Section 7.4.)
417 
418  .. [2] Smith, Steven W., "The Scientist and Engineer's Guide to Digital
419  Signal Processing", Ch. 17. http://www.dspguide.com/ch17/1.htm
420 
421  Examples
422  --------
423  A lowpass FIR filter with a response that is 1 on [0.0, 0.5], and
424  that decreases linearly on [0.5, 1.0] from 1 to 0:
425 
426  >>> from scipy import signal
427  >>> taps = signal.firwin2(150, [0.0, 0.5, 1.0], [1.0, 1.0, 0.0])
428  >>> print(taps[72:78])
429  [-0.02286961 -0.06362756 0.57310236 0.57310236 -0.06362756 -0.02286961]
430 
431  """
432 
433  if len(freq) != len(gain):
434  raise ValueError('freq and gain must be of same length.')
435 
436  if nfreqs is not None and numtaps >= nfreqs:
437  raise ValueError(('ntaps must be less than nfreqs, but firwin2 was '
438  'called with ntaps=%d and nfreqs=%s') %
439  (numtaps, nfreqs))
440 
441  if freq[0] != 0 or freq[-1] != nyq:
442  raise ValueError('freq must start with 0 and end with `nyq`.')
443  d = np.diff(freq)
444  if (d < 0).any():
445  raise ValueError('The values in freq must be nondecreasing.')
446  d2 = d[:-1] + d[1:]
447  if (d2 == 0).any():
448  raise ValueError('A value in freq must not occur more than twice.')
449 
450  if antisymmetric:
451  if numtaps % 2 == 0:
452  ftype = 4
453  else:
454  ftype = 3
455  else:
456  if numtaps % 2 == 0:
457  ftype = 2
458  else:
459  ftype = 1
460 
461  if ftype == 2 and gain[-1] != 0.0:
462  raise ValueError("A Type II filter must have zero gain at the "
463  "Nyquist rate.")
464  elif ftype == 3 and (gain[0] != 0.0 or gain[-1] != 0.0):
465  raise ValueError("A Type III filter must have zero gain at zero "
466  "and Nyquist rates.")
467  elif ftype == 4 and gain[0] != 0.0:
468  raise ValueError("A Type IV filter must have zero gain at zero rate.")
469 
470  if nfreqs is None:
471  nfreqs = 1 + 2 ** int(ceil(log(numtaps, 2)))
472 
473  # Tweak any repeated values in freq so that interp works.
474  eps = np.finfo(float).eps
475  for k in range(len(freq)):
476  if k < len(freq) - 1 and freq[k] == freq[k + 1]:
477  freq[k] = freq[k] - eps
478  freq[k + 1] = freq[k + 1] + eps
479 
480  # Linearly interpolate the desired response on a uniform mesh `x`.
481  x = np.linspace(0.0, nyq, nfreqs)
482  fx = np.interp(x, freq, gain)
483 
484  # Adjust the phases of the coefficients so that the first `ntaps` of the
485  # inverse FFT are the desired filter coefficients.
486  shift = np.exp(-(numtaps - 1) / 2. * 1.j * np.pi * x / nyq)
487  if ftype > 2:
488  shift *= 1j
489 
490  fx2 = fx * shift
491 
492  # Use irfft to compute the inverse FFT.
493  out_full = irfft(fx2)
494 
495  if window is not None:
496  # Create the window to apply to the filter coefficients.
497  from .signaltools import get_window
498  wind = get_window(window, numtaps, fftbins=False)
499  else:
500  wind = 1
501 
502  # Keep only the first `numtaps` coefficients in `out`, and multiply by
503  # the window.
504  out = out_full[:numtaps] * wind
505 
506  if ftype == 3:
507  out[out.size // 2] = 0.0
508 
509  return out

512 def remez(numtaps, bands, desired, weight=None, Hz=1, type='bandpass',
513  maxiter=25, grid_density=16):
514  """
515  Calculate the minimax optimal filter using the Remez exchange algorithm.
516 
517  Calculate the filter-coefficients for the finite impulse response
518  (FIR) filter whose transfer function minimizes the maximum error
519  between the desired gain and the realized gain in the specified
520  frequency bands using the Remez exchange algorithm.
521 
522  Parameters
523  ----------
524  numtaps : int
525  The desired number of taps in the filter. The number of taps is
526  the number of terms in the filter, or the filter order plus one.
527  bands : array_like
528  A monotonic sequence containing the band edges in Hz.
529  All elements must be non-negative and less than half the sampling
530  frequency as given by `Hz`.
531  desired : array_like
532  A sequence half the size of bands containing the desired gain
533  in each of the specified bands.
534  weight : array_like, optional
535  A relative weighting to give to each band region. The length of
536  `weight` has to be half the length of `bands`.
537  Hz : scalar, optional
538  The sampling frequency in Hz. Default is 1.
539  type : {'bandpass', 'differentiator', 'hilbert'}, optional
540  The type of filter:
541 
542  * 'bandpass' : flat response in bands. This is the default.
543 
544  * 'differentiator' : frequency proportional response in bands.
545 
546  * 'hilbert' : filter with odd symmetry, that is, type III
547  (for even order) or type IV (for odd order)
548  linear phase filters.
549 
550  maxiter : int, optional
551  Maximum number of iterations of the algorithm. Default is 25.
552  grid_density : int, optional
553  Grid density. The dense grid used in `remez` is of size
554  ``(numtaps + 1) * grid_density``. Default is 16.
555 
556  Returns
557  -------
558  out : ndarray
559  A rank-1 array containing the coefficients of the optimal
560  (in a minimax sense) filter.
561 
562  See Also
563  --------
564  firls
565  firwin
566  firwin2
567  minimum_phase
568 
569  References
570  ----------
571  .. [1] J. H. McClellan and T. W. Parks, "A unified approach to the
572  design of optimum FIR linear phase digital filters",
573  IEEE Trans. Circuit Theory, vol. CT-20, pp. 697-701, 1973.
574  .. [2] J. H. McClellan, T. W. Parks and L. R. Rabiner, "A Computer
575  Program for Designing Optimum FIR Linear Phase Digital
576  Filters", IEEE Trans. Audio Electroacoust., vol. AU-21,
577  pp. 506-525, 1973.
578 
579  Examples
580  --------
581  We want to construct a filter with a passband at 0.2-0.4 Hz, and
582  stop bands at 0-0.1 Hz and 0.45-0.5 Hz. Note that this means that the
583  behavior in the frequency ranges between those bands is unspecified and
584  may overshoot.
585 
586  >>> from scipy import signal
587  >>> bpass = signal.remez(72, [0, 0.1, 0.2, 0.4, 0.45, 0.5], [0, 1, 0])
588  >>> freq, response = signal.freqz(bpass)
589  >>> ampl = np.abs(response)
590 
591  >>> import matplotlib.pyplot as plt
592  >>> fig = plt.figure()
593  >>> ax1 = fig.add_subplot(111)
594  >>> ax1.semilogy(freq/(2*np.pi), ampl, 'b-') # freq in Hz
595  >>> plt.show()
596 
597  """
598  # Convert type
599  try:
600  tnum = {'bandpass': 1, 'differentiator': 2, 'hilbert': 3}[type]
601  except KeyError:
602  raise ValueError("Type must be 'bandpass', 'differentiator', "
603  "or 'hilbert'")
604 
605  # Convert weight
606  if weight is None:
607  weight = [1] * len(desired)
608 
609  bands = np.asarray(bands).copy()
610  return sigtools._remez(numtaps, bands, desired, weight, tnum, Hz,
611  maxiter, grid_density)

614 def firls(numtaps, bands, desired, weight=None, nyq=1.):
615  """
616  FIR filter design using least-squares error minimization.
617 
618  Calculate the filter coefficients for the linear-phase finite
619  impulse response (FIR) filter which has the best approximation
620  to the desired frequency response described by `bands` and
621  `desired` in the least squares sense (i.e., the integral of the
622  weighted mean-squared error within the specified bands is
623  minimized).
624 
625  Parameters
626  ----------
627  numtaps : int
628  The number of taps in the FIR filter. `numtaps` must be odd.
629  bands : array_like
630  A monotonic nondecreasing sequence containing the band edges in
631  Hz. All elements must be non-negative and less than or equal to
632  the Nyquist frequency given by `nyq`.
633  desired : array_like
634  A sequence the same size as `bands` containing the desired gain
635  at the start and end point of each band.
636  weight : array_like, optional
637  A relative weighting to give to each band region when solving
638  the least squares problem. `weight` has to be half the size of
639  `bands`.
640  nyq : float, optional
641  Nyquist frequency. Each frequency in `bands` must be between 0
642  and `nyq` (inclusive).
643 
644  Returns
645  -------
646  coeffs : ndarray
647  Coefficients of the optimal (in a least squares sense) FIR filter.
648 
649  See also
650  --------
651  firwin
652  firwin2
653  minimum_phase
654  remez
655 
656  Notes
657  -----
658  This implementation follows the algorithm given in [1]_.
659  As noted there, least squares design has multiple advantages:
660 
661  1. Optimal in a least-squares sense.
662  2. Simple, non-iterative method.
663  3. The general solution can obtained by solving a linear
664  system of equations.
665  4. Allows the use of a frequency dependent weighting function.
666 
667  This function constructs a Type I linear phase FIR filter, which
668  contains an odd number of `coeffs` satisfying for :math:`n < numtaps`:
669 
670  .. math:: coeffs(n) = coeffs(numtaps - 1 - n)
671 
672  The odd number of coefficients and filter symmetry avoid boundary
673  conditions that could otherwise occur at the Nyquist and 0 frequencies
674  (e.g., for Type II, III, or IV variants).
675 
676  .. versionadded:: 0.18
677 
678  References
679  ----------
680  .. [1] Ivan Selesnick, Linear-Phase Fir Filter Design By Least Squares.
681  OpenStax CNX. Aug 9, 2005.
682  http://cnx.org/contents/eb1ecb35-03a9-4610-ba87-41cd771c95f2@7

684  Examples
685  --------
686  We want to construct a band-pass filter. Note that the behavior in the
687  frequency ranges between our stop bands and pass bands is unspecified,
688  and thus may overshoot depending on the parameters of our filter:
689 
690  >>> from scipy import signal
691  >>> import matplotlib.pyplot as plt
692  >>> fig, axs = plt.subplots(2)
693  >>> nyq = 5. # Hz
694  >>> desired = (0, 0, 1, 1, 0, 0)
695  >>> for bi, bands in enumerate(((0, 1, 2, 3, 4, 5), (0, 1, 2, 4, 4.5, 5))):
696  ... fir_firls = signal.firls(73, bands, desired, nyq=nyq)
697  ... fir_remez = signal.remez(73, bands, desired[::2], Hz=2 * nyq)
698  ... fir_firwin2 = signal.firwin2(73, bands, desired, nyq=nyq)
699  ... hs = list()
700  ... ax = axs[bi]
701  ... for fir in (fir_firls, fir_remez, fir_firwin2):
702  ... freq, response = signal.freqz(fir)
703  ... hs.append(ax.semilogy(nyq*freq/(np.pi), np.abs(response))[0])
704  ... for band, gains in zip(zip(bands[::2], bands[1::2]), zip(desired[::2], desired[1::2])):
705  ... ax.semilogy(band, np.maximum(gains, 1e-7), 'k--', linewidth=2)
706  ... if bi == 0:
707  ... ax.legend(hs, ('firls', 'remez', 'firwin2'), loc='lower center', frameon=False)
708  ... else:
709  ... ax.set_xlabel('Frequency (Hz)')
710  ... ax.grid(True)
711  ... ax.set(title='Band-pass %d-%d Hz' % bands[2:4], ylabel='Magnitude')
712  ...
713  >>> fig.tight_layout()
714  >>> plt.show()

716  """ # noqa
717  numtaps = int(numtaps)
718  if numtaps % 2 == 0 or numtaps < 1:
719  raise ValueError("numtaps must be odd and >= 1")
720  M = (numtaps-1) // 2
721 
722  # normalize bands 0->1 and make it 2 columns
723  nyq = float(nyq)
724  if nyq <= 0:
725  raise ValueError('nyq must be positive, got %s <= 0.' % nyq)
726  bands = np.asarray(bands).flatten() / nyq
727  if len(bands) % 2 != 0:
728  raise ValueError("bands must contain frequency pairs.")
729  bands.shape = (-1, 2)
730 
731  # check remaining params
732  desired = np.asarray(desired).flatten()
733  if bands.size != desired.size:
734  raise ValueError("desired must have one entry per frequency, got %s "
735  "gains for %s frequencies."
736  % (desired.size, bands.size))
737  desired.shape = (-1, 2)
738  if (np.diff(bands) <= 0).any() or (np.diff(bands[:, 0]) < 0).any():
739  raise ValueError("bands must be monotonically nondecreasing and have "
740  "width > 0.")
741  if (bands[:-1, 1] > bands[1:, 0]).any():
742  raise ValueError("bands must not overlap.")
743  if (desired < 0).any():
744  raise ValueError("desired must be non-negative.")
745  if weight is None:
746  weight = np.ones(len(desired))
747  weight = np.asarray(weight).flatten()
748  if len(weight) != len(desired):
749  raise ValueError("weight must be the same size as the number of "
750  "band pairs (%s)." % (len(bands),))
751  if (weight < 0).any():
752  raise ValueError("weight must be non-negative.")
753 
754  # Set up the linear matrix equation to be solved, Qa = b
755 
756  # We can express Q(k,n) = 0.5 Q1(k,n) + 0.5 Q2(k,n)
757  # where Q1(k,n)=q(k−n) and Q2(k,n)=q(k+n), i.e. a Toeplitz plus Hankel.
758 
759  # We omit the factor of 0.5 above, instead adding it during coefficient
760  # calculation.
761 
762  # We also omit the 1/π from both Q and b equations, as they cancel
763  # during solving.
764 
765  # We have that:
766  # q(n) = 1/π ∫W(ω)cos(nω)dω (over 0->π)
767  # Using our nomalization ω=πf and with a constant weight W over each
768  # interval f1->f2 we get:
769  # q(n) = W∫cos(πnf)df (0->1) = Wf sin(πnf)/πnf
770  # integrated over each f1->f2 pair (i.e., value at f2 - value at f1).
771  n = np.arange(numtaps)[:, np.newaxis, np.newaxis]
772  q = np.dot(np.diff(np.sinc(bands * n) * bands, axis=2)[:, :, 0], weight)
773 
774  # Now we assemble our sum of Toeplitz and Hankel
775  Q1 = toeplitz(q[:M+1])
776  Q2 = hankel(q[:M+1], q[M:])
777  Q = Q1 + Q2
778 
779  # Now for b(n) we have that:
780  # b(n) = 1/π ∫ W(ω)D(ω)cos(nω)dω (over 0->π)
781  # Using our nomalization ω=πf and with a constant weight W over each
782  # interval and a linear term for D(ω) we get (over each f1->f2 interval):
783  # b(n) = W ∫ (mf+c)cos(πnf)df
784  # = f(mf+c)sin(πnf)/πnf + mf**2 cos(nπf)/(πnf)**2
785  # integrated over each f1->f2 pair (i.e., value at f2 - value at f1).
786  n = n[:M + 1] # only need this many coefficients here
787  # Choose m and c such that we are at the start and end weights
788  m = (np.diff(desired, axis=1) / np.diff(bands, axis=1))
789  c = desired[:, [0]] - bands[:, [0]] * m
790  b = bands * (m*bands + c) * np.sinc(bands * n)
791  # Use L'Hospital's rule here for cos(nπf)/(πnf)**2 @ n=0
792  b[0] -= m * bands * bands / 2.
793  b[1:] += m * np.cos(n[1:] * np.pi * bands) / (np.pi * n[1:]) ** 2
794  b = np.dot(np.diff(b, axis=2)[:, :, 0], weight)
795 
796  # Now we can solve the equation (use pinv because Q can be rank deficient)
797  a = np.dot(pinv(Q), b)
798 
799  # make coefficients symmetric (linear phase)
800  coeffs = np.hstack((a[:0:-1], 2 * a[0], a[1:]))
801  return coeffs

804 def _dhtm(mag):
805  """Compute the modified 1D discrete Hilbert transform
806 
807  Parameters
808  ----------
809  mag : ndarray
810  The magnitude spectrum. Should be 1D with an even length, and
811  preferably a fast length for FFT/IFFT.
812  """
813  # Adapted based on code by Niranjan Damera-Venkata,
814  # Brian L. Evans and Shawn R. McCaslin (see refs for `minimum_phase`)
815  sig = np.zeros(len(mag))
816  # Leave Nyquist and DC at 0, knowing np.abs(fftfreq(N)[midpt]) == 0.5
817  midpt = len(mag) // 2
818  sig[1:midpt] = 1
819  sig[midpt+1:] = -1
820  # eventually if we want to support complex filters, we will need a
821  # np.abs() on the mag inside the log, and should remove the .real
822  recon = ifft(mag * np.exp(fft(sig * ifft(np.log(mag))))).real
823  return recon

826 def minimum_phase(h, method='homomorphic', n_fft=None):
827  """Convert a linear-phase FIR filter to minimum phase
828 
829  Parameters
830  ----------
831  h : array
832  Linear-phase FIR filter coefficients.
833  method : {'hilbert', 'homomorphic'}
834  The method to use:
835 
836  'homomorphic' (default)
837  This method [4]_ [5]_ works best with filters with an
838  odd number of taps, and the resulting minimum phase filter
839  will have a magnitude response that approximates the square
840  root of the the original filter's magnitude response.
841 
842  'hilbert'
843  This method [1]_ is designed to be used with equiripple
844  filters (e.g., from `remez`) with unity or zero gain
845  regions.
846 
847  n_fft : int
848  The number of points to use for the FFT. Should be at least a
849  few times larger than the signal length (see Notes).
850 
851  Returns
852  -------
853  h_minimum : array
854  The minimum-phase version of the filter, with length
855  ``(length(h) + 1) // 2``.
856 
857  See Also
858  --------
859  firwin
860  firwin2
861  remez
862 
863  Notes
864  -----
865  Both the Hilbert [1]_ or homomorphic [4]_ [5]_ methods require selection
866  of an FFT length to estimate the complex cepstrum of the filter.
867 
868  In the case of the Hilbert method, the deviation from the ideal
869  spectrum ``epsilon`` is related to the number of stopband zeros
870  ``n_stop`` and FFT length ``n_fft`` as::
871 
872  epsilon = 2. * n_stop / n_fft
873 
874  For example, with 100 stopband zeros and a FFT length of 2048,
875  ``epsilon = 0.0976``. If we conservatively assume that the number of
876  stopband zeros is one less than the filter length, we can take the FFT
877  length to be the next power of 2 that satisfies ``epsilon=0.01`` as::
878 
879  n_fft = 2 ** int(np.ceil(np.log2(2 * (len(h) - 1) / 0.01)))
880 
881  This gives reasonable results for both the Hilbert and homomorphic
882  methods, and gives the value used when ``n_fft=None``.
883 
884  Alternative implementations exist for creating minimum-phase filters,
885  including zero inversion [2]_ and spectral factorization [3]_ [4]_.
886  For more information, see:
887 
888  http://dspguru.com/dsp/howtos/how-to-design-minimum-phase-fir-filters
889 
890  Examples
891  --------
892  Create an optimal linear-phase filter, then convert it to minimum phase:
893 
894  >>> from scipy.signal import remez, minimum_phase, freqz, group_delay
895  >>> import matplotlib.pyplot as plt
896  >>> freq = [0, 0.2, 0.3, 1.0]
897  >>> desired = [1, 0]
898  >>> h_linear = remez(151, freq, desired, Hz=2.)
899 
900  Convert it to minimum phase:
901 
902  >>> h_min_hom = minimum_phase(h_linear, method='homomorphic')
903  >>> h_min_hil = minimum_phase(h_linear, method='hilbert')
904 
905  Compare the three filters:
906 
907  >>> fig, axs = plt.subplots(4, figsize=(4, 8))
908  >>> for h, style, color in zip((h_linear, h_min_hom, h_min_hil),
909  ... ('-', '-', '--'), ('k', 'r', 'c')):
910  ... w, H = freqz(h)
911  ... w, gd = group_delay((h, 1))
912  ... w /= np.pi
913  ... axs[0].plot(h, color=color, linestyle=style)
914  ... axs[1].plot(w, np.abs(H), color=color, linestyle=style)
915  ... axs[2].plot(w, 20 * np.log10(np.abs(H)), color=color, linestyle=style)
916  ... axs[3].plot(w, gd, color=color, linestyle=style)
917  >>> for ax in axs:
918  ... ax.grid(True, color='0.5')
919  ... ax.fill_between(freq[1:3], *ax.get_ylim(), color='#ffeeaa', zorder=1)
920  >>> axs[0].set(xlim=[0, len(h_linear) - 1], ylabel='Amplitude', xlabel='Samples')
921  >>> axs[1].legend(['Linear', 'Min-Hom', 'Min-Hil'], title='Phase')
922  >>> for ax, ylim in zip(axs[1:], ([0, 1.1], [-150, 10], [-60, 60])):
923  ... ax.set(xlim=[0, 1], ylim=ylim, xlabel='Frequency')
924  >>> axs[1].set(ylabel='Magnitude')
925  >>> axs[2].set(ylabel='Magnitude (dB)')
926  >>> axs[3].set(ylabel='Group delay')
927  >>> plt.tight_layout()

945  "Discrete-Time Signal Processing," 2nd edition.
946  Upper Saddle River, N.J.: Prentice Hall, 1999.
947  """ # noqa
948  h = np.asarray(h)
949  if np.iscomplexobj(h):
950  raise ValueError('Complex filters not supported')
951  if h.ndim != 1 or h.size <= 2:
952  raise ValueError('h must be 1D and at least 2 samples long')
953  n_half = len(h) // 2
954  if not np.allclose(h[-n_half:][::-1], h[:n_half]):
955  warnings.warn('h does not appear to by symmetric, conversion may '
956  'fail', RuntimeWarning)
957  if not isinstance(method, string_types) or method not in \
958  ('homomorphic', 'hilbert',):
959  raise ValueError('method must be "homomorphic" or "hilbert", got %r'
960  % (method,))
961  if n_fft is None:
962  n_fft = 2 ** int(np.ceil(np.log2(2 * (len(h) - 1) / 0.01)))
963  n_fft = int(n_fft)
964  if n_fft < len(h):
965  raise ValueError('n_fft must be at least len(h)==%s' % len(h))
966  if method == 'hilbert':
967  w = np.arange(n_fft) * (2 * np.pi / n_fft * n_half)
968  H = np.real(fft(h, n_fft) * np.exp(1j * w))
969  dp = max(H) - 1
970  ds = 0 - min(H)
971  S = 4. / (np.sqrt(1+dp+ds) + np.sqrt(1-dp+ds)) ** 2
972  H += ds
973  H *= S
974  H = np.sqrt(H, out=H)
975  H += 1e-10 # ensure that the log does not explode
976  h_minimum = _dhtm(H)
977  else: # method == 'homomorphic'
978  # zero-pad; calculate the DFT
979  h_temp = np.abs(fft(h, n_fft))
980  # take 0.25*log(|H|**2) = 0.5*log(|H|)
981  h_temp += 1e-7 * h_temp[h_temp > 0].min() # don't let log blow up
982  np.log(h_temp, out=h_temp)
983  h_temp *= 0.5
984  # IDFT
985  h_temp = ifft(h_temp).real
986  # multiply pointwise by the homomorphic filter
987  # lmin[n] = 2u[n] - d[n]
988  win = np.zeros(n_fft)
989  win[0] = 1
990  stop = (len(h) + 1) // 2
991  win[1:stop] = 2
992  if len(h) % 2:
993  win[stop] = 1
994  h_temp *= win
995  h_temp = ifft(np.exp(fft(h_temp)))
996  h_minimum = h_temp.real
997  n_out = n_half + len(h) % 2
998  return h_minimum[:n_out]
