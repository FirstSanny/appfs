44 def pressure_network(flow_rates, Qtot, k):
45  """Evaluate non-linear equation system representing
46  the pressures and flows in a system of n parallel pipes::
47 
48  f_i = P_i - P_0, for i = 1..n
49  f_0 = sum(Q_i) - Qtot
50 
51  Where Q_i is the flow rate in pipe i and P_i the pressure in that pipe.
52  Pressure is modeled as a P=kQ**2 where k is a valve coefficient and
53  Q is the flow rate.
54 
55  Parameters
56  ----------
57  flow_rates : float
58  A 1D array of n flow rates [kg/s].
59  k : float
60  A 1D array of n valve coefficients [1/kg m].
61  Qtot : float
62  A scalar, the total input flow rate [kg/s].
63 
64  Returns
65  -------
66  F : float
67  A 1D array, F[i] == f_i.
68 
69  """
70  P = k * flow_rates**2
71  F = np.hstack((P[1:] - P[0], flow_rates.sum() - Qtot))
72  return F

75 def pressure_network_jacobian(flow_rates, Qtot, k):
76  """Return the jacobian of the equation system F(flow_rates)
77  computed by `pressure_network` with respect to
78  *flow_rates*. See `pressure_network` for the detailed
79  description of parrameters.
80 
81  Returns
82  -------
83  jac : float
84  *n* by *n* matrix ``df_i/dQ_i`` where ``n = len(flow_rates)``
85  and *f_i* and *Q_i* are described in the doc for `pressure_network`
86  """
87  n = len(flow_rates)
88  pdiff = np.diag(flow_rates[1:] * 2 * k[1:] - 2 * flow_rates[0] * k[0])
89 
90  jac = np.empty((n, n))
91  jac[:n-1, :n-1] = pdiff * 0
92  jac[:n-1, n-1] = 0
93  jac[n-1, :] = np.ones(n)
94 
95  return jac

99  return (pressure_network(flow_rates, Qtot, k),
100  pressure_network_jacobian(flow_rates, Qtot, k))
