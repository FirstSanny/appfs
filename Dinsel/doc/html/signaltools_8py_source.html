80 def _inputs_swap_needed(mode, shape1, shape2):
81  """
82  If in 'valid' mode, returns whether or not the input arrays need to be
83  swapped depending on whether `shape1` is at least as large as `shape2` in
84  every dimension.
85 
86  This is important for some of the correlation and convolution
87  implementations in this module, where the larger array input needs to come
88  before the smaller array input when operating in this mode.
89 
90  Note that if the mode provided is not 'valid', False is immediately
91  returned.
92  """
93  if mode == 'valid':
94  ok1, ok2 = True, True
95 
96  for d1, d2 in zip(shape1, shape2):
97  if not d1 >= d2:
98  ok1 = False
99  if not d2 >= d1:
100  ok2 = False
101 
102  if not (ok1 or ok2):
103  raise ValueError("For 'valid' mode, one must be at least "
104  "as large as the other in every dimension")
105 
106  return not ok1
107 
108  return False

111 def correlate(in1, in2, mode='full', method='auto'):
112  r"""
113  Cross-correlate two N-dimensional arrays.
114 
115  Cross-correlate `in1` and `in2`, with the output size determined by the
116  `mode` argument.
117 
118  Parameters
119  ----------
120  in1 : array_like
121  First input.
122  in2 : array_like
123  Second input. Should have the same number of dimensions as `in1`.
124  mode : str {'full', 'valid', 'same'}, optional
125  A string indicating the size of the output:
126 
127  ``full``
128  The output is the full discrete linear cross-correlation
129  of the inputs. (Default)
130  ``valid``
131  The output consists only of those elements that do not
132  rely on the zero-padding. In 'valid' mode, either `in1` or `in2`
133  must be at least as large as the other in every dimension.
134  ``same``
135  The output is the same size as `in1`, centered
136  with respect to the 'full' output.
137  method : str {'auto', 'direct', 'fft'}, optional
138  A string indicating which method to use to calculate the correlation.
139 
140  ``direct``
141  The correlation is determined directly from sums, the definition of
142  correlation.
143  ``fft``
144  The Fast Fourier Transform is used to perform the correlation more
145  quickly (only available for numerical arrays.)
146  ``auto``
147  Automatically chooses direct or Fourier method based on an estimate
148  of which is faster (default). See `convolve` Notes for more detail.
149 
150  .. versionadded:: 0.19.0
151 
152  Returns
153  -------
154  correlate : array
155  An N-dimensional array containing a subset of the discrete linear
156  cross-correlation of `in1` with `in2`.
157 
158  See Also
159  --------
160  choose_conv_method : contains more documentation on `method`.
161 
162  Notes
163  -----
164  The correlation z of two d-dimensional arrays x and y is defined as::
165 
166  z[...,k,...] = sum[..., i_l, ...] x[..., i_l,...] * conj(y[..., i_l - k,...])
167 
168  This way, if x and y are 1-D arrays and ``z = correlate(x, y, 'full')`` then
169 
170  .. math::
171 
172  z[k] = (x * y)(k - N + 1)
173  = \sum_{l=0}^{||x||-1}x_l y_{l-k+N-1}^{*}
174 
175  for :math:`k = 0, 1, ..., ||x|| + ||y|| - 2`
176 
177  where :math:`||x||` is the length of ``x``, :math:`N = \max(||x||,||y||)`,
178  and :math:`y_m` is 0 when m is outside the range of y.
179 
180  ``method='fft'`` only works for numerical arrays as it relies on
181  `fftconvolve`. In certain cases (i.e., arrays of objects or when
182  rounding integers can lose precision), ``method='direct'`` is always used.
183 
184  Examples
185  --------
186  Implement a matched filter using cross-correlation, to recover a signal
187  that has passed through a noisy channel.
188 
189  >>> from scipy import signal
190  >>> sig = np.repeat([0., 1., 1., 0., 1., 0., 0., 1.], 128)
191  >>> sig_noise = sig + np.random.randn(len(sig))
192  >>> corr = signal.correlate(sig_noise, np.ones(128), mode='same') / 128
193 
194  >>> import matplotlib.pyplot as plt
195  >>> clock = np.arange(64, len(sig), 128)
196  >>> fig, (ax_orig, ax_noise, ax_corr) = plt.subplots(3, 1, sharex=True)
197  >>> ax_orig.plot(sig)
198  >>> ax_orig.plot(clock, sig[clock], 'ro')
199  >>> ax_orig.set_title('Original signal')
200  >>> ax_noise.plot(sig_noise)
201  >>> ax_noise.set_title('Signal with noise')
202  >>> ax_corr.plot(corr)
203  >>> ax_corr.plot(clock, corr[clock], 'ro')
204  >>> ax_corr.axhline(0.5, ls=':')
205  >>> ax_corr.set_title('Cross-correlated with rectangular pulse')
206  >>> ax_orig.margins(0, 0.1)
207  >>> fig.tight_layout()
208  >>> fig.show()
209 
210  """
211  in1 = asarray(in1)
212  in2 = asarray(in2)
213 
214  if in1.ndim == in2.ndim == 0:
215  return in1 * in2
216  elif in1.ndim != in2.ndim:
217  raise ValueError("in1 and in2 should have the same dimensionality")
218 
219  # Don't use _valfrommode, since correlate should not accept numeric modes
220  try:
221  val = _modedict[mode]
222  except KeyError:
223  raise ValueError("Acceptable mode flags are 'valid',"
224  " 'same', or 'full'.")
225 
226  # this either calls fftconvolve or this function with method=='direct'
227  if method in ('fft', 'auto'):
228  return convolve(in1, _reverse_and_conj(in2), mode, method)
229 
230  # fastpath to faster numpy.correlate for 1d inputs when possible
231  if _np_conv_ok(in1, in2, mode):
232  return np.correlate(in1, in2, mode)
233 
234  # _correlateND is far slower when in2.size > in1.size, so swap them
235  # and then undo the effect afterward if mode == 'full'. Also, it fails
236  # with 'valid' mode if in2 is larger than in1, so swap those, too.
237  # Don't swap inputs for 'same' mode, since shape of in1 matters.
238  swapped_inputs = ((mode == 'full') and (in2.size > in1.size) or
239  _inputs_swap_needed(mode, in1.shape, in2.shape))
240 
241  if swapped_inputs:
242  in1, in2 = in2, in1
243 
244  if mode == 'valid':
245  ps = [i - j + 1 for i, j in zip(in1.shape, in2.shape)]
246  out = np.empty(ps, in1.dtype)
247 
248  z = sigtools._correlateND(in1, in2, out, val)
249 
250  else:
251  ps = [i + j - 1 for i, j in zip(in1.shape, in2.shape)]
252 
253  # zero pad input
254  in1zpadded = np.zeros(ps, in1.dtype)
255  sc = [slice(0, i) for i in in1.shape]
256  in1zpadded[sc] = in1.copy()
257 
258  if mode == 'full':
259  out = np.empty(ps, in1.dtype)
260  elif mode == 'same':
261  out = np.empty(in1.shape, in1.dtype)
262 
263  z = sigtools._correlateND(in1zpadded, in2, out, val)
264 
265  if swapped_inputs:
266  # Reverse and conjugate to undo the effect of swapping inputs
267  z = _reverse_and_conj(z)
268 
269  return z

282 def fftconvolve(in1, in2, mode="full"):
283  """Convolve two N-dimensional arrays using FFT.
284 
285  Convolve `in1` and `in2` using the fast Fourier transform method, with
286  the output size determined by the `mode` argument.
287 
288  This is generally much faster than `convolve` for large arrays (n > ~500),
289  but can be slower when only a few output values are needed, and can only
290  output float arrays (int or object array inputs will be cast to float).
291 
292  As of v0.19, `convolve` automatically chooses this method or the direct
293  method based on an estimation of which is faster.
294 
295  Parameters
296  ----------
297  in1 : array_like
298  First input.
299  in2 : array_like
300  Second input. Should have the same number of dimensions as `in1`.
301  If operating in 'valid' mode, either `in1` or `in2` must be
302  at least as large as the other in every dimension.
303  mode : str {'full', 'valid', 'same'}, optional
304  A string indicating the size of the output:
305 
306  ``full``
307  The output is the full discrete linear convolution
308  of the inputs. (Default)
309  ``valid``
310  The output consists only of those elements that do not
311  rely on the zero-padding.
312  ``same``
313  The output is the same size as `in1`, centered
314  with respect to the 'full' output.
315 
316  Returns
317  -------
318  out : array
319  An N-dimensional array containing a subset of the discrete linear
320  convolution of `in1` with `in2`.
321 
322  Examples
323  --------
324  Autocorrelation of white noise is an impulse.
325 
326  >>> from scipy import signal
327  >>> sig = np.random.randn(1000)
328  >>> autocorr = signal.fftconvolve(sig, sig[::-1], mode='full')
329 
330  >>> import matplotlib.pyplot as plt
331  >>> fig, (ax_orig, ax_mag) = plt.subplots(2, 1)
332  >>> ax_orig.plot(sig)
333  >>> ax_orig.set_title('White noise')
334  >>> ax_mag.plot(np.arange(-len(sig)+1,len(sig)), autocorr)
335  >>> ax_mag.set_title('Autocorrelation')
336  >>> fig.tight_layout()
337  >>> fig.show()
338 
339  Gaussian blur implemented using FFT convolution. Notice the dark borders
340  around the image, due to the zero-padding beyond its boundaries.
341  The `convolve2d` function allows for other types of image boundaries,
342  but is far slower.
343 
344  >>> from scipy import misc
345  >>> face = misc.face(gray=True)
346  >>> kernel = np.outer(signal.gaussian(70, 8), signal.gaussian(70, 8))
347  >>> blurred = signal.fftconvolve(face, kernel, mode='same')
348 
349  >>> fig, (ax_orig, ax_kernel, ax_blurred) = plt.subplots(3, 1,
350  ... figsize=(6, 15))
351  >>> ax_orig.imshow(face, cmap='gray')
352  >>> ax_orig.set_title('Original')
353  >>> ax_orig.set_axis_off()
354  >>> ax_kernel.imshow(kernel, cmap='gray')
355  >>> ax_kernel.set_title('Gaussian kernel')
356  >>> ax_kernel.set_axis_off()
357  >>> ax_blurred.imshow(blurred, cmap='gray')
358  >>> ax_blurred.set_title('Blurred')
359  >>> ax_blurred.set_axis_off()
360  >>> fig.show()
361 
362  """
363  in1 = asarray(in1)
364  in2 = asarray(in2)
365 
366  if in1.ndim == in2.ndim == 0: # scalar inputs
367  return in1 * in2
368  elif not in1.ndim == in2.ndim:
369  raise ValueError("in1 and in2 should have the same dimensionality")
370  elif in1.size == 0 or in2.size == 0: # empty arrays
371  return array([])
372 
373  s1 = array(in1.shape)
374  s2 = array(in2.shape)
375  complex_result = (np.issubdtype(in1.dtype, complex) or
376  np.issubdtype(in2.dtype, complex))
377  shape = s1 + s2 - 1
378 
379  # Check that input sizes are compatible with 'valid' mode
380  if _inputs_swap_needed(mode, s1, s2):
381  # Convolution is commutative; order doesn't have any effect on output
382  in1, s1, in2, s2 = in2, s2, in1, s1
383 
384  # Speed up FFT by padding to optimal size for FFTPACK
385  fshape = [fftpack.helper.next_fast_len(int(d)) for d in shape]
386  fslice = tuple([slice(0, int(sz)) for sz in shape])
387  # Pre-1.9 NumPy FFT routines are not threadsafe. For older NumPys, make
388  # sure we only call rfftn/irfftn from one thread at a time.
389  if not complex_result and (_rfft_mt_safe or _rfft_lock.acquire(False)):
390  try:
391  sp1 = np.fft.rfftn(in1, fshape)
392  sp2 = np.fft.rfftn(in2, fshape)
393  ret = (np.fft.irfftn(sp1 * sp2, fshape)[fslice].copy())
394  finally:
395  if not _rfft_mt_safe:
396  _rfft_lock.release()
397  else:
398  # If we're here, it's either because we need a complex result, or we
399  # failed to acquire _rfft_lock (meaning rfftn isn't threadsafe and
400  # is already in use by another thread). In either case, use the
401  # (threadsafe but slower) SciPy complex-FFT routines instead.
402  sp1 = fftpack.fftn(in1, fshape)
403  sp2 = fftpack.fftn(in2, fshape)
404  ret = fftpack.ifftn(sp1 * sp2)[fslice].copy()
405  if not complex_result:
406  ret = ret.real
407 
408  if mode == "full":
409  return ret
410  elif mode == "same":
411  return _centered(ret, s1)
412  elif mode == "valid":
413  return _centered(ret, s1 - s2 + 1)
414  else:
415  raise ValueError("Acceptable mode flags are 'valid',"
416  " 'same', or 'full'.")

451 def _fftconv_faster(x, h, mode):
452  """
453  See if using `fftconvolve` or `_correlateND` is faster. The boolean value
454  returned depends on the sizes and shapes of the input values.
455 
456  The big O ratios were found to hold across different machines, which makes
457  sense as it's the ratio that matters (the effective speed of the computer
458  is found in both big O constants). Regardless, this had been tuned on an
459  early 2015 MacBook Pro with 8GB RAM and an Intel i5 processor.
460  """
461  if mode == 'full':
462  out_shape = [n + k - 1 for n, k in zip(x.shape, h.shape)]
463  big_O_constant = 10963.92823819 if x.ndim == 1 else 8899.1104874
464  elif mode == 'same':
465  out_shape = x.shape
466  if x.ndim == 1:
467  if h.size <= x.size:
468  big_O_constant = 7183.41306773
469  else:
470  big_O_constant = 856.78174111
471  else:
472  big_O_constant = 34519.21021589
473  elif mode == 'valid':
474  out_shape = [n - k + 1 for n, k in zip(x.shape, h.shape)]
475  big_O_constant = 41954.28006344 if x.ndim == 1 else 66453.24316434
476  else:
477  raise ValueError('mode is invalid')
478 
479  # see whether the Fourier transform convolution method or the direct
480  # convolution method is faster (discussed in scikit-image PR #1792)
481  direct_time = (x.size * h.size * _prod(out_shape))
482  fft_time = sum(n * math.log(n) for n in (x.shape + h.shape +
483  tuple(out_shape)))
484  return big_O_constant * fft_time < direct_time

505 def _fftconvolve_valid(volume, kernel):
506  # fftconvolve doesn't support complex256
507  for not_fft_conv_supp in ["complex256", "complex192"]:
508  if hasattr(np, not_fft_conv_supp):

519  return False
520 
521  if _numeric_arrays([volume, kernel]):
522  return False

524  return True

527 def _timeit_fast(stmt="pass", setup="pass", repeat=3):
528  """
529  Returns the time the statement/function took, in seconds.
530 
531  Faster, less precise version of IPython's timeit. `stmt` can be a statement
532  written as a string or a callable.
533 
534  Will do only 1 loop (like IPython's timeit) with no repetitions
535  (unlike IPython) for very slow functions. For fast functions, only does
536  enough loops to take 5 ms, which seems to produce similar results (on
537  Windows at least), and avoids doing an extraneous cycle that isn't
538  measured.
539 
540  """
541  timer = timeit.Timer(stmt, setup)
542 
543  # determine number of calls per rep so total time for 1 rep >= 5 ms
544  x = 0
545  for p in range(0, 10):
546  number = 10**p
547  x = timer.timeit(number) # seconds
548  if x >= 5e-3 / 10: # 5 ms for final test, 1/10th that for this one
549  break
550  if x > 1: # second
551  # If it's macroscopic, don't bother with repetitions
552  best = x
553  else:
554  number *= 10
555  r = timer.repeat(repeat, number)
556  best = min(r)
557 
558  sec = best / number
559  return sec

562 def choose_conv_method(in1, in2, mode='full', measure=False):
563  """
564  Find the fastest convolution/correlation method.
565 
566  This primarily exists to be called during the ``method='auto'`` option in
567  `convolve` and `correlate`, but can also be used when performing many
568  convolutions of the same input shapes and dtypes, determining
569  which method to use for all of them, either to avoid the overhead of the
570  'auto' option or to use accurate real-world measurements.
571 
572  Parameters
573  ----------
574  in1 : array_like
575  The first argument passed into the convolution function.
576  in2 : array_like
577  The second argument passed into the convolution function.
578  mode : str {'full', 'valid', 'same'}, optional
579  A string indicating the size of the output:
580 
581  ``full``
582  The output is the full discrete linear convolution
583  of the inputs. (Default)
584  ``valid``
585  The output consists only of those elements that do not
586  rely on the zero-padding.
587  ``same``
588  The output is the same size as `in1`, centered
589  with respect to the 'full' output.
590  measure : bool, optional
591  If True, run and time the convolution of `in1` and `in2` with both
592  methods and return the fastest. If False (default), predict the fastest
593  method using precomputed values.
594 
595  Returns
596  -------
597  method : str
598  A string indicating which convolution method is fastest, either
599  'direct' or 'fft'
600  times : dict, optional
601  A dictionary containing the times (in seconds) needed for each method.
602  This value is only returned if ``measure=True``.
603 
604  See Also
605  --------
606  convolve
607  correlate
608 
609  Notes
610  -----
611  For large n, ``measure=False`` is accurate and can quickly determine the
612  fastest method to perform the convolution. However, this is not as
613  accurate for small n (when any dimension in the input or output is small).
614 
615  In practice, we found that this function estimates the faster method up to
616  a multiplicative factor of 5 (i.e., the estimated method is *at most* 5
617  times slower than the fastest method). The estimation values were tuned on
618  an early 2015 MacBook Pro with 8GB RAM but we found that the prediction
619  held *fairly* accurately across different machines.
620 
621  If ``measure=True``, time the convolutions. Because this function uses
622  `fftconvolve`, an error will be thrown if it does not support the inputs.
623  There are cases when `fftconvolve` supports the inputs but this function
624  returns `direct` (e.g., to protect against floating point integer
625  precision).

652  kernel = asarray(in2)

655  times = {}

657  times[method] = _timeit_fast(lambda: convolve(volume, kernel,

672  if any([_numeric_arrays([x], kinds='ui') for x in [volume, kernel]]):

674  max_value *= int(min(volume.size, kernel.size))
675  if max_value > 2**np.finfo('float').nmant - 1:

679  if _fftconv_faster(volume, kernel, mode):

685 def convolve(in1, in2, mode='full', method='auto'):
686  """
687  Convolve two N-dimensional arrays.
688 
689  Convolve `in1` and `in2`, with the output size determined by the
690  `mode` argument.
691 
692  Parameters
693  ----------
694  in1 : array_like
695  First input.
696  in2 : array_like
697  Second input. Should have the same number of dimensions as `in1`.
698  mode : str {'full', 'valid', 'same'}, optional
699  A string indicating the size of the output:
700 
701  ``full``
702  The output is the full discrete linear convolution
703  of the inputs. (Default)
704  ``valid``
705  The output consists only of those elements that do not
706  rely on the zero-padding. In 'valid' mode, either `in1` or `in2`
707  must be at least as large as the other in every dimension.
708  ``same``
709  The output is the same size as `in1`, centered
710  with respect to the 'full' output.
711  method : str {'auto', 'direct', 'fft'}, optional
712  A string indicating which method to use to calculate the convolution.
713 
714  ``direct``
715  The convolution is determined directly from sums, the definition of
716  convolution.
717  ``fft``
718  The Fourier Transform is used to perform the convolution by calling
719  `fftconvolve`.
720  ``auto``
721  Automatically chooses direct or Fourier method based on an estimate
722  of which is faster (default). See Notes for more detail.
723 
724  .. versionadded:: 0.19.0
725 
726  Returns
727  -------
728  convolve : array
729  An N-dimensional array containing a subset of the discrete linear
730  convolution of `in1` with `in2`.
731 
732  See Also
733  --------
734  numpy.polymul : performs polynomial multiplication (same operation, but
735  also accepts poly1d objects)
736  choose_conv_method : chooses the fastest appropriate convolution method
737  fftconvolve
738 
739  Notes
740  -----
741  By default, `convolve` and `correlate` use ``method='auto'``, which calls
742  `choose_conv_method` to choose the fastest method using pre-computed
743  values (`choose_conv_method` can also measure real-world timing with a
744  keyword argument). Because `fftconvolve` relies on floating point numbers,
745  there are certain constraints that may force `method=direct` (more detail
746  in `choose_conv_method` docstring).
747 
748  Examples
749  --------
750  Smooth a square pulse using a Hann window:
751 
752  >>> from scipy import signal
753  >>> sig = np.repeat([0., 1., 0.], 100)
754  >>> win = signal.hann(50)
755  >>> filtered = signal.convolve(sig, win, mode='same') / sum(win)
756 
757  >>> import matplotlib.pyplot as plt
758  >>> fig, (ax_orig, ax_win, ax_filt) = plt.subplots(3, 1, sharex=True)
759  >>> ax_orig.plot(sig)
760  >>> ax_orig.set_title('Original pulse')
761  >>> ax_orig.margins(0, 0.1)
762  >>> ax_win.plot(win)
763  >>> ax_win.set_title('Filter impulse response')
764  >>> ax_win.margins(0, 0.1)
765  >>> ax_filt.plot(filtered)
766  >>> ax_filt.set_title('Filtered signal')
767  >>> ax_filt.margins(0, 0.1)
768  >>> fig.tight_layout()
769  >>> fig.show()
770 
771  """
772  volume = asarray(in1)
773  kernel = asarray(in2)
774 
775  if volume.ndim == kernel.ndim == 0:
776  return volume * kernel
777 
778  if _inputs_swap_needed(mode, volume.shape, kernel.shape):
779  # Convolution is commutative; order doesn't have any effect on output
780  volume, kernel = kernel, volume
781 
782  if method == 'auto':
783  method = choose_conv_method(volume, kernel, mode=mode)
784 
785  if method == 'fft':
786  out = fftconvolve(volume, kernel, mode=mode)
787  if volume.dtype.kind in 'ui':
788  out = np.around(out)
789  return out.astype(volume.dtype)
790 
791  # fastpath to faster numpy.convolve for 1d inputs when possible
792  if _np_conv_ok(volume, kernel, mode):
793  return np.convolve(volume, kernel, mode)
794 
795  return correlate(volume, _reverse_and_conj(kernel), mode, 'direct')

902 def wiener(im, mysize=None, noise=None):
903  """
904  Perform a Wiener filter on an N-dimensional array.
905 
906  Apply a Wiener filter to the N-dimensional array `im`.
907 
908  Parameters
909  ----------
910  im : ndarray
911  An N-dimensional array.
912  mysize : int or array_like, optional
913  A scalar or an N-length list giving the size of the Wiener filter
914  window in each dimension. Elements of mysize should be odd.
915  If mysize is a scalar, then this scalar is used as the size
916  in each dimension.
917  noise : float, optional
918  The noise-power to use. If None, then noise is estimated as the
919  average of the local variance of the input.
920 
921  Returns
922  -------
923  out : ndarray
924  Wiener filtered result with the same shape as `im`.
925 
926  """
927  im = asarray(im)
928  if mysize is None:
929  mysize = [3] * im.ndim
930  mysize = asarray(mysize)
931  if mysize.shape == ():
932  mysize = np.repeat(mysize.item(), im.ndim)
933 
934  # Estimate the local mean
935  lMean = correlate(im, ones(mysize), 'same') / product(mysize, axis=0)
936 
937  # Estimate the local variance
938  lVar = (correlate(im ** 2, ones(mysize), 'same') /
939  product(mysize, axis=0) - lMean ** 2)
940 
941  # Estimate the noise power if needed.
942  if noise is None:
943  noise = mean(ravel(lVar), axis=0)
944 
945  res = (im - lMean)
946  res *= (1 - noise / lVar)
947  res += lMean
948  out = where(lVar < noise, lMean, res)
949 
950  return out

1189 def lfilter(b, a, x, axis=-1, zi=None):
1190  """
1191  Filter data along one-dimension with an IIR or FIR filter.
1192 
1193  Filter a data sequence, `x`, using a digital filter. This works for many
1194  fundamental data types (including Object type). The filter is a direct
1195  form II transposed implementation of the standard difference equation
1196  (see Notes).
1197 
1198  Parameters
1199  ----------
1200  b : array_like
1201  The numerator coefficient vector in a 1-D sequence.
1202  a : array_like
1203  The denominator coefficient vector in a 1-D sequence. If ``a[0]``
1204  is not 1, then both `a` and `b` are normalized by ``a[0]``.
1205  x : array_like
1206  An N-dimensional input array.
1207  axis : int, optional
1208  The axis of the input data array along which to apply the
1209  linear filter. The filter is applied to each subarray along
1210  this axis. Default is -1.
1211  zi : array_like, optional
1212  Initial conditions for the filter delays. It is a vector
1213  (or array of vectors for an N-dimensional input) of length
1214  ``max(len(a), len(b)) - 1``. If `zi` is None or is not given then
1215  initial rest is assumed. See `lfiltic` for more information.
1216 
1217  Returns
1218  -------
1219  y : array
1220  The output of the digital filter.
1221  zf : array, optional
1222  If `zi` is None, this is not returned, otherwise, `zf` holds the
1223  final filter delay values.
1224 
1225  See Also
1226  --------
1227  lfiltic : Construct initial conditions for `lfilter`.
1228  lfilter_zi : Compute initial state (steady state of step response) for
1229  `lfilter`.
1230  filtfilt : A forward-backward filter, to obtain a filter with linear phase.
1231  savgol_filter : A Savitzky-Golay filter.
1232  sosfilt: Filter data using cascaded second-order sections.
1233  sosfiltfilt: A forward-backward filter using second-order sections.
1234 
1235  Notes
1236  -----
1237  The filter function is implemented as a direct II transposed structure.
1238  This means that the filter implements::
1239 
1240  a[0]*y[n] = b[0]*x[n] + b[1]*x[n-1] + ... + b[M]*x[n-M]
1241  - a[1]*y[n-1] - ... - a[N]*y[n-N]
1242 
1243  where `M` is the degree of the numerator, `N` is the degree of the
1244  denominator, and `n` is the sample number. It is implemented using
1245  the following difference equations (assuming M = N)::
1246 
1247  a[0]*y[n] = b[0] * x[n] + d[0][n-1]
1248  d[0][n] = b[1] * x[n] - a[1] * y[n] + d[1][n-1]
1249  d[1][n] = b[2] * x[n] - a[2] * y[n] + d[2][n-1]
1250  ...
1251  d[N-2][n] = b[N-1]*x[n] - a[N-1]*y[n] + d[N-1][n-1]
1252  d[N-1][n] = b[N] * x[n] - a[N] * y[n]
1253 
1254  where `d` are the state variables.
1255 
1256  The rational transfer function describing this filter in the
1257  z-transform domain is::
1258 
1259  -1 -M
1260  b[0] + b[1]z + ... + b[M] z
1261  Y(z) = -------------------------------- X(z)
1262  -1 -N
1263  a[0] + a[1]z + ... + a[N] z
1264 
1265  Examples
1266  --------
1267  Generate a noisy signal to be filtered:
1268 
1269  >>> from scipy import signal

1502 def hilbert(x, N=None, axis=-1):

1505

1507 
1508  Parameters
1509  ----------
1510  x : array_like
1511  Signal data. Must be real.

1527  -----
1528  The analytic signal ``x_a(t)`` of signal ``x(t)`` is:
1529 
1530  .. math:: x_a = F^{-1}(F(x) 2U) = x + i y
1531 
1532  where `F` is the Fourier transform, `U` the unit step function,
1533  and `y` the Hilbert transform of `x`. [1]_
1534

1536  out, turning the real-valued signal into a complex signal. The Hilbert
1537  transformed signal can be obtained from ``np.imag(hilbert(x))``, and the
1538  original signal from ``np.real(hilbert(x))``.
1539

1541  ---------
1542  In this example we use the Hilbert transform to determine the amplitude
1543  envelope and instantaneous frequency of an amplitude-modulated signal.
1544 
1545  >>> import numpy as np
1546  >>> import matplotlib.pyplot as plt
1547  >>> from scipy.signal import hilbert, chirp
1548 
1549  >>> duration = 1.0
1550  >>> fs = 400.0
1551  >>> samples = int(fs*duration)
1552  >>> t = np.arange(samples) / fs

1555  apply an amplitude modulation.
1556

1558  >>> signal *= (1.0 + 0.5 * np.sin(2.0*np.pi*3.0*t) )

1561  instantaneous frequency can be obtained by differentiating the
1562  instantaneous phase in respect to time. The instantaneous phase corresponds
1563  to the phase angle of the analytic signal.
1564

1566  >>> amplitude_envelope = np.abs(analytic_signal)
1567  >>> instantaneous_phase = np.unwrap(np.angle(analytic_signal))

1569  ... (2.0*np.pi) * fs)
1570

1572  >>> ax0 = fig.add_subplot(211)
1573  >>> ax0.plot(t, signal, label='signal')
1574  >>> ax0.plot(t, amplitude_envelope, label='envelope')
1575  >>> ax0.set_xlabel("time in seconds")
1576  >>> ax0.legend()
1577  >>> ax1 = fig.add_subplot(212)
1578  >>> ax1.plot(t[1:], instantaneous_frequency)
1579  >>> ax1.set_xlabel("time in seconds")
1580  >>> ax1.set_ylim(0.0, 120.0)
1581 
1582  References
1583  ----------
1584  .. [1] Wikipedia, "Analytic signal".
1585  http://en.wikipedia.org/wiki/Analytic_signal
1586  .. [2] Leon Cohen, "Time-Frequency Analysis", 1995. Chapter 2.
1587  .. [3] Alan V. Oppenheim, Ronald W. Schafer. Discrete-Time Signal
1588  Processing, Third Edition, 2009. Chapter 12.
1589  ISBN 13: 978-1292-02572-8
1590 
1591  """

1593  if iscomplexobj(x):
1594  raise ValueError("x must be real.")
1595  if N is None:
1596  N = x.shape[axis]
1597  if N <= 0:
1598  raise ValueError("N must be positive.")

2298  is best to pass a symmetric filter with an odd number of samples if, as
2299  is usually the case, a zero-phase filter is desired.

2302  and `scipy.signal.firwin` are called to generate the appropriate filter

2306  sample of the input vector. The spacing between samples is changed

2312  sample of the next cycle for the FFT method, and gets closer to zero
2313  for the polyphase method:

2542  to the steady state of the step response.

2545  output of the filter starts at the same value as the first element of
2546  the signal to be filtered.
2547

2549  ----------

2552  information.

2558

2561  lfilter, lfiltic, filtfilt
2562

2564  -----
2565  A linear filter with order m has a state space representation (A, B, C, D),
2566  for which the output y of the filter can be expressed as::
2567 
2568  z(n+1) = A*z(n) + B*x(n)
2569  y(n) = C*z(n) + D*x(n)
2570 
2571  where z(n) is a vector of length m, A has shape (m, m), B has shape
2572  (m, 1), C has shape (1, m) and D has shape (1, 1) (assuming x(n) is
2573  a scalar). lfilter_zi solves::
2574 
2575  zi = A*zi + B
2576 
2577  In other words, it finds the initial condition for which the response
2578  to an input of all ones is a constant.
2579 
2580  Given the filter coefficients `a` and `b`, the state space matrices
2581  for the transposed direct form II implementation of the linear filter,
2582  which is the implementation used by scipy.signal.lfilter, are::
2583 
2584  A = scipy.linalg.companion(a).T
2585  B = b[1:] - a[1:]*b[0]
2586 
2587  assuming `a[0]` is 1.0; if `a[0]` is not 1, `a` and `b` are first
2588  divided by a[0].

2591  --------

2593  applies that filter to an array whose values are all 1.0; the
2594  output is also all 1.0, as expected for a lowpass filter. If the
2595  `zi` argument of `lfilter` had not been given, the output would have
2596  shown the transient signal.

2603  >>> y

2612  0.44399389, 0.35505241])

2615  `lfilter_zi` and scaled by `x[0]`. Then the output `y` has no
2616  transient until the input drops from 0.5 to 0.0.

2632  if a.ndim != 1:

2743 def _filtfilt_gust(b, a, x, axis=-1, irlen=None):
2744  """Forward-backward IIR filter that uses Gustafsson's method.
2745 
2746  Apply the IIR filter defined by `(b,a)` to `x` twice, first forward
2747  then backward, using Gustafsson's initial conditions [1]_.
2748 
2749  Let ``y_fb`` be the result of filtering first forward and then backward,
2750  and let ``y_bf`` be the result of filtering first backward then forward.
2751  Gustafsson's method is to compute initial conditions for the forward
2752  pass and the backward pass such that ``y_fb == y_bf``.
2753 
2754  Parameters
2755  ----------
2756  b : scalar or 1-D ndarray
2757  Numerator coefficients of the filter.
2758  a : scalar or 1-D ndarray
2759  Denominator coefficients of the filter.
2760  x : ndarray
2761  Data to be filtered.
2762  axis : int, optional
2763  Axis of `x` to be filtered. Default is -1.
2764  irlen : int or None, optional
2765  The length of the nonnegligible part of the impulse response.
2766  If `irlen` is None, or if the length of the signal is less than
2767  ``2 * irlen``, then no part of the impulse response is ignored.
2768 
2769  Returns
2770  -------
2771  y : ndarray
2772  The filtered data.
2773  x0 : ndarray
2774  Initial condition for the forward filter.
2775  x1 : ndarray
2776  Initial condition for the backward filter.
2777 
2778  Notes
2779  -----
2780  Typically the return values `x0` and `x1` are not needed by the
2781  caller. The intended use of these return values is in unit tests.
2782 
2783  References
2784  ----------
2785  .. [1] F. Gustaffson. Determining the initial states in forward-backward
2786  filtering. Transactions on Signal Processing, 46(4):988-992, 1996.
2787 
2788  """
2789  # In the comments, "Gustafsson's paper" and [1] refer to the
2790  # paper referenced in the docstring.
2791 
2792  b = np.atleast_1d(b)
2793  a = np.atleast_1d(a)
2794 
2795  order = max(len(b), len(a)) - 1
2796  if order == 0:
2797  # The filter is just scalar multiplication, with no state.
2798  scale = (b[0] / a[0])**2
2799  y = scale * x
2800  return y, np.array([]), np.array([])
2801 
2802  if axis != -1 or axis != x.ndim - 1:
2803  # Move the axis containing the data to the end.
2804  x = np.swapaxes(x, axis, x.ndim - 1)
2805 
2806  # n is the number of samples in the data to be filtered.
2807  n = x.shape[-1]
2808 
2809  if irlen is None or n <= 2*irlen:
2810  m = n
2811  else:
2812  m = irlen
2813 
2814  # Create Obs, the observability matrix (called O in the paper).
2815  # This matrix can be interpreted as the operator that propagates
2816  # an arbitrary initial state to the output, assuming the input is
2817  # zero.
2818  # In Gustafsson's paper, the forward and backward filters are not
2819  # necessarily the same, so he has both O_f and O_b. We use the same
2820  # filter in both directions, so we only need O. The same comment
2821  # applies to S below.
2822  Obs = np.zeros((m, order))
2823  zi = np.zeros(order)
2824  zi[0] = 1
2825  Obs[:, 0] = lfilter(b, a, np.zeros(m), zi=zi)[0]
2826  for k in range(1, order):
2827  Obs[k:, k] = Obs[:-k, 0]
2828 
2829  # Obsr is O^R (Gustafsson's notation for row-reversed O)
2830  Obsr = Obs[::-1]
2831 
2832  # Create S. S is the matrix that applies the filter to the reversed
2833  # propagated initial conditions. That is,
2834  # out = S.dot(zi)
2835  # is the same as
2836  # tmp, _ = lfilter(b, a, zeros(), zi=zi) # Propagate ICs.
2837  # out = lfilter(b, a, tmp[::-1]) # Reverse and filter.
2838 
2839  # Equations (5) & (6) of [1]
2840  S = lfilter(b, a, Obs[::-1], axis=0)
2841 
2842  # Sr is S^R (row-reversed S)
2843  Sr = S[::-1]
2844 
2845  # M is [(S^R - O), (O^R - S)]
2846  if m == n:
2847  M = np.hstack((Sr - Obs, Obsr - S))
2848  else:
2849  # Matrix described in section IV of [1].
2850  M = np.zeros((2*m, 2*order))
2851  M[:m, :order] = Sr - Obs
2852  M[m:, order:] = Obsr - S
2853 
2854  # Naive forward-backward and backward-forward filters.
2855  # These have large transients because the filters use zero initial
2856  # conditions.
2857  y_f = lfilter(b, a, x)
2858  y_fb = lfilter(b, a, y_f[..., ::-1])[..., ::-1]
2859 
2860  y_b = lfilter(b, a, x[..., ::-1])[..., ::-1]
2861  y_bf = lfilter(b, a, y_b)
2862 
2863  delta_y_bf_fb = y_bf - y_fb
2864  if m == n:
2865  delta = delta_y_bf_fb
2866  else:
2867  start_m = delta_y_bf_fb[..., :m]
2868  end_m = delta_y_bf_fb[..., -m:]
2869  delta = np.concatenate((start_m, end_m), axis=-1)
2870

2872  # The following code computes the result shown in the formula
2873  # of the paper between equations (6) and (7).
2874  if delta.ndim == 1:
2875  ic_opt = linalg.lstsq(M, delta)[0]
2876  else:

2878  # right-hand-sides in linalg.lstsq.
2879  delta2d = delta.reshape(-1, delta.shape[-1]).T
2880  ic_opt0 = linalg.lstsq(M, delta2d)[0].T
2881  ic_opt = ic_opt0.reshape(delta.shape[:-1] + (M.shape[-1],))
2882 
2883  # Now compute the filtered signal using equation (7) of [1].
2884  # First, form [S^R, O^R] and call it W.
2885  if m == n:
2886  W = np.hstack((Sr, Obsr))
2887  else:
2888  W = np.zeros((2*m, 2*order))
2889  W[:m, :order] = Sr
2890  W[m:, order:] = Obsr
2891

2893  # Y_fb^opt = Y_fb^0 + W * [x_0^opt; x_{N-1}^opt]
2894  # `wic` is (almost) the product on the right.
2895  # W has shape (m, 2*order), and ic_opt has shape (..., 2*order),
2896  # so we can't use W.dot(ic_opt). Instead, we dot ic_opt with W.T,
2897  # so wic has shape (..., m).
2898  wic = ic_opt.dot(W.T)
2899 
2900  # `wic` is "almost" the product of W and the optimal ICs in equation
2901  # (7)--if we're using a truncated impulse response (m < n), `wic`
2902  # contains only the adjustments required for the ends of the signal.
2903  # Here we form y_opt, taking this into account if necessary.
2904  y_opt = y_fb
2905  if m == n:
2906  y_opt += wic
2907  else:
2908  y_opt[..., :m] += wic[..., :m]
2909  y_opt[..., -m:] += wic[..., -m:]
2910 
2911  x0 = ic_opt[..., :order]
2912  x1 = ic_opt[..., -order:]
2913  if axis != -1 or axis != x.ndim - 1:
2914  # Restore the data axis to its original position.
2915  x0 = np.swapaxes(x0, axis, x.ndim - 1)
2916  x1 = np.swapaxes(x1, axis, x.ndim - 1)
2917  y_opt = np.swapaxes(y_opt, axis, x.ndim - 1)
2918 
2919  return y_opt, x0, x1

2922 def filtfilt(b, a, x, axis=-1, padtype='odd', padlen=None, method='pad',
2923  irlen=None):
2924  """
2925  A forward-backward filter.
2926 
2927  This function applies a linear filter twice, once forward and once
2928  backwards. The combined filter has linear phase.
2929 
2930  The function provides options for handling the edges of the signal.
2931 
2932  When `method` is "pad", the function pads the data along the given axis
2933  in one of three ways: odd, even or constant. The odd and even extensions
2934  have the corresponding symmetry about the end point of the data. The
2935  constant extension extends the data with the values at the end points. On
2936  both the forward and backward passes, the initial condition of the
2937  filter is found by using `lfilter_zi` and scaling it by the end point of
2938  the extended data.
2939 
2940  When `method` is "gust", Gustafsson's method [1]_ is used. Initial
2941  conditions are chosen for the forward and backward passes so that the
2942  forward-backward filter gives the same result as the backward-forward
2943  filter.
2944 
2945  Parameters
2946  ----------
2947  b : (N,) array_like
2948  The numerator coefficient vector of the filter.
2949  a : (N,) array_like
2950  The denominator coefficient vector of the filter. If ``a[0]``
2951  is not 1, then both `a` and `b` are normalized by ``a[0]``.
2952  x : array_like
2953  The array of data to be filtered.
2954  axis : int, optional
2955  The axis of `x` to which the filter is applied.
2956  Default is -1.
2957  padtype : str or None, optional
2958  Must be 'odd', 'even', 'constant', or None. This determines the
2959  type of extension to use for the padded signal to which the filter
2960  is applied. If `padtype` is None, no padding is used. The default
2961  is 'odd'.
2962  padlen : int or None, optional
2963  The number of elements by which to extend `x` at both ends of
2964  `axis` before applying the filter. This value must be less than
2965  ``x.shape[axis] - 1``. ``padlen=0`` implies no padding.
2966  The default value is ``3 * max(len(a), len(b))``.
2967  method : str, optional
2968  Determines the method for handling the edges of the signal, either
2969  "pad" or "gust". When `method` is "pad", the signal is padded; the
2970  type of padding is determined by `padtype` and `padlen`, and `irlen`
2971  is ignored. When `method` is "gust", Gustafsson's method is used,
2972  and `padtype` and `padlen` are ignored.
2973  irlen : int or None, optional
2974  When `method` is "gust", `irlen` specifies the length of the
2975  impulse response of the filter. If `irlen` is None, no part
2976  of the impulse response is ignored. For a long signal, specifying
2977  `irlen` can significantly improve the performance of the filter.
2978 
2979  Returns
2980  -------
2981  y : ndarray
2982  The filtered output with the same shape as `x`.
2983 
2984  See Also
2985  --------
2986  sosfiltfilt, lfilter_zi, lfilter, lfiltic, savgol_filter, sosfilt
2987 
2988  Notes
2989  -----
2990  The option to use Gustaffson's method was added in scipy version 0.16.0.
2991 
2992  References
2993  ----------
2994  .. [1] F. Gustaffson, "Determining the initial states in forward-backward
2995  filtering", Transactions on Signal Processing, Vol. 46, pp. 988-992,
2996  1996.
