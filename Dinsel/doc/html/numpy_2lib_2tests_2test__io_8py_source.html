61 class RoundtripTest(object):
62  def roundtrip(self, save_func, *args, **kwargs):
63  """
64  save_func : callable
65  Function used to save arrays to file.
66  file_on_disk : bool
67  If true, store the file on disk, instead of in a
68  string buffer.
69  save_kwds : dict
70  Parameters passed to `save_func`.
71  load_kwds : dict
72  Parameters passed to `numpy.load`.
73  args : tuple of arrays
74  Arrays stored to file.
75 
76  """
77  save_kwds = kwargs.get('save_kwds', {})
78  load_kwds = kwargs.get('load_kwds', {})
79  file_on_disk = kwargs.get('file_on_disk', False)
80 
81  if file_on_disk:
82  target_file = NamedTemporaryFile(delete=False)
83  load_file = target_file.name
84  else:
85  target_file = BytesIO()
86  load_file = target_file
87 
88  try:
89  arr = args
90 
91  save_func(target_file, *arr, **save_kwds)
92  target_file.flush()
93  target_file.seek(0)
94 
95  if sys.platform == 'win32' and not isinstance(target_file, BytesIO):
96  target_file.close()
97 
98  arr_reloaded = np.load(load_file, **load_kwds)
99 
100  self.arr = arr
101  self.arr_reloaded = arr_reloaded
102  finally:
103  if not isinstance(target_file, BytesIO):
104  target_file.close()
105  # holds an open file descriptor so it can't be deleted on win
106  if not isinstance(arr_reloaded, np.lib.npyio.NpzFile):
107  os.remove(target_file.name)
