749 def _help_expm_cond_search(A, A_norm, X, X_norm, eps, p):
750  p = np.reshape(p, A.shape)
751  p_norm = norm(p)
752  perturbation = eps * p * (A_norm / p_norm)
753  X_prime = expm(A + perturbation)
754  scaled_relative_error = norm(X_prime - X) / (X_norm * eps)
755  return -scaled_relative_error
