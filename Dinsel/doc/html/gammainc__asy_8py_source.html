22 def compute_a(n):
23  """a_k from DLMF 5.11.6"""
24  a = [mp.sqrt(2)/2]
25  for k in range(1, n):
26  ak = a[-1]/k

44  """Function from DLMF 8.12.1 shifted to be centered at 0."""
45  if lam > 0:
46  return mp.sqrt(2*(lam - mp.log(lam + 1)))

60  """d_{k, n} from DLMF 8.12.12"""
61  M = N + 2*K
62  d0 = [-mp.mpf(1)/3]
