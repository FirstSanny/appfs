35  def _get_visit_name(self):
36  """
37  return the visit name for the mixed class. When calling 'accept', the
38  method <'visit_' + name returned by this method> will be called on the
39  visitor
40  """
41  try:
42  return self.TYPE.replace('-', '_')
43  except Exception:
44  return self.__class__.__name__.lower()
45 
46  def accept(self, visitor, *args, **kwargs):
47  func = getattr(visitor, 'visit_%s' % self._get_visit_name())
48  return func(self, *args, **kwargs)
49 
50  def leave(self, visitor, *args, **kwargs):
51  func = getattr(visitor, 'leave_%s' % self._get_visit_name())
52  return func(self, *args, **kwargs)
