372 def _moment_from_stats(n, mu, mu2, g1, g2, moment_func, args):
373  if (n == 0):
374  return 1.0
375  elif (n == 1):
376  if mu is None:
377  val = moment_func(1, *args)
378  else:
379  val = mu
380  elif (n == 2):
381  if mu2 is None or mu is None:
382  val = moment_func(2, *args)
383  else:
384  val = mu2 + mu*mu
385  elif (n == 3):
386  if g1 is None or mu2 is None or mu is None:
387  val = moment_func(3, *args)
388  else:
389  mu3 = g1 * np.power(mu2, 1.5) # 3rd central moment
390  val = mu3+3*mu*mu2+mu*mu*mu # 3rd non-central moment
391  elif (n == 4):
392  if g1 is None or g2 is None or mu2 is None or mu is None:
393  val = moment_func(4, *args)
394  else:
395  mu4 = (g2+3.0)*(mu2**2.0) # 4th central moment
396  mu3 = g1*np.power(mu2, 1.5) # 3rd central moment
397  val = mu4+4*mu*mu3+6*mu*mu*mu2+mu*mu*mu*mu
398  else:
399  val = moment_func(n, *args)
400 
401  return val

567 def _ncx2_log_pdf(x, df, nc):
568  # We use (xs**2 + ns**2)/2 = (xs - ns)**2/2 + xs*ns, and include the factor
569  # of exp(-xs*ns) into the ive function to improve numerical stability
570  # at large values of xs. See also `rice.pdf`.
571  df2 = df/2.0 - 1.0
572  xs, ns = np.sqrt(x), np.sqrt(nc)
573  res = xlogy(df2/2.0, x/nc) - 0.5*(xs - ns)**2
574  res += np.log(ive(df2, xs*ns) / 2.0)
575  return res

788  def _argcheck_rvs(self, *args, **kwargs):
789  # Handle broadcasting and size validation of the rvs method.
790  # Subclasses should not have to override this method.
791  # The rule is that if `size` is not None, then `size` gives the
792  # shape of the result (integer values of `size` are treated as
793  # tuples with length 1; i.e. `size=3` is the same as `size=(3,)`.)
794  #
795  # `args` is expected to contain the shape parameters (if any), the
796  # location and the scale in a flat tuple (e.g. if there are two
797  # shape parameters `a` and `b`, `args` will be `(a, b, loc, scale)`).
798  # The only keyword argument expected is 'size'.
799  size = kwargs.get('size', None)
800  all_bcast = np.broadcast_arrays(*args)
801 
802  def squeeze_left(a):
803  while a.ndim > 0 and a.shape[0] == 1:
804  a = a[0]
805  return a
806 
807  # Eliminate trivial leading dimensions. In the convention
808  # used by numpy's random variate generators, trivial leading
809  # dimensions are effectively ignored. In other words, when `size`
810  # is given, trivial leading dimensions of the broadcast parameters
811  # in excess of the number of dimensions in size are ignored, e.g.
812  # >>> np.random.normal([[1, 3, 5]], [[[[0.01]]]], size=3)
813  # array([ 1.00104267, 3.00422496, 4.99799278])
814  # If `size` is not given, the exact broadcast shape is preserved:
815  # >>> np.random.normal([[1, 3, 5]], [[[[0.01]]]])
816  # array([[[[ 1.00862899, 3.00061431, 4.99867122]]]])
817  #
818  all_bcast = [squeeze_left(a) for a in all_bcast]
819  bcast_shape = all_bcast[0].shape
820  bcast_ndim = all_bcast[0].ndim
821 
822  if size is None:
823  size_ = bcast_shape
824  else:
825  size_ = tuple(np.atleast_1d(size))
826 
827  # Check compatibility of size_ with the broadcast shape of all
828  # the parameters. This check is intended to be consistent with
829  # how the numpy random variate generators (e.g. np.random.normal,
830  # np.random.beta) handle their arguments. The rule is that, if size
831  # is given, it determines the shape of the output. Broadcasting
832  # can't change the output size.
833 
834  # This is the standard broadcasting convention of extending the
835  # shape with fewer dimensions with enough dimensions of length 1
836  # so that the two shapes have the same number of dimensions.
837  ndiff = bcast_ndim - len(size_)
838  if ndiff < 0:
839  bcast_shape = (1,)*(-ndiff) + bcast_shape
840  elif ndiff > 0:
841  size_ = (1,)*ndiff + size_
842 
843  # This compatibility test is not standard. In "regular" broadcasting,
844  # two shapes are compatible if for each dimension, the lengths are the
845  # same or one of the lengths is 1. Here, the length of a dimension in
846  # size_ must not be less than the corresponding length in bcast_shape.
847  ok = all([bcdim == 1 or bcdim == szdim
848  for (bcdim, szdim) in zip(bcast_shape, size_)])
849  if not ok:
850  raise ValueError("size does not match the broadcast shape of "
851  "the parameters.")
852 
853  param_bcast = all_bcast[:-2]
854  loc_bcast = all_bcast[-2]
855  scale_bcast = all_bcast[-1]
856 
857  return param_bcast, loc_bcast, scale_bcast, size_

1119  """
1120  n-th order non-central moment of distribution.
1121 
1122  Parameters
1123  ----------
1124  n : int, n >= 1
1125  Order of moment.
1126  arg1, arg2, arg3,... : float
1127  The shape parameter(s) for the distribution (see docstring of the
1128  instance object for more information).
1129  loc : array_like, optional

1132  scale parameter (default=1)
1133 
1134  """

1136  if not (self._argcheck(*args) and (scale > 0)):
1137  return nan

1140  if (n < 0):
1141  raise ValueError("Moment must be positive.")
1142  mu, mu2, g1, g2 = None, None, None, None

1144  if self._stats_has_moments:
1145  mdict = {'moments': {1: 'm', 2: 'v', 3: 'vs', 4: 'vk'}[n]}
1146  else:

1153  if loc == 0:

1155  else:

1160  result += comb(n, k, exact=True)*(fac**k) * valk
