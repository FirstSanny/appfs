54 def rot90(m, k=1, axes=(0,1)):
55  """
56  Rotate an array by 90 degrees in the plane specified by axes.
57 
58  Rotation direction is from the first towards the second axis.
59 
60  .. versionadded:: 1.12.0
61 
62  Parameters
63  ----------
64  m : array_like
65  Array of two or more dimensions.
66  k : integer
67  Number of times the array is rotated by 90 degrees.
68  axes: (2,) array_like
69  The array is rotated in the plane defined by the axes.
70  Axes must be different.
71 
72  Returns
73  -------
74  y : ndarray
75  A rotated view of `m`.
76 
77  See Also
78  --------
79  flip : Reverse the order of elements in an array along the given axis.
80  fliplr : Flip an array horizontally.
81  flipud : Flip an array vertically.
82 
83  Notes
84  -----
85  rot90(m, k=1, axes=(1,0)) is the reverse of rot90(m, k=1, axes=(0,1))
86  rot90(m, k=1, axes=(1,0)) is equivalent to rot90(m, k=-1, axes=(0,1))
87 
88  Examples
89  --------
90  >>> m = np.array([[1,2],[3,4]], int)
91  >>> m
92  array([[1, 2],
93  [3, 4]])
94  >>> np.rot90(m)
95  array([[2, 4],
96  [1, 3]])
97  >>> np.rot90(m, 2)
98  array([[4, 3],
99  [2, 1]])
100  >>> m = np.arange(8).reshape((2,2,2))
101  >>> np.rot90(m, 1, (1,2))
102  array([[[1, 3],
103  [0, 2]],
104 
105  [[5, 7],
106  [4, 6]]])
107 
108  """
109  axes = tuple(axes)
110  if len(axes) != 2:
111  raise ValueError("len(axes) must be 2.")
112 
113  m = asanyarray(m)
114 
115  if axes[0] == axes[1] or absolute(axes[0] - axes[1]) == m.ndim:
116  raise ValueError("Axes must be different.")
117 
118  if (axes[0] >= m.ndim or axes[0] < -m.ndim
119  or axes[1] >= m.ndim or axes[1] < -m.ndim):
120  raise ValueError("Axes={} out of range for array of ndim={}."
121  .format(axes, m.ndim))
122 
123  k %= 4
124 
125  if k == 0:
126  return m[:]
127  if k == 2:
128  return flip(flip(m, axes[0]), axes[1])
129 
130  axes_list = arange(0, m.ndim)
131  axes_list[axes[0]], axes_list[axes[1]] = axes_list[axes[1]], axes_list[axes[0]]
132 
133  if k == 1:
134  return transpose(flip(m,axes[1]), axes_list)
135  else:
136  # k == 3
137  return flip(transpose(m, axes_list), axes[1])

140 def flip(m, axis):
141  """
142  Reverse the order of elements in an array along the given axis.
143 
144  The shape of the array is preserved, but the elements are reordered.
145 
146  .. versionadded:: 1.12.0
147 
148  Parameters
149  ----------
150  m : array_like
151  Input array.
152  axis : integer
153  Axis in array, which entries are reversed.
154 
155 
156  Returns
157  -------
158  out : array_like
159  A view of `m` with the entries of axis reversed. Since a view is
160  returned, this operation is done in constant time.
161 
162  See Also
163  --------
164  flipud : Flip an array vertically (axis=0).
165  fliplr : Flip an array horizontally (axis=1).
166 
167  Notes
168  -----
169  flip(m, 0) is equivalent to flipud(m).
170  flip(m, 1) is equivalent to fliplr(m).
171  flip(m, n) corresponds to ``m[...,::-1,...]`` with ``::-1`` at position n.
172 
173  Examples
174  --------
175  >>> A = np.arange(8).reshape((2,2,2))
176  >>> A
177  array([[[0, 1],
178  [2, 3]],
179 
180  [[4, 5],
181  [6, 7]]])
182 
183  >>> flip(A, 0)
184  array([[[4, 5],
185  [6, 7]],
186 
187  [[0, 1],
188  [2, 3]]])
189 
190  >>> flip(A, 1)
191  array([[[2, 3],
192  [0, 1]],
193 
194  [[6, 7],
195  [4, 5]]])
196 
197  >>> A = np.random.randn(3,4,5)
198  >>> np.all(flip(A,2) == A[:,:,::-1,...])
199  True
200  """
201  if not hasattr(m, 'ndim'):
202  m = asarray(m)
203  indexer = [slice(None)] * m.ndim
204  try:
205  indexer[axis] = slice(None, None, -1)
206  except IndexError:
207  raise ValueError("axis=%i is invalid for the %i-dimensional input array"
208  % (axis, m.ndim))
209  return m[tuple(indexer)]

329 def _hist_bin_doane(x):
330  """
331  Doane's histogram bin estimator.
332 
333  Improved version of Sturges' formula which works better for
334  non-normal data. See
335  http://stats.stackexchange.com/questions/55134/doanes-formula-for-histogram-binning
336 
337  Parameters
338  ----------
339  x : array_like
340  Input data that is to be histogrammed, trimmed to range. May not
341  be empty.
342 
343  Returns
344  -------
345  h : An estimate of the optimal bin width for the given data.
346  """
347  if x.size > 2:
348  sg1 = np.sqrt(6.0 * (x.size - 2) / ((x.size + 1.0) * (x.size + 3)))
349  sigma = np.std(x)
350  if sigma > 0.0:
351  # These three operations add up to
352  # g1 = np.mean(((x - np.mean(x)) / sigma)**3)
353  # but use only one temp array instead of three
354  temp = x - np.mean(x)
355  np.true_divide(temp, sigma, temp)
356  np.power(temp, 3, temp)
357  g1 = np.mean(temp)
358  return x.ptp() / (1.0 + np.log2(x.size) +
359  np.log2(1.0 + np.absolute(g1) / sg1))
360  return 0.0

363 def _hist_bin_fd(x):
364  """
365  The Freedman-Diaconis histogram bin estimator.
366 
367  The Freedman-Diaconis rule uses interquartile range (IQR) to
368  estimate binwidth. It is considered a variation of the Scott rule
369  with more robustness as the IQR is less affected by outliers than
370  the standard deviation. However, the IQR depends on fewer points
371  than the standard deviation, so it is less accurate, especially for
372  long tailed distributions.
373 
374  If the IQR is 0, this function returns 1 for the number of bins.
375  Binwidth is inversely proportional to the cube root of data size
376  (asymptotically optimal).
377 
378  Parameters
379  ----------
380  x : array_like
381  Input data that is to be histogrammed, trimmed to range. May not
382  be empty.
383 
384  Returns
385  -------
386  h : An estimate of the optimal bin width for the given data.
387  """
388  iqr = np.subtract(*np.percentile(x, [75, 25]))
389  return 2.0 * iqr * x.size ** (-1.0 / 3.0)

392 def _hist_bin_auto(x):
393  """
394  Histogram bin estimator that uses the minimum width of the
395  Freedman-Diaconis and Sturges estimators.
396 
397  The FD estimator is usually the most robust method, but its width
398  estimate tends to be too large for small `x`. The Sturges estimator
399  is quite good for small (<1000) datasets and is the default in the R
400  language. This method gives good off the shelf behaviour.

405  Input data that is to be histogrammed, trimmed to range. May not
406  be empty.
407 
408  Returns

410  h : An estimate of the optimal bin width for the given data.
411 
412  See Also
413  --------
414  _hist_bin_fd, _hist_bin_sturges
415  """

417  # vice versa. Either both are zero or neither one is.

431 def histogram(a, bins=10, range=None, normed=False, weights=None,
432  density=None):
433  r"""
434  Compute the histogram of a set of data.
435 
436  Parameters
437  ----------
438  a : array_like
439  Input data. The histogram is computed over the flattened array.
440  bins : int or sequence of scalars or str, optional
441  If `bins` is an int, it defines the number of equal-width
442  bins in the given range (10, by default). If `bins` is a
443  sequence, it defines the bin edges, including the rightmost
444  edge, allowing for non-uniform bin widths.
445 
446  .. versionadded:: 1.11.0
447 
448  If `bins` is a string from the list below, `histogram` will use
449  the method chosen to calculate the optimal bin width and
450  consequently the number of bins (see `Notes` for more detail on
451  the estimators) from the data that falls within the requested
452  range. While the bin width will be optimal for the actual data
453  in the range, the number of bins will be computed to fill the
454  entire range, including the empty portions. For visualisation,
455  using the 'auto' option is suggested. Weighted data is not
456  supported for automated bin size selection.
457 
458  'auto'
459  Maximum of the 'sturges' and 'fd' estimators. Provides good
460  all around performance.
461 
462  'fd' (Freedman Diaconis Estimator)
463  Robust (resilient to outliers) estimator that takes into
464  account data variability and data size.
465 
466  'doane'
467  An improved version of Sturges' estimator that works better
468  with non-normal datasets.
469 
470  'scott'
471  Less robust estimator that that takes into account data
472  variability and data size.
473 
474  'rice'
475  Estimator does not take variability into account, only data
476  size. Commonly overestimates number of bins required.
477 
478  'sturges'
479  R's default method, only accounts for data size. Only
480  optimal for gaussian data and underestimates number of bins
481  for large non-gaussian datasets.
482 
483  'sqrt'
484  Square root (of data size) estimator, used by Excel and
485  other programs for its speed and simplicity.
486 
487  range : (float, float), optional
488  The lower and upper range of the bins. If not provided, range
489  is simply ``(a.min(), a.max())``. Values outside the range are
490  ignored. The first element of the range must be less than or
491  equal to the second. `range` affects the automatic bin
492  computation as well. While bin width is computed to be optimal
493  based on the actual data within `range`, the bin count will fill
494  the entire range including portions containing no data.
495  normed : bool, optional
496  This keyword is deprecated in NumPy 1.6.0 due to confusing/buggy
497  behavior. It will be removed in NumPy 2.0.0. Use the ``density``
498  keyword instead. If ``False``, the result will contain the
499  number of samples in each bin. If ``True``, the result is the
500  value of the probability *density* function at the bin,
501  normalized such that the *integral* over the range is 1. Note
502  that this latter behavior is known to be buggy with unequal bin
503  widths; use ``density`` instead.
504  weights : array_like, optional
505  An array of weights, of the same shape as `a`. Each value in
506  `a` only contributes its associated weight towards the bin count
507  (instead of 1). If `density` is True, the weights are
508  normalized, so that the integral of the density over the range
509  remains 1.
510  density : bool, optional
511  If ``False``, the result will contain the number of samples in
512  each bin. If ``True``, the result is the value of the
513  probability *density* function at the bin, normalized such that
514  the *integral* over the range is 1. Note that the sum of the
515  histogram values will not be equal to 1 unless bins of unity
516  width are chosen; it is not a probability *mass* function.
517 
518  Overrides the ``normed`` keyword if given.
519 
520  Returns
521  -------
522  hist : array
523  The values of the histogram. See `density` and `weights` for a
524  description of the possible semantics.
525  bin_edges : array of dtype float
526  Return the bin edges ``(length(hist)+1)``.
527 
528 
529  See Also
530  --------
531  histogramdd, bincount, searchsorted, digitize
532 
533  Notes
534  -----
535  All but the last (righthand-most) bin is half-open. In other words,
536  if `bins` is::
537 
538  [1, 2, 3, 4]
539 
540  then the first bin is ``[1, 2)`` (including 1, but excluding 2) and
541  the second ``[2, 3)``. The last bin, however, is ``[3, 4]``, which
542  *includes* 4.
543 
544  .. versionadded:: 1.11.0
545 
546  The methods to estimate the optimal number of bins are well founded
547  in literature, and are inspired by the choices R provides for
548  histogram visualisation. Note that having the number of bins
549  proportional to :math:`n^{1/3}` is asymptotically optimal, which is
550  why it appears in most estimators. These are simply plug-in methods
551  that give good starting points for number of bins. In the equations
552  below, :math:`h` is the binwidth and :math:`n_h` is the number of
553  bins. All estimators that compute bin counts are recast to bin width
554  using the `ptp` of the data. The final bin count is obtained from
555  ``np.round(np.ceil(range / h))`.
556 
557  'Auto' (maximum of the 'Sturges' and 'FD' estimators)
558  A compromise to get a good value. For small datasets the Sturges
559  value will usually be chosen, while larger datasets will usually
560  default to FD. Avoids the overly conservative behaviour of FD
561  and Sturges for small and large datasets respectively.
562  Switchover point is usually :math:`a.size \approx 1000`.
563 
564  'FD' (Freedman Diaconis Estimator)
565  .. math:: h = 2 \frac{IQR}{n^{1/3}}
566 
567  The binwidth is proportional to the interquartile range (IQR)
568  and inversely proportional to cube root of a.size. Can be too
569  conservative for small datasets, but is quite good for large
570  datasets. The IQR is very robust to outliers.
571 
572  'Scott'
573  .. math:: h = \sigma \sqrt[3]{\frac{24 * \sqrt{\pi}}{n}}
574 
575  The binwidth is proportional to the standard deviation of the
576  data and inversely proportional to cube root of ``x.size``. Can
577  be too conservative for small datasets, but is quite good for
578  large datasets. The standard deviation is not very robust to
579  outliers. Values are very similar to the Freedman-Diaconis
580  estimator in the absence of outliers.
581 
582  'Rice'
583  .. math:: n_h = 2n^{1/3}
584 
585  The number of bins is only proportional to cube root of
586  ``a.size``. It tends to overestimate the number of bins and it
587  does not take into account data variability.
588 
589  'Sturges'
590  .. math:: n_h = \log _{2}n+1
591 
592  The number of bins is the base 2 log of ``a.size``. This
593  estimator assumes normality of data and is too conservative for
594  larger, non-normal datasets. This is the default method in R's
595  ``hist`` method.
596 
597  'Doane'
598  .. math:: n_h = 1 + \log_{2}(n) +
599  \log_{2}(1 + \frac{|g_1|}{\sigma_{g_1}})
600 
601  g_1 = mean[(\frac{x - \mu}{\sigma})^3]
602 
603  \sigma_{g_1} = \sqrt{\frac{6(n - 2)}{(n + 1)(n + 3)}}
604 
605  An improved version of Sturges' formula that produces better
606  estimates for non-normal datasets. This estimator attempts to
607  account for the skew of the data.
608 
609  'Sqrt'
610  .. math:: n_h = \sqrt n
611  The simplest and fastest estimator. Only takes into account the
612  data size.
613 
614  Examples
615  --------
616  >>> np.histogram([1, 2, 1], bins=[0, 1, 2, 3])
617  (array([0, 2, 1]), array([0, 1, 2, 3]))
618  >>> np.histogram(np.arange(4), bins=np.arange(5), density=True)
619  (array([ 0.25, 0.25, 0.25, 0.25]), array([0, 1, 2, 3, 4]))
620  >>> np.histogram([[1, 2, 1], [1, 0, 1]], bins=[0,1,2,3])
621  (array([1, 4, 1]), array([0, 1, 2, 3]))
622 
623  >>> a = np.arange(5)
624  >>> hist, bin_edges = np.histogram(a, density=True)
625  >>> hist
626  array([ 0.5, 0. , 0.5, 0. , 0. , 0.5, 0. , 0.5, 0. , 0.5])
627  >>> hist.sum()
628  2.4999999999999996
629  >>> np.sum(hist*np.diff(bin_edges))
630  1.0
631 
632  .. versionadded:: 1.11.0
633 
634  Automated Bin Selection Methods example, using 2 peak random data
635  with 2000 points:
636 
637  >>> import matplotlib.pyplot as plt
638  >>> rng = np.random.RandomState(10) # deterministic random data
639  >>> a = np.hstack((rng.normal(size=1000),
640  ... rng.normal(loc=5, scale=2, size=1000)))
641  >>> plt.hist(a, bins='auto') # plt.hist passes it's arguments to np.histogram
642  >>> plt.title("Histogram with 'auto' bins")
643  >>> plt.show()
644 
645  """
646  a = asarray(a)
647  if weights is not None:
648  weights = asarray(weights)
649  if np.any(weights.shape != a.shape):
650  raise ValueError(
651  'weights should have the same shape as a.')
652  weights = weights.ravel()
653  a = a.ravel()
654 
655  # Do not modify the original value of range so we can check for `None`
656  if range is None:
657  if a.size == 0:
658  # handle empty arrays. Can't determine range, so use 0-1.
659  mn, mx = 0.0, 1.0
660  else:
661  mn, mx = a.min() + 0.0, a.max() + 0.0
662  else:
663  mn, mx = [mi + 0.0 for mi in range]
664  if mn > mx:
665  raise ValueError(
666  'max must be larger than min in range parameter.')
667  if not np.all(np.isfinite([mn, mx])):
668  raise ValueError(
669  'range parameter must be finite.')
670  if mn == mx:
671  mn -= 0.5
672  mx += 0.5
673 
674  if isinstance(bins, basestring):
675  # if `bins` is a string for an automatic method,
676  # this will replace it with the number of bins calculated
677  if bins not in _hist_bin_selectors:
678  raise ValueError("{0} not a valid estimator for bins".format(bins))
679  if weights is not None:
680  raise TypeError("Automated estimation of the number of "
681  "bins is not supported for weighted data")
682  # Make a reference to `a`
683  b = a
684  # Update the reference if the range needs truncation
685  if range is not None:
686  keep = (a >= mn)
687  keep &= (a <= mx)
688  if not np.logical_and.reduce(keep):
689  b = a[keep]
690 
691  if b.size == 0:
692  bins = 1
693  else:
694  # Do not call selectors on empty arrays
695  width = _hist_bin_selectors[bins](b)
696  if width:
697  bins = int(np.ceil((mx - mn) / width))
698  else:
699  # Width can be zero for some estimators, e.g. FD when
700  # the IQR of the data is zero.
701  bins = 1
702 
703  # Histogram is an integer or a float array depending on the weights.
704  if weights is None:
705  ntype = np.dtype(np.intp)
706  else:
707  ntype = weights.dtype
708 
709  # We set a block size, as this allows us to iterate over chunks when
710  # computing histograms, to minimize memory usage.
711  BLOCK = 65536
712 
713  if not iterable(bins):
714  if np.isscalar(bins) and bins < 1:
715  raise ValueError(
716  '`bins` should be a positive integer.')
717  # At this point, if the weights are not integer, floating point, or
718  # complex, we have to use the slow algorithm.
719  if weights is not None and not (np.can_cast(weights.dtype, np.double) or
720  np.can_cast(weights.dtype, np.complex)):
721  bins = linspace(mn, mx, bins + 1, endpoint=True)
722 
723  if not iterable(bins):
724  # We now convert values of a to bin indices, under the assumption of
725  # equal bin widths (which is valid here).
726 
727  # Initialize empty histogram
728  n = np.zeros(bins, ntype)
729  # Pre-compute histogram scaling factor
730  norm = bins / (mx - mn)
731 
732  # Compute the bin edges for potential correction.
733  bin_edges = linspace(mn, mx, bins + 1, endpoint=True)
734 
735  # We iterate over blocks here for two reasons: the first is that for
736  # large arrays, it is actually faster (for example for a 10^8 array it
737  # is 2x as fast) and it results in a memory footprint 3x lower in the
738  # limit of large arrays.
739  for i in arange(0, len(a), BLOCK):
740  tmp_a = a[i:i+BLOCK]
741  if weights is None:
742  tmp_w = None
743  else:
744  tmp_w = weights[i:i + BLOCK]
745 
746  # Only include values in the right range
747  keep = (tmp_a >= mn)
748  keep &= (tmp_a <= mx)
749  if not np.logical_and.reduce(keep):
750  tmp_a = tmp_a[keep]
751  if tmp_w is not None:
752  tmp_w = tmp_w[keep]
753  tmp_a_data = tmp_a.astype(float)
754  tmp_a = tmp_a_data - mn
755  tmp_a *= norm
756 
757  # Compute the bin indices, and for values that lie exactly on mx we
758  # need to subtract one
759  indices = tmp_a.astype(np.intp)
760  indices[indices == bins] -= 1
761 
762  # The index computation is not guaranteed to give exactly
763  # consistent results within ~1 ULP of the bin edges.
764  decrement = tmp_a_data < bin_edges[indices]
765  indices[decrement] -= 1
766  # The last bin includes the right edge. The other bins do not.
767  increment = (tmp_a_data >= bin_edges[indices + 1]) & (indices != bins - 1)
768  indices[increment] += 1
769 
770  # We now compute the histogram using bincount
771  if ntype.kind == 'c':
772  n.real += np.bincount(indices, weights=tmp_w.real, minlength=bins)
773  n.imag += np.bincount(indices, weights=tmp_w.imag, minlength=bins)
774  else:
775  n += np.bincount(indices, weights=tmp_w, minlength=bins).astype(ntype)
776 
777  # Rename the bin edges for return.
778  bins = bin_edges
779  else:
780  bins = asarray(bins)
781  if (np.diff(bins) < 0).any():
782  raise ValueError(
783  'bins must increase monotonically.')
784 
785  # Initialize empty histogram
786  n = np.zeros(bins.shape, ntype)
787 
788  if weights is None:
789  for i in arange(0, len(a), BLOCK):
790  sa = sort(a[i:i+BLOCK])
791  n += np.r_[sa.searchsorted(bins[:-1], 'left'),
792  sa.searchsorted(bins[-1], 'right')]
793  else:
794  zero = array(0, dtype=ntype)
795  for i in arange(0, len(a), BLOCK):
796  tmp_a = a[i:i+BLOCK]
797  tmp_w = weights[i:i+BLOCK]
798  sorting_index = np.argsort(tmp_a)
799  sa = tmp_a[sorting_index]
800  sw = tmp_w[sorting_index]
801  cw = np.concatenate(([zero, ], sw.cumsum()))
802  bin_index = np.r_[sa.searchsorted(bins[:-1], 'left'),
803  sa.searchsorted(bins[-1], 'right')]
804  n += cw[bin_index]
805 
806 
807  n = np.diff(n)
808 
809  if density is not None:
810  if density:
811  db = array(np.diff(bins), float)
812  return n/db/n.sum(), bins
813  else:
814  return n, bins
815  else:
816  # deprecated, buggy behavior. Remove for NumPy 2.0.0
817  if normed:
818  db = array(np.diff(bins), float)
819  return n/(n*db).sum(), bins
820  else:

980  ni = nbin.argsort()

982  for i in arange(0, D-1):
983  xy += Ncount[ni[i]] * nbin[ni[i+1:]].prod()
984  xy += Ncount[ni[-1]]
985

987  # flattened histmat.
988  if len(xy) == 0:
989  return zeros(nbin-2, int), edges
990 
991  flatcount = bincount(xy, weights)
992  a = arange(len(flatcount))
993  hist[a] = flatcount
994 
995  # Shape into a proper matrix
996  hist = hist.reshape(sort(nbin))
997  for i in arange(nbin.size):
998  j = ni.argsort()[i]
999  hist = hist.swapaxes(i, j)
1000  ni[i], ni[j] = ni[j], ni[i]
1001 
1002  # Remove outliers (indices 0 and -1 for each dimension).
1003  core = D*[slice(1, -1)]
1004  hist = hist[core]
1005 
1006  # Normalize if normed is True
1007  if normed:
1008  s = hist.sum()
1009  for i in arange(D):
1010  shape = ones(D, int)
1011  shape[i] = nbin[i] - 2
1012  hist = hist / dedges[i].reshape(shape)
1013  hist /= s
1014 
1015  if (hist.shape != nbin - 2).any():
1016  raise RuntimeError(
1017  "Internal Shape Error")
1018  return hist, edges

1502 def gradient(f, *varargs, **kwargs):

1507  in the interior and either first differences or second order accurate
1508  one-sides (forward or backwards) differences at the boundaries. The

1525
