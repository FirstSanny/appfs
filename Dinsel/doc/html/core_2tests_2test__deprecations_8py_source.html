48  function_fails=False,

51  """Test if VisibleDeprecationWarnings are given and raised.
52 
53  This first checks if the function when called gives `num`
54  VisibleDeprecationWarnings, after that it tries to raise these
55  VisibleDeprecationWarnings and compares them with `exceptions`.
56  The exceptions can be different for cases where this code path
57  is simply not anticipated and the exception is replaced.
58 
59  Parameters
60  ----------
61  function : callable
62  The function to test
63  num : int
64  Number of VisibleDeprecationWarnings to expect. This should
65  normally be 1.
66  ignore_others : bool
67  Whether warnings of the wrong type should be ignored (note that
68  the message is not checked)
69  function_fails : bool
70  If the function would normally fail, setting this will check for
71  warnings inside a try/except block.
72  exceptions : Exception or tuple of Exceptions
73  Exception to expect when turning the warnings into an error.
74  The default checks for DeprecationWarnings. If exceptions is
75  empty the function is expected to run successfully.
76  args : tuple
77  Arguments for `function`
78  kwargs : dict
79  Keyword arguments for `function`
80  """
81  # reset the log
82  self.log[:] = []
83 
84  try:
85  function(*args, **kwargs)
86  except (Exception if function_fails else tuple()):
87  pass
88 
89  # just in case, clear the registry
90  num_found = 0
91  for warning in self.log:
92  if warning.category is np.VisibleDeprecationWarning:
93  num_found += 1
94  elif not ignore_others:
95  raise AssertionError(
96  "expected DeprecationWarning but got: %s" %
97  (warning.category,))
98  if num is not None and num_found != num:
99  msg = "%i warnings found but %i expected." % (len(self.log), num)
100  lst = [w.category for w in self.log]
101  raise AssertionError("\n".join([msg] + lst))
102 
103  with warnings.catch_warnings():
104  warnings.filterwarnings("error", message=self.message,
105  category=np.VisibleDeprecationWarning)
106  try:
107  function(*args, **kwargs)
108  if exceptions != tuple():
109  raise AssertionError(
110  "No error raised during function call")
111  except exceptions:
112  if exceptions == tuple():
113  raise AssertionError(
