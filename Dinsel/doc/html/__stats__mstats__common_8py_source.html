14 def linregress(x, y=None):
15  """
16  Calculate a linear least-squares regression for two sets of measurements.
17 
18  Parameters
19  ----------
20  x, y : array_like
21  Two sets of measurements. Both arrays should have the same length.
22  If only x is given (and y=None), then it must be a two-dimensional
23  array where one dimension has length 2. The two sets of measurements
24  are then found by splitting the array along the length-2 dimension.
25 
26  Returns
27  -------
28  slope : float
29  slope of the regression line
30  intercept : float
31  intercept of the regression line
32  rvalue : float
33  correlation coefficient
34  pvalue : float
35  two-sided p-value for a hypothesis test whose null hypothesis is
36  that the slope is zero.
37  stderr : float
38  Standard error of the estimated gradient.
39 
40  See also
41  --------
42  :func:`scipy.optimize.curve_fit` : Use non-linear
43  least squares to fit a function to data.
44  :func:`scipy.optimize.leastsq` : Minimize the sum of
45  squares of a set of equations.
46 
47  Examples
48  --------
49  >>> import matplotlib.pyplot as plt
50  >>> from scipy import stats
51  >>> np.random.seed(12345678)
52  >>> x = np.random.random(10)
53  >>> y = np.random.random(10)
54  >>> slope, intercept, r_value, p_value, std_err = stats.linregress(x, y)
55 
56  To get coefficient of determination (r_squared)
57 
58  >>> print("r-squared:", r_value**2)
59  ('r-squared:', 0.080402268539028335)
60 
61  Plot the data along with the fitted line
62 
63  >>> plt.plot(x, y, 'o', label='original data')
64  >>> plt.plot(x, intercept + slope*x, 'r', label='fitted line')
65  >>> plt.legend()
66  >>> plt.show()
67 
68  """
69  TINY = 1.0e-20
70  if y is None: # x is a (2, N) or (N, 2) shaped array_like
71  x = np.asarray(x)
72  if x.shape[0] == 2:
73  x, y = x
74  elif x.shape[1] == 2:
75  x, y = x.T
76  else:
77  msg = ("If only `x` is given as input, it has to be of shape "
78  "(2, N) or (N, 2), provided shape was %s" % str(x.shape))
79  raise ValueError(msg)
80  else:
81  x = np.asarray(x)
82  y = np.asarray(y)
83 
84  if x.size == 0 or y.size == 0:
85  raise ValueError("Inputs must not be empty.")
86 
87  n = len(x)
88  xmean = np.mean(x, None)
89  ymean = np.mean(y, None)
90 
91  # average sum of squares:
92  ssxm, ssxym, ssyxm, ssym = np.cov(x, y, bias=1).flat
93  r_num = ssxym
94  r_den = np.sqrt(ssxm * ssym)
95  if r_den == 0.0:
96  r = 0.0
97  else:
98  r = r_num / r_den
99  # test for numerical error propagation
100  if r > 1.0:
101  r = 1.0
102  elif r < -1.0:
103  r = -1.0
104 
105  df = n - 2
106  slope = r_num / ssxm
107  intercept = ymean - slope*xmean
108  if n == 2:
109  # handle case when only two points are passed in
110  if y[0] == y[1]:
111  prob = 1.0
112  else:
113  prob = 0.0
114  sterrest = 0.0
115  else:
116  t = r * np.sqrt(df / ((1.0 - r + TINY)*(1.0 + r + TINY)))
117  prob = 2 * distributions.t.sf(np.abs(t), df)
118  sterrest = np.sqrt((1 - r**2) * ssym / ssxm / df)
119 
120  return LinregressResult(slope, intercept, r, prob, sterrest)

123 def theilslopes(y, x=None, alpha=0.95):
124  r"""
125  Computes the Theil-Sen estimator for a set of points (x, y).
126 
127  `theilslopes` implements a method for robust linear regression. It
128  computes the slope as the median of all slopes between paired values.
129 
130  Parameters
131  ----------
132  y : array_like
133  Dependent variable.
134  x : array_like or None, optional
135  Independent variable. If None, use ``arange(len(y))`` instead.
136  alpha : float, optional
137  Confidence degree between 0 and 1. Default is 95% confidence.
138  Note that `alpha` is symmetric around 0.5, i.e. both 0.1 and 0.9 are
139  interpreted as "find the 90% confidence interval".
140 
141  Returns
142  -------
143  medslope : float
144  Theil slope.
145  medintercept : float
146  Intercept of the Theil line, as ``median(y) - medslope*median(x)``.
147  lo_slope : float
148  Lower bound of the confidence interval on `medslope`.
149  up_slope : float
150  Upper bound of the confidence interval on `medslope`.
151 
152  Notes
153  -----
154  The implementation of `theilslopes` follows [1]_. The intercept is
155  not defined in [1]_, and here it is defined as ``median(y) -
156  medslope*median(x)``, which is given in [3]_. Other definitions of
157  the intercept exist in the literature. A confidence interval for
158  the intercept is not given as this question is not addressed in
159  [1]_.
160 
161  References
162  ----------
163  .. [1] P.K. Sen, "Estimates of the regression coefficient based on Kendall's tau",
164  J. Am. Stat. Assoc., Vol. 63, pp. 1379-1389, 1968.
165  .. [2] H. Theil, "A rank-invariant method of linear and polynomial
166  regression analysis I, II and III", Nederl. Akad. Wetensch., Proc.
167  53:, pp. 386-392, pp. 521-525, pp. 1397-1412, 1950.
168  .. [3] W.L. Conover, "Practical nonparametric statistics", 2nd ed.,
169  John Wiley and Sons, New York, pp. 493.
170 
171  Examples
172  --------
173  >>> from scipy import stats
174  >>> import matplotlib.pyplot as plt
175 
176  >>> x = np.linspace(-5, 5, num=150)
177  >>> y = x + np.random.normal(size=x.size)
178  >>> y[11:15] += 10 # add outliers
179  >>> y[-5:] -= 7
180 
181  Compute the slope, intercept and 90% confidence interval. For comparison,
182  also compute the least-squares fit with `linregress`:
183 
184  >>> res = stats.theilslopes(y, x, 0.90)
185  >>> lsq_res = stats.linregress(x, y)
186 
187  Plot the results. The Theil-Sen regression line is shown in red, with the
188  dashed red lines illustrating the confidence interval of the slope (note
189  that the dashed red lines are not the confidence interval of the regression
190  as the confidence interval of the intercept is not included). The green
191  line shows the least-squares fit for comparison.
192 
193  >>> fig = plt.figure()
194  >>> ax = fig.add_subplot(111)
195  >>> ax.plot(x, y, 'b.')
196  >>> ax.plot(x, res[1] + res[0] * x, 'r-')
197  >>> ax.plot(x, res[1] + res[2] * x, 'r--')
198  >>> ax.plot(x, res[1] + res[3] * x, 'r--')
199  >>> ax.plot(x, lsq_res[1] + lsq_res[0] * x, 'g-')
200  >>> plt.show()
201 
202  """
203  # We copy both x and y so we can use _find_repeats.
204  y = np.array(y).flatten()
205  if x is None:
206  x = np.arange(len(y), dtype=float)
207  else:
208  x = np.array(x, dtype=float).flatten()
209  if len(x) != len(y):
210  raise ValueError("Incompatible lengths ! (%s<>%s)" % (len(y), len(x)))
211 
212  # Compute sorted slopes only when deltax > 0
213  deltax = x[:, np.newaxis] - x
214  deltay = y[:, np.newaxis] - y
215  slopes = deltay[deltax > 0] / deltax[deltax > 0]
216  slopes.sort()
217  medslope = np.median(slopes)
218  medinter = np.median(y) - medslope * np.median(x)
219  # Now compute confidence intervals
220  if alpha > 0.5:
221  alpha = 1. - alpha
222 
223  z = distributions.norm.ppf(alpha / 2.)
224  # This implements (2.6) from Sen (1968)
225  _, nxreps = _find_repeats(x)
226  _, nyreps = _find_repeats(y)
227  nt = len(slopes) # N in Sen (1968)
228  ny = len(y) # n in Sen (1968)
229  # Equation 2.6 in Sen (1968):
230  sigsq = 1/18. * (ny * (ny-1) * (2*ny+5) -
231  np.sum(k * (k-1) * (2*k + 5) for k in nxreps) -
232  np.sum(k * (k-1) * (2*k + 5) for k in nyreps))
233  # Find the confidence interval indices in `slopes`
234  sigma = np.sqrt(sigsq)
235  Ru = min(int(np.round((nt - z*sigma)/2.)), len(slopes)-1)
236  Rl = max(int(np.round((nt + z*sigma)/2.)) - 1, 0)
237  delta = slopes[[Rl, Ru]]
238  return medslope, medinter, delta[0], delta[1]

241 def _find_repeats(arr):
242  # This function assumes it may clobber its input.
243  if len(arr) == 0:
244  return np.array(0, np.float64), np.array(0, np.intp)
245 
246  # XXX This cast was previously needed for the Fortran implementation,
247  # should we ditch it?
248  arr = np.asarray(arr, np.float64).ravel()
249  arr.sort()
250 
251  # Taken from NumPy 1.9's np.unique.
252  change = np.concatenate(([True], arr[1:] != arr[:-1]))
253  unique = arr[change]
254  change_idx = np.concatenate(np.nonzero(change) + ([arr.size],))
255  freq = np.diff(change_idx)
256  atleast2 = freq > 1
257  return unique[atleast2], freq[atleast2]
