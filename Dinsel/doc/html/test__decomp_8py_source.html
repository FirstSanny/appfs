64 def symrand(dim_or_eigv):
65  """Return a random symmetric (Hermitian) matrix.
66 
67  If 'dim_or_eigv' is an integer N, return a NxN matrix, with eigenvalues
68  uniformly distributed on (-1,1).
69 
70  If 'dim_or_eigv' is 1-D real array 'a', return a matrix whose
71  eigenvalues are 'a'.
72  """
73  if isinstance(dim_or_eigv, int):
74  dim = dim_or_eigv
75  d = random(dim)*2 - 1
76  elif (isinstance(dim_or_eigv, ndarray) and
77  len(dim_or_eigv.shape) == 1):
78  dim = dim_or_eigv.shape[0]
79  d = dim_or_eigv
80  else:
81  raise TypeError("input type not supported.")
82 
83  v = random_rot(dim)
84  h = dot(dot(v.T.conj(), diag(d)), v)
85  # to avoid roundoff errors, symmetrize the matrix (again)
86  h = 0.5*(h.T+h)
87  return h

93 def random_rot(dim):
94  """Return a random rotation matrix, drawn from the Haar distribution
95  (the only uniform distribution on SO(n)).
96  The algorithm is described in the paper
97  Stewart, G.W., 'The efficient generation of random orthogonal
98  matrices with an application to condition estimators', SIAM Journal
99  on Numerical Analysis, 17(3), pp. 403-409, 1980.
100  For more information see
101  http://en.wikipedia.org/wiki/Orthogonal_matrix#Randomization"""
102  H = eye(dim)
103  D = ones((dim,))
104  for n in range(1, dim):
105  x = normal(size=(dim-n+1,))
106  D[n-1] = sign(x[0])
107  x[0] -= D[n-1]*sqrt((x*x).sum())
108  # Householder transformation
109 
110  Hx = eye(dim-n+1) - 2.*outer(x, x)/(x*x).sum()
111  mat = eye(dim)
112  mat[n-1:,n-1:] = Hx
113  H = dot(H, mat)
114  # Fix the last sign such that the determinant is 1
115  D[-1] = -D.prod()
116  H = (D*H.T).T
117  return H

370  def create_bandmat(self):
371  """Create the full matrix `self.fullmat` and
372  the corresponding band matrix `self.bandmat`."""
373  N = 10
374  self.KL = 2 # number of subdiagonals (below the diagonal)
375  self.KU = 2 # number of superdiagonals (above the diagonal)
376 
377  # symmetric band matrix
378  self.sym_mat = (diag(1.0*ones(N))
379  + diag(-1.0*ones(N-1), -1) + diag(-1.0*ones(N-1), 1)
380  + diag(-2.0*ones(N-2), -2) + diag(-2.0*ones(N-2), 2))
381 
382  # hermitian band matrix
383  self.herm_mat = (diag(-1.0*ones(N))
384  + 1j*diag(1.0*ones(N-1), -1) - 1j*diag(1.0*ones(N-1), 1)
385  + diag(-2.0*ones(N-2), -2) + diag(-2.0*ones(N-2), 2))
386 
387  # general real band matrix
388  self.real_mat = (diag(1.0*ones(N))
389  + diag(-1.0*ones(N-1), -1) + diag(-3.0*ones(N-1), 1)
390  + diag(2.0*ones(N-2), -2) + diag(-2.0*ones(N-2), 2))
391 
392  # general complex band matrix
393  self.comp_mat = (1j*diag(1.0*ones(N))
394  + diag(-1.0*ones(N-1), -1) + 1j*diag(-3.0*ones(N-1), 1)
395  + diag(2.0*ones(N-2), -2) + diag(-2.0*ones(N-2), 2))
396 
397  # Eigenvalues and -vectors from linalg.eig
398  ew, ev = linalg.eig(self.sym_mat)
399  ew = ew.real
400  args = argsort(ew)
401  self.w_sym_lin = ew[args]
402  self.evec_sym_lin = ev[:,args]
403 
404  ew, ev = linalg.eig(self.herm_mat)
405  ew = ew.real
406  args = argsort(ew)
407  self.w_herm_lin = ew[args]
408  self.evec_herm_lin = ev[:,args]
409 
410  # Extract upper bands from symmetric and hermitian band matrices
411  # (for use in dsbevd, dsbevx, zhbevd, zhbevx
412  # and their single precision versions)
413  LDAB = self.KU + 1
414  self.bandmat_sym = zeros((LDAB, N), dtype=float)
415  self.bandmat_herm = zeros((LDAB, N), dtype=complex)
416  for i in xrange(LDAB):
417  self.bandmat_sym[LDAB-i-1,i:N] = diag(self.sym_mat, i)
418  self.bandmat_herm[LDAB-i-1,i:N] = diag(self.herm_mat, i)
419 
420  # Extract bands from general real and complex band matrix
421  # (for use in dgbtrf, dgbtrs and their single precision versions)
422  LDAB = 2*self.KL + self.KU + 1
423  self.bandmat_real = zeros((LDAB, N), dtype=float)
424  self.bandmat_real[2*self.KL,:] = diag(self.real_mat) # diagonal
425  for i in xrange(self.KL):
426  # superdiagonals
427  self.bandmat_real[2*self.KL-1-i,i+1:N] = diag(self.real_mat, i+1)
428  # subdiagonals
429  self.bandmat_real[2*self.KL+1+i,0:N-1-i] = diag(self.real_mat,-i-1)
430 
431  self.bandmat_comp = zeros((LDAB, N), dtype=complex)
432  self.bandmat_comp[2*self.KL,:] = diag(self.comp_mat) # diagonal
433  for i in xrange(self.KL):
434  # superdiagonals
435  self.bandmat_comp[2*self.KL-1-i,i+1:N] = diag(self.comp_mat, i+1)
436  # subdiagonals
437  self.bandmat_comp[2*self.KL+1+i,0:N-1-i] = diag(self.comp_mat,-i-1)

677  overwrite, lower, turbo,
678  eigenvalues):
679  """Solve a standard eigenvalue problem."""
680  if iscomplex(empty(1, dtype=dtype)):
681  a = _complex_symrand(dim, dtype)

697  overwrite, lower, turbo,
