14 def lpgen_2d(m,n):
15  """ -> A b c LP test: m*n vars, m+n constraints
16  row sums == n/m, col sums == 1
17  https://gist.github.com/denis-bz/8647461
18  """
19  np.random.seed(0)
20  c = - np.random.exponential(size=(m,n))
21  Arow = np.zeros((m,m*n))
22  brow = np.zeros(m)
23  for j in range(m):
24  j1 = j + 1
25  Arow[j,j*n:j1*n] = 1
26  brow[j] = n/m
27 
28  Acol = np.zeros((n,m*n))
29  bcol = np.zeros(n)
30  for j in range(n):
31  j1 = j + 1
32  Acol[j,j::n] = 1
33  bcol[j] = 1
34 
35  A = np.vstack((Arow,Acol))
36  b = np.hstack((brow,bcol))
37 
38  return A, b, c.ravel()
