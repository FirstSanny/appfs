38 def _make_tag(base_dt, val, mdtype, sde=False):
39  ''' Makes a simple matlab tag, full or sde '''
40  base_dt = np.dtype(base_dt)
41  bo = boc.to_numpy_code(base_dt.byteorder)
42  byte_count = base_dt.itemsize
43  if not sde:
44  udt = bo + 'u4'
45  padding = 8 - (byte_count % 8)
46  all_dt = [('mdtype', udt),
47  ('byte_count', udt),
48  ('val', base_dt)]
49  if padding:
50  all_dt.append(('padding', 'u1', padding))
51  else: # is sde
52  udt = bo + 'u2'
53  padding = 4-byte_count
54  if bo == '<': # little endian
55  all_dt = [('mdtype', udt),
56  ('byte_count', udt),
57  ('val', base_dt)]
58  else: # big endian
59  all_dt = [('byte_count', udt),
60  ('mdtype', udt),
61  ('val', base_dt)]
62  if padding:
63  all_dt.append(('padding', 'u1', padding))
64  tag = np.zeros((1,), dtype=all_dt)
65  tag['mdtype'] = mdtype
66  tag['byte_count'] = byte_count
67  tag['val'] = val
68  return tag
