18 def parse_info(wininfo_name, egginfo_name):
19  """Extract metadata from filenames.
20 
21  Extracts the 4 metadataitems needed (name, version, pyversion, arch) from
22  the installer filename and the name of the egg-info directory embedded in
23  the zipfile (if any).
24 
25  The egginfo filename has the format::
26 
27  name-ver(-pyver)(-arch).egg-info
28 
29  The installer filename has the format::
30 
31  name-ver.arch(-pyver).exe
32 
33  Some things to note:
34 
35  1. The installer filename is not definitive. An installer can be renamed
36  and work perfectly well as an installer. So more reliable data should
37  be used whenever possible.
38  2. The egg-info data should be preferred for the name and version, because
39  these come straight from the distutils metadata, and are mandatory.
40  3. The pyver from the egg-info data should be ignored, as it is
41  constructed from the version of Python used to build the installer,
42  which is irrelevant - the installer filename is correct here (even to
43  the point that when it's not there, any version is implied).
44  4. The architecture must be taken from the installer filename, as it is
45  not included in the egg-info data.
46  5. Architecture-neutral installers still have an architecture because the
47  installer format itself (being executable) is architecture-specific. We
48  should therefore ignore the architecture if the content is pure-python.
49  """
50 
51  egginfo = None
52  if egginfo_name:
53  egginfo = egg_info_re.search(egginfo_name)
54  if not egginfo:
55  raise ValueError("Egg info filename %s is not valid" %
56  (egginfo_name,))
57 
58  # Parse the wininst filename
59  # 1. Distribution name (up to the first '-')
60  w_name, sep, rest = wininfo_name.partition('-')
61  if not sep:
62  raise ValueError("Installer filename %s is not valid" %
63  (wininfo_name,))
64  # Strip '.exe'
65  rest = rest[:-4]
66  # 2. Python version (from the last '-', must start with 'py')
67  rest2, sep, w_pyver = rest.rpartition('-')
68  if sep and w_pyver.startswith('py'):
69  rest = rest2
70  w_pyver = w_pyver.replace('.', '')
71  else:
72  # Not version specific - use py2.py3. While it is possible that
73  # pure-Python code is not compatible with both Python 2 and 3, there
74  # is no way of knowing from the wininst format, so we assume the best
75  # here (the user can always manually rename the wheel to be more
76  # restrictive if needed).
77  w_pyver = 'py2.py3'
78  # 3. Version and architecture
79  w_ver, sep, w_arch = rest.rpartition('.')
80  if not sep:
81  raise ValueError("Installer filename %s is not valid" %
82  (wininfo_name,))
83 
84  if egginfo:
85  w_name = egginfo.group('name')
86  w_ver = egginfo.group('ver')
87 
88  return dict(name=w_name, ver=w_ver, arch=w_arch, pyver=w_pyver)

95  for filename in bdw.namelist():

97  egginfo_name = filename
98  break
99 
100  info = parse_info(os.path.basename(path), egginfo_name)
101 
102  root_is_purelib = True
103  for zipinfo in bdw.infolist():
104  if zipinfo.filename.startswith('PLATLIB'):
105  root_is_purelib = False
106  break
107  if root_is_purelib:
108  paths = {'purelib': ''}
109  else:
110  paths = {'platlib': ''}
111 
112  dist_info = "%(name)s-%(ver)s" % info
113  datadir = "%s.data/" % dist_info
114 
115  # rewrite paths to trick ZipFile into extracting an egg
116  # XXX grab wininst .ini - between .exe, padding, and first zip file.
117  members = []
118  egginfo_name = ''
119  for zipinfo in bdw.infolist():
120  key, basename = zipinfo.filename.split('/', 1)
121  key = key.lower()
122  basepath = paths.get(key, None)
123  if basepath is None:
124  basepath = datadir + key.lower() + '/'
125  oldname = zipinfo.filename
126  newname = basepath + basename
127  zipinfo.filename = newname
128  del bdw.NameToInfo[oldname]
129  bdw.NameToInfo[newname] = zipinfo
130  # Collect member names, but omit '' (from an entry like "PLATLIB/"
131  if newname:
132  members.append(newname)
133  # Remember egg-info name for the egg2dist call below
134  if not egginfo_name:
135  if newname.endswith('.egg-info'):
136  egginfo_name = newname
137  elif '.egg-info/' in newname:
138  egginfo_name, sep, _ = newname.rpartition('/')
139  dir = tempfile.mkdtemp(suffix="_b2w")
140  bdw.extractall(dir, members)
141 
142  # egg2wheel
143  abi = 'none'
144  pyver = info['pyver']
145  arch = (info['arch'] or 'any').replace('.', '_').replace('-', '_')
146  # Wininst installers always have arch even if they are not
147  # architecture-specific (because the format itself is).
148  # So, assume the content is architecture-neutral if root is purelib.
149  if root_is_purelib:
150  arch = 'any'
151  # If the installer is architecture-specific, it's almost certainly also
152  # CPython-specific.
153  if arch != 'any':
154  pyver = pyver.replace('py', 'cp')
155  wheel_name = '-'.join((
156  dist_info,
157  pyver,
158  abi,
159  arch
160  ))
161  if root_is_purelib:
162  bw = wheel.bdist_wheel.bdist_wheel(distutils.dist.Distribution())
163  else:
164  bw = _bdist_wheel_tag(distutils.dist.Distribution())
165 
166  bw.root_is_pure = root_is_purelib
167  bw.python_tag = pyver
168  bw.plat_name_supplied = True
169  bw.plat_name = info['arch'] or 'any'
170 
171  if not root_is_purelib:
172  bw.full_tag_supplied = True
173  bw.full_tag = (pyver, abi, arch)
174 
175  dist_info_dir = os.path.join(dir, '%s.dist-info' % dist_info)
176  bw.egg2dist(os.path.join(dir, egginfo_name), dist_info_dir)
177  bw.write_wheelfile(dist_info_dir, generator='wininst2wheel')
178  bw.write_record(dir, dist_info_dir)
179 
180  archive_wheelfile(os.path.join(dest_dir, wheel_name), dir)
181  rmtree(dir)
