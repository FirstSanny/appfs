19 def _open_file(file_like, appendmat):
20  ''' Open `file_like` and return as file-like object '''
21  if isinstance(file_like, string_types):
22  try:
23  return open(file_like, 'rb')
24  except IOError as e:
25  if appendmat and not file_like.endswith('.mat'):
26  file_like += '.mat'
27  try:
28  return open(file_like, 'rb')
29  except IOError:
30  pass # Rethrow the original exception.
31  raise
32  # not a string - maybe file-like object
33  try:
34  file_like.read(0)
35  except AttributeError:
36  raise IOError('Reader needs file name or open file-like object')
37  return file_like

41 def mat_reader_factory(file_name, appendmat=True, **kwargs):
42  """
43  Create reader for matlab .mat format files.
44 
45  Parameters
46  ----------
47  %(file_arg)s
48  %(append_arg)s
49  %(load_args)s
50  %(struct_arg)s
51 
52  Returns
53  -------
54  matreader : MatFileReader object
55  Initialized instance of MatFileReader class matching the mat file
56  type detected in `filename`.
57  """
58  byte_stream = _open_file(file_name, appendmat)
59  mjv, mnv = get_matfile_version(byte_stream)
60  if mjv == 0:
61  return MatFile4Reader(byte_stream, **kwargs)
62  elif mjv == 1:
63  return MatFile5Reader(byte_stream, **kwargs)
64  elif mjv == 2:
65  raise NotImplementedError('Please use HDF reader for matlab v7.3 files')
66  else:
67  raise TypeError('Did not recognize version %s' % mjv)

70 @docfiller
71 def loadmat(file_name, mdict=None, appendmat=True, **kwargs):
72  """
73  Load MATLAB file.
74 
75  Parameters
76  ----------
77  file_name : str
78  Name of the mat file (do not need .mat extension if
79  appendmat==True). Can also pass open file-like object.
80  mdict : dict, optional
81  Dictionary in which to insert matfile variables.
82  appendmat : bool, optional
83  True to append the .mat extension to the end of the given
84  filename, if not already present.
85  byte_order : str or None, optional
86  None by default, implying byte order guessed from mat
87  file. Otherwise can be one of ('native', '=', 'little', '<',
88  'BIG', '>').
89  mat_dtype : bool, optional
90  If True, return arrays in same dtype as would be loaded into
91  MATLAB (instead of the dtype with which they are saved).
92  squeeze_me : bool, optional
93  Whether to squeeze unit matrix dimensions or not.
94  chars_as_strings : bool, optional
95  Whether to convert char arrays to string arrays.
96  matlab_compatible : bool, optional
97  Returns matrices as would be loaded by MATLAB (implies
98  squeeze_me=False, chars_as_strings=False, mat_dtype=True,
99  struct_as_record=True).
100  struct_as_record : bool, optional
101  Whether to load MATLAB structs as numpy record arrays, or as
102  old-style numpy arrays with dtype=object. Setting this flag to
103  False replicates the behavior of scipy version 0.7.x (returning
104  numpy object arrays). The default setting is True, because it
105  allows easier round-trip load and save of MATLAB files.
106  verify_compressed_data_integrity : bool, optional
107  Whether the length of compressed sequences in the MATLAB file
108  should be checked, to ensure that they are not longer than we expect.
109  It is advisable to enable this (the default) because overlong
110  compressed sequences in MATLAB files generally indicate that the
111  files have experienced some sort of corruption.
112  variable_names : None or sequence
113  If None (the default) - read all variables in file. Otherwise
114  `variable_names` should be a sequence of strings, giving names of the
115  matlab variables to read from the file. The reader will skip any
116  variable with a name not in this sequence, possibly saving some read
117  processing.
118 
119  Returns
120  -------
121  mat_dict : dict
122  dictionary with variable names as keys, and loaded matrices as
123  values.
124 
125  Notes
126  -----
127  v4 (Level 1.0), v6 and v7 to 7.2 matfiles are supported.
128 
129  You will need an HDF5 python library to read matlab 7.3 format mat
130  files. Because scipy does not supply one, we do not implement the
131  HDF5 / 7.3 interface here.
132 
133  """
134  variable_names = kwargs.pop('variable_names', None)
135  MR = mat_reader_factory(file_name, appendmat, **kwargs)
136  matfile_dict = MR.get_variables(variable_names)
137  if mdict is not None:
138  mdict.update(matfile_dict)
139  else:
140  mdict = matfile_dict
141  if isinstance(file_name, string_types):
142  MR.mat_stream.close()
143  return mdict

146 @docfiller
147 def savemat(file_name, mdict,
148  appendmat=True,
149  format='5',
150  long_field_names=False,
151  do_compression=False,
152  oned_as='row'):
153  """
154  Save a dictionary of names and arrays into a MATLAB-style .mat file.
155 
156  This saves the array objects in the given dictionary to a MATLAB-
157  style .mat file.
158 
159  Parameters
160  ----------
161  file_name : str or file-like object
162  Name of the .mat file (.mat extension not needed if ``appendmat ==
163  True``).
164  Can also pass open file_like object.
165  mdict : dict
166  Dictionary from which to save matfile variables.
167  appendmat : bool, optional
168  True (the default) to append the .mat extension to the end of the
169  given filename, if not already present.
170  format : {'5', '4'}, string, optional
171  '5' (the default) for MATLAB 5 and up (to 7.2),
172  '4' for MATLAB 4 .mat files.
173  long_field_names : bool, optional
174  False (the default) - maximum field name length in a structure is
175  31 characters which is the documented maximum length.
176  True - maximum field name length in a structure is 63 characters
177  which works for MATLAB 7.6+.
178  do_compression : bool, optional
179  Whether or not to compress matrices on write. Default is False.
180  oned_as : {'row', 'column'}, optional
181  If 'column', write 1-D numpy arrays as column vectors.
182  If 'row', write 1-D numpy arrays as row vectors.
183 
184  See also
185  --------
186  mio4.MatFile4Writer
187  mio5.MatFile5Writer
188  """
189  file_is_string = isinstance(file_name, string_types)
190  if file_is_string:
191  if appendmat and file_name[-4:] != ".mat":
192  file_name = file_name + ".mat"
193  file_stream = open(file_name, 'wb')
194  else:
195  if not hasattr(file_name, 'write'):
196  raise IOError('Writer needs file name or writeable '
197  'file-like object')
198  file_stream = file_name
199  if format == '4':
200  if long_field_names:
201  raise ValueError("Long field names are not available for version 4 files")
202  MW = MatFile4Writer(file_stream, oned_as)
203  elif format == '5':
204  MW = MatFile5Writer(file_stream,
205  do_compression=do_compression,
206  unicode_strings=True,
207  long_field_names=long_field_names,
208  oned_as=oned_as)
209  else:
210  raise ValueError("Format should be '4' or '5'")
211  MW.put_variables(mdict)
212  if file_is_string:
213  file_stream.close()

217 def whosmat(file_name, appendmat=True, **kwargs):
218  """
219  List variables inside a MATLAB file.
220 
221  Parameters
222  ----------
223  %(file_arg)s
224  %(append_arg)s
225  %(load_args)s
226  %(struct_arg)s
227 
228  Returns
229  -------
230  variables : list of tuples
231  A list of tuples, where each tuple holds the matrix name (a string),
232  its shape (tuple of ints), and its data class (a string).
233  Possible data classes are: int8, uint8, int16, uint16, int32, uint32,
234  int64, uint64, single, double, cell, struct, object, char, sparse,
235  function, opaque, logical, unknown.
236 
237  Notes
238  -----
239  v4 (Level 1.0), v6 and v7 to 7.2 matfiles are supported.
240 
241  You will need an HDF5 python library to read matlab 7.3 format mat
242  files. Because scipy does not supply one, we do not implement the
243  HDF5 / 7.3 interface here.
244 
245  .. versionadded:: 0.12.0
246 
247  """
248  ML = mat_reader_factory(file_name, **kwargs)
249  variables = ML.list_variables()
250  if isinstance(file_name, string_types):
251  ML.mat_stream.close()
252  return variables
