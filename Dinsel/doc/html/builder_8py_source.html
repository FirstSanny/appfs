90 class AstroidBuilder(raw_building.InspectBuilder):
91  """Class for building an astroid tree from source code or from a live module.
92 
93  The param *manager* specifies the manager class which should be used.
94  If no manager is given, then the default one will be used. The
95  param *apply_transforms* determines if the transforms should be
96  applied after the tree was built from source or from a live object,
97  by default being True.
98  """
99  # pylint: disable=redefined-outer-name
100  def __init__(self, manager=None, apply_transforms=True):
101  super(AstroidBuilder, self).__init__()
102  self._manager = manager or MANAGER
103  self._apply_transforms = apply_transforms

105  def module_build(self, module, modname=None):
106  """Build an astroid from a living module instance."""
107  node = None
108  path = getattr(module, '__file__', None)
109  if path is not None:
110  path_, ext = os.path.splitext(modutils._path_from_filename(path))
111  if ext in ('.py', '.pyc', '.pyo') and os.path.exists(path_ + '.py'):
112  node = self.file_build(path_ + '.py', modname)
113  if node is None:
114  # this is a built-in module
115  # get a partial representation by introspection
116  node = self.inspect_build(module, modname=modname, path=path)
117  if self._apply_transforms:
118  # We have to handle transformation by ourselves since the
119  # rebuilder isn't called for builtin nodes
120  node = self._manager.visit_transforms(node)
121  return node

123  def file_build(self, path, modname=None):
124  """Build astroid from a source code file (i.e. from an ast)
125 
126  *path* is expected to be a python source file
127  """
128  try:
129  stream, encoding, data = open_source_file(path)
130  except IOError as exc:
131  util.reraise(exceptions.AstroidBuildingError(
132  'Unable to load file {path}:\n{error}',
133  modname=modname, path=path, error=exc))
134  except (SyntaxError, LookupError) as exc:
135  util.reraise(exceptions.AstroidSyntaxError(
136  'Python 3 encoding specification error or unknown encoding:\n'
137  '{error}', modname=modname, path=path, error=exc))
138  except UnicodeError: # wrong encoding
139  # detect_encoding returns utf-8 if no encoding specified
140  util.reraise(exceptions.AstroidBuildingError(
141  'Wrong or no encoding specified for {filename}.',
142  filename=path))
143  with stream:
144  # get module name if necessary
145  if modname is None:
146  try:
147  modname = '.'.join(modutils.modpath_from_file(path))
148  except ImportError:
149  modname = os.path.splitext(os.path.basename(path))[0]
150  # build astroid representation
151  module = self._data_build(data, modname, path)
152  return self._post_build(module, encoding)

160  def _post_build(self, module, encoding):
161  """Handles encoding and delayed nodes after a module has been built"""
162  module.file_encoding = encoding
163  self._manager.cache_module(module)
164  # post tree building steps after we stored the module in the cache:
165  for from_node in module._import_from_nodes:
166  if from_node.modname == '__future__':
167  for symbol, _ in from_node.names:
168  module.future_imports.add(symbol)
169  self.add_from_names_to_locals(from_node)
170  # handle delayed assattr nodes
171  for delayed in module._delayed_assattr:
172  self.delayed_assattr(delayed)
173 
174  # Visit the transforms
175  if self._apply_transforms:
176  module = self._manager.visit_transforms(module)
177  return module

179  def _data_build(self, data, modname, path):
180  """Build tree node from data and add some informations"""
181  try:
182  node = _parse(data + '\n')
183  except (TypeError, ValueError, SyntaxError) as exc:
184  util.reraise(exceptions.AstroidSyntaxError(
185  'Parsing Python code failed:\n{error}',
186  source=data, modname=modname, path=path, error=exc))
187  if path is not None:
188  node_file = os.path.abspath(path)
189  else:
190  node_file = '<?>'
191  if modname.endswith('.__init__'):
192  modname = modname[:-9]
193  package = True
194  else:
195  package = path is not None and os.path.splitext(os.path.basename(path))[0] == '__init__'
196  builder = rebuilder.TreeRebuilder(self._manager)
197  module = builder.visit_module(node, modname, node_file, package)
198  module._import_from_nodes = builder._import_from_nodes
199  module._delayed_assattr = builder._delayed_assattr
200  return module

202  def add_from_names_to_locals(self, node):
203  """Store imported names to the locals
204 
205  Resort the locals if coming from a delayed node
206  """
207  _key_func = lambda node: node.fromlineno
208  def sort_locals(my_list):
209  my_list.sort(key=_key_func)
210 
211  for (name, asname) in node.names:
212  if name == '*':
213  try:
214  imported = node.do_import_module()
215  except exceptions.AstroidBuildingError:
216  continue
217  for name in imported.public_names():
218  node.parent.set_local(name, node)
219  sort_locals(node.parent.scope().locals[name])
220  else:
221  node.parent.set_local(asname or name, node)
222  sort_locals(node.parent.scope().locals[asname or name])

224  def delayed_assattr(self, node):
225  """Visit a AssAttr node
226 
227  This adds name to locals and handle members definition.
228  """
229  try:
230  frame = node.frame()
231  for inferred in node.expr.infer():
232  if inferred is util.Uninferable:
233  continue
234  try:
235  if inferred.__class__ is bases.Instance:
236  inferred = inferred._proxied
237  iattrs = inferred.instance_attrs
238  if not _can_assign_attr(inferred, node.attrname):
239  continue
240  elif isinstance(inferred, bases.Instance):
241  # Const, Tuple, ... we may be wrong, may be not, but
242  # anyway we don't want to pollute builtin's namespace
243  continue
244  elif inferred.is_function:
245  iattrs = inferred.instance_attrs
246  else:
247  iattrs = inferred.locals
248  except AttributeError:
249  # XXX log error
250  continue
251  values = iattrs.setdefault(node.attrname, [])
252  if node in values:
253  continue
254  # get assign in __init__ first XXX useful ?
255  if (frame.name == '__init__' and values and
256  values[0].frame().name != '__init__'):
257  values.insert(0, node)
258  else:
259  values.append(node)
260  except exceptions.InferenceError:
261  pass

284 def _extract_expressions(node):
285  """Find expressions in a call to _TRANSIENT_FUNCTION and extract them.
286 
287  The function walks the AST recursively to search for expressions that
288  are wrapped into a call to _TRANSIENT_FUNCTION. If it finds such an
289  expression, it completely removes the function call node from the tree,
290  replacing it by the wrapped expression inside the parent.
291 
292  :param node: An astroid node.
293  :type node: astroid.bases.NodeNG
294  :yields: The sequence of wrapped expressions on the modified tree
295  expression can be found.
296  """
297  if (isinstance(node, nodes.Call)
298  and isinstance(node.func, nodes.Name)
299  and node.func.name == _TRANSIENT_FUNCTION):
300  real_expr = node.args[0]
301  real_expr.parent = node.parent
302  # Search for node in all _astng_fields (the fields checked when
303  # get_children is called) of its parent. Some of those fields may
304  # be lists or tuples, in which case the elements need to be checked.
305  # When we find it, replace it by real_expr, so that the AST looks
306  # like no call to _TRANSIENT_FUNCTION ever took place.
307  for name in node.parent._astroid_fields:
308  child = getattr(node.parent, name)
309  if isinstance(child, (list, tuple)):
310  for idx, compound_child in enumerate(child):
311  if compound_child is node:
312  child[idx] = real_expr
313  elif child is node:
314  setattr(node.parent, name, real_expr)
315  yield real_expr
316  else:
317  for child in node.get_children():
318  for result in _extract_expressions(child):
319  yield result

322 def _find_statement_by_line(node, line):
323  """Extracts the statement on a specific line from an AST.
324 
325  If the line number of node matches line, it will be returned;
326  otherwise its children are iterated and the function is called
327  recursively.
328 
329  :param node: An astroid node.
330  :type node: astroid.bases.NodeNG
331  :param line: The line number of the statement to extract.
332  :type line: int
333  :returns: The statement on the line, or None if no statement for the line
334  can be found.
335  :rtype: astroid.bases.NodeNG or None
336  """
337  if isinstance(node, (nodes.ClassDef, nodes.FunctionDef)):
338  # This is an inaccuracy in the AST: the nodes that can be
339  # decorated do not carry explicit information on which line
340  # the actual definition (class/def), but .fromline seems to
341  # be close enough.
342  node_line = node.fromlineno
343  else:
344  node_line = node.lineno
345 
346  if node_line == line:
347  return node
348 
349  for child in node.get_children():
350  result = _find_statement_by_line(child, line)
351  if result:
352  return result
353 
354  return None

357 def extract_node(code, module_name=''):
358  """Parses some Python code as a module and extracts a designated AST node.
359 
360  Statements:
361  To extract one or more statement nodes, append #@ to the end of the line
362 
363  Examples:
364  >>> def x():
365  >>> def y():
366  >>> return 1 #@
367 
368  The return statement will be extracted.
369 
370  >>> class X(object):
371  >>> def meth(self): #@
372  >>> pass
373 
374  The function object 'meth' will be extracted.
375 
376  Expressions:
377  To extract arbitrary expressions, surround them with the fake
378  function call __(...). After parsing, the surrounded expression
379  will be returned and the whole AST (accessible via the returned
380  node's parent attribute) will look like the function call was
381  never there in the first place.
382 
383  Examples:
384  >>> a = __(1)
385 
386  The const node will be extracted.
387 
388  >>> def x(d=__(foo.bar)): pass
389 
390  The node containing the default argument will be extracted.
391 
392  >>> def foo(a, b):
393  >>> return 0 < __(len(a)) < b
394 
395  The node containing the function call 'len' will be extracted.
396 
397  If no statements or expressions are selected, the last toplevel
398  statement will be returned.
399 
400  If the selected statement is a discard statement, (i.e. an expression
401  turned into a statement), the wrapped expression is returned instead.
402 
403  For convenience, singleton lists are unpacked.
404 
405  :param str code: A piece of Python code that is parsed as
406  a module. Will be passed through textwrap.dedent first.
407  :param str module_name: The name of the module.
408  :returns: The designated node from the parse tree, or a list of nodes.
409  :rtype: astroid.bases.NodeNG, or a list of nodes.
410  """
411  def _extract(node):
412  if isinstance(node, nodes.Expr):
413  return node.value
414 
415  return node
416 
417  requested_lines = []
418  for idx, line in enumerate(code.splitlines()):
419  if line.strip().endswith(_STATEMENT_SELECTOR):
420  requested_lines.append(idx + 1)
421 
422  tree = parse(code, module_name=module_name)
423  extracted = []
424  if requested_lines:
425  for line in requested_lines:
426  extracted.append(_find_statement_by_line(tree, line))
427 
428  # Modifies the tree.
429  extracted.extend(_extract_expressions(tree))
430 
431  if not extracted:
432  extracted.append(tree.body[-1])
433 
434  extracted = [_extract(node) for node in extracted]
435  if len(extracted) == 1:
436  return extracted[0]
437  return extracted
