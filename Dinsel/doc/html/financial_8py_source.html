36 def fv(rate, nper, pmt, pv, when='end'):
37  """
38  Compute the future value.
39 
40  Given:
41  * a present value, `pv`
42  * an interest `rate` compounded once per period, of which
43  there are
44  * `nper` total
45  * a (fixed) payment, `pmt`, paid either
46  * at the beginning (`when` = {'begin', 1}) or the end
47  (`when` = {'end', 0}) of each period
48 
49  Return:
50  the value at the end of the `nper` periods
51 
52  Parameters
53  ----------
54  rate : scalar or array_like of shape(M, )
55  Rate of interest as decimal (not per cent) per period
56  nper : scalar or array_like of shape(M, )
57  Number of compounding periods
58  pmt : scalar or array_like of shape(M, )
59  Payment
60  pv : scalar or array_like of shape(M, )
61  Present value
62  when : {{'begin', 1}, {'end', 0}}, {string, int}, optional
63  When payments are due ('begin' (1) or 'end' (0)).
64  Defaults to {'end', 0}.
65 
66  Returns
67  -------
68  out : ndarray
69  Future values. If all input is scalar, returns a scalar float. If
70  any input is array_like, returns future values for each input element.
71  If multiple inputs are array_like, they all must have the same shape.
72 
73  Notes
74  -----
75  The future value is computed by solving the equation::
76 
77  fv +
78  pv*(1+rate)**nper +
79  pmt*(1 + rate*when)/rate*((1 + rate)**nper - 1) == 0
80 
81  or, when ``rate == 0``::
82 
83  fv + pv + pmt * nper == 0
84 
85  References
86  ----------
87  .. [WRW] Wheeler, D. A., E. Rathke, and R. Weir (Eds.) (2009, May).
88  Open Document Format for Office Applications (OpenDocument)v1.2,
89  Part 2: Recalculated Formula (OpenFormula) Format - Annotated Version,
90  Pre-Draft 12. Organization for the Advancement of Structured Information
91  Standards (OASIS). Billerica, MA, USA. [ODT Document].
92  Available:
93  http://www.oasis-open.org/committees/documents.php?wg_abbrev=office-formula
94  OpenDocument-formula-20090508.odt
95 
96  Examples
97  --------
98  What is the future value after 10 years of saving $100 now, with
99  an additional monthly savings of $100. Assume the interest rate is
100  5% (annually) compounded monthly?
101 
102  >>> np.fv(0.05/12, 10*12, -100, -100)
103  15692.928894335748
104 
105  By convention, the negative sign represents cash flow out (i.e. money not
106  available today). Thus, saving $100 a month at 5% annual interest leads
107  to $15,692.93 available to spend in 10 years.
108 
109  If any input is array_like, returns an array of equal shape. Let's
110  compare different interest rates from the example above.
111 
112  >>> a = np.array((0.05, 0.06, 0.07))/12
113  >>> np.fv(a, 10*12, -100, -100)
114  array([ 15692.92889434, 16569.87435405, 17509.44688102])
115 
116  """
117  when = _convert_when(when)
118  (rate, nper, pmt, pv, when) = map(np.asarray, [rate, nper, pmt, pv, when])
119  temp = (1+rate)**nper
120  miter = np.broadcast(rate, nper, pmt, pv, when)
121  zer = np.zeros(miter.shape)
122  fact = np.where(rate == zer, nper + zer,
123  (1 + rate*when)*(temp - 1)/rate + zer)
124  return -(pv*temp + pmt*fact)

126 def pmt(rate, nper, pv, fv=0, when='end'):
127  """
128  Compute the payment against loan principal plus interest.
129 
130  Given:
131  * a present value, `pv` (e.g., an amount borrowed)
132  * a future value, `fv` (e.g., 0)
133  * an interest `rate` compounded once per period, of which
134  there are
135  * `nper` total
136  * and (optional) specification of whether payment is made
137  at the beginning (`when` = {'begin', 1}) or the end
138  (`when` = {'end', 0}) of each period
139 
140  Return:
141  the (fixed) periodic payment.
142 
143  Parameters
144  ----------
145  rate : array_like
146  Rate of interest (per period)
147  nper : array_like
148  Number of compounding periods
149  pv : array_like
150  Present value
151  fv : array_like, optional
152  Future value (default = 0)
153  when : {{'begin', 1}, {'end', 0}}, {string, int}
154  When payments are due ('begin' (1) or 'end' (0))
155 
156  Returns
157  -------
158  out : ndarray
159  Payment against loan plus interest. If all input is scalar, returns a
160  scalar float. If any input is array_like, returns payment for each
161  input element. If multiple inputs are array_like, they all must have
162  the same shape.
163 
164  Notes
165  -----
166  The payment is computed by solving the equation::
167 
168  fv +
169  pv*(1 + rate)**nper +
170  pmt*(1 + rate*when)/rate*((1 + rate)**nper - 1) == 0
171 
172  or, when ``rate == 0``::
173 
174  fv + pv + pmt * nper == 0
175 
176  for ``pmt``.
177 
178  Note that computing a monthly mortgage payment is only
179  one use for this function. For example, pmt returns the
180  periodic deposit one must make to achieve a specified
181  future balance given an initial deposit, a fixed,
182  periodically compounded interest rate, and the total
183  number of periods.
184 
185  References
186  ----------
187  .. [WRW] Wheeler, D. A., E. Rathke, and R. Weir (Eds.) (2009, May).
188  Open Document Format for Office Applications (OpenDocument)v1.2,
189  Part 2: Recalculated Formula (OpenFormula) Format - Annotated Version,
190  Pre-Draft 12. Organization for the Advancement of Structured Information
191  Standards (OASIS). Billerica, MA, USA. [ODT Document].
192  Available:
193  http://www.oasis-open.org/committees/documents.php
194  ?wg_abbrev=office-formulaOpenDocument-formula-20090508.odt
195 
196  Examples
197  --------
198  What is the monthly payment needed to pay off a $200,000 loan in 15
199  years at an annual interest rate of 7.5%?
200 
201  >>> np.pmt(0.075/12, 12*15, 200000)
202  -1854.0247200054619
203 
204  In order to pay-off (i.e., have a future-value of 0) the $200,000 obtained
205  today, a monthly payment of $1,854.02 would be required. Note that this
206  example illustrates usage of `fv` having a default value of 0.
207 
208  """
209  when = _convert_when(when)
210  (rate, nper, pv, fv, when) = map(np.array, [rate, nper, pv, fv, when])
211  temp = (1 + rate)**nper
212  mask = (rate == 0.0)
213  masked_rate = np.where(mask, 1.0, rate)
214  z = np.zeros(np.broadcast(masked_rate, nper, pv, fv, when).shape)
215  fact = np.where(mask != z, nper + z,
216  (1 + masked_rate*when)*(temp - 1)/masked_rate + z)
217  return -(fv + pv*temp) / fact

219 def nper(rate, pmt, pv, fv=0, when='end'):
220  """
221  Compute the number of periodic payments.
222 
223  Parameters
224  ----------
225  rate : array_like
226  Rate of interest (per period)
227  pmt : array_like
228  Payment
229  pv : array_like
230  Present value
231  fv : array_like, optional
232  Future value
233  when : {{'begin', 1}, {'end', 0}}, {string, int}, optional
234  When payments are due ('begin' (1) or 'end' (0))
235 
236  Notes
237  -----
238  The number of periods ``nper`` is computed by solving the equation::
239 
240  fv + pv*(1+rate)**nper + pmt*(1+rate*when)/rate*((1+rate)**nper-1) = 0
241 
242  but if ``rate = 0`` then::
243 
244  fv + pv + pmt*nper = 0
245 
246  Examples
247  --------
248  If you only had $150/month to pay towards the loan, how long would it take
249  to pay-off a loan of $8,000 at 7% annual interest?
250 
251  >>> print(round(np.nper(0.07/12, -150, 8000), 5))
252  64.07335
253 
254  So, over 64 months would be required to pay off the loan.
255 
256  The same analysis could be done with several different interest rates
257  and/or payments and/or total amounts to produce an entire table.
258 
259  >>> np.nper(*(np.ogrid[0.07/12: 0.08/12: 0.01/12,
260  ... -150 : -99 : 50 ,
261  ... 8000 : 9001 : 1000]))
262  array([[[ 64.07334877, 74.06368256],
263  [ 108.07548412, 127.99022654]],
264  [[ 66.12443902, 76.87897353],
265  [ 114.70165583, 137.90124779]]])
266 
267  """
268  when = _convert_when(when)
269  (rate, pmt, pv, fv, when) = map(np.asarray, [rate, pmt, pv, fv, when])
270 
271  use_zero_rate = False
272  with np.errstate(divide="raise"):
273  try:
274  z = pmt*(1.0+rate*when)/rate
275  except FloatingPointError:
276  use_zero_rate = True
277 
278  if use_zero_rate:
279  return (-fv + pv) / (pmt + 0.0)
280  else:
281  A = -(fv + pv)/(pmt+0.0)
282  B = np.log((-fv+z) / (pv+z))/np.log(1.0+rate)
283  miter = np.broadcast(rate, pmt, pv, fv, when)
284  zer = np.zeros(miter.shape)
285  return np.where(rate == zer, A + zer, B + zer) + 0.0

287 def ipmt(rate, per, nper, pv, fv=0.0, when='end'):
288  """
289  Compute the interest portion of a payment.
290 
291  Parameters
292  ----------
293  rate : scalar or array_like of shape(M, )
294  Rate of interest as decimal (not per cent) per period
295  per : scalar or array_like of shape(M, )
296  Interest paid against the loan changes during the life or the loan.
297  The `per` is the payment period to calculate the interest amount.
298  nper : scalar or array_like of shape(M, )
299  Number of compounding periods
300  pv : scalar or array_like of shape(M, )
301  Present value
302  fv : scalar or array_like of shape(M, ), optional
303  Future value
304  when : {{'begin', 1}, {'end', 0}}, {string, int}, optional
305  When payments are due ('begin' (1) or 'end' (0)).
306  Defaults to {'end', 0}.
307 
308  Returns
309  -------
310  out : ndarray
311  Interest portion of payment. If all input is scalar, returns a scalar
312  float. If any input is array_like, returns interest payment for each
313  input element. If multiple inputs are array_like, they all must have
314  the same shape.
315 
316  See Also
317  --------
318  ppmt, pmt, pv
319 
320  Notes
321  -----
322  The total payment is made up of payment against principal plus interest.
323 
324  ``pmt = ppmt + ipmt``
325 
326  Examples
327  --------
328  What is the amortization schedule for a 1 year loan of $2500 at
329  8.24% interest per year compounded monthly?
330 
331  >>> principal = 2500.00
332 
333  The 'per' variable represents the periods of the loan. Remember that
334  financial equations start the period count at 1!
335 
336  >>> per = np.arange(1*12) + 1
337  >>> ipmt = np.ipmt(0.0824/12, per, 1*12, principal)
338  >>> ppmt = np.ppmt(0.0824/12, per, 1*12, principal)
339 
340  Each element of the sum of the 'ipmt' and 'ppmt' arrays should equal
341  'pmt'.
342 
343  >>> pmt = np.pmt(0.0824/12, 1*12, principal)
344  >>> np.allclose(ipmt + ppmt, pmt)
345  True
346 
347  >>> fmt = '{0:2d} {1:8.2f} {2:8.2f} {3:8.2f}'
348  >>> for payment in per:
349  ... index = payment - 1
350  ... principal = principal + ppmt[index]
351  ... print(fmt.format(payment, ppmt[index], ipmt[index], principal))
352  1 -200.58 -17.17 2299.42
353  2 -201.96 -15.79 2097.46
354  3 -203.35 -14.40 1894.11
355  4 -204.74 -13.01 1689.37
356  5 -206.15 -11.60 1483.22
357  6 -207.56 -10.18 1275.66
358  7 -208.99 -8.76 1066.67
359  8 -210.42 -7.32 856.25
360  9 -211.87 -5.88 644.38
361  10 -213.32 -4.42 431.05
362  11 -214.79 -2.96 216.26
363  12 -216.26 -1.49 -0.00
364 
365  >>> interestpd = np.sum(ipmt)
366  >>> np.round(interestpd, 2)
367  -112.98
368

370  when = _convert_when(when)
371  rate, per, nper, pv, fv, when = np.broadcast_arrays(rate, per, nper,
372  pv, fv, when)
373  total_pmt = pmt(rate, nper, pv, fv, when)
374  ipmt = _rbl(rate, per, total_pmt, pv, when)*rate
375  try:
376  ipmt = np.where(when == 1, ipmt/(1 + rate), ipmt)
377  ipmt = np.where(np.logical_and(when == 1, per == 1), 0.0, ipmt)
378  except IndexError:
379  pass
380  return ipmt

419 def pv(rate, nper, pmt, fv=0.0, when='end'):
420  """
421  Compute the present value.
422 
423  Given:
424  * a future value, `fv`
425  * an interest `rate` compounded once per period, of which
426  there are
427  * `nper` total
428  * a (fixed) payment, `pmt`, paid either
429  * at the beginning (`when` = {'begin', 1}) or the end
430  (`when` = {'end', 0}) of each period
431 
432  Return:
433  the value now
434 
435  Parameters
436  ----------
437  rate : array_like
438  Rate of interest (per period)
439  nper : array_like
440  Number of compounding periods
441  pmt : array_like
442  Payment
443  fv : array_like, optional
444  Future value
445  when : {{'begin', 1}, {'end', 0}}, {string, int}, optional
446  When payments are due ('begin' (1) or 'end' (0))
447 
448  Returns
449  -------
450  out : ndarray, float
451  Present value of a series of payments or investments.
452 
453  Notes
454  -----
455  The present value is computed by solving the equation::
456 
457  fv +
458  pv*(1 + rate)**nper +
459  pmt*(1 + rate*when)/rate*((1 + rate)**nper - 1) = 0
460 
461  or, when ``rate = 0``::
462 
463  fv + pv + pmt * nper = 0
464 
465  for `pv`, which is then returned.
466 
467  References
468  ----------
469  .. [WRW] Wheeler, D. A., E. Rathke, and R. Weir (Eds.) (2009, May).
470  Open Document Format for Office Applications (OpenDocument)v1.2,
471  Part 2: Recalculated Formula (OpenFormula) Format - Annotated Version,
472  Pre-Draft 12. Organization for the Advancement of Structured Information
473  Standards (OASIS). Billerica, MA, USA. [ODT Document].
474  Available:
475  http://www.oasis-open.org/committees/documents.php?wg_abbrev=office-formula
476  OpenDocument-formula-20090508.odt
477 
478  Examples
479  --------
480  What is the present value (e.g., the initial investment)
481  of an investment that needs to total $15692.93
482  after 10 years of saving $100 every month? Assume the
483  interest rate is 5% (annually) compounded monthly.
484 
485  >>> np.pv(0.05/12, 10*12, -100, 15692.93)
486  -100.00067131625819
487 
488  By convention, the negative sign represents cash flow out
489  (i.e., money not available today). Thus, to end up with
490  $15,692.93 in 10 years saving $100 a month at 5% annual
491  interest, one's initial deposit should also be $100.
492 
493  If any input is array_like, ``pv`` returns an array of equal shape.
494  Let's compare different interest rates in the example above:
495 
496  >>> a = np.array((0.05, 0.04, 0.03))/12
497  >>> np.pv(a, 10*12, -100, 15692.93)
498  array([ -100.00067132, -649.26771385, -1273.78633713])
499 
500  So, to end up with the same $15692.93 under the same $100 per month
501  "savings plan," for annual interest rates of 4% and 3%, one would
502  need initial investments of $649.27 and $1273.79, respectively.
503 
504  """
505  when = _convert_when(when)
506  (rate, nper, pmt, fv, when) = map(np.asarray, [rate, nper, pmt, fv, when])
507  temp = (1+rate)**nper
508  miter = np.broadcast(rate, nper, pmt, fv, when)
509  zer = np.zeros(miter.shape)
510  fact = np.where(rate == zer, nper+zer, (1+rate*when)*(temp-1)/rate+zer)
511  return -(fv + pmt*fact)/temp
512 
513 # Computed with Sage
514 # (y + (r + 1)^n*x + p*((r + 1)^n - 1)*(r*w + 1)/r)/(n*(r + 1)^(n - 1)*x -
515 # p*((r + 1)^n - 1)*(r*w + 1)/r^2 + n*p*(r + 1)^(n - 1)*(r*w + 1)/r +
516 # p*((r + 1)^n - 1)*w/r)
517 
518 def _g_div_gp(r, n, p, x, y, w):
519  t1 = (r+1)**n
520  t2 = (r+1)**(n-1)
521  return ((y + t1*x + p*(t1 - 1)*(r*w + 1)/r) /
522  (n*t2*x - p*(t1 - 1)*(r*w + 1)/(r**2) + n*p*t2*(r*w + 1)/r +
523  p*(t1 - 1)*w/r))
524 
525 # Use Newton's iteration until the change is less than 1e-6
526 # for all values or a maximum of 100 iterations is reached.
527 # Newton's rule is
528 # r_{n+1} = r_{n} - g(r_n)/g'(r_n)
529 # where
530 # g(r) is the formula
531 # g'(r) is the derivative with respect to r.
532 def rate(nper, pmt, pv, fv, when='end', guess=0.10, tol=1e-6, maxiter=100):
533  """
534  Compute the rate of interest per period.
535 
536  Parameters
537  ----------
538  nper : array_like
539  Number of compounding periods
540  pmt : array_like
541  Payment
542  pv : array_like
543  Present value
544  fv : array_like
545  Future value
546  when : {{'begin', 1}, {'end', 0}}, {string, int}, optional
547  When payments are due ('begin' (1) or 'end' (0))
548  guess : float, optional
549  Starting guess for solving the rate of interest
550  tol : float, optional
551  Required tolerance for the solution
552  maxiter : int, optional
553  Maximum iterations in finding the solution
554 
555  Notes
556  -----
557  The rate of interest is computed by iteratively solving the
558  (non-linear) equation::
559 
560  fv + pv*(1+rate)**nper + pmt*(1+rate*when)/rate * ((1+rate)**nper - 1) = 0
561 
562  for ``rate``.
563 
564  References
565  ----------
566  Wheeler, D. A., E. Rathke, and R. Weir (Eds.) (2009, May). Open Document
567  Format for Office Applications (OpenDocument)v1.2, Part 2: Recalculated
568  Formula (OpenFormula) Format - Annotated Version, Pre-Draft 12.
569  Organization for the Advancement of Structured Information Standards
570  (OASIS). Billerica, MA, USA. [ODT Document]. Available:
571  http://www.oasis-open.org/committees/documents.php?wg_abbrev=office-formula
572  OpenDocument-formula-20090508.odt
573 
574  """
575  when = _convert_when(when)
576  (nper, pmt, pv, fv, when) = map(np.asarray, [nper, pmt, pv, fv, when])
577  rn = guess
578  iter = 0
579  close = False
580  while (iter < maxiter) and not close:
581  rnp1 = rn - _g_div_gp(rn, nper, pmt, pv, fv, when)
582  diff = abs(rnp1-rn)
583  close = np.all(diff < tol)
584  iter += 1
585  rn = rnp1
586  if not close:
587  # Return nan's in array of the same shape as rn
588  return np.nan + rn
589  else:
590  return rn

592 def irr(values):
593  """
594  Return the Internal Rate of Return (IRR).
595 
596  This is the "average" periodically compounded rate of return
597  that gives a net present value of 0.0; for a more complete explanation,
598  see Notes below.
599 
600  Parameters
601  ----------
602  values : array_like, shape(N,)
603  Input cash flows per time period. By convention, net "deposits"
604  are negative and net "withdrawals" are positive. Thus, for
605  example, at least the first element of `values`, which represents
606  the initial investment, will typically be negative.
607 
608  Returns
609  -------
610  out : float
611  Internal Rate of Return for periodic input values.
612 
613  Notes
614  -----
615  The IRR is perhaps best understood through an example (illustrated
616  using np.irr in the Examples section below). Suppose one invests 100
617  units and then makes the following withdrawals at regular (fixed)
618  intervals: 39, 59, 55, 20. Assuming the ending value is 0, one's 100
619  unit investment yields 173 units; however, due to the combination of
620  compounding and the periodic withdrawals, the "average" rate of return
621  is neither simply 0.73/4 nor (1.73)^0.25-1. Rather, it is the solution
622  (for :math:`r`) of the equation:
623 
624  .. math:: -100 + \\frac{39}{1+r} + \\frac{59}{(1+r)^2}
625  + \\frac{55}{(1+r)^3} + \\frac{20}{(1+r)^4} = 0
626 
627  In general, for `values` :math:`= [v_0, v_1, ... v_M]`,
628  irr is the solution of the equation: [G]_
629 
630  .. math:: \\sum_{t=0}^M{\\frac{v_t}{(1+irr)^{t}}} = 0
631 
632  References
633  ----------
634  .. [G] L. J. Gitman, "Principles of Managerial Finance, Brief," 3rd ed.,
635  Addison-Wesley, 2003, pg. 348.
636 
637  Examples
638  --------
639  >>> round(irr([-100, 39, 59, 55, 20]), 5)
640  0.28095
641  >>> round(irr([-100, 0, 0, 74]), 5)
642  -0.0955
643  >>> round(irr([-100, 100, 0, -7]), 5)
644  -0.0833
645  >>> round(irr([-100, 100, 0, 7]), 5)
646  0.06206
647  >>> round(irr([-5, 10.5, 1, -8, 1]), 5)
648  0.0886
649 
650  (Compare with the Example given for numpy.lib.financial.npv)
651 
652  """
653  res = np.roots(values[::-1])
654  mask = (res.imag == 0) & (res.real > 0)
655  if not mask.any():
656  return np.nan
657  res = res[mask].real
658  # NPV(rate) = 0 can have more than one solution so we return
659  # only the solution closest to zero.
660  rate = 1.0/res - 1
661  rate = rate.item(np.argmin(np.abs(rate)))
662  return rate

664 def npv(rate, values):
665  """
666  Returns the NPV (Net Present Value) of a cash flow series.
667 
668  Parameters
669  ----------
670  rate : scalar
671  The discount rate.
672  values : array_like, shape(M, )
673  The values of the time series of cash flows. The (fixed) time
674  interval between cash flow "events" must be the same as that for
675  which `rate` is given (i.e., if `rate` is per year, then precisely
676  a year is understood to elapse between each cash flow event). By
677  convention, investments or "deposits" are negative, income or
678  "withdrawals" are positive; `values` must begin with the initial
679  investment, thus `values[0]` will typically be negative.
680 
681  Returns
682  -------
683  out : float
684  The NPV of the input cash flow series `values` at the discount
685  `rate`.
686 
687  Notes
688  -----
689  Returns the result of: [G]_
690 
691  .. math :: \\sum_{t=0}^{M-1}{\\frac{values_t}{(1+rate)^{t}}}
692 
693  References
694  ----------
695  .. [G] L. J. Gitman, "Principles of Managerial Finance, Brief," 3rd ed.,
696  Addison-Wesley, 2003, pg. 346.
697 
698  Examples
699  --------
700  >>> np.npv(0.281,[-100, 39, 59, 55, 20])
701  -0.0084785916384548798
702 
703  (Compare with the Example given for numpy.lib.financial.irr)
704 
705  """
706  values = np.asarray(values)
707  return (values / (1+rate)**np.arange(0, len(values))).sum(axis=0)

709 def mirr(values, finance_rate, reinvest_rate):
710  """
711  Modified internal rate of return.
712 
713  Parameters
714  ----------
715  values : array_like
716  Cash flows (must contain at least one positive and one negative
717  value) or nan is returned. The first value is considered a sunk
718  cost at time zero.
719  finance_rate : scalar
720  Interest rate paid on the cash flows
721  reinvest_rate : scalar
722  Interest rate received on the cash flows upon reinvestment
723 
724  Returns
725  -------
726  out : float
727  Modified internal rate of return
728 
729  """
730  values = np.asarray(values, dtype=np.double)
731  n = values.size
732  pos = values > 0
733  neg = values < 0
734  if not (pos.any() and neg.any()):
735  return np.nan
736  numer = np.abs(npv(reinvest_rate, values*pos))
737  denom = np.abs(npv(finance_rate, values*neg))
738  return (numer/denom)**(1.0/(n - 1))*(1 + reinvest_rate) - 1
