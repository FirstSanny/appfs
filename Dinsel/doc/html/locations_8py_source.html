124 def distutils_scheme(dist_name, user=False, home=None, root=None,
125  isolated=False, prefix=None):
126  """
127  Return a distutils install scheme
128  """
129  from distutils.dist import Distribution
130 
131  scheme = {}
132 
133  if isolated:
134  extra_dist_args = {"script_args": ["--no-user-cfg"]}
135  else:
136  extra_dist_args = {}
137  dist_args = {'name': dist_name}
138  dist_args.update(extra_dist_args)
139 
140  d = Distribution(dist_args)
141  d.parse_config_files()
142  i = d.get_command_obj('install', create=True)
143  # NOTE: setting user or home has the side-effect of creating the home dir
144  # or user base for installations during finalize_options()
145  # ideally, we'd prefer a scheme class that has no side-effects.
146  assert not (user and prefix), "user={0} prefix={1}".format(user, prefix)
147  i.user = user or i.user
148  if user:
149  i.prefix = ""
150  i.prefix = prefix or i.prefix
151  i.home = home or i.home
152  i.root = root or i.root
153  i.finalize_options()
154  for key in SCHEME_KEYS:
155  scheme[key] = getattr(i, 'install_' + key)
156 
157  # install_lib specified in setup.cfg should install *everything*
158  # into there (i.e. it takes precedence over both purelib and
159  # platlib). Note, i.install_lib is *always* set after
160  # finalize_options(); we only want to override here if the user
161  # has explicitly requested it hence going back to the config
162  if 'install_lib' in d.get_option_dict('install'):
163  scheme.update(dict(purelib=i.install_lib, platlib=i.install_lib))
164 
165  if running_under_virtualenv():
166  scheme['headers'] = os.path.join(
167  sys.prefix,
168  'include',
169  'site',
170  'python' + sys.version[:3],
171  dist_name,
172  )
173 
174  if root is not None:
175  path_no_drive = os.path.splitdrive(
176  os.path.abspath(scheme["headers"]))[1]
177  scheme["headers"] = os.path.join(
178  root,
179  path_no_drive[1:],
180  )
181 
182  return scheme
