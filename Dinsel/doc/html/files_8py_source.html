53 @contract(returns='unicode')
54 def canonical_filename(filename):
55  """Return a canonical file name for `filename`.
56 
57  An absolute path with no redundant components and normalized case.
58 
59  """
60  if filename not in CANONICAL_FILENAME_CACHE:
61  if not os.path.isabs(filename):
62  for path in [os.curdir] + sys.path:
63  if path is None:
64  continue
65  f = os.path.join(path, filename)
66  try:
67  exists = os.path.exists(f)
68  except UnicodeError:
69  exists = False
70  if exists:
71  filename = f
72  break
73  cf = abs_file(filename)
74  CANONICAL_FILENAME_CACHE[filename] = cf
75  return CANONICAL_FILENAME_CACHE[filename]

193 class TreeMatcher(object):
194  """A matcher for files in a tree.
195 
196  Construct with a list of paths, either files or directories. Paths match
197  with the `match` method if they are one of the files, or if they are
198  somewhere in a subtree rooted at one of the directories.
199 
200  """
201  def __init__(self, paths):
202  self.paths = list(paths)
203 
204  def __repr__(self):
205  return "<TreeMatcher %r>" % self.paths
206 
207  def info(self):
208  """A list of strings for displaying when dumping state."""
209  return self.paths

211  def match(self, fpath):
212  """Does `fpath` indicate a file in one of our trees?"""
213  for p in self.paths:
214  if fpath.startswith(p):
215  if fpath == p:
216  # This is the same file!
217  return True
218  if fpath[len(p)] == os.sep:
219  # This is a file in the directory
220  return True
221  return False

252 class FnmatchMatcher(object):
253  """A matcher for files by file name pattern."""
254  def __init__(self, pats):
255  self.pats = pats[:]
256  # fnmatch is platform-specific. On Windows, it does the Windows thing
257  # of treating / and \ as equivalent. But on other platforms, we need to
258  # take care of that ourselves.
259  fnpats = (fnmatch.translate(p) for p in pats)
260  fnpats = (p.replace(r"\/", r"[\\/]") for p in fnpats)
261  if env.WINDOWS:
262  # Windows is also case-insensitive. BTW: the regex docs say that
263  # flags like (?i) have to be at the beginning, but fnmatch puts
264  # them at the end, and having two there seems to work fine.
265  fnpats = (p + "(?i)" for p in fnpats)
266  self.re = re.compile(join_regex(fnpats))
267 
268  def __repr__(self):
269  return "<FnmatchMatcher %r>" % self.pats
270 
271  def info(self):
272  """A list of strings for displaying when dumping state."""
273  return self.pats
274 
275  def match(self, fpath):
276  """Does `fpath` match one of our file name patterns?"""
277  return self.re.match(fpath) is not None

290 class PathAliases(object):
291  """A collection of aliases for paths.
292 
293  When combining data files from remote machines, often the paths to source
294  code are different, for example, due to OS differences, or because of
295  serialized checkouts on continuous integration machines.
296 
297  A `PathAliases` object tracks a list of pattern/result pairs, and can
298  map a path through those aliases to produce a unified path.
299 
300  """
301  def __init__(self):
302  self.aliases = []
303 
304  def add(self, pattern, result):
305  """Add the `pattern`/`result` pair to the list of aliases.
306 
307  `pattern` is an `fnmatch`-style pattern. `result` is a simple
308  string. When mapping paths, if a path starts with a match against
309  `pattern`, then that match is replaced with `result`. This models
310  isomorphic source trees being rooted at different places on two
311  different machines.
312 
313  `pattern` can't end with a wildcard component, since that would
314  match an entire tree, and not just its root.
315 
316  """
317  # The pattern can't end with a wildcard component.
318  pattern = pattern.rstrip(r"\/")
319  if pattern.endswith("*"):
320  raise CoverageException("Pattern must not end with wildcards.")
321  pattern_sep = sep(pattern)
322 
323  # The pattern is meant to match a filepath. Let's make it absolute
324  # unless it already is, or is meant to match any prefix.
325  if not pattern.startswith('*') and not isabs_anywhere(pattern):
326  pattern = abs_file(pattern)
327  pattern += pattern_sep
328 
329  # Make a regex from the pattern. fnmatch always adds a \Z to
330  # match the whole string, which we don't want, so we remove the \Z.
331  # While removing it, we only replace \Z if followed by paren, or at
332  # end, to keep from destroying a literal \Z in the pattern.
333  regex_pat = fnmatch.translate(pattern)
334  regex_pat = re.sub(r'\\Z(\(|$)', r'\1', regex_pat)
335 
336  # We want */a/b.py to match on Windows too, so change slash to match
337  # either separator.
338  regex_pat = regex_pat.replace(r"\/", r"[\\/]")
339  # We want case-insensitive matching, so add that flag.
340  regex = re.compile(r"(?i)" + regex_pat)
341 
342  # Normalize the result: it must end with a path separator.
343  result_sep = sep(result)
344  result = result.rstrip(r"\/") + result_sep
345  self.aliases.append((regex, result, pattern_sep, result_sep))
346 
347  def map(self, path):
348  """Map `path` through the aliases.
349 
350  `path` is checked against all of the patterns. The first pattern to
351  match is used to replace the root of the path with the result root.
352  Only one pattern is ever used. If no patterns match, `path` is
353  returned unchanged.
354 
355  The separator style in the result is made to match that of the result
356  in the alias.
357 
358  Returns the mapped path. If a mapping has happened, this is a
359  canonical path. If no mapping has happened, it is the original value
360  of `path` unchanged.
361 
362  """
363  for regex, result, pattern_sep, result_sep in self.aliases:
364  m = regex.match(path)
365  if m:
366  new = path.replace(m.group(0), result)
367  if pattern_sep != result_sep:
368  new = new.replace(pattern_sep, result_sep)
369  new = canonical_filename(new)
370  return new
371  return path

374 def find_python_files(dirname):
375  """Yield all of the importable Python files in `dirname`, recursively.
376 
377  To be importable, the files have to be in a directory with a __init__.py,
378  except for `dirname` itself, which isn't required to have one. The
379  assumption is that `dirname` was specified directly, so the user knows
380  best, but sub-directories are checked for a __init__.py to be sure we only
381  find the importable files.
382 
383  """
384  for i, (dirpath, dirnames, filenames) in enumerate(os.walk(dirname)):
385  if i > 0 and '__init__.py' not in filenames:
386  # If a directory doesn't have __init__.py, then it isn't
387  # importable and neither are its files
388  del dirnames[:]
389  continue
390  for filename in filenames:
391  # We're only interested in files that look like reasonable Python
392  # files: Must end with .py or .pyw, and must not have certain funny
393  # characters that probably mean they are editor junk.
394  if re.match(r"^[^.#~!$@%^&*()+=,]+\.pyw?$", filename):
395  yield os.path.join(dirpath, filename)
