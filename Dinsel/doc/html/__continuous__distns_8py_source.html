327 def _beta_mle_a(a, b, n, s1):
328  # The zeros of this function give the MLE for `a`, with
329  # `b`, `n` and `s1` given. `s1` is the sum of the logs of
330  # the data. `n` is the number of data points.
331  psiab = sc.psi(a + b)
332  func = s1 - n * (-psiab + sc.psi(a))
333  return func

336 def _beta_mle_ab(theta, n, s1, s2):
337  # Zeros of this function are critical points of
338  # the maximum likelihood function. Solving this system
339  # for theta (which contains a and b) gives the MLE for a and b
340  # given `n`, `s1` and `s2`. `s1` is the sum of the logs of the data,
341  # and `s2` is the sum of the logs of 1 - data. `n` is the number
342  # of data points.
343  a, b = theta
344  psiab = sc.psi(a + b)
345  func = [s1 - n * (-psiab + sc.psi(a)),
346  s2 - n * (-psiab + sc.psi(b))]
347  return func

352

354 
355  Notes
356  -----
357  The probability density function for `beta` is::
358 
359  gamma(a+b) * x**(a-1) * (1-x)**(b-1)
360  beta.pdf(x, a, b) = ------------------------------------
361  gamma(a)*gamma(b)

363  for ``0 < x < 1``, ``a > 0``, ``b > 0``, where ``gamma(z)`` is the gamma
364  function (`scipy.special.gamma`).
365 
366  `beta` takes ``a`` and ``b`` as shape parameters.
367 
368  %(after_notes)s
369 
370  %(example)s
371 
372  """

377  return np.exp(self._logpdf(x, a, b))
378 
379  def _logpdf(self, x, a, b):
380  lPx = sc.xlog1py(b - 1.0, -x) + sc.xlogy(a - 1.0, x)
381  lPx -= sc.betaln(a, b)
382  return lPx

392  var = (a*b*1.0)/(a+b+1.0)/(a+b)**2.0
393  g1 = 2.0*(b-a)*np.sqrt((1.0+a+b)/(a*b)) / (2+a+b)
394  g2 = 6.0*(a**3 + a**2*(1-2*b) + b**2*(1+b) - 2*a*b*(2+b))

412  @inherit_docstring_from(rv_continuous)
413  def fit(self, data, *args, **kwds):
414  """%(super)s
415  In the special case where both `floc` and `fscale` are given, a
416  `ValueError` is raised if any value `x` in `data` does not satisfy
417  `floc < x < floc + fscale`.
418  """
419  # Override rv_continuous.fit, so we can more efficiently handle the
420  # case where floc and fscale are given.
421 
422  f0 = (kwds.get('f0', None) or kwds.get('fa', None) or
423  kwds.get('fix_a', None))
424  f1 = (kwds.get('f1', None) or kwds.get('fb', None) or
425  kwds.get('fix_b', None))
426  floc = kwds.get('floc', None)
427  fscale = kwds.get('fscale', None)
428 
429  if floc is None or fscale is None:
430  # do general fit
431  return super(beta_gen, self).fit(data, *args, **kwds)
432 
433  if f0 is not None and f1 is not None:
434  # This check is for consistency with `rv_continuous.fit`.
435  raise ValueError("All parameters fixed. There is nothing to "
436  "optimize.")
437 
438  # Special case: loc and scale are constrained, so we are fitting
439  # just the shape parameters. This can be done much more efficiently
440  # than the method used in `rv_continuous.fit`. (See the subsection
441  # "Two unknown parameters" in the section "Maximum likelihood" of
442  # the Wikipedia article on the Beta distribution for the formulas.)
443 
444  # Normalize the data to the interval [0, 1].
445  data = (np.ravel(data) - floc) / fscale
446  if np.any(data <= 0) or np.any(data >= 1):
447  raise FitDataError("beta", lower=floc, upper=floc + fscale)
448  xbar = data.mean()
449 
450  if f0 is not None or f1 is not None:
451  # One of the shape parameters is fixed.
452 
453  if f0 is not None:
454  # The shape parameter a is fixed, so swap the parameters
455  # and flip the data. We always solve for `a`. The result
456  # will be swapped back before returning.
457  b = f0
458  data = 1 - data
459  xbar = 1 - xbar
460  else:
461  b = f1
462 
463  # Initial guess for a. Use the formula for the mean of the beta
464  # distribution, E[x] = a / (a + b), to generate a reasonable
465  # starting point based on the mean of the data and the given
466  # value of b.
467  a = b * xbar / (1 - xbar)
468 
469  # Compute the MLE for `a` by solving _beta_mle_a.
470  theta, info, ier, mesg = optimize.fsolve(
471  _beta_mle_a, a,
472  args=(b, len(data), np.log(data).sum()),
473  full_output=True
474  )
475  if ier != 1:
476  raise FitSolverError(mesg=mesg)
477  a = theta[0]
478 
479  if f0 is not None:
480  # The shape parameter a was fixed, so swap back the
481  # parameters.
482  a, b = b, a
483 
484  else:
485  # Neither of the shape parameters is fixed.
486 
487  # s1 and s2 are used in the extra arguments passed to _beta_mle_ab
488  # by optimize.fsolve.
489  s1 = np.log(data).sum()
490  s2 = sc.log1p(-data).sum()
491 
492  # Use the "method of moments" to estimate the initial
493  # guess for a and b.
494  fac = xbar * (1 - xbar) / data.var(ddof=0) - 1
495  a = xbar * fac
496  b = (1 - xbar) * fac
497 
498  # Compute the MLE for a and b by solving _beta_mle_ab.
499  theta, info, ier, mesg = optimize.fsolve(
500  _beta_mle_ab, [a, b],
501  args=(len(data), s1, s2),
502  full_output=True
503  )
504  if ier != 1:
505  raise FitSolverError(mesg=mesg)
506  a, b = theta
507 
508  return a, b, floc, fscale
