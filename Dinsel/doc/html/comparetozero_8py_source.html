23  """Checks for comparisons to zero.

25  An exception to this rule is when 0 is allowed in the program and has a

41  @utils.check_messages('compare-to-zero')
42  def visit_compare(self, node):
43  _operators = ['!=', '==', 'is not', 'is']
44  # note: astroid.Compare has the left most operand in node.left
45  # while the rest are a list of tuples in node.ops
46  # the format of the tuple is ('compare operator sign', node)
47  # here we squash everything into `ops` to make it easier for processing later
48  ops = [('', node.left)]
49  ops.extend(node.ops)
50  ops = list(itertools.chain(*ops))
51 
52  for ops_idx in range(len(ops) - 2):
53  op_1 = ops[ops_idx]
54  op_2 = ops[ops_idx + 1]
55  op_3 = ops[ops_idx + 2]
56  error_detected = False
57 
58  # 0 ?? X
59  if _is_constant_zero(op_1) and op_2 in _operators + ['<']:
60  error_detected = True
61  # X ?? 0
62  elif op_2 in _operators + ['>'] and _is_constant_zero(op_3):
63  error_detected = True
64 
65  if error_detected:
66  self.add_message('compare-to-zero', node=node)
