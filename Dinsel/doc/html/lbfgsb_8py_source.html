370 class LbfgsInvHessProduct(LinearOperator):
371  """Linear operator for the L-BFGS approximate inverse Hessian.
372 
373  This operator computes the product of a vector with the approximate inverse
374  of the Hessian of the objective function, using the L-BFGS limited
375  memory approximation to the inverse Hessian, accumulated during the
376  optimization.
377 
378  Objects of this class implement the ``scipy.sparse.linalg.LinearOperator``
379  interface.
380 
381  Parameters
382  ----------
383  sk : array_like, shape=(n_corr, n)
384  Array of `n_corr` most recent updates to the solution vector.
385  (See [1]).
386  yk : array_like, shape=(n_corr, n)
387  Array of `n_corr` most recent updates to the gradient. (See [1]).
388 
389  References
390  ----------
391  .. [1] Nocedal, Jorge. "Updating quasi-Newton matrices with limited
392  storage." Mathematics of computation 35.151 (1980): 773-782.
393 
394  """
395  def __init__(self, sk, yk):
396  """Construct the operator."""
397  if sk.shape != yk.shape or sk.ndim != 2:
398  raise ValueError('sk and yk must have matching shape, (n_corrs, n)')
399  n_corrs, n = sk.shape
400 
401  super(LbfgsInvHessProduct, self).__init__(
402  dtype=np.float64, shape=(n, n))
403 
404  self.sk = sk
405  self.yk = yk
406  self.n_corrs = n_corrs
407  self.rho = 1 / np.einsum('ij,ij->i', sk, yk)
408 
409  def _matvec(self, x):
410  """Efficient matrix-vector multiply with the BFGS matrices.
411 
412  This calculation is described in Section (4) of [1].
413 
414  Parameters
415  ----------
416  x : ndarray
417  An array with shape (n,) or (n,1).
418 
419  Returns
420  -------
421  y : ndarray
422  The matrix-vector product
423 
424  """
425  s, y, n_corrs, rho = self.sk, self.yk, self.n_corrs, self.rho
426  q = np.array(x, dtype=self.dtype, copy=True)
427  if q.ndim == 2 and q.shape[1] == 1:
428  q = q.reshape(-1)
429 
430  alpha = np.zeros(n_corrs)
431 
432  for i in range(n_corrs-1, -1, -1):
433  alpha[i] = rho[i] * np.dot(s[i], q)
434  q = q - alpha[i]*y[i]
435 
436  r = q
437  for i in range(n_corrs):
438  beta = rho[i] * np.dot(y[i], r)
439  r = r + s[i] * (alpha[i] - beta)
440 
441  return r

443  def todense(self):
444  """Return a dense array representation of this operator.
445 
446  Returns
447  -------
448  arr : ndarray, shape=(n, n)
449  An array with the same shape and containing
450  the same data represented by this `LinearOperator`.
451 
452  """
453  s, y, n_corrs, rho = self.sk, self.yk, self.n_corrs, self.rho
454  I = np.eye(*self.shape, dtype=self.dtype)
455  Hk = I
456 
457  for i in range(n_corrs):
458  A1 = I - s[i][:, np.newaxis] * y[i][np.newaxis, :] * rho[i]
459  A2 = I - y[i][:, np.newaxis] * s[i][np.newaxis, :] * rho[i]
460 
461  Hk = np.dot(A1, np.dot(Hk, A2)) + (rho[i] * s[i][:, np.newaxis] *
462  s[i][np.newaxis, :])
