141 def interpret_distro_name(
142  location, basename, metadata, py_version=None, precedence=SOURCE_DIST,
143  platform=None
144  ):
145  """Generate alternative interpretations of a source distro name
146 
147  Note: if `location` is a filesystem filename, you should call
148  ``pkg_resources.normalize_path()`` on it before passing it to this
149  routine!
150  """
151  # Generate alternative interpretations of a source distro name
152  # Because some packages are ambiguous as to name/versions split
153  # e.g. "adns-python-1.1.0", "egenix-mx-commercial", etc.
154  # So, we generate each possible interepretation (e.g. "adns, python-1.1.0"
155  # "adns-python, 1.1.0", and "adns-python-1.1.0, no version"). In practice,
156  # the spurious interpretations should be ignored, because in the event
157  # there's also an "adns" package, the spurious "python-1.1.0" version will
158  # compare lower than any numeric version number, and is therefore unlikely
159  # to match a request for it. It's still a potential problem, though, and
160  # in the long run PyPI and the distutils should go for "safe" names and
161  # versions in distribution archive names (sdist and bdist).
162 
163  parts = basename.split('-')
164  if not py_version and any(re.match(r'py\d\.\d$', p) for p in parts[2:]):
165  # it is a bdist_dumb, not an sdist -- bail out
166  return
167 
168  for p in range(1, len(parts) + 1):
169  yield Distribution(
170  location, metadata, '-'.join(parts[:p]), '-'.join(parts[p:]),
171  py_version=py_version, precedence=precedence,
172  platform=platform
173  )

177 def unique_everseen(iterable, key=None):
178  "List unique elements, preserving order. Remember all elements ever seen."
179  # unique_everseen('AAAABBBCCDAABBB') --> A B C D
180  # unique_everseen('ABBCcAD', str.lower) --> A B C D
181  seen = set()
182  seen_add = seen.add
183  if key is None:
184  for element in six.moves.filterfalse(seen.__contains__, iterable):
185  seen_add(element)

187  else:
188  for element in iterable:
189  k = key(element)

544  def download(self, spec, tmpdir):
545  """Locate and/or download `spec` to `tmpdir`, returning a local path
546 
547  `spec` may be a ``Requirement`` object, or a string containing a URL,
548  an existing local filename, or a project/version requirement spec
549  (i.e. the string form of a ``Requirement`` object). If it is the URL
550  of a .py file with an unambiguous ``#egg=name-version`` tag (i.e., one
551  that escapes ``-`` as ``_`` throughout), a trivial ``setup.py`` is
552  automatically created alongside the downloaded file.
553 
554  If `spec` is a ``Requirement`` object or a string containing a
555  project/version requirement spec, this method returns the location of
556  a matching distribution (possibly after downloading it to `tmpdir`).
557  If `spec` is a locally existing file or directory name, it is simply
558  returned unchanged. If `spec` is a URL, it is downloaded to a subpath
559  of `tmpdir`, and the local filename is returned. Various errors may be
560  raised if a problem occurs during downloading.
561  """
562  if not isinstance(spec, Requirement):
563  scheme = URL_SCHEME(spec)
564  if scheme:
565  # It's a url, download it to tmpdir
566  found = self._download_url(scheme.group(1), spec, tmpdir)
567  base, fragment = egg_info_for_url(spec)
568  if base.endswith('.py'):
569  found = self.gen_setup(found, fragment, tmpdir)
570  return found
571  elif os.path.exists(spec):
572  # Existing file or directory, just return it
573  return spec
574  else:
575  spec = parse_requirement_arg(spec)
576  return getattr(self.fetch_distribution(spec, tmpdir), 'location', None)

578  def fetch_distribution(
579  self, requirement, tmpdir, force_scan=False, source=False,
580  develop_ok=False, local_index=None
581  ):
582  """Obtain a distribution suitable for fulfilling `requirement`
583 
584  `requirement` must be a ``pkg_resources.Requirement`` instance.
585  If necessary, or if the `force_scan` flag is set, the requirement is
586  searched for in the (online) package index as well as the locally
587  installed packages. If a distribution matching `requirement` is found,
588  the returned distribution's ``location`` is the value you would have
589  gotten from calling the ``download()`` method with the matching
590  distribution's URL or filename. If no matching distribution is found,
591  ``None`` is returned.
592 
593  If the `source` flag is set, only source distributions and source
594  checkout links will be considered. Unless the `develop_ok` flag is
595  set, development and system eggs (i.e., those using the ``.egg-info``
596  format) will be ignored.
597  """
598  # process a Requirement
599  self.info("Searching for %s", requirement)
600  skipped = {}
601  dist = None
602 
603  def find(req, env=None):
604  if env is None:
605  env = self
606  # Find a matching distribution; may be called more than once
607 
608  for dist in env[req.key]:
609 
610  if dist.precedence == DEVELOP_DIST and not develop_ok:
611  if dist not in skipped:
612  self.warn("Skipping development or system egg: %s", dist)
613  skipped[dist] = 1
614  continue
615 
616  if dist in req and (dist.precedence <= SOURCE_DIST or not source):
617  dist.download_location = self.download(dist.location, tmpdir)
618  if os.path.exists(dist.download_location):
619  return dist
620 
621  if force_scan:
622  self.prescan()
623  self.find_packages(requirement)
624  dist = find(requirement)
625 
626  if not dist and local_index is not None:
627  dist = find(requirement, local_index)
628 
629  if dist is None:
630  if self.to_scan is not None:
631  self.prescan()
632  dist = find(requirement)
633 
634  if dist is None and not force_scan:
635  self.find_packages(requirement)

957 def _encode_auth(auth):
958  """
959  A function compatible with Python 2.3-3.3 that will encode
960  auth from a URL suitable for an HTTP header.
961  >>> str(_encode_auth('username%3Apassword'))
962  'dXNlcm5hbWU6cGFzc3dvcmQ='
