180 def check_discrete_chisquare(distfn, arg, rvs, alpha, msg):
181  """Perform chisquare test for random sample of a discrete distribution
182 
183  Parameters
184  ----------
185  distname : string
186  name of distribution function
187  arg : sequence
188  parameters of distribution
189  alpha : float
190  significance level, threshold for p-value
191 
192  Returns
193  -------
194  result : bool
195  0 if test passes, 1 if test fails
196 
197  """
198  wsupp = 0.05
199 
200  # construct intervals with minimum mass `wsupp`.
201  # intervals are left-half-open as in a cdf difference
202  lo = int(max(distfn.a, -1000))
203  distsupport = xrange(lo, int(min(distfn.b, 1000)) + 1)
204  last = 0
205  distsupp = [lo]
206  distmass = []
207  for ii in distsupport:
208  current = distfn.cdf(ii, *arg)
209  if current - last >= wsupp - 1e-14:
210  distsupp.append(ii)
211  distmass.append(current - last)
212  last = current
213  if current > (1 - wsupp):
214  break
215  if distsupp[-1] < distfn.b:
216  distsupp.append(distfn.b)
217  distmass.append(1 - last)
218  distsupp = np.array(distsupp)
219  distmass = np.array(distmass)
220 
221  # convert intervals to right-half-open as required by histogram
222  histsupp = distsupp + 1e-8
223  histsupp[0] = distfn.a
224 
225  # find sample frequencies and perform chisquare test
226  freq, hsupp = np.histogram(rvs, histsupp)
227  chis, pval = stats.chisquare(np.array(freq), len(rvs)*distmass)
228 
229  npt.assert_(pval > alpha,
230  'chisquare - test for %s at arg = %s with pval = %s' %
231  (msg, str(arg), str(pval)))
