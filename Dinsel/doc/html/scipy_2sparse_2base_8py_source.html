342  def __mul__(self, other):
343  """interpret other and call one of the following
344 
345  self._mul_scalar()
346  self._mul_vector()
347  self._mul_multivector()
348  self._mul_sparse_matrix()
349  """
350 
351  M, N = self.shape
352 
353  if other.__class__ is np.ndarray:
354  # Fast path for the most common case
355  if other.shape == (N,):
356  return self._mul_vector(other)
357  elif other.shape == (N, 1):
358  return self._mul_vector(other.ravel()).reshape(M, 1)
359  elif other.ndim == 2 and other.shape[0] == N:
360  return self._mul_multivector(other)
361 
362  if isscalarlike(other):
363  # scalar value
364  return self._mul_scalar(other)
365 
366  if issparse(other):
367  if self.shape[1] != other.shape[0]:
368  raise ValueError('dimension mismatch')
369  return self._mul_sparse_matrix(other)
370 
371  # If it's a list or whatever, treat it like a matrix
372  other_a = np.asanyarray(other)
373 
374  if other_a.ndim == 0 and other_a.dtype == np.object_:
375  # Not interpretable as an array; return NotImplemented so that
376  # other's __rmul__ can kick in if that's implemented.
377  return NotImplemented
378 
379  try:
380  other.shape
381  except AttributeError:
382  other = other_a
383 
384  if other.ndim == 1 or other.ndim == 2 and other.shape[1] == 1:
385  # dense row or column vector
386  if other.shape != (N,) and other.shape != (N, 1):
387  raise ValueError('dimension mismatch')
388 
389  result = self._mul_vector(np.ravel(other))
390 
391  if isinstance(other, np.matrix):
392  result = np.asmatrix(result)
393 
394  if other.ndim == 2 and other.shape[1] == 1:
395  # If 'other' was an (nx1) column vector, reshape the result
396  result = result.reshape(-1, 1)
397 
398  return result
399 
400  elif other.ndim == 2:
401 
403 
404  if other.shape[0] != self.shape[1]:
405  raise ValueError('dimension mismatch')
406 
407  result = self._mul_multivector(np.asarray(other))
408 
409  if isinstance(other, np.matrix):
410  result = np.asmatrix(result)
411 
412  return result

654  def getcol(self, j):
655  """Returns a copy of column j of the matrix, as an (m x 1) sparse

657  """
658  # Spmatrix subclasses should override this method for efficiency.
659  # Post-multiply by a (n x 1) column vector 'a' containing all zeros
660  # except for a_j = 1
