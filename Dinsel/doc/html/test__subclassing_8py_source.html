96 # Also a subclass that overrides __str__, __repr__ and __setitem__, disallowing
97 # setting to non-class values (and thus np.ma.core.masked_print_option)
98 # and overrides __array_wrap__, updating the info dict, to check that this
99 # doesn't get destroyed by MaskedArray._update_from. But this one also needs
100 # its own iterator...
101 class CSAIterator(object):
102  """
103  Flat iterator object that uses its own setter/getter
104  (works around ndarray.flat not propagating subclass setters/getters
105  see https://github.com/numpy/numpy/issues/4564)
106  roughly following MaskedIterator
107  """
108  def __init__(self, a):
109  self._original = a
110  self._dataiter = a.view(np.ndarray).flat
111 
112  def __iter__(self):
113  return self
114 
115  def __getitem__(self, indx):
116  out = self._dataiter.__getitem__(indx)
117  if not isinstance(out, np.ndarray):
118  out = out.__array__()
119  out = out.view(type(self._original))
120  return out
121 
122  def __setitem__(self, index, value):
123  self._dataiter[index] = self._original._validate_input(value)
124 
125  def __next__(self):
126  return next(self._dataiter).__array__().view(type(self._original))
127 
128  next = __next__

131 class ComplicatedSubArray(SubArray):
132 
133  def __str__(self):
134  return 'myprefix {0} mypostfix'.format(self.view(SubArray))
135 
136  def __repr__(self):
137  # Return a repr that does not start with 'name('
138  return '<{0} {1}>'.format(self.__class__.__name__, self)
139 
140  def _validate_input(self, value):
141  if not isinstance(value, ComplicatedSubArray):
142  raise ValueError("Can only set to MySubArray values")
143  return value
144 
145  def __setitem__(self, item, value):
146  # validation ensures direct assignment with ndarray or
147  # masked_print_option will fail
148  super(ComplicatedSubArray, self).__setitem__(
149  item, self._validate_input(value))
150 
151  def __getitem__(self, item):
152  # ensure getter returns our own class also for scalars
153  value = super(ComplicatedSubArray, self).__getitem__(item)
154  if not isinstance(value, np.ndarray): # scalar
155  value = value.__array__().view(ComplicatedSubArray)
156  return value
157 
158  @property
159  def flat(self):
160  return CSAIterator(self)
161 
162  @flat.setter
163  def flat(self, value):
164  y = self.ravel()
165  y[:] = value
166 
167  def __array_wrap__(self, obj, context=None):
168  obj = super(ComplicatedSubArray, self).__array_wrap__(obj, context)
169  if context is not None and context[0] is np.multiply:
170  obj.info['multiplied'] = obj.info.get('multiplied', 0) + 1
171 
172  return obj
