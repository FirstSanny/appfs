87  xp = linspace(-8, 13, 100)

89  xp_zeros[np.logical_or(xp_zeros < 0., xp_zeros > 4.)] = 0
90  xp_clip = xp.copy()
91  xp_clip[xp_clip < x[0]] = x[0]
92  xp_clip[xp_clip > x[-1]] = x[-1]
93 
94  for cls in [UnivariateSpline, InterpolatedUnivariateSpline]:
95  spl = cls(x=x, y=y)
96  for ext in [0, 'extrapolate']:
97  assert_allclose(spl(xp, ext=ext), xp**3, atol=1e-16)
98  assert_allclose(cls(x, y, ext=ext)(xp), xp**3, atol=1e-16)
99  for ext in [1, 'zeros']:
100  assert_allclose(spl(xp, ext=ext), xp_zeros**3, atol=1e-16)
101  assert_allclose(cls(x, y, ext=ext)(xp), xp_zeros**3, atol=1e-16)
102  for ext in [2, 'raise']:
103  assert_raises(ValueError, spl, xp, **dict(ext=ext))
104  for ext in [3, 'const']:
105  assert_allclose(spl(xp, ext=ext), xp_clip**3, atol=1e-16)
106  assert_allclose(cls(x, y, ext=ext)(xp), xp_clip**3, atol=1e-16)

477 def _numdiff_2d(func, x, y, dx=0, dy=0, eps=1e-8):
478  if dx == 0 and dy == 0:
479  return func(x, y)
480  elif dx == 1 and dy == 0:
481  return (func(x + eps, y) - func(x - eps, y)) / (2*eps)
482  elif dx == 0 and dy == 1:
483  return (func(x, y + eps) - func(x, y - eps)) / (2*eps)
484  elif dx == 1 and dy == 1:
485  return (func(x + eps, y + eps) - func(x - eps, y + eps)
486  - func(x + eps, y - eps) + func(x - eps, y - eps)) / (2*eps)**2
487  else:
488  raise ValueError("invalid derivative order")
