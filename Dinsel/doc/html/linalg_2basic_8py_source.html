25 def solve(a, b, sym_pos=False, lower=False, overwrite_a=False,
26  overwrite_b=False, debug=None, check_finite=True, assume_a='gen',
27  transposed=False):
28  """
29  Solves the linear equation set ``a * x = b`` for the unknown ``x``
30  for square ``a`` matrix.
31 
32  If the data matrix is known to be a particular type then supplying the
33  corresponding string to ``assume_a`` key chooses the dedicated solver.
34  The available options are
35 
36  =================== ========
37  generic matrix 'gen'
38  symmetric 'sym'
39  hermitian 'her'
40  positive definite 'pos'
41  =================== ========
42 
43  If omitted, ``'gen'`` is the default structure.
44 
45  The datatype of the arrays define which solver is called regardless
46  of the values. In other words, even when the complex array entries have
47  precisely zero imaginary parts, the complex solver will be called based
48  on the data type of the array.
49 
50  Parameters
51  ----------
52  a : (N, N) array_like
53  Square input data
54  b : (N, NRHS) array_like
55  Input data for the right hand side.
56  sym_pos : bool, optional
57  Assume `a` is symmetric and positive definite. This key is deprecated
58  and assume_a = 'pos' keyword is recommended instead. The functionality
59  is the same. It will be removed in the future.
60  lower : bool, optional
61  If True, only the data contained in the lower triangle of `a`. Default
62  is to use upper triangle. (ignored for ``'gen'``)
63  overwrite_a : bool, optional
64  Allow overwriting data in `a` (may enhance performance).
65  Default is False.
66  overwrite_b : bool, optional
67  Allow overwriting data in `b` (may enhance performance).
68  Default is False.
69  check_finite : bool, optional
70  Whether to check that the input matrices contain only finite numbers.
71  Disabling may give a performance gain, but may result in problems
72  (crashes, non-termination) if the inputs do contain infinities or NaNs.
73  assume_a : str, optional
74  Valid entries are explained above.
75  transposed: bool, optional
76  If True, depending on the data type ``a^T x = b`` or ``a^H x = b`` is
77  solved (only taken into account for ``'gen'``).
78 
79  Returns
80  -------
81  x : (N, NRHS) ndarray
82  The solution array.
83 
84  Raises
85  ------
86  ValueError
87  If size mismatches detected or input a is not square.
88  LinAlgError
89  If the matrix is singular.
90  RuntimeWarning
91  If an ill-conditioned input a is detected.
92 
93  Examples
94  --------
95  Given `a` and `b`, solve for `x`:
96 
97  >>> a = np.array([[3, 2, 0], [1, -1, 0], [0, 5, 1]])
98  >>> b = np.array([2, 4, -1])
99  >>> from scipy import linalg
100  >>> x = linalg.solve(a, b)
101  >>> x
102  array([ 2., -2., 9.])
103  >>> np.dot(a, x) == b
104  array([ True, True, True], dtype=bool)
105 
106  Notes
107  -----
108  If the input b matrix is a 1D array with N elements, when supplied
109  together with an NxN input a, it is assumed as a valid column vector
110  despite the apparent size mismatch. This is compatible with the
111  numpy.dot() behavior and the returned result is still 1D array.
112 
113  The generic, symmetric, hermitian and positive definite solutions are
114  obtained via calling ?GESVX, ?SYSVX, ?HESVX, and ?POSVX routines of
115  LAPACK respectively.
116  """
117  # Flags for 1D or nD right hand side
118  b_is_1D = False
119  b_is_ND = False
120 
121  a1 = atleast_2d(_asarray_validated(a, check_finite=check_finite))
122  b1 = atleast_1d(_asarray_validated(b, check_finite=check_finite))
123  n = a1.shape[0]
124 
125  overwrite_a = overwrite_a or _datacopied(a1, a)
126  overwrite_b = overwrite_b or _datacopied(b1, b)
127 
128  if a1.shape[0] != a1.shape[1]:
129  raise ValueError('Input a needs to be a square matrix.')
130 
131  if n != b1.shape[0]:
132  # Last chance to catch 1x1 scalar a and 1D b arrays
133  if not (n == 1 and b1.size != 0):
134  raise ValueError('Input b has to have same number of rows as '
135  'input a')
136 
137  # accomodate empty arrays
138  if b1.size == 0:
139  return np.asfortranarray(b1.copy())
140 
141  # regularize 1D b arrays to 2D and catch nD RHS arrays
142  if b1.ndim == 1:
143  if n == 1:
144  b1 = b1[None, :]
145  else:
146  b1 = b1[:, None]
147  b_is_1D = True
148  elif b1.ndim > 2:
149  b_is_ND = True
150 
151  r_or_c = complex if np.iscomplexobj(a1) else float
152 
153  if assume_a in ('gen', 'sym', 'her', 'pos'):
154  _structure = assume_a
155  else:
156  raise ValueError('{} is not a recognized matrix structure'
157  ''.format(assume_a))
158 
159  # Deprecate keyword "debug"
160  if debug is not None:
161  warnings.warn('Use of the "debug" keyword is deprecated '
162  'and this keyword will be removed in the future '
163  'versions of SciPy.', DeprecationWarning)
164 
165  # Backwards compatibility - old keyword.
166  if sym_pos:
167  assume_a = 'pos'
168 
169  if _structure == 'gen':
170  gesvx = get_lapack_funcs('gesvx', (a1, b1))
171  trans_conj = 'N'
172  if transposed:
173  trans_conj = 'T' if r_or_c is float else 'H'
174  (_, _, _, _, _, _, _,
175  x, rcond, _, _, info) = gesvx(a1, b1,
176  trans=trans_conj,
177  overwrite_a=overwrite_a,
178  overwrite_b=overwrite_b
179  )
180  elif _structure == 'sym':
181  sysvx, sysvx_lw = get_lapack_funcs(('sysvx', 'sysvx_lwork'), (a1, b1))
182  lwork = _compute_lwork(sysvx_lw, n, lower)
183  _, _, _, _, x, rcond, _, _, info = sysvx(a1, b1, lwork=lwork,
184  lower=lower,
185  overwrite_a=overwrite_a,
186  overwrite_b=overwrite_b
187  )
188  elif _structure == 'her':
189  hesvx, hesvx_lw = get_lapack_funcs(('hesvx', 'hesvx_lwork'), (a1, b1))
190  lwork = _compute_lwork(hesvx_lw, n, lower)
191  _, _, x, rcond, _, _, info = hesvx(a1, b1, lwork=lwork,
192  lower=lower,
193  overwrite_a=overwrite_a,
194  overwrite_b=overwrite_b
195  )
196  else:
197  posvx = get_lapack_funcs('posvx', (a1, b1))
198  _, _, _, _, _, x, rcond, _, _, info = posvx(a1, b1,
199  lower=lower,
200  overwrite_a=overwrite_a,
201  overwrite_b=overwrite_b
202  )
203 
204  # Unlike ?xxSV, ?xxSVX writes the solution x to a separate array, and
205  # overwrites b with its scaled version which is thrown away. Thus, the
206  # solution does not admit the same shape with the original b. For
207  # backwards compatibility, we reshape it manually.
208  if b_is_1D:
209  x = x.ravel()
210  if b_is_ND:
211  x = x.reshape(*b1.shape, order='F')
212 
213  if info < 0:
214  raise ValueError('LAPACK reported an illegal value in {}-th argument'
215  '.'.format(-info))
216  elif info == 0:
217  return x
218  elif 0 < info <= n:
219  raise LinAlgError('Matrix is singular.')
220  elif info > n:
221  warnings.warn('scipy.linalg.solve\nIll-conditioned matrix detected.'
222  ' Result is not guaranteed to be accurate.\nReciprocal'
223  ' condition number: {}'.format(rcond), RuntimeWarning)
224  return x

227 def solve_triangular(a, b, trans=0, lower=False, unit_diagonal=False,
228  overwrite_b=False, debug=None, check_finite=True):
229  """
230  Solve the equation `a x = b` for `x`, assuming a is a triangular matrix.
231 
232  Parameters
233  ----------
234  a : (M, M) array_like
235  A triangular matrix
236  b : (M,) or (M, N) array_like
237  Right-hand side matrix in `a x = b`
238  lower : bool, optional
239  Use only data contained in the lower triangle of `a`.
240  Default is to use upper triangle.
241  trans : {0, 1, 2, 'N', 'T', 'C'}, optional
242  Type of system to solve:
243 
244  ======== =========
245  trans system
246  ======== =========
247  0 or 'N' a x = b
248  1 or 'T' a^T x = b
249  2 or 'C' a^H x = b
250  ======== =========
251  unit_diagonal : bool, optional
252  If True, diagonal elements of `a` are assumed to be 1 and
253  will not be referenced.
254  overwrite_b : bool, optional
255  Allow overwriting data in `b` (may enhance performance)
256  check_finite : bool, optional
257  Whether to check that the input matrices contain only finite numbers.
258  Disabling may give a performance gain, but may result in problems
259  (crashes, non-termination) if the inputs do contain infinities or NaNs.
260 
261  Returns
262  -------
263  x : (M,) or (M, N) ndarray
264  Solution to the system `a x = b`. Shape of return matches `b`.
265 
266  Raises
267  ------
268  LinAlgError
269  If `a` is singular
270 
271  Notes
272  -----
273  .. versionadded:: 0.9.0
274 
275  """
276 
277  # Deprecate keyword "debug"
278  if debug is not None:
279  warnings.warn('Use of the "debug" keyword is deprecated '
280  'and this keyword will be removed in the future '
281  'versions of SciPy.', DeprecationWarning)
282 
283  a1 = _asarray_validated(a, check_finite=check_finite)
284  b1 = _asarray_validated(b, check_finite=check_finite)
285  if len(a1.shape) != 2 or a1.shape[0] != a1.shape[1]:
286  raise ValueError('expected square matrix')
287  if a1.shape[0] != b1.shape[0]:
288  raise ValueError('incompatible dimensions')
289  overwrite_b = overwrite_b or _datacopied(b1, b)
290  if debug:
291  print('solve:overwrite_b=', overwrite_b)
292  trans = {'N': 0, 'T': 1, 'C': 2}.get(trans, trans)
293  trtrs, = get_lapack_funcs(('trtrs',), (a1, b1))
294  x, info = trtrs(a1, b1, overwrite_b=overwrite_b, lower=lower,
295  trans=trans, unitdiag=unit_diagonal)
296 
297  if info == 0:
298  return x
299  if info > 0:
300  raise LinAlgError("singular matrix: resolution failed at diagonal %d" %
301  (info-1))
302  raise ValueError('illegal value in %d-th argument of internal trtrs' %
303  (-info))

306 def solve_banded(l_and_u, ab, b, overwrite_ab=False, overwrite_b=False,
307  debug=None, check_finite=True):
308  """
309  Solve the equation a x = b for x, assuming a is banded matrix.
310 
311  The matrix a is stored in `ab` using the matrix diagonal ordered form::
312 
313  ab[u + i - j, j] == a[i,j]
314 
315  Example of `ab` (shape of a is (6,6), `u` =1, `l` =2)::
316 
317  * a01 a12 a23 a34 a45
318  a00 a11 a22 a33 a44 a55
319  a10 a21 a32 a43 a54 *
320  a20 a31 a42 a53 * *
321 
322  Parameters
323  ----------
324  (l, u) : (integer, integer)
325  Number of non-zero lower and upper diagonals
326  ab : (`l` + `u` + 1, M) array_like
327  Banded matrix
328  b : (M,) or (M, K) array_like
329  Right-hand side
330  overwrite_ab : bool, optional
331  Discard data in `ab` (may enhance performance)
332  overwrite_b : bool, optional
333  Discard data in `b` (may enhance performance)
334  check_finite : bool, optional
335  Whether to check that the input matrices contain only finite numbers.
336  Disabling may give a performance gain, but may result in problems
337  (crashes, non-termination) if the inputs do contain infinities or NaNs.
338 
339  Returns
340  -------
341  x : (M,) or (M, K) ndarray
342  The solution to the system a x = b. Returned shape depends on the
343  shape of `b`.
344 
345  """
346 
347  # Deprecate keyword "debug"
348  if debug is not None:
349  warnings.warn('Use of the "debug" keyword is deprecated '
350  'and this keyword will be removed in the future '
351  'versions of SciPy.', DeprecationWarning)
352 
353  a1 = _asarray_validated(ab, check_finite=check_finite, as_inexact=True)
354  b1 = _asarray_validated(b, check_finite=check_finite, as_inexact=True)
355  # Validate shapes.
356  if a1.shape[-1] != b1.shape[0]:
357  raise ValueError("shapes of ab and b are not compatible.")
358  (l, u) = l_and_u
359  if l + u + 1 != a1.shape[0]:
360  raise ValueError("invalid values for the number of lower and upper "
361  "diagonals: l+u+1 (%d) does not equal ab.shape[0] "
362  "(%d)" % (l+u+1, ab.shape[0]))
363 
364  overwrite_b = overwrite_b or _datacopied(b1, b)
365  if a1.shape[-1] == 1:
366  b2 = np.array(b1, copy=(not overwrite_b))
367  b2 /= a1[1, 0]
368  return b2
369  if l == u == 1:
370  overwrite_ab = overwrite_ab or _datacopied(a1, ab)
371  gtsv, = get_lapack_funcs(('gtsv',), (a1, b1))
372  du = a1[0, 1:]
373  d = a1[1, :]
374  dl = a1[2, :-1]
375  du2, d, du, x, info = gtsv(dl, d, du, b1, overwrite_ab, overwrite_ab,
376  overwrite_ab, overwrite_b)
377  else:
378  gbsv, = get_lapack_funcs(('gbsv',), (a1, b1))
379  a2 = np.zeros((2*l+u+1, a1.shape[1]), dtype=gbsv.dtype)
380  a2[l:, :] = a1
381  lu, piv, x, info = gbsv(l, u, a2, b1, overwrite_ab=True,
382  overwrite_b=overwrite_b)
383  if info == 0:
384  return x
385  if info > 0:
386  raise LinAlgError("singular matrix")
387  raise ValueError('illegal value in %d-th argument of internal '
388  'gbsv/gtsv' % -info)

391 def solveh_banded(ab, b, overwrite_ab=False, overwrite_b=False, lower=False,
392  check_finite=True):
393  """
394  Solve equation a x = b. a is Hermitian positive-definite banded matrix.
395 
396  The matrix a is stored in `ab` either in lower diagonal or upper
397  diagonal ordered form:
398 
399  ab[u + i - j, j] == a[i,j] (if upper form; i <= j)
400  ab[ i - j, j] == a[i,j] (if lower form; i >= j)
401 
402  Example of `ab` (shape of a is (6, 6), `u` =2)::
403 
404  upper form:
405  * * a02 a13 a24 a35
406  * a01 a12 a23 a34 a45
407  a00 a11 a22 a33 a44 a55
408 
409  lower form:
410  a00 a11 a22 a33 a44 a55
411  a10 a21 a32 a43 a54 *
412  a20 a31 a42 a53 * *
413 
414  Cells marked with * are not used.
415 
416  Parameters
417  ----------
418  ab : (`u` + 1, M) array_like
419  Banded matrix
420  b : (M,) or (M, K) array_like
421  Right-hand side
422  overwrite_ab : bool, optional
423  Discard data in `ab` (may enhance performance)
424  overwrite_b : bool, optional
425  Discard data in `b` (may enhance performance)
426  lower : bool, optional
427  Is the matrix in the lower form. (Default is upper form)
428  check_finite : bool, optional
429  Whether to check that the input matrices contain only finite numbers.
430  Disabling may give a performance gain, but may result in problems
431  (crashes, non-termination) if the inputs do contain infinities or NaNs.
432 
433  Returns
434  -------
435  x : (M,) or (M, K) ndarray
436  The solution to the system a x = b. Shape of return matches shape
437  of `b`.
438 
439  """
440  a1 = _asarray_validated(ab, check_finite=check_finite)
441  b1 = _asarray_validated(b, check_finite=check_finite)
442  # Validate shapes.
443  if a1.shape[-1] != b1.shape[0]:
444  raise ValueError("shapes of ab and b are not compatible.")
445 
446  overwrite_b = overwrite_b or _datacopied(b1, b)
447  overwrite_ab = overwrite_ab or _datacopied(a1, ab)
448 
449  if a1.shape[0] == 2:
450  ptsv, = get_lapack_funcs(('ptsv',), (a1, b1))
451  if lower:
452  d = a1[0, :].real
453  e = a1[1, :-1]
454  else:
455  d = a1[1, :].real
456  e = a1[0, 1:].conj()
457  d, du, x, info = ptsv(d, e, b1, overwrite_ab, overwrite_ab,
458  overwrite_b)
459  else:
460  pbsv, = get_lapack_funcs(('pbsv',), (a1, b1))
461  c, x, info = pbsv(a1, b1, lower=lower, overwrite_ab=overwrite_ab,
462  overwrite_b=overwrite_b)
463  if info > 0:
464  raise LinAlgError("%d-th leading minor not positive definite" % info)
465  if info < 0:
466  raise ValueError('illegal value in %d-th argument of internal '
467  'pbsv' % -info)
468  return x

471 def solve_toeplitz(c_or_cr, b, check_finite=True):
472  """Solve a Toeplitz system using Levinson Recursion
473 
474  The Toeplitz matrix has constant diagonals, with c as its first column
475  and r as its first row. If r is not given, ``r == conjugate(c)`` is
476  assumed.
477 
478  Parameters
479  ----------
480  c_or_cr : array_like or tuple of (array_like, array_like)
481  The vector ``c``, or a tuple of arrays (``c``, ``r``). Whatever the
482  actual shape of ``c``, it will be converted to a 1-D array. If not
483  supplied, ``r = conjugate(c)`` is assumed; in this case, if c[0] is
484  real, the Toeplitz matrix is Hermitian. r[0] is ignored; the first row
485  of the Toeplitz matrix is ``[c[0], r[1:]]``. Whatever the actual shape
486  of ``r``, it will be converted to a 1-D array.
487  b : (M,) or (M, K) array_like
488  Right-hand side in ``T x = b``.
489  check_finite : bool, optional
490  Whether to check that the input matrices contain only finite numbers.
491  Disabling may give a performance gain, but may result in problems
492  (result entirely NaNs) if the inputs do contain infinities or NaNs.
493 
494  Returns
495  -------
496  x : (M,) or (M, K) ndarray
497  The solution to the system ``T x = b``. Shape of return matches shape
498  of `b`.
499 
500  Notes
501  -----
502  The solution is computed using Levinson-Durbin recursion, which is faster
503  than generic least-squares methods, but can be less numerically stable.
504  """
505  # If numerical stability of this algorithm is a problem, a future
506  # developer might consider implementing other O(N^2) Toeplitz solvers,
507  # such as GKO (http://www.jstor.org/stable/2153371) or Bareiss.
508  if isinstance(c_or_cr, tuple):
509  c, r = c_or_cr
510  c = _asarray_validated(c, check_finite=check_finite).ravel()
511  r = _asarray_validated(r, check_finite=check_finite).ravel()
512  else:
513  c = _asarray_validated(c_or_cr, check_finite=check_finite).ravel()
514  r = c.conjugate()
515 
516  # Form a 1D array of values to be used in the matrix, containing a reversed
517  # copy of r[1:], followed by c.
518  vals = np.concatenate((r[-1:0:-1], c))
519  if b is None:
520  raise ValueError('illegal value, `b` is a required argument')
521 
522  b = _asarray_validated(b)
523  if vals.shape[0] != (2*b.shape[0] - 1):
524  raise ValueError('incompatible dimensions')
525  if np.iscomplexobj(vals) or np.iscomplexobj(b):
526  vals = np.asarray(vals, dtype=np.complex128, order='c')
527  b = np.asarray(b, dtype=np.complex128)
528  else:
529  vals = np.asarray(vals, dtype=np.double, order='c')
530  b = np.asarray(b, dtype=np.double)
531 
532  if b.ndim == 1:
533  x, _ = levinson(vals, np.ascontiguousarray(b))
534  else:
535  b_shape = b.shape
536  b = b.reshape(b.shape[0], -1)
537  x = np.column_stack(
538  (levinson(vals, np.ascontiguousarray(b[:, i]))[0])
539  for i in range(b.shape[1]))
540  x = x.reshape(*b_shape)
541 
542  return x

554 def solve_circulant(c, b, singular='raise', tol=None,
555  caxis=-1, baxis=0, outaxis=0):
556  """Solve C x = b for x, where C is a circulant matrix.
557 
558  `C` is the circulant matrix associated with the vector `c`.
559 
560  The system is solved by doing division in Fourier space. The
561  calculation is::
562 
563  x = ifft(fft(b) / fft(c))
564 
565  where `fft` and `ifft` are the fast Fourier transform and its inverse,
566  respectively. For a large vector `c`, this is *much* faster than
567  solving the system with the full circulant matrix.
568 
569  Parameters
570  ----------
571  c : array_like
572  The coefficients of the circulant matrix.
573  b : array_like
574  Right-hand side matrix in ``a x = b``.
575  singular : str, optional
576  This argument controls how a near singular circulant matrix is
577  handled. If `singular` is "raise" and the circulant matrix is
578  near singular, a `LinAlgError` is raised. If `singular` is
579  "lstsq", the least squares solution is returned. Default is "raise".
580  tol : float, optional
581  If any eigenvalue of the circulant matrix has an absolute value
582  that is less than or equal to `tol`, the matrix is considered to be
583  near singular. If not given, `tol` is set to::
584 
585  tol = abs_eigs.max() * abs_eigs.size * np.finfo(np.float64).eps
586 
587  where `abs_eigs` is the array of absolute values of the eigenvalues
588  of the circulant matrix.
589  caxis : int
590  When `c` has dimension greater than 1, it is viewed as a collection
591  of circulant vectors. In this case, `caxis` is the axis of `c` that
592  holds the vectors of circulant coefficients.
593  baxis : int
594  When `b` has dimension greater than 1, it is viewed as a collection
595  of vectors. In this case, `baxis` is the axis of `b` that holds the
596  right-hand side vectors.
597  outaxis : int
598  When `c` or `b` are multidimensional, the value returned by
599  `solve_circulant` is multidimensional. In this case, `outaxis` is
600  the axis of the result that holds the solution vectors.
601 
602  Returns
603  -------
604  x : ndarray
605  Solution to the system ``C x = b``.
606 
607  Raises
608  ------
609  LinAlgError
610  If the circulant matrix associated with `c` is near singular.
611 
612  See Also
613  --------
614  circulant
615 
616  Notes
617  -----
618  For a one-dimensional vector `c` with length `m`, and an array `b`
619  with shape ``(m, ...)``,
620 
621  solve_circulant(c, b)
622 
623  returns the same result as
624 
625  solve(circulant(c), b)
626 
627  where `solve` and `circulant` are from `scipy.linalg`.
628 
629  .. versionadded:: 0.16.0
630 
631  Examples
632  --------
633  >>> from scipy.linalg import solve_circulant, solve, circulant, lstsq
634 
635  >>> c = np.array([2, 2, 4])
636  >>> b = np.array([1, 2, 3])
637  >>> solve_circulant(c, b)
638  array([ 0.75, -0.25, 0.25])
639 
640  Compare that result to solving the system with `scipy.linalg.solve`:
641 
642  >>> solve(circulant(c), b)
643  array([ 0.75, -0.25, 0.25])
644 
645  A singular example:
646 
647  >>> c = np.array([1, 1, 0, 0])
648  >>> b = np.array([1, 2, 3, 4])
649 
650  Calling ``solve_circulant(c, b)`` will raise a `LinAlgError`. For the
651  least square solution, use the option ``singular='lstsq'``:
652 
653  >>> solve_circulant(c, b, singular='lstsq')
654  array([ 0.25, 1.25, 2.25, 1.25])
655 
656  Compare to `scipy.linalg.lstsq`:
657 
658  >>> x, resid, rnk, s = lstsq(circulant(c), b)
659  >>> x
660  array([ 0.25, 1.25, 2.25, 1.25])
661 
662  A broadcasting example:
663 
664  Suppose we have the vectors of two circulant matrices stored in an array
665  with shape (2, 5), and three `b` vectors stored in an array with shape
666  (3, 5). For example,
667 
668  >>> c = np.array([[1.5, 2, 3, 0, 0], [1, 1, 4, 3, 2]])
669  >>> b = np.arange(15).reshape(-1, 5)
670 
671  We want to solve all combinations of circulant matrices and `b` vectors,
672  with the result stored in an array with shape (2, 3, 5). When we
673  disregard the axes of `c` and `b` that hold the vectors of coefficients,
674  the shapes of the collections are (2,) and (3,), respectively, which are
675  not compatible for broadcasting. To have a broadcast result with shape
676  (2, 3), we add a trivial dimension to `c`: ``c[:, np.newaxis, :]`` has
677  shape (2, 1, 5). The last dimension holds the coefficients of the
678  circulant matrices, so when we call `solve_circulant`, we can use the
679  default ``caxis=-1``. The coefficients of the `b` vectors are in the last
680  dimension of the array `b`, so we use ``baxis=-1``. If we use the
681  default `outaxis`, the result will have shape (5, 2, 3), so we'll use
682  ``outaxis=-1`` to put the solution vectors in the last dimension.
683 
684  >>> x = solve_circulant(c[:, np.newaxis, :], b, baxis=-1, outaxis=-1)
685  >>> x.shape
686  (2, 3, 5)
687  >>> np.set_printoptions(precision=3) # For compact output of numbers.
688  >>> x
689  array([[[-0.118, 0.22 , 1.277, -0.142, 0.302],
690  [ 0.651, 0.989, 2.046, 0.627, 1.072],
691  [ 1.42 , 1.758, 2.816, 1.396, 1.841]],
692  [[ 0.401, 0.304, 0.694, -0.867, 0.377],
693  [ 0.856, 0.758, 1.149, -0.412, 0.831],
694  [ 1.31 , 1.213, 1.603, 0.042, 1.286]]])
695 
696  Check by solving one pair of `c` and `b` vectors (cf. ``x[1, 1, :]``):
697 
698  >>> solve_circulant(c[1], b[1, :])
699  array([ 0.856, 0.758, 1.149, -0.412, 0.831])
700 
701  """
702  c = np.atleast_1d(c)
703  nc = _get_axis_len("c", c, caxis)
704  b = np.atleast_1d(b)
705  nb = _get_axis_len("b", b, baxis)
706  if nc != nb:
707  raise ValueError('Incompatible c and b axis lengths')
708 
709  fc = np.fft.fft(np.rollaxis(c, caxis, c.ndim), axis=-1)
710  abs_fc = np.abs(fc)
711  if tol is None:
712  # This is the same tolerance as used in np.linalg.matrix_rank.
713  tol = abs_fc.max(axis=-1) * nc * np.finfo(np.float64).eps
714  if tol.shape != ():
715  tol.shape = tol.shape + (1,)
716  else:
717  tol = np.atleast_1d(tol)
718 
719  near_zeros = abs_fc <= tol
720  is_near_singular = np.any(near_zeros)
721  if is_near_singular:
722  if singular == 'raise':
723  raise LinAlgError("near singular circulant matrix.")
724  else:
725  # Replace the small values with 1 to avoid errors in the
726  # division fb/fc below.
727  fc[near_zeros] = 1
728 
729  fb = np.fft.fft(np.rollaxis(b, baxis, b.ndim), axis=-1)
730 
731  q = fb / fc
732 
733  if is_near_singular:
734  # `near_zeros` is a boolean array, same shape as `c`, that is
735  # True where `fc` is (near) zero. `q` is the broadcasted result
736  # of fb / fc, so to set the values of `q` to 0 where `fc` is near
737  # zero, we use a mask that is the broadcast result of an array
738  # of True values shaped like `b` with `near_zeros`.
739  mask = np.ones_like(b, dtype=bool) & near_zeros
740  q[mask] = 0
741 
742  x = np.fft.ifft(q, axis=-1)
743  if not (np.iscomplexobj(c) or np.iscomplexobj(b)):
744  x = x.real
745  if outaxis != -1:
746  x = np.rollaxis(x, -1, outaxis)
747  return x

892 def lstsq(a, b, cond=None, overwrite_a=False, overwrite_b=False,
893  check_finite=True, lapack_driver=None):
894  """
895  Compute least-squares solution to equation Ax = b.
896 
897  Compute a vector x such that the 2-norm ``|b - A x|`` is minimized.
898 
899  Parameters
900  ----------
901  a : (M, N) array_like
902  Left hand side matrix (2-D array).
903  b : (M,) or (M, K) array_like
904  Right hand side matrix or vector (1-D or 2-D array).
905  cond : float, optional
906  Cutoff for 'small' singular values; used to determine effective
907  rank of a. Singular values smaller than
908  ``rcond * largest_singular_value`` are considered zero.
909  overwrite_a : bool, optional
910  Discard data in `a` (may enhance performance). Default is False.
911  overwrite_b : bool, optional
912  Discard data in `b` (may enhance performance). Default is False.
913  check_finite : bool, optional
914  Whether to check that the input matrices contain only finite numbers.
915  Disabling may give a performance gain, but may result in problems
916  (crashes, non-termination) if the inputs do contain infinities or NaNs.
917  lapack_driver: str, optional
918  Which LAPACK driver is used to solve the least-squares problem.
919  Options are ``'gelsd'``, ``'gelsy'``, ``'gelss'``. Default
920  (``'gelsd'``) is a good choice. However, ``'gelsy'`` can be slightly
921  faster on many problems. ``'gelss'`` was used historically. It is
922  generally slow but uses less memory.
923 
924  .. versionadded:: 0.17.0
925 
926  Returns
927  -------
928  x : (N,) or (N, K) ndarray
929  Least-squares solution. Return shape matches shape of `b`.
930  residues : () or (1,) or (K,) ndarray
931  Sums of residues, squared 2-norm for each column in ``b - a x``.
932  If rank of matrix a is ``< N`` or ``> M``, or ``'gelsy'`` is used,
933  this is an empty array. If b was 1-D, this is an (1,) shape array,
934  otherwise the shape is (K,).
935  rank : int
936  Effective rank of matrix `a`.
937  s : (min(M,N),) ndarray or None
938  Singular values of `a`. The condition number of a is
939  ``abs(s[0] / s[-1])``. None is returned when ``'gelsy'`` is used.
940 
941  Raises
942  ------
943  LinAlgError
944  If computation does not converge.
945 
946  ValueError
947  When parameters are wrong.
948 
949  See Also
950  --------
951  optimize.nnls : linear least squares with non-negativity constraint
952 
953  """
954  a1 = _asarray_validated(a, check_finite=check_finite)
955  b1 = _asarray_validated(b, check_finite=check_finite)
956  if len(a1.shape) != 2:
957  raise ValueError('expected matrix')
958  m, n = a1.shape
959  if len(b1.shape) == 2:
960  nrhs = b1.shape[1]
961  else:
962  nrhs = 1
963  if m != b1.shape[0]:
964  raise ValueError('incompatible dimensions')
965 
966  driver = lapack_driver
967  if driver is None:
968  driver = lstsq.default_lapack_driver
969  if driver not in ('gelsd', 'gelsy', 'gelss'):
970  raise ValueError('LAPACK driver "%s" is not found' % driver)
971 
972  lapack_func, lapack_lwork = get_lapack_funcs((driver,
973  '%s_lwork' % driver),
974  (a1, b1))
975  real_data = True if (lapack_func.dtype.kind == 'f') else False
976 
977  if m < n:
978  # need to extend b matrix as it will be filled with
979  # a larger solution matrix
980  if len(b1.shape) == 2:
981  b2 = np.zeros((n, nrhs), dtype=lapack_func.dtype)
982  b2[:m, :] = b1
983  else:
984  b2 = np.zeros(n, dtype=lapack_func.dtype)
985  b2[:m] = b1
986  b1 = b2
987 
988  overwrite_a = overwrite_a or _datacopied(a1, a)
989  overwrite_b = overwrite_b or _datacopied(b1, b)
990 
991  if cond is None:
992  cond = np.finfo(lapack_func.dtype).eps
993 
994  if driver in ('gelss', 'gelsd'):
995  if driver == 'gelss':
996  lwork = _compute_lwork(lapack_lwork, m, n, nrhs, cond)
997  v, x, s, rank, work, info = lapack_func(a1, b1, cond, lwork,
998  overwrite_a=overwrite_a,
999  overwrite_b=overwrite_b)
1000 
1001  elif driver == 'gelsd':
1002  if real_data:
1003  lwork, iwork = _compute_lwork(lapack_lwork, m, n, nrhs, cond)
1004  if iwork == 0:
1005  # this is LAPACK bug 0038: dgelsd does not provide the
1006  # size of the iwork array in query mode. This bug was
1007  # fixed in LAPACK 3.2.2, released July 21, 2010.
1008  mesg = ("internal gelsd driver lwork query error, "
1009  "required iwork dimension not returned. "
1010  "This is likely the result of LAPACK bug "
1011  "0038, fixed in LAPACK 3.2.2 (released "
1012  "July 21, 2010). ")
1013 
1014  if lapack_driver is None:
1015  # restart with gelss
1016  lstsq.default_lapack_driver = 'gelss'
1017  mesg += "Falling back to 'gelss' driver."
1018  warnings.warn(mesg, RuntimeWarning)
1019  return lstsq(a, b, cond, overwrite_a, overwrite_b,
1020  check_finite, lapack_driver='gelss')
1021

1023  mesg += ("Use a different lapack_driver when calling lstsq"
1024  " or upgrade LAPACK.")
1025  raise LstsqLapackError(mesg)
1026 
1027  x, s, rank, info = lapack_func(a1, b1, lwork,
1028  iwork, cond, False, False)
1029  else: # complex data
1030  lwork, rwork, iwork = _compute_lwork(lapack_lwork, m, n,
1031  nrhs, cond)
1032  x, s, rank, info = lapack_func(a1, b1, lwork, rwork, iwork,
1033  cond, False, False)
1034  if info > 0:
1035  raise LinAlgError("SVD did not converge in Linear Least Squares")
1036  if info < 0:
1037  raise ValueError('illegal value in %d-th argument of internal %s'
1038  % (-info, lapack_driver))

1040  if m > n:
1041  x1 = x[:n]
1042  if rank == n:
1043  resids = np.sum(np.abs(x[n:])**2, axis=0)
1044  x = x1

1048  lwork = _compute_lwork(lapack_lwork, m, n, nrhs, cond)

1050  v, x, j, rank, info = lapack_func(a1, b1, jptv, cond,
1051  lwork, False, False)
1052  if info < 0:
1053  raise ValueError("illegal value in %d-th argument of internal "
1054  "gelsy" % -info)
1055  if m > n:
1056  x1 = x[:n]
1057  x = x1
1058  return x, np.array([], x.dtype), rank, None

1121  """
1122  Compute the (Moore-Penrose) pseudo-inverse of a matrix.
1123 
1124  Calculate a generalized inverse of a matrix using its
1125  singular-value decomposition and including all 'large' singular
1126  values.
1127 
1128  Parameters
1129  ----------
1130  a : (M, N) array_like

1136  If None or -1, suitable machine precision is used.
1137  return_rank : bool, optional
1138  if True, return the effective rank of the matrix
1139  check_finite : bool, optional
1140  Whether to check that the input matrix contains only finite numbers.
1141  Disabling may give a performance gain, but may result in problems
1142  (crashes, non-termination) if the inputs do contain infinities or NaNs.
1143 
1144  Returns
1145  -------
1146  B : (N, M) ndarray
1147  The pseudo-inverse of matrix `a`.
1148  rank : int
1149  The effective rank of the matrix. Returned if return_rank == True
1150 
1151  Raises
1152  ------
1153  LinAlgError
1154  If SVD computation does not converge.
1155 
1156  Examples
1157  --------
1158  >>> from scipy import linalg
1159  >>> a = np.random.randn(9, 6)
1160  >>> B = linalg.pinv2(a)
1161  >>> np.allclose(a, np.dot(a, np.dot(B, a)))
1162  True
1163  >>> np.allclose(B, np.dot(B, np.dot(a, B)))
1164  True

1170  if rcond is not None:
1171  cond = rcond
1172  if cond in [None, -1]:

1174  factor = {'f': 1E3, 'd': 1E6}
1175  cond = factor[t] * np.finfo(t).eps

1266  overwrite_a=False):
1267  """
1268  A wrapper around LAPACK's xGEBAL routine family for matrix balancing.
1269 
1270  The balancing tries to equalize the row and column 1-norms by applying
1271  a similarity transformation such that the magnitude variation of the
1272  matrix entries is reflected to the scaling matrices.
1273 
1274  Moreover, if enabled, the matrix is first permuted to isolate the upper
1275  triangular parts of the matrix and, again if scaling is also enabled,
1276  only the remaining subblocks are subjected to scaling.
1277 
1278  The balanced matrix satisfies the following equality
1279 
1280  .. math::
1281 
1282  B = T^{-1} A T
1283 
1284  The scaling coefficients are approximated to the nearest power of 2
1285  to avoid round-off errors.
1286 
1287  Parameters
1288  ----------
1289  A : (n, n) array_like
1290  Square data matrix for the balancing.
1291  permute : bool, optional
1292  The selector to define whether permutation of A is also performed
1293  prior to scaling.
1294  scale : bool, optional
1295  The selector to turn on and off the scaling. If False, the matrix
1296  will not be scaled.
1297  separate : bool, optional

1299  to a tuple of two separate 1D permutation and scaling arrays.
1300  overwrite_a : bool, optional
1301  This is passed to xGEBAL directly. Essentially, overwrites the result
1302  to the data. It might increase the space efficiency. See LAPACK manual
1303  for details. This is False by default.

1307  B : (n, n) ndarray
1308  Balanced matrix

1314  ``T`` above, the scaling and the permutation vector is given

1324  LAPACK routines.

1327  been modified to account for special cases. See [2]_ for details
1328  which have been implemented since LAPACK v3.5.0. Before this version

1330  conditioning. See [3]_ for such examples.
