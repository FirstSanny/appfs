1 """
2 ====================================================================
3 K-means clustering and vector quantization (:mod:`scipy.cluster.vq`)
4 ====================================================================
5 
6 Provides routines for k-means clustering, generating code books
7 from k-means models, and quantizing vectors by comparing them with
8 centroids in a code book.
9 
10 .. autosummary::
11  :toctree: generated/
12 
13  whiten -- Normalize a group of observations so each feature has unit variance
14  vq -- Calculate code book membership of a set of observation vectors
15  kmeans -- Performs k-means on a set of observation vectors forming k clusters
16  kmeans2 -- A different implementation of k-means with more methods
17  -- for initializing centroids
18 
19 Background information
20 ======================
21 The k-means algorithm takes as input the number of clusters to
22 generate, k, and a set of observation vectors to cluster. It
23 returns a set of centroids, one for each of the k clusters. An
24 observation vector is classified with the cluster number or
25 centroid index of the centroid closest to it.
26 
27 A vector v belongs to cluster i if it is closer to centroid i than
28 any other centroids. If v belongs to i, we say centroid i is the
29 dominating centroid of v. The k-means algorithm tries to
30 minimize distortion, which is defined as the sum of the squared distances
31 between each observation vector and its dominating centroid. Each
32 step of the k-means algorithm refines the choices of centroids to
33 reduce distortion. The change in distortion is used as a
34 stopping criterion: when the change is lower than a threshold, the
35 k-means algorithm is not making sufficient progress and
36 terminates. One can also define a maximum number of iterations.
37 
38 Since vector quantization is a natural application for k-means,
39 information theory terminology is often used. The centroid index
40 or cluster index is also referred to as a "code" and the table
41 mapping codes to centroids and vice versa is often referred as a
42 "code book". The result of k-means, a set of centroids, can be
43 used to quantize vectors. Quantization aims to find an encoding of
44 vectors that reduces the expected distortion.
45 
46 All routines expect obs to be a M by N array where the rows are
47 the observation vectors. The codebook is a k by N array where the
48 i'th row is the centroid of code word i. The observation vectors
49 and centroids have the same feature dimension.
50 
51 As an example, suppose we wish to compress a 24-bit color image
52 (each pixel is represented by one byte for red, one for blue, and
53 one for green) before sending it over the web. By using a smaller
54 8-bit encoding, we can reduce the amount of data by two
55 thirds. Ideally, the colors for each of the 256 possible 8-bit
56 encoding values should be chosen to minimize distortion of the
57 color. Running k-means with k=256 generates a code book of 256
58 codes, which fills up all possible 8-bit sequences. Instead of
59 sending a 3-byte value for each pixel, the 8-bit centroid index
60 (or code word) of the dominating centroid is transmitted. The code
61 book is also sent over the wire so each 8-bit code can be
62 translated back to a 24-bit pixel value representation. If the
63 image of interest was of an ocean, we would expect many 24-bit
64 blues to be represented by 8-bit codes. If it was an image of a
65 human face, more flesh tone colors would be represented in the
66 code book.
67

95 def whiten(obs, check_finite=True):
96  """
97  Normalize a group of observations on a per feature basis.
98 
99  Before running k-means, it is beneficial to rescale each feature
100  dimension of the observation set with whitening. Each feature is
101  divided by its standard deviation across all observations to give
102  it unit variance.
103 
104  Parameters
105  ----------
106  obs : ndarray
107  Each row of the array is an observation. The
108  columns are the features seen during each observation.
109 
110  >>> # f0 f1 f2
111  >>> obs = [[ 1., 1., 1.], #o0
112  ... [ 2., 2., 2.], #o1
113  ... [ 3., 3., 3.], #o2
114  ... [ 4., 4., 4.]] #o3
115 
116  check_finite : bool, optional
117  Whether to check that the input matrices contain only finite numbers.
118  Disabling may give a performance gain, but may result in problems
119  (crashes, non-termination) if the inputs do contain infinities or NaNs.
120  Default: True
121 
122  Returns
123  -------
124  result : ndarray
125  Contains the values in `obs` scaled by the standard deviation
126  of each column.
127 
128  Examples
129  --------
130  >>> from scipy.cluster.vq import whiten
131  >>> features = np.array([[1.9, 2.3, 1.7],
132  ... [1.5, 2.5, 2.2],
133  ... [0.8, 0.6, 1.7,]])
134  >>> whiten(features)
135  array([[ 4.17944278, 2.69811351, 7.21248917],
136  [ 3.29956009, 2.93273208, 9.33380951],
137  [ 1.75976538, 0.7038557 , 7.21248917]])
138 
139  """
140  obs = _asarray_validated(obs, check_finite=check_finite)
141  std_dev = np.std(obs, axis=0)
142  zero_std_mask = std_dev == 0
143  if zero_std_mask.any():
144  std_dev[zero_std_mask] = 1.0
145  warnings.warn("Some columns have standard deviation zero. "
146  "The values of these columns will not change.",
147  RuntimeWarning)
148  return obs / std_dev

151 def vq(obs, code_book, check_finite=True):
152  """
153  Assign codes from a code book to observations.
154 
155  Assigns a code from a code book to each observation. Each
156  observation vector in the 'M' by 'N' `obs` array is compared with the
157  centroids in the code book and assigned the code of the closest
158  centroid.
159 
160  The features in `obs` should have unit variance, which can be
161  achieved by passing them through the whiten function. The code
162  book can be created with the k-means algorithm or a different
163  encoding algorithm.
164 
165  Parameters
166  ----------
167  obs : ndarray
168  Each row of the 'M' x 'N' array is an observation. The columns are
169  the "features" seen during each observation. The features must be
170  whitened first using the whiten function or something equivalent.
171  code_book : ndarray
172  The code book is usually generated using the k-means algorithm.
173  Each row of the array holds a different code, and the columns are
174  the features of the code.
175 
176  >>> # f0 f1 f2 f3
177  >>> code_book = [
178  ... [ 1., 2., 3., 4.], #c0
179  ... [ 1., 2., 3., 4.], #c1
180  ... [ 1., 2., 3., 4.]] #c2
181 
182  check_finite : bool, optional
183  Whether to check that the input matrices contain only finite numbers.
184  Disabling may give a performance gain, but may result in problems
185  (crashes, non-termination) if the inputs do contain infinities or NaNs.
186  Default: True
187 
188  Returns
189  -------
190  code : ndarray
191  A length M array holding the code book index for each observation.
192  dist : ndarray
193  The distortion (distance) between the observation and its nearest
194  code.
195 
196  Examples
197  --------
198  >>> from numpy import array
199  >>> from scipy.cluster.vq import vq
200  >>> code_book = array([[1.,1.,1.],
201  ... [2.,2.,2.]])
202  >>> features = array([[ 1.9,2.3,1.7],
203  ... [ 1.5,2.5,2.2],
204  ... [ 0.8,0.6,1.7]])
205  >>> vq(features,code_book)
206  (array([1, 1, 0],'i'), array([ 0.43588989, 0.73484692, 0.83066239]))
207 
208  """
209  obs = _asarray_validated(obs, check_finite=check_finite)
210  code_book = _asarray_validated(code_book, check_finite=check_finite)
211  ct = np.common_type(obs, code_book)
212 
213  c_obs = obs.astype(ct, copy=False)
214 
215  if code_book.dtype != ct:
216  c_code_book = code_book.astype(ct)
217  else:
218  c_code_book = code_book
219 
220  if ct in (np.float32, np.float64):
221  results = _vq.vq(c_obs, c_code_book)
222  else:
223  results = py_vq(obs, code_book)
224  return results

227 def py_vq(obs, code_book, check_finite=True):
228  """ Python version of vq algorithm.
229 
230  The algorithm computes the euclidian distance between each
231  observation and every frame in the code_book.
232 
233  Parameters
234  ----------
235  obs : ndarray
236  Expects a rank 2 array. Each row is one observation.
237  code_book : ndarray
238  Code book to use. Same format than obs. Should have same number of
239  features (eg columns) than obs.
240  check_finite : bool, optional
241  Whether to check that the input matrices contain only finite numbers.
242  Disabling may give a performance gain, but may result in problems
243  (crashes, non-termination) if the inputs do contain infinities or NaNs.
244  Default: True
245 
246  Returns
247  -------
248  code : ndarray
249  code[i] gives the label of the ith obversation, that its code is
250  code_book[code[i]].
251  mind_dist : ndarray
252  min_dist[i] gives the distance between the ith observation and its
253  corresponding code.
254 
255  Notes
256  -----
257  This function is slower than the C version but works for
258  all input types. If the inputs have the wrong types for the
259  C versions of the function, this one is called as a last resort.
260 
261  It is about 20 times slower than the C version.
262 
263  """
264  obs = _asarray_validated(obs, check_finite=check_finite)
265  code_book = _asarray_validated(code_book, check_finite=check_finite)
266 
267  # n = number of observations
268  # d = number of features
269  if np.ndim(obs) == 1:
270  if not np.ndim(obs) == np.ndim(code_book):
271  raise ValueError(
272  "Observation and code_book should have the same rank")
273  else:
274  return _py_vq_1d(obs, code_book)
275  else:
276  (n, d) = np.shape(obs)
277 
278  # code books and observations should have same number of features and same
279  # shape
280  if not np.ndim(obs) == np.ndim(code_book):
281  raise ValueError("Observation and code_book should have the same rank")
282  elif not d == code_book.shape[1]:
283  raise ValueError("Code book(%d) and obs(%d) should have the same "
284  "number of features (eg columns)""" %
285  (code_book.shape[1], d))
286 
287  code = np.zeros(n, dtype=int)
288  min_dist = np.zeros(n)
289  for i in range(n):
290  dist = np.sum((obs[i] - code_book) ** 2, 1)
291  code[i] = np.argmin(dist)
292  min_dist[i] = dist[code[i]]
293 
294  return code, np.sqrt(min_dist)

330 def py_vq2(obs, code_book, check_finite=True):
331  """2nd Python version of vq algorithm.
332 
333  The algorithm simply computes the euclidian distance between each
334  observation and every frame in the code_book/
335 
336  Parameters
337  ----------
338  obs : ndarray
339  Expect a rank 2 array. Each row is one observation.
340  code_book : ndarray
341  Code book to use. Same format than obs. Should have same number of
342  features (eg columns) than obs.
343  check_finite : bool, optional
344  Whether to check that the input matrices contain only finite numbers.
345  Disabling may give a performance gain, but may result in problems
346  (crashes, non-termination) if the inputs do contain infinities or NaNs.
347  Default: True
348 
349  Returns
350  -------
351  code : ndarray
352  code[i] gives the label of the ith obversation, that its code is
353  code_book[code[i]].
354  mind_dist : ndarray
355  min_dist[i] gives the distance between the ith observation and its
356  corresponding code.
357 
358  Notes
359  -----
360  This could be faster when number of codebooks is small, but it
361  becomes a real memory hog when codebook is large. It requires
362  N by M by O storage where N=number of obs, M = number of
363  features, and O = number of codes.
364 
365  """

367  code_book = _asarray_validated(code_book, check_finite=check_finite)
368  d = np.shape(obs)[1]

380  # The next line I think is equivalent and should be faster than the one
381  # above, but in practice didn't seem to make much difference:
382  # min_dist = choose(code,dist)
383  return code, min_dist

386 def _kmeans(obs, guess, thresh=1e-5):
387  """ "raw" version of k-means.
388 
389  Returns
390  -------
391  code_book
392  the lowest distortion codebook found.
393  avg_dist
394  the average distance a observation is from a code in the book.
395  Lower means the code_book matches the data better.
396 
397  See Also
398  --------
399  kmeans : wrapper around k-means
400 
401  Examples
402  --------
403  Note: not whitened in this example.
404 
405  >>> from numpy import array
406  >>> from scipy.cluster.vq import _kmeans
407  >>> features = array([[ 1.9,2.3],
408  ... [ 1.5,2.5],
409  ... [ 0.8,0.6],
410  ... [ 0.4,1.8],
411  ... [ 1.0,1.0]])
412  >>> book = array((features[0],features[2]))
413  >>> _kmeans(features,book)
414  (array([[ 1.7 , 2.4 ],
415  [ 0.73333333, 1.13333333]]), 0.40563916697728591)
416 
417  """
418 
419  code_book = np.array(guess, copy=True)
420  avg_dist = []
421  diff = np.inf
422  while diff > thresh:
423  nc = code_book.shape[0]
424  # compute membership and distances between obs and code_book
425  obs_code, distort = vq(obs, code_book)
426  avg_dist.append(np.mean(distort, axis=-1))
427  # recalc code_book as centroids of associated obs
428  if(diff > thresh):
429  code_book, has_members = _vq.update_cluster_means(obs, obs_code, nc)
430  code_book = code_book.compress(has_members, axis=0)
431  if len(avg_dist) > 1:
432  diff = avg_dist[-2] - avg_dist[-1]
433 
434  return code_book, avg_dist[-1]

437 def kmeans(obs, k_or_guess, iter=20, thresh=1e-5, check_finite=True):
438  """
439  Performs k-means on a set of observation vectors forming k clusters.
440 
441  The k-means algorithm adjusts the centroids until sufficient
442  progress cannot be made, i.e. the change in distortion since
443  the last iteration is less than some threshold. This yields
444  a code book mapping centroids to codes and vice versa.
445 
446  Distortion is defined as the sum of the squared differences
447  between the observations and the corresponding centroid.
448 
449  Parameters
450  ----------
451  obs : ndarray
452  Each row of the M by N array is an observation vector. The
453  columns are the features seen during each observation.
454  The features must be whitened first with the `whiten` function.
455 
456  k_or_guess : int or ndarray
457  The number of centroids to generate. A code is assigned to
458  each centroid, which is also the row index of the centroid
459  in the code_book matrix generated.
460 
461  The initial k centroids are chosen by randomly selecting
462  observations from the observation matrix. Alternatively,
463  passing a k by N array specifies the initial k centroids.
464 
465  iter : int, optional
466  The number of times to run k-means, returning the codebook
467  with the lowest distortion. This argument is ignored if
468  initial centroids are specified with an array for the
469  ``k_or_guess`` parameter. This parameter does not represent the
470  number of iterations of the k-means algorithm.
471 
472  thresh : float, optional
473  Terminates the k-means algorithm if the change in
474  distortion since the last k-means iteration is less than
475  or equal to thresh.
476 
477  check_finite : bool, optional
478  Whether to check that the input matrices contain only finite numbers.
479  Disabling may give a performance gain, but may result in problems
480  (crashes, non-termination) if the inputs do contain infinities or NaNs.
481  Default: True
482 
483  Returns
484  -------
485  codebook : ndarray
486  A k by N array of k centroids. The i'th centroid
487  codebook[i] is represented with the code i. The centroids
488  and codes generated represent the lowest distortion seen,
489  not necessarily the globally minimal distortion.
490 
491  distortion : float
492  The distortion between the observations passed and the
493  centroids generated.
494 
495  See Also
496  --------
497  kmeans2 : a different implementation of k-means clustering
498  with more methods for generating initial centroids but without
499  using a distortion change threshold as a stopping criterion.
500 
501  whiten : must be called prior to passing an observation matrix
502  to kmeans.
503 
504  Examples
505  --------
506  >>> from numpy import array
507  >>> from scipy.cluster.vq import vq, kmeans, whiten
508  >>> features = array([[ 1.9,2.3],
509  ... [ 1.5,2.5],
510  ... [ 0.8,0.6],
511  ... [ 0.4,1.8],
512  ... [ 0.1,0.1],
513  ... [ 0.2,1.8],
514  ... [ 2.0,0.5],
515  ... [ 0.3,1.5],
516  ... [ 1.0,1.0]])
517  >>> whitened = whiten(features)
518  >>> book = np.array((whitened[0],whitened[2]))
519  >>> kmeans(whitened,book)
520  (array([[ 2.3110306 , 2.86287398], # random
521  [ 0.93218041, 1.24398691]]), 0.85684700941625547)
522 
523  >>> from numpy import random
524  >>> random.seed((1000,2000))
525  >>> codes = 3
526  >>> kmeans(whitened,codes)
527  (array([[ 2.3110306 , 2.86287398], # random
528  [ 1.32544402, 0.65607529],
529  [ 0.40782893, 2.02786907]]), 0.5196582527686241)
530 
531  """
532  obs = _asarray_validated(obs, check_finite=check_finite)
533  if int(iter) < 1:
534  raise ValueError('iter must be at least 1.')
535 
536  # Determine whether a count (scalar) or an initial guess (array) was passed.
537  k = None
538  guess = None
539  try:
540  k = int(k_or_guess)
541  except TypeError:
542  guess = _asarray_validated(k_or_guess, check_finite=check_finite)
543 
544  if guess is not None:
545  if guess.size < 1:
546  raise ValueError("Asked for 0 cluster ? initial book was %s" %
547  guess)
548  result = _kmeans(obs, guess, thresh=thresh)
549  else:
550  if k != k_or_guess:
551  raise ValueError('if k_or_guess is a scalar, it must be an integer')
552  # initialize best distance value to a large value
553  best_dist = np.inf
554  No = obs.shape[0]
555  k = k_or_guess
556  if k < 1:
557  raise ValueError("Asked for 0 cluster ? ")
558  for i in range(iter):
559  # the initial code book is randomly selected from observations
560  k_random_indices = np.random.randint(0, No, k)
561  if np.any(_numpy_compat.unique(k_random_indices,
562  return_counts=True)[1] > 1):
563  # randint can give duplicates, which is incorrect. Only fix
564  # the issue if it occurs, to not change results for users who
565  # use a random seed and get no duplicates.
566  k_random_indices = np.random.permutation(No)[:k]
567 
568  guess = np.take(obs, k_random_indices, 0)
569  book, dist = _kmeans(obs, guess, thresh=thresh)
570  if dist < best_dist:
571  best_book = book
572  best_dist = dist
573  result = best_book, best_dist
574  return result

577 def _kpoints(data, k):
578  """Pick k points at random in data (one row = one observation).
579 
580  This is done by taking the k first values of a random permutation of 1..N
581  where N is the number of observation.
582 
583  Parameters
584  ----------
585  data : ndarray
586  Expect a rank 1 or 2 array. Rank 1 are assumed to describe one
587  dimensional data, rank 2 multidimensional data, in which case one
588  row is one observation.
589  k : int
590  Number of samples to generate.
591 
592  """
593  if data.ndim > 1:
594  n = data.shape[0]
595  else:
596  n = data.size
597 
598  p = np.random.permutation(n)
599  x = data[p[:k], :].copy()
600 
601  return x

604 def _krandinit(data, k):
605  """Returns k samples of a random variable which parameters depend on data.
606 
607  More precisely, it returns k observations sampled from a Gaussian random
608  variable which mean and covariances are the one estimated from data.
609 
610  Parameters
611  ----------
612  data : ndarray
613  Expect a rank 1 or 2 array. Rank 1 are assumed to describe one
614  dimensional data, rank 2 multidimensional data, in which case one
615  row is one observation.
616  k : int
617  Number of samples to generate.
618 
619  """
620  def init_rank1(data):
621  mu = np.mean(data)
622  cov = np.cov(data)
623  x = np.random.randn(k)
624  x *= np.sqrt(cov)
625  x += mu
626  return x
627 
628  def init_rankn(data):
629  mu = np.mean(data, 0)
630  cov = np.atleast_2d(np.cov(data, rowvar=0))
631 
632  # k rows, d cols (one row = one obs)
633  # Generate k sample of a random variable ~ Gaussian(mu, cov)
634  x = np.random.randn(k, mu.size)
635  x = np.dot(x, np.linalg.cholesky(cov).T) + mu
636  return x
637 
638  def init_rank_def(data):
639  # initialize when the covariance matrix is rank deficient
640  mu = np.mean(data, axis=0)
641  _, s, vh = np.linalg.svd(data - mu, full_matrices=False)
642  x = np.random.randn(k, s.size)
643  sVh = s[:, None] * vh / np.sqrt(data.shape[0] - 1)
644  x = np.dot(x, sVh) + mu
645  return x
646 
647  nd = np.ndim(data)
648  if nd == 1:
649  return init_rank1(data)
650  elif data.shape[1] > data.shape[0]:
651  return init_rank_def(data)
652  else:
653  return init_rankn(data)
