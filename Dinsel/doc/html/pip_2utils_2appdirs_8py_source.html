14 def user_cache_dir(appname):
15  r"""
16  Return full path to the user-specific cache dir for this application.
17 
18  "appname" is the name of application.
19 
20  Typical user cache directories are:
21  macOS: ~/Library/Caches/<AppName>
22  Unix: ~/.cache/<AppName> (XDG default)
23  Windows: C:\Users<username>\AppData\Local<AppName>\Cache
24 
25  On Windows the only suggestion in the MSDN docs is that local settings go
26  in the `CSIDL_LOCAL_APPDATA` directory. This is identical to the
27  non-roaming app data dir (the default returned by `user_data_dir`). Apps
28  typically put cache data somewhere *under* the given dir here. Some
29  examples:
30  ...\Mozilla\Firefox\Profiles<ProfileName>\Cache
31  ...\Acme\SuperApp\Cache\1.0
32 
33  OPINION: This function appends "Cache" to the `CSIDL_LOCAL_APPDATA` value.
34  """
35  if WINDOWS:
36  # Get the base path
37  path = os.path.normpath(_get_win_folder("CSIDL_LOCAL_APPDATA"))
38 
39  # When using Python 2, return paths as bytes on Windows like we do on
40  # other operating systems. See helper function docs for more details.
41  if PY2 and isinstance(path, text_type):
42  path = _win_path_to_bytes(path)
43 
44  # Add our app name and Cache directory to it
45  path = os.path.join(path, appname, "Cache")
46  elif sys.platform == "darwin":
47  # Get the base path
48  path = expanduser("~/Library/Caches")
49 
50  # Add our app name to it
51  path = os.path.join(path, appname)
52  else:
53  # Get the base path
54  path = os.getenv("XDG_CACHE_HOME", expanduser("~/.cache"))
55 
56  # Add our app name to it
57  path = os.path.join(path, appname)
58 
59  return path

62 def user_data_dir(appname, roaming=False):
63  """
64  Return full path to the user-specific data dir for this application.
65 
66  "appname" is the name of application.
67  If None, just the system directory is returned.
68  "roaming" (boolean, default False) can be set True to use the Windows
69  roaming appdata directory. That means that for users on a Windows
70  network setup for roaming profiles, this user data will be
71  sync'd on login. See
72  <http://technet.microsoft.com/en-us/library/cc766489(WS.10).aspx>
73  for a discussion of issues.
74 
75  Typical user data directories are:
76  macOS: ~/Library/Application Support/<AppName>
77  Unix: ~/.local/share/<AppName> # or in
78  $XDG_DATA_HOME, if defined
79  Win XP (not roaming): C:\Documents and Settings<username>\ ...
80  ...Application Data<AppName>
81  Win XP (roaming): C:\Documents and Settings<username>\Local ...
82  ...Settings\Application Data<AppName>
83  Win 7 (not roaming): C:\\Users<username>\AppData\Local<AppName>
84  Win 7 (roaming): C:\\Users<username>\AppData\Roaming<AppName>
85 
86  For Unix, we follow the XDG spec and support $XDG_DATA_HOME.
87  That means, by default "~/.local/share/<AppName>".
88  """
89  if WINDOWS:
90  const = roaming and "CSIDL_APPDATA" or "CSIDL_LOCAL_APPDATA"
91  path = os.path.join(os.path.normpath(_get_win_folder(const)), appname)
92  elif sys.platform == "darwin":
93  path = os.path.join(
94  expanduser('~/Library/Application Support/'),
95  appname,
96  )
97  else:
98  path = os.path.join(
99  os.getenv('XDG_DATA_HOME', expanduser("~/.local/share")),
100  appname,
101  )
102 
103  return path

106 def user_config_dir(appname, roaming=True):
107  """Return full path to the user-specific config dir for this application.
108 
109  "appname" is the name of application.
110  If None, just the system directory is returned.
111  "roaming" (boolean, default True) can be set False to not use the
112  Windows roaming appdata directory. That means that for users on a
113  Windows network setup for roaming profiles, this user data will be
114  sync'd on login. See
115  <http://technet.microsoft.com/en-us/library/cc766489(WS.10).aspx>
116  for a discussion of issues.
117 
118  Typical user data directories are:
119  macOS: same as user_data_dir
120  Unix: ~/.config/<AppName>
121  Win *: same as user_data_dir
122 
123  For Unix, we follow the XDG spec and support $XDG_CONFIG_HOME.
124  That means, by default "~/.config/<AppName>".
125  """
126  if WINDOWS:
127  path = user_data_dir(appname, roaming=roaming)
128  elif sys.platform == "darwin":
129  path = user_data_dir(appname)
130  else:
131  path = os.getenv('XDG_CONFIG_HOME', expanduser("~/.config"))
132  path = os.path.join(path, appname)
133 
134  return path

139 def site_config_dirs(appname):
140  """Return a list of potential user-shared config dirs for this application.
141 
142  "appname" is the name of application.
143 
144  Typical user config directories are:
145  macOS: /Library/Application Support/<AppName>/
146  Unix: /etc or $XDG_CONFIG_DIRS[i]/<AppName>/ for each value in
147  $XDG_CONFIG_DIRS
148  Win XP: C:\Documents and Settings\All Users\Application ...
149  ...Data<AppName>\
150  Vista: (Fail! "C:\ProgramData" is a hidden *system* directory
151  on Vista.)
152  Win 7: Hidden, but writeable on Win 7:
153  C:\ProgramData<AppName>\
154  """
155  if WINDOWS:
156  path = os.path.normpath(_get_win_folder("CSIDL_COMMON_APPDATA"))
157  pathlist = [os.path.join(path, appname)]
158  elif sys.platform == 'darwin':
159  pathlist = [os.path.join('/Library/Application Support', appname)]
160  else:
161  # try looking in $XDG_CONFIG_DIRS
162  xdg_config_dirs = os.getenv('XDG_CONFIG_DIRS', '/etc/xdg')
163  if xdg_config_dirs:
164  pathlist = [
165  os.path.join(expanduser(x), appname)
166  for x in xdg_config_dirs.split(os.pathsep)
167  ]
168  else:
169  pathlist = []
170 
171  # always look in /etc directly as well
172  pathlist.append('/etc')
173 
174  return pathlist

179 def _get_win_folder_from_registry(csidl_name):
180  """
181  This is a fallback technique at best. I'm not sure if using the
182  registry for this guarantees us the correct answer for all CSIDL_*
183  names.
184  """
185  import _winreg
186 
187  shell_folder_name = {
188  "CSIDL_APPDATA": "AppData",

198  return directory

201 def _get_win_folder_with_ctypes(csidl_name):
202  csidl_const = {
203  "CSIDL_APPDATA": 26,
204  "CSIDL_COMMON_APPDATA": 35,
205  "CSIDL_LOCAL_APPDATA": 28,
206  }[csidl_name]
207 
208  buf = ctypes.create_unicode_buffer(1024)
209  ctypes.windll.shell32.SHGetFolderPathW(None, csidl_const, None, 0, buf)
210 
211  # Downgrade to short path name if have highbit chars. See
212  # <http://bugs.activestate.com/show_bug.cgi?id=85099>.
213  has_high_char = False
214  for c in buf:
215  if ord(c) > 255:
216  has_high_char = True
217  break
218  if has_high_char:
219  buf2 = ctypes.create_unicode_buffer(1024)
220  if ctypes.windll.kernel32.GetShortPathNameW(buf.value, buf2, 1024):
221  buf = buf2
222 
223  return buf.value

233 def _win_path_to_bytes(path):
234  """Encode Windows paths to bytes. Only used on Python 2.
235 
236  Motivation is to be consistent with other operating systems where paths
237  are also returned as bytes. This avoids problems mixing bytes and Unicode
238  elsewhere in the codebase. For more details and discussion see
239  <https://github.com/pypa/pip/issues/3463>.
240 
241  If encoding using ASCII and MBCS fails, return the original Unicode path.
242  """
243  for encoding in ('ASCII', 'MBCS'):
244  try:
245  return path.encode(encoding)
246  except (UnicodeEncodeError, LookupError):
247  pass
248  return path
